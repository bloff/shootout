    6 function Ack () {
    7     if [ $1 -eq 0 ]; then
    8         Ack=$[ $2 + 1 ]
    9     elif [ $2 -eq 0 ]; then
   10         Ack $[ $1 - 1 ] 1
   11     else
   12         Ack $1 $[ $2 - 1 ]
   13         local SubAck=$Ack
   14         Ack $[ $1 - 1 ] $SubAck
   15     fi
   16 }
   17 
   18 n=${1:-1}
   19 Ack 3 $n
   20 echo "Ack(3,$n): $Ack"
