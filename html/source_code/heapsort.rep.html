1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: heapsort.rep.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   (define IM     139968)
   16   (define IA       3877)
   17   (define IC      29573)
   18 
   19   (define LAST 42)
   20   (define (gen_random max)
   21     (/ (* max (setq LAST (modulo (+ (* LAST IA) IC) IM))) IM))
   22 
   23   (define (heapsort n ra)
   24     (let ((ir n)
   25       (l (+ (lsh n -1) 1))
   26       (i 0)
   27       (j 0)
   28       (rra 0.0))
   29       (catch 'return
   30     (while #t
   31       (cond ((> l 1)
   32          (setq rra (aref ra (setq l (- l 1)))))
   33         (t
   34          (setq rra (aref ra ir))
   35          (aset ra ir (aref ra 1))
   36          (setq ir (- ir 1))
   37          (cond ((= ir 1)
   38             (aset ra 1 rra)
   39             (throw 'return)))))
   40       (setq i l)
   41       (setq j (lsh l 1))
   42       (while (<= j ir)
   43         (cond ((and (< j ir) (< (aref ra j) (aref ra (+ j 1))))
   44            (setq j (+ j 1))))
   45         (cond ((< rra (aref ra j))
   46            (aset ra i (aref ra j))
   47            (setq j (+ j (setq i j))))
   48           (t
   49            (setq j (+ ir 1)))))
   50       (aset ra i rra)))))
   51 
   52   (let* ((arg (get-command-line-option "--num" t))
   53      (n (if arg (read-from-string arg) 1))
   54      (last (+ n 1))
   55      (ary (make-vector last 0)))
   56     (do ((i 1 (+ i 1)))
   57     ((= i last))
   58       (aset ary i (gen_random 1.0)))
   59     (heapsort n ary)
   60     (let ((result (aref ary n)))
   61       (setq result (/ (round (* result 10000000000)) 10000000000))
   62       (format standard-output "%12d\n" result))))
