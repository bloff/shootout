    5 -- uses `call' to catch errors; return the error message
    6 -- (or nil if there are no errors)
    7 
    8 function try (f, arg)
    9   local status, err = pcall(f, arg)
   10   if not status then return err end
   11 end
   12 
   13 local HI = 0
   14 local LO = 0
   15 
   16 function some_function (n)
   17   local res = try(hi_function, n)
   18   if res then print("We shouldn't get here: " .. res) end
   19 end
   20 
   21 
   22 function hi_function (n)
   23   local res = try(lo_function, n)
   24   if res == "Hi_Exception" then HI = HI+1
   25   elseif res then error(res, 0)  -- rethrow
   26   end
   27 end
   28 
   29 
   30 function lo_function (n)
   31   local res = try(blowup, n)
   32   if res == "Lo_Exception" then LO = LO+1
   33   elseif res then error(res, 0)  -- rethrow
   34   end
   35 end
   36 
   37 
   38 function blowup (n)
   39   if math.mod(n,2) ~= 0 then error("Lo_Exception", 0)
   40   else error("Hi_Exception",0)
   41   end
   42 end
   43 
   44 
   45 N = (arg and arg[1]) or 1
   46 for i=1,N do
   47   some_function(i)
   48 end
   49 
   50 print(string.format("Exceptions: HI=%d / LO=%d", HI, LO))
