    6 #include <iostream>
    7 #include <string>
    8 using namespace std;
    9 
   10 int main(int argc, char *argv[])
   11 {
   12     int i, n = ((argc == 2) ? atoi(argv[1]) : 1);
   13     string str;
   14     size_t capacity = 31;
   15     str.reserve(capacity); // as per C-string
   16     size_t newLength = 6;
   17     for (i = 0; i < n; i++)
   18     {
   19     if(newLength > capacity)
   20     {
   21         capacity *= 2;
   22         str.reserve(capacity);
   23     }
   24     str += "hello\n";
   25     newLength += 6;
   26     }
   27     cout << str.length() << endl;
   28     return 0;
   29 }
