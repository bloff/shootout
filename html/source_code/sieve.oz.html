    7 %%  Usage: start from command line with
    8 %%     ozc -x sieve.oz -o sieve.oz.exe
    9 %%     sieve.oz.exe 900
   10 
   11 functor
   12 import System Application
   13 
   14 define Args N Flags Start Stop in
   15 
   16     [Args] = {Application.getArgs plain}
   17     N = {String.toInt Args}
   18 
   19     Start = 2
   20     Stop = 8192
   21 
   22     Flags = {BitArray.new Start Stop}
   23     for I in Start..Stop do {BitArray.set Flags I} end
   24 
   25     for I in 1..N do
   26            for J in Start..Stop do
   27             if {BitArray.test Flags J} then
   28                 for K in J+J..Stop;J do {BitArray.clear Flags K} end
   29             end
   30         end
   31     end
   32 
   33    {System.showInfo "Count: "#{BitArray.card Flags}}
   34 
   35    {Application.exit 0}
   36 end
