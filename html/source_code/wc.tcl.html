    5 # this program is modified from:
    6 #   http://cm.bell-labs.com/cm/cs/who/bwk/interps/pap.html
    7 # Timing Trials, or, the Trials of Timing: Experiments with Scripting
    8 # and User-Interface Languages</a> by Brian W. Kernighan and
    9 # Christopher J. Van Wyk.
   10 
   11 # Modified by Miguel Sofer
   12 
   13 proc main {} {
   14     set nl 0
   15     set nc 0
   16     set nw 0
   17 
   18     while {1} {
   19     set data [read stdin 4096]
   20     if {![string length $data]} {break}
   21     if {[gets stdin extra] >= 0} {
   22         append data $extra
   23         incr nc
   24     }
   25     incr nc [string length $data]
   26     incr nw [regexp -all {(?:^|\s)\S} $data]
   27     incr nl [regexp -all -line {^} $data]
   28     }
   29     puts "$nl $nw $nc"
   30 }
   31 
   32 main
