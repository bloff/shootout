   17 Start world   
   18    # (console, world) = stdio world 
   19    #! (nl, nw, nc) = countchars console 0 0 0 False 
   20    = toString nl +++ " " +++ toString nw +++ " " +++ toString nc +++ "\n"

   22 countchars :: !*File !Int !Int !Int !.Bool -> .(Int,Int,Int) 
   23 countchars f nc nl nw insideWord 
   24    # (ok,c,f) = freadc f 
   25    | not ok = (nl, nw, nc) 
   26    #! (insideWord, nw) = wordcount c 
   27    = countchars f (nc+1) (linecount c) nw insideWord 
   28     
   29       where  
   30       linecount c = if (c=='\n') (nl+1) nl 
   31        
   32       wordcount c =  
   33          if (c==' '|| c=='\r' || c=='\n' || c=='\t')  
   34             (False, nw) 
   35             (if (not insideWord)(True,nw+1)(insideWord,nw))
