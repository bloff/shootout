    4 import java.io.*;
    5 import java.util.*;
    6 import java.text.*;
    7 
    8 public class sumcol {
    9     public static void main(String[] args) {
   10     int sum = 0;
   11     String line;
   12         try {
   13             BufferedReader in = new BufferedReader(new InputStreamReader(
   14                                 System.in));
   15             while ((line = in.readLine()) != null) {
   16         sum = sum + Integer.parseInt(line);
   17             }
   18         } catch (IOException e) {
   19             System.err.println(e);
   20             return;
   21         }
   22     System.out.println(Integer.toString(sum));
   23     }
   24 }
