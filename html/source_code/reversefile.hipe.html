    5 -module(reversefile).
    6 -export([main/0, main/1]).
    7 
    8 main() -> main(['1']).
    9 main(Args) ->
   10     Port = open_port({fd, 0, 1}, [eof, {line, 128}]),
   11     read_and_print(Port),
   12     halt(0).
   13 
   14 read_and_print(Port) ->
   15     receive
   16     {Port, {_, {_, Line}}} ->
   17         read_and_print(Port),
   18         io:fwrite([Line, $\n]);
   19     {Port, eof} ->
   20         ok
   21     end.
