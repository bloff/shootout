    5 ;;; $Id: prodcons.guile.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $

    7 
    8 (use-modules (ice-9 threads))
    9 
   10 (define mutex (make-mutex))
   11 (define access (make-condition-variable))
   12 (define count 0)
   13 (define data 0)
   14 (define produced 0)
   15 (define consumed 0)
   16 
   17 ;; the consumer thread definition seems wrong
   18 ;; how does it ever stop/get joined?
   19 (define (consumer n)
   20   (let ((i 0))
   21     (while #t
   22      (lock-mutex mutex)
   23      (while (= count 0)
   24         (wait-condition-variable access mutex))
   25      (set! i data)
   26      (set! count 0)
   27      (signal-condition-variable access)
   28      (unlock-mutex mutex)
   29      (set! consumed (+ consumed 1)))))
   30 
   31 (define (producer n)
   32   (do ((i 1 (+ i 1)))
   33       ((> i n))
   34     (lock-mutex mutex)
   35     (while (= count 1)
   36       (wait-condition-variable access mutex))
   37     (set! data i)
   38     (set! count 1)
   39     (signal-condition-variable access)
   40     (unlock-mutex mutex)
   41     (set! produced (+ produced 1))))
   42 
   43 (define (main args)
   44   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   45     (let ((c (make-thread (lambda () (consumer n)))))
   46       (producer n)
   47       (join-thread c)
   48       (display produced) (display " ") (display consumed) (newline))))
