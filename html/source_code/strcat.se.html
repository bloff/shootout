    6 class STRCAT
    7 
    8 creation
    9     make
   10 
   11 feature
   12 
   13     concat_string : STRING is "hello%N";
   14 
   15     make is
   16         local
   17             i,n : INTEGER;
   18             str: STRING;
   19         do
   20             if argument_count = 1 then
   21                 n := argument(1).to_integer
   22             else
   23                 n := 1
   24             end
   25 
   26             from i := 1
   27                 !!str.make(100);
   28             until i > n
   29             loop
   30                 str.append(concat_string);
   31                 i := i + 1;
   32             end;
   33             io.put_integer(str.count);
   34             io.put_character ('%N');
   35         sleep(1);
   36         end;
   37 
   38 feature {NONE}
   39 
   40    -- helper function to call Unix sleep().
   41    -- we do this so we can make the process hang around long
   42    -- enough so the parent can measure its memory size.
   43    sleep(i: INTEGER) is
   44       external "C"
   45       end;
   46 
   47 end
