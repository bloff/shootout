    5 class HASH
    6 
    7 creation make
    8 
    9 feature -- Initialization
   10 
   11    make is
   12       local
   13      n : INTEGER
   14      i, j, c : INTEGER
   15      s : STRING
   16      ht : DICTIONARY[INTEGER, STRING]
   17 
   18      arg: ARGUMENTS
   19       do
   20      n := argument(1).to_integer
   21 
   22      !!ht.with_capacity(n);
   23 
   24      from  i := 1
   25      until i > n
   26      loop
   27         s := i.to_hexadecimal
   28         from  j := 1
   29         until s @ j /= '0'
   30         loop  j := j + 1
   31         end
   32         ht.put (i, s.substring(j,s.count))
   33         i := i+1
   34      end
   35 
   36      from  i := n
   37      until i = 0
   38      loop
   39         s := i.to_string
   40          if ht.has (s) then
   41            c := c + 1
   42         end
   43         i := i - 1
   44      end
   45 
   46      print (c.out + "%N")
   47 
   48       end
   49 
   50 end -- class HASH
