    7 # List.
    8 - : int = 45

   10 
   11 open List
   12 open Printf
   13 
   14 let seq n =
   15   let rec _seq m l = if m = 0 then l else _seq (m - 1) (m::l) in
   16   _seq n []
   17 
   18 let pop_head l =
   19   let item = hd(!l) in l := tl(!l); item
   20 
   21 let push_head l item =
   22   l := item :: !l
   23 
   24 let _ =
   25   let n =
   26     try int_of_string Sys.argv.(1)
   27     with Invalid_argument _ -> 1 in
   28 
   29   (* create list (A) of integers from 1 through N *)
   30   let a = ref (seq n)
   31 
   32   (* create empty list B *)
   33   and b = ref [] in
   34 
   35   (* move each individual item from A to B, in a loop, reversing order *)
   36   let rec loop a b =
   37     match !a with
   38       [] -> ()
   39     | _ -> begin push_head b (pop_head a) ; loop a b end
   40   in loop a b;
   41 
   42   (* print first 2 items in B *)
   43   printf "%d %d\n" (hd !b) (hd (tl !b));
   44 
   45   (* reverse B (can be done in place) *)
   46   b := rev !b;
   47 
   48   (* is 0 a member of B? *)
   49   if mem 0 !b then print_endline "true" else print_endline "false";
   50 
   51   (* is N a member of B? *)
   52   if mem n !b then print_endline "true" else print_endline "false";
   53 
   54   (* filter values from B to A that are less than N/2, preserving order *)
   55   let mid = n/2 in
   56   a := filter (fun x -> x < mid) !b;
   57 
   58   (* print first ten items of A *)
   59   let rec print_n n l =
   60     match l with
   61       [] -> ()
   62     | _ -> if n > 0 then begin
   63     printf "%d%s" (hd l) (if n > 1 then " " else "\n");
   64     print_n (n - 1) (tl l)
   65     end
   66   in print_n 10 !a;
   67 
   68   (* print sum of items in A that are less than 1000 *)
   69   printf "%d\n" (fold_left (fun ac i -> if i < 1000 then ac + i else ac) 0
   70                  !a);
   71 
   72   (* append B to end of A *)
   73   a := append !a !b;
   74 
   75   (* print length and last element of A *)
   76   printf "%d %d\n" (length !a) (fold_left (fun ac i -> i) 0 !a);
