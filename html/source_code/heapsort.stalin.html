    5 (define IM   139968)
    6 (define IA     3877)
    7 (define IC    29573)
    8 
    9 (define LAST 42)
   10 (define (gen_random max)
   11   (set! LAST (modulo (+ (* LAST IA) IC) IM))
   12   (/ (* max LAST) IM))
   13 
   14 (define (heapsort n ra)
   15   (let ((ir n)
   16     (l (+ (bit-rsh n 1) 1))
   17     (i 0)
   18     (j 0)
   19     (rra 0.0))
   20     (bind-exit (return)
   21            ;; how to do an infinite loop in bigloo?
   22            ;; (while #t
   23            (do ((bar #t))
   24            ((= 1 0))
   25          (cond ((> l 1)
   26             (set! l (- l 1))
   27             (set! rra (vector-ref ra l)))
   28                (else
   29             (set! rra (vector-ref ra ir))
   30             (vector-set! ra ir (vector-ref ra 1))
   31             (set! ir (- ir 1))
   32             (cond ((= ir 1)
   33                    (vector-set! ra 1 rra)
   34                    (return #t)))))
   35          (set! i l)
   36          (set! j (bit-lsh l 1))
   37          ;; how to do a while loop in bigloo?
   38          ;; (while (<= j ir)
   39          (do ((foo #t))
   40              ((> j ir))
   41            (cond ((and (< j ir) (< (vector-ref ra j) (vector-ref ra (+ j 1))))
   42               (set! j (+ j 1))))
   43            (cond ((< rra (vector-ref ra j))
   44               (vector-set! ra i (vector-ref ra j))
   45               (set! i j)
   46               (set! j (+ j i)))
   47              (else
   48               (set! j (+ ir 1)))))
   49          (vector-set! ra i rra)))))
   50 
   51 (define (main args)
   52   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   53      (last (+ n 1))
   54      (ary (make-vector last 0)))
   55     (do ((i 1 (+ i 1)))
   56     ((= i last))
   57       (vector-set! ary i (gen_random 1.0)))
   58     (heapsort n ary)
   59     (display (format "~,10F\n" (vector-ref ary n)))))
   60 
   61 (main (vector->list argv))
