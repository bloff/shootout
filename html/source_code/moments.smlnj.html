    8 
    9 structure Test : sig
   10     val main : (string * string list) -> OS.Process.status
   11 end = struct
   12 
   13 val ins = TextIO.stdIn
   14 
   15 fun loop (nums,sum) =
   16    case TextIO.inputLine ins of
   17       NONE => (nums,sum)
   18     | SOME(instr) =>
   19         (case instr of
   20               "" => (nums,sum)
   21         | l => (case Real.fromString l of
   22               NONE => raise Fail "invalid input"
   23             | SOME num => loop(num::nums,sum+num)))
   24 
   25 fun printl [] = print "\n" | printl(h::t) = ( print h ; printl t )
   26 
   27 fun r2s (x: real): string =
   28    if Real.== (x, 0.0) then "0.000000"
   29    else String.translate
   30      (fn #"~" => "-" | c => str c)
   31      (Real.fmt (StringCvt.FIX (SOME 6)) x)
   32 
   33 fun main(name, args) =  let
   34 
   35     val (nums,sum) = loop ([],0.0)
   36     val nums = Array.fromList nums
   37     val n = Array.length nums
   38     val n_float = real n
   39     val mean = sum / n_float
   40 
   41     fun moments (x,{average_deviation,variance,skew,kurtosis}) = let
   42       val deviation = x - mean
   43       val average_deviation =
   44     average_deviation + abs(deviation)
   45       val dev2 = deviation * deviation
   46       val variance = variance + dev2
   47       val dev3 = dev2 * deviation
   48       val skew = skew + dev3
   49     val dev4 = dev3 * deviation
   50     val kurtosis = kurtosis + dev4
   51     in {average_deviation=average_deviation,
   52     variance=variance,
   53     skew=skew,
   54     kurtosis=kurtosis}
   55     end
   56     val init = {average_deviation=0.0,
   57         variance=0.0,
   58         skew=0.0,
   59         kurtosis=0.0}
   60 
   61     val {average_deviation,variance,skew,kurtosis} =
   62       Array.foldl moments init nums
   63 
   64     val average_deviation = average_deviation / n_float
   65     val variance = variance /  real (n - 1);
   66     val standard_deviation = Real.Math.sqrt (variance)
   67     val {skew,kurtosis} =
   68       if variance > 0.0
   69     then {skew=skew / n_float / variance / standard_deviation,
   70           kurtosis=kurtosis / n_float / variance / variance - 3.0}
   71       else {skew=skew,kurtosis=kurtosis}
   72 
   73     val _ = ArrayQSort.sort Real.compare nums
   74     val mid = Int.quot (n, 2)
   75     val median =
   76       if Int.rem (n, 2) = 1
   77     then Array.sub (nums, mid)
   78       else (Array.sub (nums, mid) +
   79         Array.sub (nums, mid - 1)) / 2.0
   80 in
   81   printl ["n:                  ", Int.toString n, "\n",
   82       "median:             ", r2s median, "\n",
   83       "mean:               ", r2s mean, "\n",
   84       "average_deviation:  ", r2s average_deviation, "\n",
   85       "standard_deviation: ", r2s standard_deviation, "\n",
   86       "variance:           ", r2s variance, "\n",
   87       "skew:               ", r2s skew, "\n",
   88       "kurtosis:           ", r2s kurtosis];
   89   OS.Process.success
   90 end
   91 end
   92 
   93 val _ = SMLofNJ.exportFn("moments", Test.main);
