1 #!/usr/bin/guile \
    2 -e main -s
    3 !#
    4 ;;; $Id: ackermann.guile.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
    5 ;;; http://www.bagley.org/~doug/shootout/
    6 
    7 (debug-set! stack 0)
    8 
    9 (define (main args)
   10   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   11     (print-list "Ack(3," n "): " (ack 3 n))))
   12 
   13 (define (ack m n)
   14   (cond ((zero? m) (+ n 1))
   15     ((zero? n) (ack (- m 1) 1))
   16     (else      (ack (- m 1) (ack m (- n 1))))))
   17 
   18 (define (print-list . items) (for-each display items) (newline))
