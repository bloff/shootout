    6 import sys
    7 
    8 def main():
    9     nl = nw = nc = 0
   10     rl = sys.stdin.readlines
   11     lines = rl(4096)
   12 
   13     while lines:
   14         nl += len(lines)
   15         for line in lines:
   16             nc += len(line)
   17             nw += len(line.split())
   18         lines = rl(4096)
   19 
   20     print "%d %d %d" % (nl, nw, nc)
   21 
   22 main()
