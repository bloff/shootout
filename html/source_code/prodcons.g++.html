    5 #include <iostream>
    6 #include <stdio.h>
    7 #include <stdlib.h>
    8 #include <string.h>
    9 #include <unistd.h>
   10 #include <signal.h>
   11 #include <errno.h>
   12 #include <sys/types.h>
   13 #include <pthread.h>
   14 
   15 using namespace std;
   16 
   17 pthread_mutex_t mutex;
   18 pthread_cond_t control;
   19 void producer(int *arg);
   20 void consumer(int *arg);
   21 int pcount, data, consumed, produced;
   22 
   23 
   24 int
   25 main(int argc, char *argv[]) {
   26     int n = ((argc == 2) ? atoi(argv[1]) : 1);
   27     pthread_t t1, t2;
   28 
   29     pcount = data = consumed = produced = 0;
   30 
   31     if (pthread_mutex_init(&mutex, NULL)) {
   32     perror("pthread_mutex_init");
   33     exit(1);
   34     }
   35     if (pthread_cond_init(&control, NULL)) {
   36     perror("pthread_cond_init");
   37     exit(1);
   38     }
   39     if (pthread_create(&t1, (pthread_attr_t *)NULL,
   40                (void * (*)(void *))producer, (void *)&n)) {
   41     perror("pthread_create");
   42     exit(1);
   43     }
   44     if (pthread_create(&t2, (pthread_attr_t *)NULL,
   45                (void * (*)(void *))consumer, (void *)&n)) {
   46     perror("pthread_create");
   47     exit(1);
   48     }
   49 
   50     pthread_join(t1, NULL);
   51     pthread_join(t2, NULL);
   52     cout << produced << " " << consumed << endl;
   53     return(0);
   54 }
   55 
   56 
   57 void producer(int *arg) {
   58     int i, n = *arg;
   59     for (i=1; i<=n; i++) {
   60     pthread_mutex_lock(&mutex);
   61     while (pcount == 1) {
   62         pthread_cond_wait(&control, &mutex);
   63     }
   64     data = i;
   65     pcount = 1;
   66     pthread_cond_signal(&control);
   67     pthread_mutex_unlock(&mutex);
   68     produced++;
   69     }
   70 }
   71 
   72 
   73 void consumer(int *arg) {
   74     int i = 0, n = *arg;
   75     while (1) {
   76     pthread_mutex_lock(&mutex);
   77     while (pcount == 0) {
   78         pthread_cond_wait(&control, &mutex);
   79     }
   80     i = data;
   81     pcount = 0;
   82     pthread_cond_signal(&control);
   83     pthread_mutex_unlock(&mutex);
   84     consumed++;
   85     if (i == n) return;
   86     }
   87 }
