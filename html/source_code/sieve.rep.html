1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: sieve.rep.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   (let* ((arg (get-command-line-option "--num" t))
   16      (num (if arg (read-from-string arg) 1))
   17      (count 0))
   18     (while (> num 0)
   19       (setq num (1- num))
   20       (let ((flags (make-vector 8192 t)))
   21     (setq count 0)
   22     (do ((i 2 (1+ i)))
   23         ((>= i 8192))
   24       (if (aref flags i)
   25           (progn
   26            (do ((k (+ i i) (+ k i)))
   27            ((>= k 8192))
   28          (aset flags k nil))
   29            (setq count (1+ count)))))))
   30     (format standard-output "Count: %d\n" count)))
