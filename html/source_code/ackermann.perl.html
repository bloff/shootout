   15 # in our quest for speed, we must get ugly:
   16 # it helps reduce stack frame size a little bit
   17 # from Leif Stensson
   18 sub Ack {
   19     return $_[0] ? ($_[1] ? Ack($_[0]-1, Ack($_[0], $_[1]-1))
   20             : Ack($_[0]-1, 1))
   21     : $_[1]+1;
   22 }
   23 
   24 my $NUM = $ARGV[0];
   25 $NUM = 1 if ($NUM < 1);
   26 my $ack = Ack(3, $NUM);
   27 print "Ack(3,$NUM): $ack\n";
