1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: spellcheck.rep.html,v 1.1 2004-06-14 23:09:29 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep
   14     rep.system
   15     rep.io.files
   16     rep.regexp
   17     rep.data.tables)
   18 
   19   (let* ((arg (get-command-line-option "--num" t))
   20      (n (if arg (read-from-string arg) 1))
   21      (dict (make-table string-hash eq-hash))
   22      (line ""))
   23     (let ((dictfile (open-file "Usr.Dict.Words" 'read)))
   24       (when dictfile
   25     (unwind-protect
   26         (while (setq line (read-line dictfile))
   27           (when (string-match "\\S+" line)
   28         (table-set dict (substring line (match-start) (match-end)) #t)))
   29       (close-file dictfile))))
   30     (while (setq line (read-line))
   31       (when (string-match "\\S+" line)
   32     (let ((word (substring line (match-start) (match-end))))
   33       (when (not (table-ref dict word))
   34         (format standard-output "%s\n" word)))))))
