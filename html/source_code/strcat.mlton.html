    7 fun for (start, stop, f) =
    8    let
    9       fun loop i =
   10      if i > stop
   11         then ()
   12      else (f i; loop (i + 1))
   13    in
   14       loop start
   15    end

   17 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0

   22 structure Buffer:
   23    sig
   24       type 'a t
   25 
   26       val add: 'a t * 'a array -> unit
   27       val length: 'a t -> int
   28       val new: 'a -> 'a t
   29    end =
   30    struct
   31       datatype 'a t = T of {dummy: 'a,
   32                 length: int ref,
   33                 elts: 'a array ref}
   34 
   35       fun add (T {dummy, elts, length}, a) =
   36      let
   37         val l = !length
   38         val e = !elts
   39         val en = Array.length e
   40         val an = Array.length a
   41         val e =
   42            if l + an >= en
   43           then
   44              let
   45             val e = Array.tabulate (2 * en, fn i =>
   46                         if i < l
   47                            then Array.sub (e, i)
   48                         else dummy)
   49             val _ = elts := e
   50              in e
   51              end
   52            else e
   53         val _ =
   54            Array.copy {src = a, dst = e, di = l}
   55         val _ = length := l + an
   56      in ()
   57      end
   58 
   59       fun new (dummy: 'a) = T {dummy = dummy,
   60                    length = ref 0,
   61                    elts = ref (Array.array (32, dummy))}
   62 
   63       fun length (T {length, ...}) = !length
   64    end
   65 
   66 fun main (name, args) =
   67    let
   68       val stuff =
   69      Array.tabulate (String.size stuff, fn i => String.sub (stuff, i))
   70       val n = atoi (hd (args @ ["1"]))
   71       val b = Buffer.new #"\000"
   72       val _ = for (1, n, fn _ => Buffer.add (b, stuff))
   73       val _ = printl [Int.toString (Buffer.length b)]
   74    in
   75       ()
   76    end
