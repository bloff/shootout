    6 %%  Section 11.5 of the Oz Tutorial provides these
    7 %%  implementations of Event and UnitBufferM and states:
    8 %%  in Oz, it is very rare to write programs in the
    9 %%  (traditional) monitor style shown above. In general
   10 %%  it is very awkward.
   11 %%
   12 %%  There's an extensive treatment of Oz concurrency in
   13 %%  the book 'Concepts, Techniques, and Models of Computer
   14 %%  Programming' - find it online with google.
   15 %%
   16 %%  Usage: start from command line with
   17 %%     ozc -x prodcons.oz -o prodcons.oz.exe
   18 %%     prodcons.oz.exe 100000
   19 
   20 functor
   21 import System Application
   22 
   23 define
   24 
   25 Produced
   26 Consumed
   27 
   28 class Event from BaseObject
   29    prop locking
   30    attr f r
   31    meth init
   32       X in f <- X r <- X
   33    end
   34    meth put(I)
   35       X in lock @r=I|X r<-X end
   36    end
   37    meth get(?I)
   38       X in lock @f=I|X f<-X end {Wait I}
   39    end
   40    meth wait
   41       {self get(_)}
   42    end
   43    meth notify
   44       {self put(unit)}
   45    end
   46 end
   47 
   48 
   49 class UnitBufferM
   50    attr item empty psignal csignal
   51    prop locking
   52    meth init
   53       empty <- true
   54       psignal <- {New Event init}
   55       csignal <- {New Event init}
   56    end
   57    meth put(I)
   58       X in
   59       lock
   60          if @empty then
   61             item <- I
   62             empty <- false
   63             X = yes
   64             {@csignal notify}
   65          else X = no end
   66       end
   67       if X == no then
   68          {@psignal wait}
   69          {self put(I)}
   70       end
   71    end
   72    meth get(I)
   73       X in
   74       lock
   75          if {Not @empty} then
   76             I = @item
   77             empty <- true
   78             {@psignal notify}
   79             X = yes
   80          else X = no end
   81       end
   82       if X == no then
   83          {@csignal wait}
   84          {self get(I)}
   85       end
   86    end
   87 end
   88 
   89 
   90 proc {Producer N I B}
   91    if N > 0 then
   92       {B put(I)}
   93       %% {System.showInfo 'Produced '#I} %% just to check synchronization
   94       {Producer N-1 I+1 B}
   95    else Produced = {NewCell I} end
   96 end
   97 
   98 
   99 proc {Consumer N I B}
  100    if N > 0 then
  101       {B get(I)}
  102       %% {System.showInfo 'Consumed '#I} %% just to check synchronization
  103       {Consumer N-1 I+1 B}
  104    else Consumed = {NewCell I} end
  105 end
  106 
  107 
  108 in
  109    local Args N UB in
  110       [Args] = {Application.getArgs plain}
  111       N = {String.toInt Args}
  112 
  113       UB = {New UnitBufferM init}
  114       thread {Producer N 0 UB} end
  115       thread {Consumer N 0 UB} end
  116 
  117          %% Oz is a dataflow language.
  118          %% The main thread will wait until logic variables
  119          %% Produced and Consumed have been given values
  120       {System.showInfo {Access Produced}#' '#{Access Consumed}}
  121    end
  122    {Application.exit 0}
  123 end
