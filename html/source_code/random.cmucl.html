    7 (defconstant IM     139968)
    8 (defconstant IA       3877)
    9 (defconstant IC     29573)
   10 
   11 (defvar LAST 42)
   12 
   13 (declaim (inline gen_random))
   14 (defun gen_random (max)
   15   (declare (type (signed-byte 32) IM IA IC LAST))
   16   (declare (double-float max))
   17   (setq LAST (mod (+ (the fixnum (* LAST IA)) IC) IM))
   18   (/ (* max LAST) IM))
   19 
   20 (defun main ()
   21   (let ((n (parse-integer (or (car (last extensions:*command-line-strings*)
   22    ) "1"))))
   23     (loop for i fixnum from 1 below n do
   24       (gen_random 100.0d0))
   25     (format t "~,9F~%" (gen_random 100.0d0))))
