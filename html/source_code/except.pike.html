    9 class HiException( mixed value ) { constant IsHi = 1; }
   10 class LoException( mixed value ) { constant IsLo = 1; }
   11 
   12 void some_function( int num )
   13 {
   14     if( mixed e = catch(  hi_function( num ) ) )
   15     error( "We shouldn't get here (%s)", describe_error( e ) );
   16 }
   17 
   18 int HI, LO;
   19 
   20 void hi_function(int num)
   21 {
   22     if( mixed e = catch( lo_function( num ) ) )
   23     if( e->IsHi )
   24         HI++;
   25     else
   26         throw( e );
   27 }
   28 
   29 void lo_function(int num)
   30 {
   31     if( mixed e = catch(  blowup(num) ) )
   32     if( e->IsLo )
   33         LO++;
   34     else
   35         throw( e );
   36 }
   37 
   38 
   39 void blowup(int num)
   40 {
   41     if( num & 1 )
   42     throw( LoException(num) );
   43     else
   44     throw( HiException(num) );
   45 }
   46 
   47 void main(int argc, array argv)
   48 {
   49     int num = (int)argv[-1];
   50     if( num < 1 )
   51     num = 1;
   52     while(num)
   53     some_function( num-- );
   54     write( "Exceptions: HI=%d / LO=%d\n" , HI, LO );
   55 }
