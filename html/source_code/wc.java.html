    5 import java.io.*;
    6 import java.util.*;
    7 import java.text.*;
    8 
    9 // this program modified from:
   10 //   http://cm.bell-labs.com/cm/cs/who/bwk/interps/pap.html
   11 // Timing Trials, or, the Trials of Timing: Experiments with Scripting
   12 // and User-Interface Languages</a> by Brian W. Kernighan and
   13 // Christopher J. Van Wyk.
   14 
   15 public class wc {
   16     public static void main(String[] args) {
   17         int nl = 0, nw = 0, nc = 0;
   18 
   19         try {
   20             byte[] buff = new byte[4096];
   21             boolean inword = false;
   22             int length;
   23 
   24             while ((length = System.in.read(buff)) != -1) {
   25                 nc += length;
   26                 for(int i = 0; i < length; i++) {
   27                     char c = (char)buff[i];
   28                     if (c == '\n')
   29                         ++nl;
   30                     if (Character.isWhitespace(c))
   31                         inword = false;
   32                     else if (inword == false) {
   33                         ++nw;
   34                         inword = true;
   35                     }
   36                 }
   37             }
   38         } catch (IOException e) {
   39             System.err.println(e);
   40             return;
   41         }
   42         System.out.println(Integer.toString(nl) + " " +
   43                            Integer.toString(nw) + " " +
   44                            Integer.toString(nc));
   45     }
   46 }
