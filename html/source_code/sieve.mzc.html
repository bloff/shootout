1 #!/usr/bin/mzscheme -r
    2 
    3 (define (main args)
    4   (let ((n (if (= (vector-length args) 0)
    5                1
    6                (string->number (vector-ref args 0))))
    7     (count 0)
    8     (flags (make-vector 8192)))
    9     (let loop ((iter n))
   10       (if (> iter 0)
   11       (begin
   12             (do ((i 0 (+ i 1))) ((>= i 8192)) (vector-set! flags i #t))
   13             (set! count 0)
   14             (do ((i 2 (+ 1 i)))
   15               ((>= i 8192))
   16               (if (vector-ref flags i)
   17                   (begin
   18                     (do ((k (+ i i) (+ k i)))
   19                       ((>= k 8192))
   20                       (vector-set! flags k #f))
   21                     (set! count (+ 1 count)))))
   22         (loop (- iter 1)))))
   23     (display "Count: ") (display count) (newline)))
   24 
   25 (main (current-command-line-arguments))
