    5 -- this version uses a custom string buffer

    7 ------------------------------------------------------------------
    8 -- Buffer library
    9 ------------------------------------------------------------------
   10 
   11 Buffer = {""}
   12 
   13 function Buffer:new ()
   14   local new = {}
   15   self.__index = self
   16   setmetatable(new, self)
   17   return new
   18 end
   19 
   20 function Buffer:add (s)
   21   table.insert(self, s)    -- push 's' into the the stack
   22   for i=table.getn(self)-1, 1, -1 do
   23     if string.len(self[i]) > string.len(self[i+1]) then
   24       break
   25     end
   26     self[i] = self[i] .. table.remove(self)
   27   end
   28 end
   29 
   30 function Buffer:close ()
   31   self[1] = table.concat(self)
   32   table.setn(self, 1)   -- now there is only one element
   33   return self[1]
   34 end
