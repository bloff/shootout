    5 import java.io.*;
    6 import java.net.*;
    7 
    8 public class echo {
    9     public static void main(String[] args) throws Exception {
   10     int iIterations = 1;
   11     try {
   12         iIterations = Integer.parseInt(args[0]);
   13     } catch(Exception e) { }
   14 
   15     EchoServer esServer = new EchoServer(0);
   16     new EchoClient(InetAddress.getLocalHost(), esServer.getPort(),
   17                 iIterations);
   18     }
   19 }
   20 
   21 class EchoClient extends Thread {
   22     private static final String GREETING = "Hello there sailor\n";
   23     private final InetAddress inetaServer;
   24     private final int         iPort;
   25     private final int         iIterations;
   26 
   27     public EchoClient(InetAddress inetaServer, int iPort, int iIterations)
   28                       {
   29     this.inetaServer = inetaServer;
   30     this.iPort = iPort;
   31     this.iIterations = iIterations;
   32     start();
   33     }
   34 
   35     public void run() {
   36     Socket socketFromServer = null;
   37     try {
   38         socketFromServer = new Socket(inetaServer, iPort);
   39         BufferedReader in = new BufferedReader(new InputStreamReader(
   40                          socketFromServer.getInputStream()));
   41         OutputStream out = socketFromServer.getOutputStream();
   42 
   43         byte[] bytesOut = GREETING.getBytes();
   44         String strIn = GREETING.trim();
   45         for(int i = 0; i < iIterations; ++i) {
   46         out.write(bytesOut);
   47         out.flush();
   48         String strRead = in.readLine();
   49         if(!strRead.equals(strIn))
   50             throw new RuntimeException("client: \"" + strIn + "\"  ne \"" +
   51                                  strRead + "\"");
   52         }
   53     } catch(Exception e) {
   54         e.printStackTrace();
   55     }
   56 
   57     try {
   58         socketFromServer.close();
   59     } catch(Exception e) { }
   60     }
   61 }
   62 
   63 class EchoServer extends Thread {
   64     private static final int   BUFFER_SIZE = 1024;
   65     private final ServerSocket ssAccepting;
   66     private final int          iPort;
   67 
   68     public EchoServer(int iPort) throws IOException {
   69     ssAccepting = new ServerSocket(iPort);
   70     this.iPort = ssAccepting.getLocalPort();
   71     start();
   72     }
   73 
   74     public final int getPort() {
   75     return iPort;
   76     }
   77 
   78     public void run() {
   79     byte bytesIn[] = new byte[BUFFER_SIZE];
   80     try {
   81         Socket socketClient = ssAccepting.accept();
   82         InputStream in = socketClient.getInputStream();
   83         OutputStream out = socketClient.getOutputStream();
   84         int iLength, iCount = 0;
   85         while ((iLength = in.read(bytesIn)) != -1) {
   86         out.write(bytesIn, 0, iLength);
   87         out.flush();
   88         iCount += iLength;
   89         }
   90         System.out.println("server processed " + iCount + " bytes");
   91     } catch (Exception e) {
   92         e.printStackTrace();
   93     }
   94     }
   95 }
