    6 (defun main ()
    7   (let* ((n (parse-integer (or (car (last extensions:*command-line-
    8    strings*)) "1")))
    9      (A '())
   10      ;; create empty list B
   11      (B '())
   12      (mid (/ N 2))
   13      (sum 0))
   14 
   15     ;; create list (A) of integers from 1 through N
   16     (do ((i N (- i 1)))
   17     ((= i 0))
   18       (setq A (cons i A)))
   19 
   20     ;; move each individual item from A to B, in a loop, reversing order
   21     (do ((empty (null A) (null A)))
   22     (empty)
   23       (setq B (cons (car A) B))
   24       (setq A (cdr A)))
   25 
   26     ;; print first 2 elements of B
   27     (format t "~a ~a~%" (car B) (cadr B))
   28 
   29     ;; reverse B (can be done in place)
   30     (setq B (reverse B))
   31 
   32     ;; is 0 a member of B?
   33     (format t "~a~%" (if (member 0 B) "true" "false"))
   34 
   35     ;; is N a member of B?
   36     (format t "~a~%" (if (member N B) "true" "false"))
   37 
   38     ;; filter values from B to A that are less than N/2, preserving order
   39     (loop for x in B do (if (< x mid) (setq A (cons x A))))
   40     (setq A (reverse A))
   41 
   42     ;; print first ten items of A
   43     (do ((i 0 (+ i 1))
   44      (x A (cdr x)))
   45     ((or (= i 10) (null x)))
   46       (format t  (if (< i 9) "~a " "~a~%") (car x)))
   47 
   48     ;; print sum of items in A that are less than 1000
   49     (loop for x in A do (if (< x 1000) (setq sum (+ x sum))))
   50     (format t "~a~%" sum)
   51 
   52     ;; append B to end of A
   53     ;(nconc A B)
   54     (setf (cdr (last A)) B)
   55 
   56     ;; print length of A
   57     (format t "~a ~a~%" (length A) (car (last A)))))
