    5 BUFSIZE = 2^12
    6 
    7 local cc,lc,wc = 0,0,0
    8 while 1 do
    9     local lines, rest = io.read(BUFSIZE, "*l")
   10     if lines == nil then break end
   11     if rest then lines = lines..rest..'\n' end
   12     cc = cc+string.len(lines)
   13     local _,t = string.gsub(lines, "%S+", "")   -- count words in the line
   14     wc = wc+t
   15     _,t = string.gsub(lines, "\n", "\n")   -- count newlines in the line
   16     lc = lc+t
   17 end
   18 
   19 io.write(lc, " ", wc, " ", cc, "\n")
