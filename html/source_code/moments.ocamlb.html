    7 let cutoff = 5
    8 
    9 (* Unscrupulously stolen from Array.stable_sort (OCaml standard
   10  distribution) *)
   11 let sort_floats len (ar : float array) =
   12   let merge src1ofs src1len src2 src2ofs src2len dst dstofs =
   13     let src1r = src1ofs + src1len and src2r = src2ofs + src2len in
   14     let rec loop i1 s1 i2 s2 d =
   15       if s1 <= s2 then begin
   16         dst.(d) <- s1;
   17         let i1 = i1 + 1 in
   18         if i1 < src1r then loop i1 ar.(i1) i2 s2 (d + 1)
   19         else Array.blit src2 i2 dst (d + 1) (src2r - i2)
   20       end else begin
   21         dst.(d) <- s2;
   22         let i2 = i2 + 1 in
   23         if i2 < src2r then loop i1 s1 i2 src2.(i2) (d + 1)
   24         else Array.blit ar i1 dst (d + 1) (src1r - i1)
   25       end in
   26     loop src1ofs ar.(src1ofs) src2ofs src2.(src2ofs) dstofs in
   27   let isortto srcofs dst dstofs len =
   28     for i = 0 to len - 1 do
   29       let e = ar.(srcofs + i) and j = ref (dstofs + i - 1) in
   30       while !j >= dstofs && e <= dst.(!j) do
   31         dst.(!j + 1) <- dst.(!j);
   32         decr j;
   33       done;
   34       dst.(!j + 1) <- e
   35     done in
   36   let rec sortto srcofs dst dstofs len =
   37     if len <= cutoff then isortto srcofs dst dstofs len else begin
   38       let l1 = len / 2 in
   39       let l2 = len - l1 in
   40       sortto (srcofs + l1) dst (dstofs + l1) l2;
   41       sortto srcofs ar (srcofs + l2) l1;
   42       merge (srcofs + l2) l1 dst (dstofs + l1) l2 dst dstofs
   43     end in
   44   if len <= cutoff then isortto 0 ar 0 len else begin
   45     let l1 = len / 2 in
   46     let l2 = len - l1 in
   47     let t = Array.make l2 ar.(0) in
   48     sortto l1 t 0 l2;
   49     sortto 0 ar l2 l1;
   50     merge l2 l1 t 0 l2 ar 0
   51   end
   52 
   53 let n_ref = ref 0 and sum_ref = ref 0.0 and size_ref = ref 4096
   54 let nums_ref = ref (Array.create !size_ref 0.0);;
   55 
   56 try
   57   while true do
   58     let num = float_of_string (input_line stdin) in
   59     !nums_ref.(!n_ref) <- num;
   60     sum_ref := !sum_ref +. num;
   61     incr n_ref;
   62     let size = !size_ref in
   63     if !n_ref = size then begin
   64       let new_size = size lsl 2 in
   65       size_ref := new_size;
   66       let new_ar = Array.create new_size 0.0 in
   67       Array.blit !nums_ref 0 new_ar 0 size;
   68       nums_ref := new_ar
   69     end
   70   done
   71 with End_of_file -> ();
   72 
   73 let n = !n_ref and nums = !nums_ref in
   74 let n_float = float n in
   75 let mean = !sum_ref /. n_float
   76 and avg_dev_ref = ref 0.0 and std_dev_ref = ref 0.0 and var_ref = ref 0.0
   77 and skew_ref = ref 0.0 and kurt_ref = ref 0.0 in
   78 
   79 for i = 0 to n - 1 do
   80   let dev = nums.(i) -. mean in
   81   avg_dev_ref := !avg_dev_ref +. abs_float dev;
   82   let dev2 = dev *. dev in
   83   var_ref := !var_ref +. dev2;
   84   let dev3 = dev2 *. dev in
   85   skew_ref := !skew_ref +. dev3;
   86   kurt_ref := !kurt_ref +. dev3 *. dev
   87 done;
   88 
   89 let avg_dev = !avg_dev_ref /. n_float and var = !var_ref /. float (n - 1)
   90               in
   91 let std_dev = sqrt var in
   92 
   93 if var > 0.0 then begin
   94   skew_ref := !skew_ref /. n_float /. var /. std_dev;
   95   kurt_ref := !kurt_ref /. n_float /. var /. var -. 3.0
   96 end;
   97 
   98 sort_floats n nums;
   99 
  100 let median =
  101   let mid = n lsr 1 in
  102   if n mod 2 = 1 then nums.(mid)
  103   else (nums.(mid) +. nums.(mid - 1)) /. 2.0 in
  104 
  105 Printf.printf "n:                  %d\n" n;
  106 Printf.printf "median:             %f\n" median;
  107 Printf.printf "mean:               %f\n" mean;
  108 Printf.printf "average_deviation:  %f\n" avg_dev;
  109 Printf.printf "standard_deviation: %f\n" std_dev;
  110 Printf.printf "variance:           %f\n" var;
  111 Printf.printf "skew:               %f\n" !skew_ref;
  112 Printf.printf "kurtosis:           %f\n" !kurt_ref
