    6 proc main {} {
    7     global argv
    8     set n [lindex $argv 0]
    9     set x 0
   10     incr n 1
   11     set a $n
   12     while {[incr a -1]} {
   13     set b $n
   14     while {[incr b -1]} {
   15         set c $n
   16         while {[incr c -1]} {
   17         set d $n
   18         while {[incr d -1]} {
   19             set e $n
   20             while {[incr e -1]} {
   21             set f $n
   22             while {[incr f -1]} {
   23                 incr x
   24             }
   25             }
   26         }
   27         }
   28     }
   29     }
   30     puts $x
   31 }
   32 
   33 main
