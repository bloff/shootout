1 #!/bin/bash
    2 # $Id: hash2.bash.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 # from Steve Fink
    5 
    6 n=${1:-1}
    7 
    8 while [ $n -gt 0 ]; do
    9     # Initialize the array
   10     i=0
   11     while [ $i -lt 10000 ]; do
   12         eval H1_foo_$i=$i
   13         i=$[ $i + 1 ]
   14     done
   15 
   16     # Iterate through the hash values, attempting to keep to the spirit of
   17     # eg perl's keys %h
   18     for assign in $(  # Get around the subshell problem
   19         set | grep -a '^H1_' | cut -d= -f1 | cut -c4- | while read; do
   20             eval h2=\${H2_${REPLY}:-0}
   21             eval h1=\$H1_$REPLY
   22             echo H2_$REPLY=$[ $h2 + $h1 ]
   23         done
   24     ); do
   25         eval $assign
   26     done
   27 
   28     n=$[ $n - 1 ]
   29 done
   30 
   31 echo "$H1_foo_1 $H1_foo_9999 $H2_foo_1 $H2_foo_9999"
