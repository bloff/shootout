    7 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
    8         1
    9 and hash1 = Hashtbl.create 10000 and hash2 = Hashtbl.create 10000 in
   10 let update_hash2 k v =
   11   try let valref = Hashtbl.find hash2 k in valref := !valref + !v
   12   with Not_found -> Hashtbl.add hash2 k (ref !v) in
   13 for i = 0 to 9999 do Hashtbl.add hash1 ("foo_" ^ string_of_int i) (ref i)
   14         done;
   15 for i = 1 to n do Hashtbl.iter update_hash2 hash1 done;
   16 Printf.printf "%d %d %d %d\n"
   17   !(Hashtbl.find hash1 "foo_1") !(Hashtbl.find hash1 "foo_9999")
   18   !(Hashtbl.find hash2 "foo_1") !(Hashtbl.find hash2 "foo_9999")
