    6 (declaim (optimize (speed 3) (debug 0) (safety 0) (space 0) (compilation-
    7  speed 0)))
    8 
    9 (defun main ()
   10   (let* ((start 0)
   11      (current #\X)		    	;junk char save me the locally
   12      (end 0)
   13      (nc 0)
   14      (buffer (make-string 4096)))
   15     (declare (type (simple-base-string 4096) buffer) (fixnum start end nc)
   16          (base-char current))
   17     (labels
   18     ((get-char ()
   19            (when (= start end)
   20              (setf start 0)
   21              (setf end (read-sequence buffer *standard-input*))
   22              (incf nc end)
   23              (when (zerop end)
   24                (return-from get-char nil)))
   25            (setf current (schar buffer start))
   26            (incf start)))
   27       (let ((nl 0)
   28         (nw 0)
   29         (inword nil))
   30     (declare (fixnum nl nw))
   31     (loop while (get-char) do
   32       (cond ((char= current #\newline)
   33          (incf nl)
   34          (setq inword nil))
   35         ((or (char= current #\space) (char= current #\tab))
   36          (setq inword nil))
   37         ((not inword) ;; only tested if we have a word constituent
   38          (incf nw) (setq inword t))))
   39     (format t "~A ~A ~A~%" nl nw nc)))))
