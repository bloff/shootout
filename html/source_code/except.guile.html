    5 ;;; $Id: except.guile.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $

    7 
    8 (use-modules (ice-9 format))
    9 
   10 (define HI 0)
   11 (define LO 0)
   12 
   13 (define (some_fun n)
   14   (catch #t
   15      (lambda () (hi_fun n))
   16      (lambda args #f)))
   17 
   18 (define (hi_fun n)
   19   (catch 'Hi_Exception
   20      (lambda () (lo_fun n))
   21      (lambda args (set! HI (+ HI 1)))))
   22 
   23 (define (lo_fun n)
   24   (catch 'Lo_Exception
   25      (lambda () (blowup n))
   26      (lambda args (set! LO (+ LO 1)))))
   27 
   28 (define (blowup n)
   29   (if (= 0 (modulo n 2))
   30       (throw 'Hi_Exception)
   31       (throw 'Lo_Exception)))
   32 
   33 (define (main args)
   34   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   35     (do ((i 0 (+ i 1)))
   36     ((= i n))
   37       (some_fun i)))
   38   (display (format "Exceptions: HI=~D / LO=~D\n" HI LO)))
   39
