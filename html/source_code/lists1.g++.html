    5 #include <iostream>
    6 #include <list>
    7 #include <numeric>
    8 #include <ext/numeric>
    9 
   10 using namespace std;
   11 
   12 void list_print_n (list<int> L, int n) {
   13     int c, lastc = n - 1;
   14     list<int>::iterator i;
   15     for (c = 0, i = L.begin(); i != L.end(), c < n; ++i, ++c) {
   16     cout << (*i);
   17     if (c < lastc) cout << " ";
   18     }
   19     cout << endl;
   20 }
   21 
   22 int main(int argc, char* argv[]) {
   23     int N = (argc == 2 ? (atoi(argv[1]) < 1 ? 1 : atoi(argv[1])): 1);
   24     list<int>::iterator i;
   25 
   26     // create empty list B
   27     list<int> B;
   28 
   29     // create list (A) of integers from 1 through N
   30     list<int> A(N);
   31     __gnu_cxx::iota(A.begin(), A.end(), 1);
   32 
   33     // move each individual item from A to B, in a loop, reversing order
   34     while (! A.empty()) {
   35         B.push_front(A.front());
   36         A.pop_front();
   37     }
   38 
   39     // print first 2 elements of B
   40     list_print_n(B, 2);
   41 
   42     // reverse B (can be done in place)
   43     B.reverse();
   44     // reverse(B.begin(), B.end());
   45 
   46     // is 0 a member of B?
   47     cout << ((find(B.begin(), B.end(), 0) == B.end()) ? "false" : "true") <
   48              < endl;
   49 
   50     // is N a member of B?
   51     cout << ((find(B.begin(), B.end(), N) == B.end()) ? "false" : "true") <
   52              < endl;
   53 
   54     // filter values from B to A that are less than N/2, preserving order
   55     int mid = N/2;
   56     for (i = B.begin(); i != B.end(); ++i) {
   57     if ((*i) < mid) A.push_back(*i);
   58     }
   59 
   60     // print first ten items of A
   61     list_print_n(A, 10);
   62 
   63     // print sum of items in A that are less than 1000
   64     int sum = 0;
   65     for (i = A.begin(); i != A.end(); ++i) {
   66     if ((*i) < 1000) sum += (*i);
   67     }
   68     cout << sum << endl;
   69 
   70     // append B to end of A
   71     A.splice(A.end(), B);
   72 
   73     // print length and last element of A
   74     cout << A.size() << " " << A.back() << endl;
   75 }
