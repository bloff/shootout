    4 import IO
    5 import System (getArgs)
    6 import RegexString (matchRegex, mkRegex)
    7 
    8 phoneMatch = matchRegex $ mkRegex regex
    9          where regex =
   10                "(^|[^0-9\\(]+)" ++
   11                "(" ++
   12                    "\\(([0-9][0-9][0-9])\\)" ++
   13                    "|" ++
   14                    "([0-9][0-9][0-9])" ++
   15                    ")" ++
   16                    " " ++
   17                    "([0-9][0-9][0-9])" ++
   18                    "[ -]"  ++
   19                    "([0-9][0-9][0-9][0-9])" ++
   20                    "[^0-9]"
   21 
   22 
   23 loadFile ls  = do
   24            eof <- hIsEOF stdin
   25            if (not eof)
   26           then do
   27                  inStr <- hGetLine stdin
   28              loadFile ((inStr ++ "\n") : ls)
   29           else
   30                 return $ reverse ls
   31 
   32 canonicalPhone matchGrps = "(" ++ areacode ++ ") " ++ phnum
   33                            where areacode = let (_,b) = splitAt 2
   34                                             matchGrps in
   35                         (head b) ++ (head . tail $ b)
   36                  phnum = let (_,b) = splitAt 4 matchGrps in
   37                      (head b) ++ "-" ++ (head . tail) b
   38 
   39 findPhoneNums [] = []
   40 findPhoneNums (l:ls) = case matchInLine l of
   41                [] -> findPhoneNums ls
   42                x  -> [x] ++ findPhoneNums ls
   43                        where matchInLine = maybe [] canonicalPhone .
   44                                            phoneMatch
   45 
   46 main = do
   47        args <- getArgs
   48        lines <- loadFile []
   49        let n = read . head $ args in
   50            sequence_ $ map displayLn $ zip [1..] $ doIterations n lines
   51        return ()
   52        where displayLn (n,l) = do putStrLn $ show n ++ ": " ++ l
   53          doIterations n lines = last $ take n (repeat (findPhoneNums lines))
