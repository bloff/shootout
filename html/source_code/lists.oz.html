1 %%  Always 10000's times faster to append to the head of 
    2 %%  the list (and then Reverse) rather than traversing 
    3 %%  the entire list in each append before adding to tail. 
    4 %% 
    5 %%  http://shootout.alioth.debian.org/ 
    6 %% 
    7 %%  contributed by Isaac Gouy 
    8 %% 
    9 %%  Usage: start from command line with 
   10 %%     ozc -x lists.oz -o lists.oz.exe 
   11 %%     lists.oz.exe 16 
   12  
   13 functor 
   14 import System Application 
   15  
   16 define 
   17    L1 L2 L3 
   18     
   19    fun {MoveLeftRight Li2 Li3} 
   20       %% list|item requires list to be traversed to end 
   21       %% before appending item. item|list always faster. 
   22       %% {Reverse item|list} result same as list|item 
   23       local H T in  
   24      if Li2 == nil then Li2|[{Reverse Li3}] 
   25      else 
   26         H|T = Li2               %% take from left 
   27         {MoveLeftRight T H|Li3} %% append to left, reverse later 
   28      end 
   29       end 
   30    end 
   31  
   32    proc {AssignL2L3 L} 
   33       %% Factor out the assignment to allow simple reuse 
   34       {Assign L2 {Nth L 1}} 
   35       {Assign L3 {List.last L}} 
   36    end 
   37     
   38 in 
   39    local Args N Size in 
   40       [Args] = {Application.getArgs plain} 
   41       N = {String.toInt Args} 
   42       Size = 10000 
   43       {NewCell nil L1}{NewCell nil L2}{NewCell nil L3}     
   44       for I in 1..N do 
   45         %% Create L1 
   46      {Assign L1 {List.number 1 Size 1} } 
   47         %% Copy L1 to L2 
   48      {Assign L2 {List.take {Access L1} Size} } 
   49      {Assign L3 nil} 
   50  
   51         %% Move items from L2 to L3 
   52      {AssignL2L3 {MoveLeftRight {Access L2}{Access L3} }} 
   53         %% Move items from L3 to L2 
   54      {AssignL2L3 {Reverse {MoveLeftRight {Reverse {Access L3}}{Access L2}} }}
   55         %% Reverse L1 
   56      {Assign L1 {Reverse {Access L1}}} 
   57          if {Not {Nth {Access L1} 1} == Size andthen {Access L1} == {
   58              Access L2}} then 
   59             {System.showInfo 'L1 != L2'} 
   60          end 
   61       end 
   62       {System.showInfo {Length {Access L1}}} 
   63    {Application.exit 0} 
   64    end 
   65 end
