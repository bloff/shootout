    5 (module sieve
    6     (main main)
    7     (extern (macro printf::int (string int) "printf")))
    8 
    9 (define (main args)
   10   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   11     (count 0)
   12     (flags (make-vector 8192)))
   13     (let loop ((iter n))
   14       (if (> iter 0)
   15       (begin
   16               (do ((i 0 (+ i 1)))
   17           ((>= i 8192))
   18         (vector-set! flags i #t))
   19           (set! count 0)
   20           (do ((i 2 (+ 1 i)))
   21           ((>= i 8192))
   22         (if (vector-ref flags i)
   23             (begin
   24               (do ((k (+ i i) (+ k i)))
   25               ((>= k 8192))
   26             (vector-set! flags k #f))
   27               (set! count (+ 1 count)))))
   28         (loop (- iter 1)))))
   29     (display "Count: ") (display count) (newline)))
