    6 (define (main args)
    7   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
    8     (count 0)
    9     (flags (make-vector 8192)))
   10     (let loop ((iter n))
   11       (if (> iter 0)
   12       (begin
   13               (do ((i 0 (+ i 1))) ((>= i 8192)) (vector-set! flags i #t))
   14           (set! count 0)
   15           (do ((i 2 (+ 1 i)))
   16           ((>= i 8192))
   17         (if (vector-ref flags i)
   18             (begin
   19               (do ((k (+ i i) (+ k i)))
   20               ((>= k 8192))
   21             (vector-set! flags k #f))
   22               (set! count (+ 1 count)))))
   23         (loop (- iter 1)))))
   24     (display "Count: ") (display count) (newline)))
   25 
   26 (main (vector->list argv))
