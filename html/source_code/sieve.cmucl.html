    6 (declaim (optimize (speed 3) (safety 0) (debug 0) (space 0) (compilation-
    7  speed 0)))
    8 (defun main ()
    9   (let ((n (parse-integer (or (car (last ext:*command-line-strings*)) "1"))
   10    )
   11         (flags (make-array 8193 :element-type 'fixnum :initial-element 1)))
   12     (loop repeat n of-type fixnum for count of-type fixnum = 0 then 0 do
   13        (loop for i fixnum from 2 upto 8192 do
   14           (unless (zerop (aref flags i))
   15             (loop for k fixnum from (* 2 i) upto 8192 by i do
   16                   (setf (aref flags k) 0))
   17             (incf count)))
   18        finally (format t "Count: ~D~%" count))))
