   15 (defun ih-sigpipe (signal code scp)
   16   (declare (ignore signal code scp))
   17   (signal 'sigpipe))

   19 (defun main ()
   20   (let ((lsock (ext:create-inet-listener 0))
   21     (n (parse-integer (or (car (last extensions:*command-line-strings*)) "1"))
   22   ))
   23     (multiple-value-bind (host port) (get-socket-host-and-port lsock)
   24       (declare (ignore host))
   25       (let ((fork-res (unix:unix-fork)))
   26     (if fork-res
   27         (if (zerop fork-res)
   28         (echo-client port n)
   29           (let ((stream (sys:make-fd-stream
   30                  (ext:accept-tcp-connection lsock)
   31                  :output t :input t :buffering :line))
   32             (buffer (make-string 64))
   33             (insize 0)
   34             (sum 0))
   35         (declare (fixnum insize sum))
   36         (sys:enable-interrupt UNIX:sigpipe #'ih-sigpipe)
   37         (handler-case
   38             (progn
   39               (loop ;; loop seems to chew up my declarations!?
   40             while (not (zerop (setf (the fixnum insize)
   41                         (the fixnum (sys:read-n-bytes stream buffer 0 64 nil)))))
   42             do (write-sequence buffer stream :start 0 :end insize)
   43             (incf sum insize))
   44               (unix::unix-wait)
   45               (format t "server processed ~D bytes~%" sum))
   46           (sigpipe (foo) (declare (ignore foo)) (error-return "Kid died
   47      prematurely")) ;; as good as catching sigchild for this app
   48           (end-of-file (foo) (declare (ignore foo))(error-return "EOF signalled.
   49      Huh???"))))))))))

   51 (defun error-return (str)
   52   (format *standard-output* "Error: ~s ~%" str)
   53   (quit))

   55 (defun echo-client (port iter)
   56   (declare (fixnum iter))
   57   (let* ((stream (sys:make-fd-stream
   58           (ext:connect-to-inet-socket "127.0.0.1" port)
   59           :output t :input t :buffering :line))
   60      (estr "Hello there sailor
   61 ")
   62        (len (length estr))
   63        (buffer (make-string len)))
   64     (dotimes (i iter)
   65       (declare (fixnum i len) (simple-base-string buffer)
   66            (inline write-sequence sys:read-n-bytes string=))
   67       (write-sequence estr stream :start 0 :end len )
   68       (sys:read-n-bytes stream buffer 0 len nil)
   69       (unless (string= estr buffer)
   70     (format t "client did not receive what it sent ~%")))))
