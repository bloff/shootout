    6 %% Quick and Dirty transliteration from the Mercury solution 
    7 %% with +1 adjustment for array indexes.  
    8 %% Mercury uses 0..N-1 and Erlang uses 1..N 
    9 %% 
   10 %% Usage: start from command line with 
   11 %%     erlc heapsort.erl 
   12 %%     erl -noinput -s heapsort main 10000 
   13  
   14 -module(heapsort).  
   15 -export([main/1]).  
   16  
   17 random_heap(I, Seed, H) -> 
   18     if  
   19         I < size(H) ->  
   20             {NextSeed, R} = gen_random(Seed), 
   21             random_heap(I+1, NextSeed, up_heap(I, R, H)); 
   22         true -> H 
   23     end. 
   24  
   25  
   26 up_heap(N, Y, H) -> 
   27     HalfN = N div 2, 
   28     X = element(HalfN+1, H), %%%% +1 
   29     Condition = 0 < N andalso X < Y, 
   30     if  
   31         Condition -> up_heap(HalfN, Y, setelement(N+1, H, X)); %%%% +1 
   32         true -> setelement(N+1, H, Y) %%%% +1 
   33     end. 
   34  
   35  
   36 heapsort(0, H) -> H; 
   37 heapsort(N, H) -> heapsort(N-1, remove_greatest(N, H)). 
   38  
   39  
   40 remove_greatest(N, H) -> 
   41     X = element(0+1, H), %%%% +1 
   42     Y = element(N+1, H), %%%% +1 
   43     down_heap(0, N-1, Y, setelement(N+1, H, X)). %%%% +1 
   44  
   45  
   46 down_heap(I, N, X, H) ->  
   47     L = I + I + 1, 
   48     R = L + 1, 
   49     if 
   50         N < L ->  
   51             setelement(I+1, H, X); %%%% +1 
   52         true -> 
   53             Condition = R < N andalso element(R+1, H) > element(L+1, H),
   54                         %%%% +1 
   55             J = if  
   56                    Condition -> R; 
   57                    true -> L 
   58                 end, 
   59             Y = element(J+1, H), 
   60             if 
   61                 X > Y -> setelement(I+1, H, X); %%%% +1 
   62                 true -> down_heap(J, N, X, setelement(I+1, H, Y)) %%%% +1 
   63             end 
   64     end. 
   65  
   66  
   67 gen_random(Seed) -> 
   68     IM = 139968, IA = 3877, IC = 29573, 
   69     S = ((Seed * IA) + IC) rem IM, 
   70     {S, S/IM}. 
   71  
   72  
   73 main([Arg]) -> 
   74     N = list_to_integer(atom_to_list(Arg)), 
   75     Seed = 42, 
   76     RandomHeap = random_heap(0, Seed, erlang:make_tuple(N, 0.0)), 
   77     SortedHeap = heapsort(N-1, RandomHeap), 
   78     io:fwrite("~.10f~n", [element(N, SortedHeap)]),             
   79     halt(0).
