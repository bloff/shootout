    8 def main():

   10     for foo in xrange(0,NUM):
   11         flags = (8192+1) * [1]
   12         count = 0
   13         for i in xrange(2,8192+1):
   14             if flags[i]:
   15                 # remove all multiples of prime: i
   16                 k = i + i
   17                 while k <= 8192:
   18                     flags[k] = 0
   19                     k = k + i
   20                 count = count + 1
   21     print "Count:", count
