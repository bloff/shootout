1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; TBD: figure out how to format results better
   10 
   11 ;;; $Id: moments.rep.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
   12 ;;; http://www.bagley.org/~doug/shootout/
   13 
   14 (define-structure main ()
   15   (open rep rep.system rep.regexp)
   16 
   17   ;; I can't think of an easier way to format a floating point number
   18   ;; and be able to specify number of digits following decimal point.
   19   (define (fltfmt decpts number)
   20     (let* ((x (expt 10 decpts))
   21        (pad (format #f "%d" x))
   22        (numlist (string-split "\\." (format #f "%d" (/ (round (* number x)) x)
   23      )))
   24        (zeros (substring pad (- (length pad) (- decpts (length (cadr numlist))
   25      ))))
   26        (numstr (concat (car numlist) "." (cadr numlist) zeros)))
   27       numstr))
   28 
   29   ;; eat the unused command line arg
   30   (get-command-line-option "--num" t)
   31 
   32   (let* ((sum 0.0)
   33      (numlist (let loop ((line (read-line))
   34                  (numlist '()))
   35                (if (not line)
   36                numlist
   37              (let ((num (read-from-string line)))
   38                (setq sum (+ num sum))
   39                (loop (read-line) (cons num numlist)))))))
   40     (let ((n (length numlist)))
   41       (let ((mean (/ sum n))
   42         (average_deviation 0.0)
   43         (standard_deviation 0.0)
   44         (variance 0.0)
   45         (skew 0.0)
   46         (kurtosis 0.0)
   47         (median 0.0)
   48         (deviation 0.0))
   49     (let loop ((nums numlist))
   50          (if (not (null nums))
   51          (progn
   52            (setq deviation (- (car nums) mean))
   53            (setq average_deviation (+ average_deviation (abs deviation)))
   54            (setq variance (+ variance (expt deviation 2.0)))
   55            (setq skew (+ skew (expt deviation 3.0)))
   56            (setq kurtosis (+ kurtosis (expt deviation 4)))
   57            (loop (cdr nums)))))
   58 
   59     (setq average_deviation (/ average_deviation n))
   60     (setq variance (/ variance (1- n)))
   61     (setq standard_deviation (sqrt variance))
   62 
   63     (cond ((> variance 0.0)
   64            (setq skew (/ skew (* n variance standard_deviation)))
   65            (setq kurtosis (- (/ kurtosis (* n variance variance))
   66                  3.0))))
   67 
   68     (setq numlist (sort numlist))
   69     (let ((mid (/ n 2)))
   70       (if (zerop (mod n 2))
   71           (setq median (/ (+ (elt numlist mid)
   72                  (elt numlist (1- mid)))
   73                   2.0))
   74         (setq median (elt numlist mid))))
   75 
   76 
   77     (setq standard_deviation (/ (round (* standard_deviation 1000000))
   78   1000000))
   79 
   80     (format standard-output "n:                  %d\n" n)
   81     (format standard-output "median:             %s\n" (fltfmt 6 median))
   82     (format standard-output "mean:               %s\n" (fltfmt 6 mean))
   83     (format standard-output "average_deviation:  %s\n" (fltfmt 6
   84   average_deviation))
   85     (format standard-output "standard_deviation: %s\n" (fltfmt 6
   86   standard_deviation))
   87     (format standard-output "variance:           %s\n" (fltfmt 6 variance))
   88     (format standard-output "skew:               %s\n" (fltfmt 6 skew))
   89     (format standard-output "kurtosis:           %s\n" (fltfmt 6 kurtosis)))))
   90   )
