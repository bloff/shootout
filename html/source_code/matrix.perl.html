    9 # modified to pass rows and cols, and avoid matrix size checks
   10 # I've sped up the original quite a bit by removing some loop
   11 # invariants and declaring "use integer"
   12 
   13 use strict;
   14 use integer;
   15 
   16 my $size = 30;
   17 
   18 sub mkmatrix {
   19     my($rows, $cols) = @_;
   20     --$rows; --$cols;
   21     my $count = 1;
   22     my @mx = ();
   23     foreach (0 .. $rows) {
   24     my @row = ();
   25     $row[$_] = $count++ foreach (0 .. $cols);
   26     push(@mx, \@row);
   27     }
   28     return(\@mx);
   29 }
   30 
   31 sub mmult {
   32     my ($rows, $cols, $m1, $m2) = @_;
   33     my @m3 = ();
   34     --$rows; --$cols;
   35     for my $i (0 .. $rows) {
   36     my @row = ();
   37     my $m1i = $m1->[$i];
   38     for my $j (0 .. $cols) {
   39         my $val = 0;
   40         for my $k (0 .. $cols) {
   41         $val += $m1i->[$k] * $m2->[$k]->[$j];
   42         }
   43         push(@row, $val);
   44     }
   45     push(@m3, \@row);
   46     }
   47     return(\@m3);
   48 }
   49 
   50 my $N = $ARGV[0] || 1;
   51 
   52 my $m1 = mkmatrix($size, $size);
   53 my $m2 = mkmatrix($size, $size);
   54 my $mm;
   55 while ($N--) {
   56     $mm = mmult($size, $size, $m1, $m2);
   57 }
   58 print "$mm->[0]->[0] $mm->[2]->[3] $mm->[3]->[2] $mm->[4]->[4]\n";
