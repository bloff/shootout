1 #!/usr/bin/mzscheme -r
    2 ;; Moments.scm
    3 
    4 (require (lib "list.ss"))  ; quiksort and mergesort
    5 
    6 (define (main args)
    7   (let* ((sum 0.0)
    8   (numlist (let loop ((line (read-line)) (numlist '()))
    9       (cond ((eof-object? line) numlist)
   10      (else
   11       (let ((num (string->number line)))
   12         (set! sum (+ num sum))
   13         (loop (read-line) (cons num numlist))))))))
   14     (let ((n (length numlist)))
   15       (let ((mean (/ sum n))
   16      (average_deviation 0.0)
   17      (standard_deviation 0.0)
   18      (variance 0.0)
   19      (skew 0.0)
   20      (kurtosis 0.0)
   21      (median 0.0)
   22      (deviation 0.0))
   23  (let loop ((nums numlist))
   24       (if (not (null? nums))
   25    (begin
   26      (set! deviation (- (car nums) mean))
   27      (set! average_deviation (+ average_deviation (abs deviation)))
   28      (set! variance (+ variance (expt deviation 2.0)))
   29      (set! skew (+ skew (expt deviation 3.0)))
   30      (set! kurtosis (+ kurtosis (expt deviation 4)))
   31      (loop (cdr nums)))))
   32 
   33  (set! average_deviation (/ average_deviation (exact->inexact n)))
   34  (set! variance (/ variance (- n 1)))
   35  (set! standard_deviation (sqrt variance))
   36 
   37  (cond ((> variance 0.0)
   38         (set! skew (/ skew (* n variance standard_deviation)))
   39         (set! kurtosis (- (/ kurtosis (* n variance variance))
   40      3.0))))
   41 
   42  (set! numlist (quicksort numlist (lambda (x y) (< x y))))
   43 
   44  (let ((mid (quotient n 2)))
   45    (if (zero? (modulo n 2))
   46        (set! median (/ (+ (car (list-tail numlist mid))
   47      (car (list-tail numlist (- mid 1))))
   48          2.0))
   49      (set! median (car (list-tail numlist mid)))))
   50 
   51 
   52  (set! standard_deviation (/ (round (* standard_deviation 1000000))
   53 1000000))
   54 
   55  (for-each display
   56                   `("n:                  " ,n                   "\n"
   57                     "median:             " ,median              "\n"
   58                     "mean:               " ,mean                "\n"
   59                     "average_deviation:  " ,average_deviation   "\n"
   60                     "standard_deviation: " ,standard_deviation  "\n"
   61                     "variance:           " ,variance            "\n"
   62                     "skew:               " ,skew                "\n"
   63                     "kurtosis:           " ,kurtosis            "\n" ))))))
   64 
   65 
   66
