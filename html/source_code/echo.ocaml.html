    7 open Unix
    8 
    9 let data = "Hello there sailor\n"
   10 
   11 let rec sock_write sock buf offset = function
   12   | 0 -> ()
   13   | len ->
   14       let nwritten = write sock buf offset len in
   15       sock_write sock buf (offset + nwritten) (len - nwritten)
   16 
   17 let sock_readline buf sock =
   18   let offset = ref (read sock buf 0 64) in
   19   while buf.[!offset - 1] <> '\n' do
   20     offset := !offset + read sock buf !offset 64
   21   done;
   22   !offset
   23 
   24 let rec buf_ok buf n = n <= 0 || buf.[n] = data.[n] && buf_ok buf (n - 1)
   25 
   26 let echo_client n port =
   27   let sock = socket PF_INET SOCK_STREAM 0 in
   28   connect sock (ADDR_INET (inet_addr_of_string "127.0.0.1", port));
   29   let len = String.length data and buf = String.create 64 in
   30   for i = 1 to n do
   31     sock_write sock data 0 len;
   32     let ans_len = sock_readline buf sock in
   33     if ans_len <> len || not (buf_ok buf (len - 1)) then
   34       failwith ("client got bad data: " ^ String.sub buf 0 ans_len)
   35   done;
   36   close sock
   37 
   38 let ssock =
   39   let ssock = socket PF_INET SOCK_STREAM 0
   40   and addr = inet_addr_of_string "127.0.0.1" in
   41   bind ssock (ADDR_INET (addr, 0));
   42   setsockopt ssock SO_REUSEADDR true;
   43   listen ssock 2;
   44   ssock
   45 
   46 let get_port sock =
   47   match getsockname sock with
   48   | ADDR_INET (_, port) -> port
   49   | ADDR_UNIX _ -> failwith "getsockname"
   50 
   51 let echo_server n =
   52   let port = get_port ssock and pid = fork() in
   53   if pid <> 0 then begin
   54     let csock, addr = accept ssock
   55     and buf = String.create 64 and len = ref 0 and nread = ref 1 in
   56     while !nread > 0 do
   57       nread := read csock buf 0 64;
   58       sock_write csock buf 0 !nread;
   59       len := !len + !nread
   60     done;
   61     ignore (wait ());
   62     Printf.printf "server processed %d bytes\n" !len end
   63   else echo_client n port
   64 
   65 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
   66         1;;
   67 echo_server n
