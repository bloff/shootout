1 #!/usr/bin/guile \
    2 -e main -s
    3 !#
    4 
    5 ;;; $Id: heapsort.guile.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 
    8 (use-modules (ice-9 format))
    9 
   10 (define IM     139968)
   11 (define IA       3877)
   12 (define IC      29573)
   13 
   14 (define LAST 42)
   15 (define (gen_random max)
   16   (set! LAST (modulo (+ (* LAST IA) IC) IM))
   17   (/ (* max LAST) IM))
   18 
   19 (define (heapsort n ra)
   20   (let ((ir n)
   21     (l (+ (ash n -1) 1))
   22     (i 0)
   23     (j 0)
   24     (rra 0.0))
   25     (define (heapsortloop)
   26        (while #t
   27           (cond ((> l 1)
   28              (set! l (- l 1))
   29              (set! rra (vector-ref ra l)))
   30             (else
   31              (set! rra (vector-ref ra ir))
   32              (vector-set! ra ir (vector-ref ra 1))
   33              (set! ir (- ir 1))
   34              (cond ((= ir 1)
   35                 (vector-set! ra 1 rra)
   36                 (throw 'return)))))
   37           (set! i l)
   38           (set! j (ash l 1))
   39           (while (<= j ir)
   40              (cond ((and (< j ir) (< (vector-ref ra j) (vector-ref ra (+ j 1))))
   41                 (set! j (+ j 1))))
   42              (cond ((< rra (vector-ref ra j))
   43                 (vector-set! ra i (vector-ref ra j))
   44                 (set! i j)
   45                 (set! j (+ j i)))
   46                    (else
   47                 (set! j (+ ir 1)))))
   48           (vector-set! ra i rra)))
   49     (catch 'return
   50        heapsortloop
   51        (lambda args #t))))
   52 
   53 (define (main args)
   54   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   55      (last (+ n 1))
   56      (ary (make-vector last 0)))
   57     (do ((i 1 (+ i 1)))
   58     ((= i last))
   59       (vector-set! ary i (gen_random 1.0)))
   60     (heapsort n ary)
   61     (display (format "~,10F\n" (vector-ref ary n)))))
