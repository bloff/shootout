1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: wc.rep.html,v 1.1 2004-06-14 23:09:29 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system rep.regexp)
   14 
   15   (defun my-string-split (regexp string)
   16     "Return a list of substrings of STRING, each delimited by REGEXP.
   17 null-strings are removed."
   18     (let loop ((point 0)
   19            (parts '()))
   20      (if (string-match regexp string point)
   21          (loop (match-end)
   22            (if (< point (match-start))
   23            (cons (substring string point (match-start)) parts)
   24          parts))
   25        (let ((part (substring string point)))
   26          (if (> (length part) 0)
   27          (cons part parts)
   28            parts)))))
   29 
   30   ;; eat the unused command line arg
   31   (get-command-line-option "--num" t)
   32   (let ((nl 0)
   33     (nw 0)
   34     (nc 0)
   35     (line))
   36     (while (setq line (read-line))
   37       (setq nl (+ nl 1))
   38       (setq nw (+ nw (length (my-string-split "[ \t\n]+" line))))
   39       (setq nc (+ nc (length line))))
   40     (format standard-output "%d %d %d\n" nl nw nc)))
   41 
   42
