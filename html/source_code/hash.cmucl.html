    6 (declaim (optimize (speed 3) (space 0) (safety 0) (debug 0) (compilation-
    7  speed 0)))
    8 (defun command-line-argument ()
    9   (parse-integer (or (car (last ext:*command-line-strings*)) "1")))
   10 
   11 
   12 (defconstant +digit+ "0123456789ABCDEF")
   13 
   14 (defconstant +digits-needed+
   15   #((10 100 1000 10000 100000 10000000 100000000 536870911)
   16     (16 256 4096 65536 1048576 16777216 268435456 4294967296 536870911)))
   17 
   18 (defun fixnum-to-string (n base)
   19   (declare (fixnum n base))
   20   (let* ((size (position-if (lambda (x) (> (the fixnum x) n))
   21                 (aref +digits-needed+ (ash base -4))))
   22      (result (make-string (1+ size))))
   23     (loop for i fixnum from size downto 0 with q fixnum = n and r fixnum =
   24      0
   25       do (multiple-value-setq (q r) (floor q base))
   26          (setf (schar result i) (aref +digit+ r)))
   27     result))
   28 
   29 (defun main (&optional (n (command-line-argument)))
   30   (let ((hash (make-hash-table :test 'equal :size n)))
   31     (macrolet ((hash (i base) `(gethash (fixnum-to-string ,i ,base) hash)))
   32       (loop for i fixnum from 1 to n do (setf (hash i 16) i))
   33       (format t "~a~%" (loop for i fixnum from n downto 1 count (hash i 10)
   34        )))))
