    5 \ read NUM from last command line argument
    6 0. argc @ 1- arg >number 2drop drop constant NUM
    7 
    8 variable hsiz    32                       hsiz !  \ buffer can hold this
    9 much
   10 variable hbuf    hsiz @ allocate throw    hbuf !  \ start of buffer
   11 variable hoff    0                        hoff !  \ current offset in
   12 buffer
   13 
   14 : STUFF s" hello." ;
   15 
   16 : strcat ( straddr strlen -- )
   17     dup                              \ dup strlen on stack
   18     hsiz @ hoff @ - >                \ if strlen > remaining space
   19     if                               \ reallocate buffer
   20     hsiz @ 2* hsiz !             \ double size
   21     hbuf @ hsiz @ resize throw   \ reallocate buffer
   22     hbuf !                       \ store (possibly new) buffer start
   23     then
   24     swap over                        \ stack: strlen straddr strlen
   25     hbuf @ hoff @ +
   26     swap cmove>                      \ append from straddr to hbuf+hoff
   27     hoff @ + hoff !                  \ update hoff
   28     ;
   29 
   30 : main ( -- )
   31     NUM 0
   32     do
   33     STUFF strcat
   34     loop
   35     \ as a final result push the resultant string on the stack as if we
   36     \ were going to use it for something.
   37     hbuf @ hoff @
   38     \ and print out the length
   39     1 u.r cr drop ;
   40 
   41 main
   42 
   43 bye \ th-th-that's all folks!
