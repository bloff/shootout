    5 import sys
    6 
    7 IM = 139968
    8 IA =   3877
    9 IC =  29573
   10 
   11 LAST = 42
   12 def gen_random(max):
   13     global LAST
   14     LAST = (LAST * IA + IC) % IM
   15     return( (max * LAST) / IM )
   16 
   17 def heapsort(n, ra):
   18     rra = i = j = 0
   19     l = (n >> 1) + 1
   20     ir = n
   21 
   22     while (1):
   23         if (l > 1):
   24             l -= 1
   25             rra = ra[l]
   26         else:
   27             rra = ra[ir]
   28             ra[ir] = ra[1]
   29             ir -= 1
   30             if (ir == 1):
   31                 ra[1] = rra
   32                 return
   33         i = l
   34         j = l << 1
   35         while (j <= ir):
   36             if ((j < ir) and (ra[j] < ra[j+1])):
   37                 j += 1
   38             if (rra < ra[j]):
   39                 ra[i] = ra[j]
   40                 i = j
   41                 j += i
   42             else:
   43                 j = ir + 1
   44         ra[i] = rra
   45 
   46 def main():
   47     N = int(sys.argv[1])
   48     if N < 1:
   49         N = 1
   50 
   51     ary = range(N+1)
   52     for i in xrange(1,N+1):
   53         ary[i] = gen_random(1.0)
   54 
   55     heapsort(N, ary)
   56 
   57     print "%.10f" % ary[N]
   58 
   59 main()
