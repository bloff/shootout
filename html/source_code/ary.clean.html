   14 module ary 
   15 import StdEnv, StdArrayExtensions, LanguageShootout 
   16  
   17 Start = toString y`.[0] +++ " " +++ toString y`.[n-1] +++ "\n" 
   18    where 
   19    y` = loop x y (n-1) 1000        
   20     
   21    n = argi     
   22    x = createX n   
   23    y = createUnboxedIntArr n 0 
   24     
   25    loop :: !{#Int} !*{#Int} !Int !Int -> .{#Int}    
   26    loop x y n rep 
   27       | rep==0   = y 
   28                  = loop x (inc x y n) n (rep-1)             
   29                                                    
   30    inc x=:{[j]=xj} y=:{[j]=yj} j 
   31       | j<0   = y 
   32               = inc x {y&[j] = yj + xj} (j-1)       
   33                                      
   34 createX :: !Int -> {#Int} 
   35 createX n = {i \\ i <- [1..n]}
