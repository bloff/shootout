    6 (module wc
    7     (main main)
    8     (option (set! *genericity* #f)))
    9 
   10 (define (main args)
   11   (let* ((num-char 0)
   12          (num-word 0)
   13          (num-line 0)
   14          (g (regular-grammar
   15              ()
   16              ((+ #\newline)
   17               (set! num-char (+ num-char (the-length)))
   18               (set! num-line (+ num-line (the-length)))
   19               (ignore))
   20              ((+ (in #\space #\tab))
   21               (set! num-char (+ num-char (the-length)))
   22               (ignore))
   23              ((+ (out #\newline #\space #\tab))
   24               (set! num-char (+ num-char (the-length)))
   25               (set! num-word (+ 1 num-word))
   26               (ignore)))))
   27     (read/rp g (current-input-port))
   28     (display* num-line " " num-word " " num-char #"\n")))
