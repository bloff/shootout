    7 enum State { Outside, Inside };
    8 
    9 void main() {
   10     int nl = 0, nw = 0, nc = 0; // line, word and character counters
   11     int sl; // Size of input
   12 
   13     State state = Outside; // Inside or outside word
   14 
   15     string buf;
   16     string rest="";
   17     array l;
   18     do {
   19     buf = replace(Stdio.stdin.read( 4196 ), "\t", " ");
   20     if(strlen(buf)) {
   21         nc += sizeof(buf);
   22         l = (rest+ buf) / "\n";
   23         nl += sizeof(l)-1;
   24         foreach(l[..sizeof(l)-2], rest) {
   25         nw += sizeof(rest / " " - ({""}));
   26         }
   27         if(sizeof(l)>1) {
   28         rest = l[-1];
   29         } //else rest="";
   30     } else {
   31         nw += sizeof(rest / " " - ({""}));
   32         break;
   33     }
   34     } while( 1 );
   35     write("%d %d %d\n", nl, nw, nc);
   36 }
