    7 void main(int argc, array(string) argv)
    8 {
    9     int i, c = 0;
   10     int n = (int)argv[-1];
   11     if (n < 1) n = 1;
   12 
   13     mapping(string:int) X = ([]);
   14 
   15     for (i=1; i<=n; i++) {
   16     X[sprintf("%x", i)] = i;
   17     }
   18     for (i=n; i>0; i--) {
   19     if (X[(string)i]) c++;
   20     }
   21     write("%d\n", c);
   22 }
