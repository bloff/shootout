    5 #include <iostream>
    6 #include <stdlib.h>
    7 
    8 using namespace std;
    9 
   10 int main(int argc, char *argv[]) {
   11     int n = ((argc == 2) ? atoi(argv[1]) : 1);
   12     int a, b, c, d, e, f, x=0;
   13 
   14     for (a=0; a<n; a++)
   15     for (b=0; b<n; b++)
   16         for (c=0; c<n; c++)
   17         for (d=0; d<n; d++)
   18             for (e=0; e<n; e++)
   19             for (f=0; f<n; f++)
   20                 x++;
   21 
   22     cout << x << endl;
   23     return(0);
   24 }
