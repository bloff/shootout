    6 (module strcat
    7     (main main))
    8 
    9 (define hello #"hello\n")
   10 
   11 (define (main args)
   12   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   13      (str ""))
   14     (do ((i 0 (+ i 1)))
   15     ((= i n))
   16       (set! str (string-append str hello)))
   17     (display (string-length str))
   18     (newline)))
