    7 let max = 4096
    8 let buf = String.create max
    9 let count = Hashtbl.create 75000 and wbuf = Buffer.create 64
   10 let to_list w c l = Printf.sprintf "%7d\t%s" !c w :: l
   11 
   12 let rec scan_words i n inword =
   13   if i < n then
   14     let c = buf.[i] in
   15     if 'a' <= c && c <= 'z' then begin
   16       Buffer.add_char wbuf c;
   17       scan_words (i + 1) n true
   18     end
   19     else if 'A' <= c && c <= 'Z' then begin
   20       Buffer.add_char wbuf (Char.unsafe_chr (Char.code c + 32));
   21       scan_words (i + 1) n true
   22     end
   23     else if inword then begin
   24       let word = Buffer.contents wbuf in
   25       begin try incr (Hashtbl.find count word)
   26       with Not_found -> Hashtbl.add count word (ref 1) end;
   27       Buffer.clear wbuf;
   28       scan_words (i + 1) n false
   29     end
   30     else scan_words (i + 1) n false
   31   else
   32     let nread = input stdin buf 0 max in
   33     if nread <> 0 then scan_words 0 nread inword;;
   34 
   35 scan_words 0 0 false;
   36 let out_lines = Hashtbl.fold to_list count [] in
   37 List.iter print_endline (List.sort (fun a b -> compare b a) out_lines)
