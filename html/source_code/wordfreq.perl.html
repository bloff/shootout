    5 # Tony Bowden suggested using tr versus lc and split(/[^a-z]/)
    6 
    7 use strict;
    8 
    9 my %count = ();
   10 while (read(STDIN, $_, 4095) and $_ .= <STDIN>) {
   11     tr/A-Za-z/ /cs;
   12     ++$count{$_} foreach split(' ', lc $_);
   13 }
   14 
   15 my @lines = ();
   16 my ($w, $c);
   17 push(@lines, sprintf("%7d\t%s\n", $c, $w)) while (($w, $c) = each(%count));
   18 print sort { $b cmp $a } @lines;
