    6 def ack(m, n)
    7     if m == 0 then
    8     n + 1
    9     elsif n == 0 then
   10     ack(m - 1, 1)
   11     else
   12     ack(m - 1, ack(m, n - 1))
   13     end
   14 end
   15 
   16 NUM = Integer(ARGV.shift || 1)
   17 print "Ack(3,", NUM, "): ", ack(3, NUM), "\n"
