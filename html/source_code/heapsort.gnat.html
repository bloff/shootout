    8 with System, Ada.Command_Line, Text_IO;
    9 
   10 procedure Heapsort is
   11    type Real is digits Positive'Max (15, System.Max_Digits);
   12    package Rio is new Text_IO.Float_IO (Num => Real);
   13 
   14    package Random_Real is
   15       function Gen_Random (Supr : Real) return Real;
   16       pragma Inline (Gen_Random);
   17    end Random_Real;
   18 
   19    package body Random_Real is
   20       IM          : constant Positive := 139968;
   21       IA          : constant Integer := 3877;
   22       IC          : constant Integer := 29573;
   23       Last        : Integer := 42;
   24 
   25       function Gen_Random (Supr : Real) return Real is
   26          pragma Suppress (Overflow_Check);
   27          pragma Suppress (Range_Check);
   28       begin
   29          Last := (Last * IA + IC) mod IM;
   30          return Supr * Real (Last) / Real (IM);
   31       end Gen_Random;
   32    end Random_Real;
   33 
   34    type Range_Int is new Integer;
   35    subtype Offset_Int is Range_Int;
   36 
   37    type Real_Array is array (Range_Int range <>) of Real;
   38    type Real_Array_Ptr is access Real_Array;
   39 
   40    procedure Heapsort (A : in out Real_Array) is
   41       pragma Suppress (Overflow_Check);
   42       pragma Suppress (Index_Check);
   43       pragma Suppress (Range_Check);
   44       subtype Range_Positive is Range_Int;
   45       First          : constant Range_Int := A'First;    --  might be <= -
   46                                              1
   47       IR             : Range_Positive;
   48       One            : constant Offset_Int := 1;
   49       Minus_One      : constant Offset_Int := -1;
   50       First_Minus_1  : constant Range_Int := First + Minus_One;
   51       First_Plus_1   : constant Range_Int := First + One;
   52       RRA            : Real;
   53       L              : Offset_Int := One + (A'Length / 2);
   54    begin
   55       if A'Length <= 0 then
   56          return;
   57       end if;
   58       IR := A'Last;
   59       loop
   60          if L > One then
   61             L := L - One;
   62             RRA := A (First_Minus_1 + L);
   63          else
   64             RRA := A (IR);
   65             A (IR) := A (First);
   66             if IR <= First_Plus_1 then
   67                A (First) := RRA;
   68                exit;
   69             else
   70                IR := IR + Minus_One;
   71             end if;
   72          end if;
   73          declare
   74             K1    : Range_Positive := First_Minus_1 + L;
   75             K2    : Range_Positive := K1 + L;
   76          begin
   77             while K2 <= IR loop
   78                if K2 < IR then
   79                   if A (K2) < A (K2 + One) then
   80                      K2 := K2 + One;
   81                   end if;
   82                end if;
   83                if RRA < A (K2) then
   84                   A (K1) := A (K2);
   85                   K1 := K2;
   86                   K2 := K1 + (K1 - First_Minus_1);
   87                else
   88                   K2 := IR + One;
   89                end if;
   90             end loop;
   91             A (K1) := RRA;
   92          end;
   93       end loop;
   94    end Heapsort;
   95 
   96    N           : Range_Int;
   97    No_Verify   : constant Boolean := True;
   98    Chk         : Real := 0.0;
   99    X           : Real_Array_Ptr;
  100 begin
  101    begin
  102       N := Range_Int'Max (1, Range_Int'Value (Ada.Command_Line.Argument (1)
  103            ));
  104    exception
  105       when Constraint_Error => N := 1;
  106    end;
  107    X := new Real_Array (0 .. N - 1);   --  3% slower than 'declare' (stack)
  108    for Iter in X'Range loop
  109       X (Iter) := Random_Real.Gen_Random (Supr => 1.0);
  110    end loop;
  111    if No_Verify then
  112       Heapsort (A => X.all);
  113       Rio.Put (X (X'Last), Fore => 0, Aft => 10, Exp => 0);
  114       Text_IO.New_Line;
  115    else
  116       for Iter in X'Range loop Chk := Chk + X (Iter); end loop;
  117       Heapsort (A => X.all);
  118       for K in X'Range loop
  119          pragma Assert (K + 1 = X'Last or else X (K) <= X (K + 1));
  120          Chk := Chk - X (K);
  121       end loop;
  122       pragma Assert (abs Chk < 50.0 * Real (N) * Real'Model_Epsilon);
  123    end if;
  124 end Heapsort;
