    6 -module(ary).
    7 -export([main/1]).
    8 
    9 main() -> main(['1']).
   10 main([Arg]) ->
   11     Num = list_to_integer(atom_to_list(Arg)),
   12     ets:new(y, [set, private, named_table]),
   13     clear_y_array(Num),
   14     X = list_to_tuple(lists:seq(1, Num)),
   15     repeat(X, Num, 1000),
   16     [{_,First}] = ets:lookup(y, 1),
   17     [{_,Last}]  = ets:lookup(y, Num),
   18     io:fwrite("~w ~w~n", [First, Last]),
   19     ets:delete(y),
   20     halt(0).
   21 
   22 clear_y_array(0) -> ok;
   23 clear_y_array(I) ->
   24     ets:insert(y, {I,0}),
   25     clear_y_array(I - 1).
   26 
   27 repeat(X, N, 0) -> ok;
   28 repeat(X, N, K) ->
   29     calc(X, N),
   30     repeat(X, N, K - 1).
   31 
   32 calc(X, 0) -> ok;
   33 calc(X, N) ->
   34     ets:update_counter(y, N, element(N, X)),
   35     calc(X, N - 1).
