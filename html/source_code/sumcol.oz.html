    3 functor
    4 import
    5     System(printInfo)
    6     Application(exit)
    7     Open(text file)
    8     OS
    9 define
   10 class TextFile
   11   from Open.file Open.text
   12 end
   13 
   14 fun {SumLines FILE SUM}
   15     {System.printInfo "file.atEnd="}
   16     if {FILE atEnd($)} == true then
   17         {System.printInfo "Y"}
   18     else
   19         {System.printInfo "N"}
   20     end
   21     {System.printInfo "\n"}
   22     case {FILE getS($)} of false then
   23         {System.printInfo "file terminated, returning "}
   24         {System.printInfo SUM}
   25         {System.printInfo "\n"}
   26         SUM
   27     elseof LINE then
   28         {System.printInfo "got "}
   29         {System.printInfo LINE}
   30         {System.printInfo "\n"}
   31         {SumLines FILE SUM+{String.toInt LINE}}
   32     end
   33 end
   34 
   35 in
   36     local STDIN SUM in
   37         STDIN = {New TextFile init(name:stdin)}
   38         SUM = {SumLines STDIN 0}
   39         {System.printInfo SUM}
   40     end
   41     {Application.exit 0}
   42 end
