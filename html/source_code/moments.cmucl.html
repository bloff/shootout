    9 (defun quicksort (vec lo hi) ;; modified from from Roger Corman's posting
   10  in cll
   11   (declare (fixnum hi lo) (type (simple-array double-float) vec))
   12     (if (> hi lo)
   13         (let* ((mid (round (+ lo hi) 2))
   14                (i lo)
   15                (j (+ hi 1))
   16                (p (aref vec mid)))
   17       (declare (fixnum i j) (double-float p))
   18             (rotatef (aref vec mid) (aref vec lo)) ;; swap mid element to
   19              first
   20             (loop
   21                 (loop do (incf i)
   22                     until (or (> i hi) (> p (aref vec i))))
   23                 (loop do (decf j)
   24                     until (or (<= j lo) (> (aref vec j) p)))
   25         (if (< j i) (return))
   26                 (rotatef (aref vec i)(aref vec j)))
   27   (rotatef (aref vec lo) (aref vec j)) ;;put partition element in place
   28   (quicksort vec lo (- j 1))  (quicksort vec i hi))) vec)

   30 (defun do-moments (data n mean)
   31   (declare (fixnum n) (double-float mean) (type (simple-array double-float)
   32    data))
   33   (let ((average_deviation 0.0d0)
   34     (standard_deviation 0.0d0)
   35     (variance 0.0d0)
   36     (skew 0.0d0)
   37     (kurtosis 0.0d0)
   38     (median 0.0d0))
   39     (declare (double-float mean average_deviation standard_deviation
   40                variance skew kurtosis median))
   41     (declare (inline quicksort))
   42     (loop for i fixnum from 0 below n do
   43       (let* ((deviation (- (the double-float (aref data i)) mean))
   44          (deviation2 (* deviation deviation))
   45          (deviation3 (* deviation deviation2))
   46          (deviation4 (* deviation deviation3)))
   47     (incf average_deviation (abs deviation))
   48     (incf variance deviation2)
   49     (incf skew deviation3)
   50     (incf kurtosis deviation4)))
   51 
   52     (setq average_deviation (/ average_deviation n))
   53     (setq variance (/ variance (1- n)))
   54     (setq standard_deviation (sqrt variance))
   55 
   56     (cond ((> variance 0.0d0)
   57        (setq skew (/ skew (* n variance standard_deviation)))
   58        (setq kurtosis (- (/ kurtosis (* (coerce n 'double-float)
   59                         variance variance))
   60                  3.0d0))))
   61     (setf data (quicksort data 0 (1- n)))
   62     (let ((mid (/ n 2)))
   63       (fixnum mid)
   64       (if (zerop (mod n 2))
   65       (setq median (/ (+ (the double-float (aref data mid))
   66                  (the double-float (aref data (1- mid))))
   67               2.0d0))
   68     (setq median (aref data mid))))
   69     (format t "n:                  ~A~%" n)
   70     (format t "median:             ~,6F~%" median)
   71     (format t "mean:               ~,6F~%" mean)
   72     (format t "average_deviation:  ~,6F~%" average_deviation)
   73     (format t "standard_deviation: ~,6F~%" standard_deviation)
   74     (format t "variance:           ~,6F~%" variance)
   75     (format t "skew:               ~,6F~%" skew)
   76     (format t "kurtosis:           ~,6F~%" kurtosis)))
