    7 structure Test : sig
    8     val main : (string * string list) -> OS.Process.status
    9 end = struct
   10 
   11 fun incr r = r := !r + 1
   12 fun print_endline s = (print s; print "\n")
   13 fun for (start, stop, f) =
   14    let
   15       fun loop i =
   16      if i > stop
   17         then ()
   18      else (f i; loop (i + 1))
   19    in
   20       loop start
   21    end
   22 
   23 exception HiException of int
   24 exception LoException of int
   25 
   26 val hi = ref 0
   27 val lo = ref 0
   28 
   29 fun blowup n =
   30   if n mod 2 = 0 then raise (LoException n)
   31   else raise (HiException n)
   32 
   33 fun lo_fun n =
   34   blowup n
   35   handle LoException ex => incr lo
   36 
   37 fun hi_fun n =
   38   lo_fun n
   39   handle HiException ex => incr hi
   40 
   41 fun some_fun n =
   42   hi_fun n
   43   handle x =>
   44      (print_endline "Should not get here.";
   45       raise x)
   46 
   47 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   48 fun printl [] = print "\n" | printl(h::t) = ( print h ; printl t );
   49 
   50 fun main (name, args) =
   51    let
   52       val n = atoi (hd (args @ ["1"]))
   53       val _ = for (1, n, some_fun)
   54    in
   55       printl ["Exceptions: HI=",
   56           Int.toString (!hi),
   57           " / LO=",
   58           Int.toString (!lo)];
   59       OS.Process.success
   60    end
   61 end
   62 
   63 val _ = SMLofNJ.exportFn("except", Test.main);
