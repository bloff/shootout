1 #!/usr/bin/guile \
    2 -e main -s
    3 !#
    4 
    5 ;;; $Id: random.guile.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 
    8 (use-modules (ice-9 format))
    9 
   10 (define IM   139968)
   11 (define IA     3877)
   12 (define IC    29573)
   13 
   14 (define LAST 42)
   15 (define (gen_random max)
   16   (set! LAST (modulo (+ (* LAST IA) IC) IM))
   17   (/ (* max LAST) IM))
   18 
   19 (define (main args)
   20   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   21     (let loop ((iter n))
   22       (if (> iter 1)
   23       (begin
   24         (gen_random 100.0)
   25         (loop (- iter 1)))))
   26     (display (format "~,9F\n" (gen_random 100.0)))))
