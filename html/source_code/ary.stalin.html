   11 (define (string->exact-integer string)
   12  (let ((n (string->number string)))
   13   (if (integer? n)
   14       (if (exact? n)
   15       n
   16       (panic "STRING->EXACT-INTEGER yields non-EXACT-INTEGER"))
   17       (panic "STRING->EXACT-INTEGER yields non-EXACT-INTEGER"))))
   18 
   19 (define (main args)
   20   (let* ((n (or (and (= (length args) 2) (string->exact-integer (cadr args)
   21    )) 1))
   22      (x (make-vector n 0))
   23      (y (make-vector n 0))
   24      (last (- n 1)))
   25     (do ((i 0 (+ i 1)))
   26     ((= i n))
   27       (vector-set! x i (+ i 1)))
   28     (do ((k 0 (+ k 1)))
   29     ((= k 1000))
   30       (do ((i last (- i 1)))
   31       ((< i 0))
   32     (vector-set! y i (+ (vector-ref x i) (vector-ref y i)))))
   33     (print-list (vector-ref y 0) " " (vector-ref y last))))
   34 
   35 (define (print-list . items) (for-each display items) (newline))
   36 
   37 (main (vector->list argv))
