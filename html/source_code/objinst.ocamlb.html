    8 class toggle start_state = object (self)
    9   val mutable state = start_state
   10 
   11   method value = state
   12   method activate = state <- not state; self
   13 end
   14 
   15 class nth_toggle start_state max_counter = object (self)
   16   inherit toggle start_state
   17 
   18   val count_max = max_counter
   19   val mutable counter = 0
   20 
   21   method activate =
   22     counter <- counter + 1;
   23     if counter >= count_max then begin
   24       state <- not state;
   25       counter <- 0
   26     end;
   27     self
   28 end
   29 
   30 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
   31         1
   32 let tog = new toggle true;;
   33 for i = 1 to 5 do Printf.printf "%b\n" tog#activate#value done;
   34 for i = 1 to n do ignore (new toggle true) done;
   35 print_newline ();
   36 let ntog = new nth_toggle true 3 in
   37 for i = 1 to 8 do Printf.printf "%b\n" ntog#activate#value done;
   38 for i = 1 to n do ignore (new nth_toggle true 3) done
