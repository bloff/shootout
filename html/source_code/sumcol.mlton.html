    6 fun sumlines sum =
    7   if TextIO.endOfStream TextIO.stdIn
    8   then (print (Int.toString sum); print "\n")
    9   else case (TextIO.inputLine TextIO.stdIn) of
   10       NONE => sumlines sum
   11     | SOME(str) =>
   12         (case (Int.fromString str) of
   13         NONE => sumlines sum
   14       | SOME(i) => sumlines (sum + i));
   15 
   16 fun main(name, args) = (sumlines 0; OS.Process.success);
