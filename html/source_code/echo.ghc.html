   20 server_sock = do
   21     s <- socket AF_INET Stream 6
   22     setSocketOption s ReuseAddr 1
   23     bindSocket s (SockAddrInet portnum iNADDR_ANY)
   24     listen s 2
   25     return s
   26 
   27 echo_server s = do
   28     (s', clientAddr) <- accept s
   29     proc <- read_data s' 0
   30     putStrLn ("server processed "++(show proc)++" bytes")
   31     sClose s'
   32     where
   33        read_data sock totalbytes = do
   34          str <- catch (recv sock 19) (\e -> if isEOFError e then return ""
   35                     else ioError e)
   36          if ((length str) >= 19)
   37             then (do
   38               writ <- send sock str
   39           read_data sock (totalbytes+(length str)))
   40             else (return totalbytes)
   41 
   42 local        = "127.0.0.1"
   43 message        = "Hello there sailor\n"
   44 portnum :: PortNumber  = 7001
   45 
   46 client_sock = do
   47     s <- socket AF_INET Stream 6
   48     ia <- inet_addr local
   49     connect s (SockAddrInet portnum ia)
   50     return s
   51 
   52 echo_client n = do
   53     s <- client_sock
   54     drop <- server_echo s n
   55     sClose s
   56     where
   57       server_echo sock n
   58           | n > 0 =(do
   59                        send sock message
   60                str <- catch (recv sock 19) (\e -> if isEOFError e then return ""
   61                        else ioError e)
   62                        if (str /= message) then (do exitFailure)
   63                        else server_echo sock (n-1))
   64           | otherwise = (return [])
   65 
   66 main = do
   67      ~[n] <- getArgs
   68      -- server & client semaphores
   69          --get the server socket
   70          ssock <- server_sock
   71      -- fork off the server
   72      s <- myForkIO (echo_server ssock)
   73      -- fork off the client
   74      c <- myForkIO (echo_client (read n::Int))
   75      -- let 'em run until they've signaled they're done
   76          join s ; join c
   77      hFlush stdout
   78 
   79 -- these are used to make the main thread wait until
   80 -- the child threads have exited
   81 myForkIO :: IO () -> IO (MVar ())
   82 myForkIO io = do
   83   mvar <- newEmptyMVar
   84   forkIO (io `finally` putMVar mvar ())
   85   return mvar
   86 
   87 join :: MVar () -> IO ()
   88 join mvar = readMVar mvar
