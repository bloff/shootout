    5 import java.io.*;
    6 import java.util.*;
    7 import java.text.*;
    8 
    9 class Counter {
   10     int count = 1;
   11 }
   12 
   13 public class wordfreq {
   14 
   15     public static void main(String[] args) {
   16     wf();
   17     }
   18 
   19     public static String padleft(String s,int n,char c) {
   20         int len = s.length();
   21         if( len>=n ) return s;
   22         char[] buf = new char[n];
   23         for( int i=0;i<n-len;i++ ) buf[i]=c;
   24         s.getChars(0,len,buf,n-len);
   25         return new String(buf);
   26     }
   27 
   28     public static void wf() {
   29         HashMap map = new HashMap();
   30         try {
   31             Reader r = new BufferedReader(new InputStreamReader(System.in))
   32                        ;
   33             StreamTokenizer st = new StreamTokenizer(r);
   34             st.lowerCaseMode(true);
   35             st.whitespaceChars( 0, 64 );
   36             st.wordChars(65, 90);
   37             st.whitespaceChars( 91, 96 );
   38             st.wordChars(97, 122);
   39             st.whitespaceChars( 123, 255 );
   40             int tt = st.nextToken();
   41             while (tt != StreamTokenizer.TT_EOF) {
   42                 if (tt == StreamTokenizer.TT_WORD) {
   43                     if (map.containsKey(st.sval)) {
   44                         ((Counter)map.get(st.sval)).count++;
   45                     } else {
   46                         map.put(st.sval, new Counter());
   47                     }
   48                 }
   49                 tt = st.nextToken();
   50             }
   51         } catch (IOException e) {
   52             System.err.println(e);
   53             return;
   54         }
   55 
   56         Collection entries = map.entrySet();
   57     // flatten the entries set into a vector for sorting
   58     Vector rev_wf = new Vector(entries);
   59 
   60         // Sort the vector according to its value
   61         Collections.sort(rev_wf, new Comparator() {
   62         public int compare(Object o1, Object o2) {
   63             // First sort by frequency
   64             int c = ((Counter)((Map.Entry)o2).getValue()).count - ((Counter)((
   65               Map.Entry)o1).getValue()).count;
   66             if (c == 0) { // Second sort by lexicographical order
   67             c = ((String)((Map.Entry)o2).getKey()).compareTo((String)((Map.Entry)o1)
   68        .getKey());
   69             }
   70             return c;
   71         }
   72         }
   73              );
   74 
   75         Iterator it = rev_wf.iterator();
   76     Map.Entry ent;
   77     String word;
   78     int count;
   79         while(it.hasNext()) {
   80             ent = (Map.Entry)it.next();
   81         word = ((String)ent.getKey());
   82         count = ((Counter)ent.getValue()).count;
   83         System.out.println(padleft(Integer.toString(count),7,' ') + "\t" +
   84                         word);
   85         }
   86     }
   87 }
