    5 ;;; $Id: ary.guile.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $

    7 
    8 (define (main args)
    9   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   10      (x (make-vector n 0))
   11      (y (make-vector n 0))
   12      (last (- n 1)))
   13     (do ((i 0 (+ i 1)))
   14     ((= i n))
   15       (vector-set! x i (+ i 1)))
   16     (do ((k 0 (+ k 1)))
   17     ((= k 1000))
   18       (do ((i last (- i 1)))
   19       ((< i 0))
   20     (vector-set! y i (+ (vector-ref x i) (vector-ref y i)))))
   21     (print-list (vector-ref y 0) " " (vector-ref y last))))
   22 
   23 (define (print-list . items) (for-each display items) (newline))
