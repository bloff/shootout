    2 -e main -s
    3 !#

    5 ;;; $Id: strcat.guile.html,v 1.1 2004-06-14 23:09:29 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 ;;; from Benedikt Rosenau
    8 
    9 (use-modules (oop goops))
   10 
   11 (define-class <buffer> ()
   12   (siz #:getter buffer-size #:init-value 64)
   13   (len #:getter buffer-length #:init-value 0)
   14   (field #:init-value (make-string 64)))
   15 
   16 (define-method (buffer->string (b <buffer>))
   17   (substring (slot-ref b 'field) 0 (buffer-length b)))
   18 
   19 (define-method (buffer-append! (b <buffer>) (s <string>))
   20   (let* ((length-b (buffer-length b))
   21          (size-b (buffer-size b))
   22          (length-s (string-length s))
   23          (new-length (+ length-b length-s)))
   24     (if (> new-length size-b)
   25       (let* ((new-size (+ size-b (max length-b length-s)))
   26              (new-field (make-string new-size)))
   27         (substring-move-left! (slot-ref b 'field) 0 length-b new-field 0)
   28         (slot-set! b 'field new-field)
   29         (slot-set! b 'siz new-size)))
   30     (substring-move-left! s 0 length-s (slot-ref b 'field) length-b)
   31     (slot-set! b 'len new-length)
   32      b))
   33 
   34 
   35 (define-method (main (args <list>))
   36   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   37     (buf (make <buffer>)))
   38     (do ((i 0 (+ i 1))) ((= i n))
   39         (buffer-append! buf "hello\n"))
   40     (display (buffer-length buf))
   41     (newline)))
