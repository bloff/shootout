    5 import sys
    6 
    7 def main():
    8     N = int(sys.argv[1])
    9     if N < 1:
   10         N = 1
   11 
   12     # create list (A) of integers from 1 through N
   13     A = range(1, N + 1)
   14 
   15     # create empty list B
   16     B = []
   17 
   18     # move each individual item from head of A to head of B, in a loop
   19     # the result should be in reverse order
   20     while A:
   21         B.append(A.pop())
   22 
   23     # print first 2 elements of B
   24     print B[0], B[1]
   25 
   26     # reverse B (can be done in place)
   27     B.reverse()
   28 
   29     # is 0 a member of B?
   30     if 0 in B:
   31         print "true"
   32     else:
   33         print "false"
   34 
   35     # is N a member of B?
   36     if N in B:
   37         print "true"
   38     else:
   39         print "false"
   40 
   41     # filter values from B to A that are less than N/2, preserving order
   42     mid = N/2
   43     #A = filter(lambda item: item < mid, B)
   44     for item in B:
   45         if item < mid:
   46             A.append(item)
   47 
   48     # print first ten items of A
   49     print (" ").join(map(lambda x: "%d" % x, A[0:10]))
   50 
   51     # print sum of items in A that are less than 1000
   52     print reduce(lambda x, y: x + y, (filter(lambda z: z < 1000, A)))
   53 
   54     # append B to end of A
   55     A = A + B
   56 
   57     # print length and last element of A
   58     print len(A), A[-1]
   59 
   60 main()
