   12 : build ( -- )
   13     get-current hash1 set-current
   14     10000 0 ?do
   15     i 0 <# #s '_ hold 'o hold 'o hold 'f hold #> nextname i constant
   16     loop
   17     set-current ;
   18 
   19 : search-new ( addr u wid -- xt )
   20     >r 2dup r@ search-wordlist if
   21     rdrop nip nip
   22     else
   23     nextname get-current r> set-current 0 constant set-current
   24     lastxt
   25     endif ;
   26 
   27 : add-to-hash2 ( nt -- )
   28     dup name>int execute ( nt n )
   29     swap name>string hash2 search-new >body +! ;
   30 
   31 : build2 ( -- )
   32     hash1 wordlist-id begin
   33     @ dup
   34     while
   35     dup add-to-hash2
   36     repeat
   37     drop ;
   38 
   39 : countdecs ( -- )
   40     NUM 0 ?do
   41     build2
   42     loop ;
