    5 with Text_IO, Ada.Strings.Fixed, Ada.Command_Line;
    6 
    7 procedure Matrix is
    8    function L_Trim (Source : String; Side : Ada.Strings.Trim_End :=
    9                Ada.Strings.Left) return String renames Ada.Strings.Fixed.
   10                Trim;
   11 
   12    type Int is new Integer;
   13    type Int_Matrix is array (Positive range <>, Positive range <>) of Int;
   14 
   15    function Mk_Matrix (NRows, NCols : Natural) return Int_Matrix is
   16       Count    : Int := 1;
   17       M        : Int_Matrix (1 .. NRows, 1 .. NCols);
   18    begin
   19       for I in M'Range (1) loop
   20          for J in M'Range (2) loop
   21             M (I, J) := Count;
   22             Count := Count + 1;
   23          end loop;
   24       end loop;
   25       return M;
   26    end Mk_Matrix;
   27 
   28    procedure M_Mult (M1, M2 : Int_Matrix; MM : in out Int_Matrix) is
   29       pragma Inline (M_Mult);
   30       pragma Suppress (Index_Check);
   31       Sum      : Int;
   32    begin
   33       if not (M1'First (2) = M2'First (1) and M1'Last (2) = M2'Last (1)
   34               and
   35                M1'First (1) = MM'First (1) and M1'Last (1) = MM'Last (1)
   36                          and
   37                M2'First (2) = MM'First (2) and M2'Last (2) = MM'Last (2))
   38                          then
   39          raise Constraint_Error;
   40       end if;
   41       for I in M1'Range (1) loop
   42          for J in M2'Range (2) loop
   43             Sum := 0;
   44             for KK in M1'Range (2) loop
   45                Sum := Sum + M1 (I, KK) * M2 (KK, J);
   46             end loop;
   47             MM (I, J) := Sum;
   48          end loop;
   49       end loop;
   50    end M_Mult;
   51 
   52    Size     : constant Natural := 30;
   53    M1, M2, MM : Int_Matrix (1 .. Size, 1 .. Size);
   54    N        : Positive := 1;
   55 begin
   56    begin
   57       N := Positive'Value (Ada.Command_Line.Argument (1));
   58    exception
   59       when Constraint_Error => null;
   60    end;
   61    M1 := Mk_Matrix (Size, Size);
   62    M2 := Mk_Matrix (Size, Size);
   63    for Iter in 1 .. N loop
   64       M_Mult (M1, M2, MM);
   65    end loop;
   66    Text_IO.Put_Line (L_Trim (Int'Image (MM (1, 1))) & Int'Image (MM (3, 4))
   67                      &
   68             Int'Image (MM (4, 3)) & Int'Image (MM (5, 5)));
   69 end Matrix;
