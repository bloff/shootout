    5 import sys
    6 from threading import *
    7 
    8 access = Condition()
    9 count = 0
   10 consumed = 0
   11 produced = 0
   12 data = 0
   13 
   14 def consumer(n):
   15     global count, data, consumed
   16     while 1:
   17         access.acquire()
   18         while count == 0:
   19             access.wait()
   20         i = data
   21         count = 0
   22         access.notify()
   23         access.release()
   24         consumed += 1
   25         if i == n:
   26             break
   27 
   28 def producer(n):
   29     global count, data, produced
   30     for i in xrange(1,n+1):
   31         access.acquire()
   32         while count == 1:
   33             access.wait()
   34         data = i
   35         count = 1
   36         access.notify()
   37         access.release()
   38         produced += 1
   39 
   40 def main(n):
   41     t1 = Thread(target=producer, args=(n,))
   42     t2 = Thread(target=consumer, args=(n,))
   43     t1.start()
   44     t2.start()
   45     t1.join()
   46     t2.join()
   47     print produced, consumed
   48 
   49 main(int(sys.argv[1]))
