    4 import System(getArgs, getProgName, exitWith, ExitCode(..))
    5 
    6 main = do
    7        arg <- getArgs
    8        case arg of
    9          [number] -> putStrLn (show (fib (read number)))
   10          _        -> do
   11                      progname <- getProgName
   12                      putStrLn ("Usage: " ++ progname ++ " number")
   13                      exitWith (ExitFailure 1)
   14 
   15 
   16 fib :: Int -> Int
   17 fib 0 = 1
   18 fib 1 = 1
   19 fib n = fib (n-2) + fib (n-1)
