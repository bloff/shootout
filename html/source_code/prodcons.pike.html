    6 inherit Thread.Condition: access;
    7 inherit Thread.Mutex: mutex;
    8 int data, consumed, produced, count;
    9 
   10 void producer(int n) {
   11     for (int i=1; i<=n; i++) {
   12     object mtx = mutex::lock();
   13     while (count != 0) access::wait(mtx);
   14     data = i;
   15     count += 1;
   16     destruct(mtx);
   17     access::signal();
   18     produced += 1;
   19     }
   20 }
   21 
   22 void consumer(int n) {
   23     while (1) {
   24     object mtx = mutex::lock();
   25     while (count == 0) access::wait(mtx);
   26     int i = data;
   27     count -= 1;
   28     access::signal();
   29     destruct(mtx);
   30     consumed += 1;
   31     if (i == n) break;
   32     }
   33 }
   34 
   35 void main(int argc, array(string) argv) {
   36     int n = (int)argv[-1];
   37     if (n < 1) n = 1;
   38     data = consumed = produced = count = 0;
   39     thread_create(producer, n);
   40     consumer(n);
   41     write("%d %d\n", produced, consumed);
   42 }
