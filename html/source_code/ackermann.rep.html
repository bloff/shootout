1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: ackermann.rep.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   (define (ack M N)
   16     (cond ((= M 0) (+ N 1))
   17       ((= N 0) (ack (- M 1) 1))
   18       (t (ack (- M 1) (ack M (- N 1))))))
   19 
   20   (let* ((arg (get-command-line-option "--num" t))
   21      (n (if arg (read-from-string arg) 1)))
   22     (max-lisp-depth 2500)
   23     (format standard-output "Ack(3,%d): %d\n" n (ack 3 n))))
