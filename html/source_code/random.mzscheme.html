    7 #! /bin/sh
    8 #|
    9 exec mzscheme -qr "$0" ${1+"$@"}
   10 |#
   11 
   12 (module random mzscheme
   13    (provide main)
   14    ; (define-constant name val) "replaces" all occurences
   15    ; of name with val.
   16    (define-syntax (define-constant stx)
   17      (syntax-case stx ()
   18        ((define-constant id val)
   19         (syntax (define-syntax id (syntax-id-rules () (_ val)))))))
   20 
   21    (define-constant IM 139968)
   22    (define-constant IA   3877)
   23    (define-constant IC  29573)
   24 
   25    (define gen_random
   26      (let ((LAST 42))
   27        (lambda (max)
   28          (set! LAST (modulo (+ (* LAST IA) IC) IM))
   29          (/ (* max LAST) IM))))
   30 
   31    (define (main args)
   32      (let ((n (if (= (vector-length args) 0)
   33                   1
   34                   (string->number (vector-ref args 0)))))
   35        (let loop ((iter n))
   36          (if (> iter 1)
   37              (begin
   38                (gen_random 100.0)
   39                (loop (- iter 1)))))
   40        (display (gen_random 100.0))
   41        (newline))))
   42 
   43 (require random)
   44 (main (current-command-line-arguments))
