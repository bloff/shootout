1 #!/usr/bin/mzscheme -r
    2 ;; Matrix.scm
    3 
    4 (define size 30)
    5 
    6 (define (1+ x) (+ x 1))
    7 
    8 (define (mkmatrix rows cols)
    9   (let ((mx (make-vector rows 0))
   10  (count 1))
   11     (do ((i 0 (1+ i)))
   12  ((= i rows))
   13       (let ((row (make-vector cols 0)))
   14  (do ((j 0 (1+ j)))
   15      ((= j cols))
   16    (vector-set! row j count)
   17    (set! count (+ count 1)))
   18  (vector-set! mx i row)))
   19     mx))
   20 
   21 (define (num-cols mx)
   22   (let ((row (vector-ref mx 0)))
   23     (vector-length row)))
   24 
   25 (define (num-rows mx)
   26   (vector-length mx))
   27 
   28 (define (mmult rows cols m1 m2)
   29   (let ((m3 (make-vector rows 0)))
   30     (do ((i 0 (1+ i)))
   31  ((= i rows))
   32       (let ((m1i (vector-ref m1 i))
   33      (row (make-vector cols 0)))
   34  (do ((j 0 (1+ j)))
   35      ((= j cols))
   36    (let ((val 0))
   37      (do ((k 0 (1+ k)))
   38   ((= k cols))
   39        (set! val (+ val (* (vector-ref m1i k)
   40       (vector-ref (vector-ref m2 k) j)))))
   41      (vector-set! row j val)))
   42  (vector-set! m3 i row)))
   43     m3))
   44 
   45 (define (matrix-print m)
   46   (do ((i 0 (1+ i)))
   47       ((= i (num-rows m)))
   48     (let ((row (vector-ref m i)))
   49       (do ((j 0 (1+ j)))
   50    ((= j (num-cols m)))
   51  (display (vector-ref row j))
   52  (if (< j (num-cols m))
   53      (display " ")))
   54       (newline))))
   55 
   56 (define (print-list . items) (for-each display items) (newline))
   57 
   58 (define (main args)
   59   (let ((n (or (and (= (vector-length args) 1) (string->number (vector-ref
   60 args 0)))
   61                1)))
   62     (let ((mm 0)
   63    (m1 (mkmatrix size size))
   64    (m2 (mkmatrix size size)))
   65       (let loop ((iter n))
   66  (cond ((> iter 0)
   67         (set! mm (mmult size size m1 m2))
   68         (loop (- iter 1)))))
   69       (let ((r0 (vector-ref mm 0))
   70      (r2 (vector-ref mm 2))
   71      (r3 (vector-ref mm 3))
   72      (r4 (vector-ref mm 4)))
   73  (print-list (vector-ref r0 0) " " (vector-ref r2 3) " "
   74       (vector-ref r3 2) " " (vector-ref r4 4))))))
   75 
   76 (main (current-command-line-arguments))
