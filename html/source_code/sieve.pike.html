    7 void main(int argc, array(string) argv)
    8 {
    9     array(int) flags;
   10     int i, k, count, num;
   11 
   12     num = (int)argv[-1];
   13     if (num < 1)
   14     num = 1;
   15 
   16     while (num--) {
   17     count = 0;
   18     flags = ({ 1 })*8193;
   19     for (i=2; i <= 8192; i++) {
   20         if (flags[i]) {
   21         for (k=i+i; k <= 8192; k+=i)
   22             flags[k] = 0;
   23         count++;
   24         }
   25     }
   26     }
   27     write("Count: %d\n", count);
   28 }
