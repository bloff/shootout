1 #!/usr/bin/mzscheme -r
    2 
    3 (define HI 0)
    4 (define LO 0)
    5 
    6 (define (hi-excp? x) (eq? x 'Hi_Exception))
    7 (define (lo-excp? x) (eq? x 'Lo_Exception))
    8 (define (true? x) (if (boolean? x) x #t))
    9 
   10 (define (some_fun n)
   11   (with-handlers
   12     ([true? (lambda (exn) #f)])
   13     (hi_fun n)))
   14 
   15 (define (hi_fun n)
   16   (with-handlers
   17     ([hi-excp? (lambda (exn) (set! HI (+ HI 1))) ])
   18     (lo_fun n)))
   19 
   20 (define (lo_fun n)
   21   (with-handlers
   22     ([lo-excp? (lambda (exn) (set! LO (+ LO 1))) ])
   23     (blowup n)))
   24 
   25 (define (blowup n)
   26   (if (= 0 (modulo n 2))
   27       (raise 'Hi_Exception)
   28       (raise 'Lo_Exception)))
   29 
   30 (define (main args)
   31   (let* ((n (if (= (vector-length args) 1) (string->number (vector-ref
   32    args 0)) 1)))
   33     (do ((i 0 (+ i 1)))
   34       ((= i n))
   35       (some_fun i)))
   36     (printf "Exceptions: HI=~a / LO=~a~n" HI LO))
   37 
   38 (main (current-command-line-arguments))
