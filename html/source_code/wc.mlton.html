    8 fun incr r = r := !r + 1

   10 val nl = ref 0
   11 val nw = ref 0
   12 val nc = ref 0
   13 val max = 4096
   14 val buf = Word8Array.array (max, 0w0)
   15 val sub = Word8Array.sub
   16 
   17 fun readblock scanfun =
   18    let
   19       val nread = Posix.IO.readArr (Posix.FileSys.stdin,
   20                     Word8ArraySlice.slice(buf, 0, NONE))
   21    in
   22       if nread = 0
   23      then ()
   24       else (nc := !nc + nread;
   25         scanfun (0, nread))
   26    end

   28 val c2b = Byte.charToByte
   29 val newline = c2b #"\n"
   30 val space = c2b #" "
   31 val tab = c2b #"\t"
   32 
   33 fun scan_out_of_word (i, n) =
   34    if i < n
   35       then
   36      let
   37         val c = sub (buf, i)
   38      in
   39         if c = newline
   40            then (incr nl; scan_out_of_word (i + 1, n))
   41         else if c = space orelse c = tab
   42             then scan_out_of_word (i + 1, n)
   43          else (incr nw; scan_in_word (i + 1, n))
   44      end
   45    else readblock scan_out_of_word
   46 
   47 and scan_in_word (i, n) =
   48   if i < n then
   49      let
   50     val c = sub (buf, i)
   51      in
   52     if c = newline
   53        then (incr nl; scan_out_of_word (i + 1, n))
   54     else if c = space orelse c = tab
   55         then scan_out_of_word (i + 1, n)
   56          else scan_in_word (i + 1, n)
   57      end
   58   else readblock scan_in_word
