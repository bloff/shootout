    7 open HashTable;
    8 
    9 fun hashtest2 n =
   10     let
   11     exception NotFound
   12     val h1 = mkTable (HashString.hashString, op =) (10000, NotFound)
   13     val h2 = mkTable (HashString.hashString, op =) (10000, NotFound)
   14     fun doinserts1 i =
   15         if i < 10000 then (
   16         insert h1 ("foo_" ^ (Int.toString i), i);
   17         doinserts1 (i+1)
   18         ) else ()
   19     fun addinto h k v1 =
   20         case find h k of
   21         SOME valref => valref := (!valref) + v1
   22           | NONE => insert h (k, ref v1)
   23     fun doinserts2 i =
   24         if i < n then (
   25         appi (fn (k,v) => (addinto h2 k v)) h1;
   26         doinserts2 (i+1)
   27         ) else ()
   28     in (
   29     doinserts1 0;
   30     doinserts2 0;
   31     print (Int.toString (lookup h1 "foo_1")); print " ";
   32     print (Int.toString (lookup h1 "foo_9999")); print " ";
   33     print (Int.toString (!(lookup h2 "foo_1"))); print " ";
   34     print (Int.toString (!(lookup h2 "foo_9999")));
   35     print "\n"
   36     ) end;
   37 
   38 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   39 
   40 fun main(name, args) =
   41     let
   42     val arg = hd(args @ ["1"])
   43     val num = atoi arg
   44     in
   45     hashtest2 num;
   46     OS.Process.success
   47     end;
   48 
   49 
   50 val _ = main( CommandLine.name(), CommandLine.arguments() );
