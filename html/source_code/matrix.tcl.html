    9 # modified to avoid matrix size checks
   10 # --Doug
   11 
   12 # additional speedups by Kristoffer Lawson and Miguel Sofer
   13 
   14 set size 30;
   15 
   16 proc mkmatrix {rows cols} {
   17     set count 1;
   18     set mx [list]
   19     for { set i 0 } { $i < $rows } { incr i } {
   20     set row [list]
   21     for { set j 0 } { $j < $cols } { incr j } {
   22         lappend row $count;
   23         incr count;
   24     }
   25     lappend mx $row;
   26     }
   27     return $mx;
   28 }
   29 
   30 proc mmult {m1 m2} {
   31     set cols [lindex $m2 0]
   32     foreach row1 $m1 {
   33         set row [list]
   34         set i 0
   35         foreach - $cols {
   36             set elem 0
   37             foreach elem1 $row1 row2 $m2 {
   38                 set elem [expr {$elem + $elem1 * [lindex $row2 $i]}]
   39             }
   40             lappend row $elem
   41             incr i
   42         }
   43         lappend result $row
   44     }
   45     return $result
   46 }
   47 
   48 proc main {} {
   49     global argv size
   50     set num [lindex $argv 0]
   51     if {$num < 1} {
   52     set num 1
   53     }
   54 
   55     set m1 [mkmatrix $size $size]
   56     set m2 [mkmatrix $size $size]
   57     while {$num > 0} {
   58         incr num -1
   59         set m [mmult $m1 $m2]
   60     }
   61 
   62     puts "[lindex [lindex $m 0] 0] [lindex [lindex $m 2] 3] [lindex [
   63     lindex $m 3] 2] [lindex [lindex $m 4] 4]"
   64 }
   65 
   66 main
