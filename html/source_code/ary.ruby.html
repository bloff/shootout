    7 n = Integer(ARGV.shift || 1)
    8 
    9 x = Array.new(n)
   10 y = Array.new(n, 0)
   11 
   12 for i in 0 ... n
   13   x[i] = i + 1
   14 end
   15 
   16 for k in 0 .. 999
   17   (n-1).step(0,-1) do |i|
   18     y[i] = y.at(i) + x.at(i)
   19   end
   20 end
   21 
   22 puts "#{y.first} #{y.last}"
