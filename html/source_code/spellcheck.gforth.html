    6 wordlist constant dict
    7 
    8 32 constant max-word
    9 
   10 create line max-word 2 + allot
   11 
   12 : read-dict ( -- )
   13     get-current dict set-current
   14     s" Usr.Dict.Words" r/o open-file throw
   15     begin
   16     line max-word 2 pick read-line throw
   17     while
   18     line swap nextname create
   19     repeat
   20     2drop set-current ;
   21 
   22 : spellcheck ( fid -- )
   23     begin
   24     line max-word 2 pick read-line throw
   25     while
   26     line swap 2dup dict search-wordlist if
   27         drop 2drop
   28     else
   29         type cr
   30     endif
   31     repeat
   32     2drop ;
   33 
   34 read-dict stdin spellcheck bye
