    8 module Main where
    9 import List(sortBy)
   10 import Char(toLower,isLower)
   11 import FiniteMap(fmToList,emptyFM,addToFM_C)
   12 
   13 main = interact $ unlines . pretty . sort . fmToList . makemap . cwords .
   14        lower
   15  where
   16        pretty l  = [pad 7 (show n) ++ "\t" ++ w | (w,n) <- l]
   17                    where
   18                          pad n s = replicate (n - length s) ' ' ++ s
   19 
   20        sort      = sortBy (\(w0,n0) (w1,n1) -> case compare n1 n0 of
   21                                                EQ -> compare w1 w0
   22                            x  -> x)
   23 
   24        makemap   = addFM emptyFM
   25                where addFM fm [] = fm
   26                  addFM fm (x:xs) = addFM (addToFM_C (+) fm x 1) xs
   27 
   28        cwords s  = case dropWhile (not . isLower) s of
   29                 "" -> []
   30                         s' -> w : (cwords s'')
   31                               where (w, s'') = span isLower s'
   32 
   33        lower     = map toLower
