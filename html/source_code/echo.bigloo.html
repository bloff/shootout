    5 (module echo
    6     ;(option (set! *genericity* #f))
    7     (extern (macro fork::int () "fork"))
    8     (extern (macro wait::int () "wait"))
    9     (main main))
   10 
   11 ;; we'll use bigloo functions that automatically add/remove newline
   12 (define DATA "Hello there sailor")
   13 
   14 (define (echo-client n port)
   15   (let* ((sock (make-client-socket "localhost" port))
   16      (sk-out (socket-output sock))
   17      (sk-in (socket-input sock)))
   18     (do ((i 0 (+ i 1)))
   19     ((= i n))
   20       (fprint sk-out DATA)
   21       (flush-output-port sk-out)
   22       (let ((ans (read-line sk-in)))
   23     (if (not (string=? DATA ans))
   24         (error "echo-client" "BAD reply" ans))))))
   25 
   26 (define (echo-server n)
   27   (let* ((ssock (make-server-socket))
   28      (port (socket-port-number ssock))
   29      (pid (fork)))
   30     (cond ((> pid 0)            ;parent
   31        (socket-accept-connection ssock)
   32        (let ((sk-in (socket-input ssock))
   33          (sk-out (socket-output ssock))
   34          (bytes 0))
   35          (let loop ((line (read-line sk-in)))
   36            (if (not (eof-object? line))
   37            (begin
   38              (fprint sk-out line)
   39              (flush-output-port sk-out)
   40              (set! bytes (+ bytes (string-length line) 1))
   41              (loop (read-line sk-in)))))
   42          (print "server processed " bytes " bytes"))
   43        (wait))
   44       (else                ;child
   45        (echo-client n port)))))
   46 
   47 
   48 (define (main args)
   49   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   50     (echo-server n)))
