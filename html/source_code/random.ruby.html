    2 # -*- mode: ruby -*-

    5 
    6 IM = 139968
    7 IA = 3877
    8 IC = 29573
    9 
   10 $last = 42.0
   11 def gen_random (max) (max * ($last = ($last * IA + IC) % IM)) / IM end
   12 
   13 N = Integer(ARGV.shift || 1) - 1
   14 N.times do
   15     gen_random(100.0)
   16 end
   17 printf "%.9f\n", gen_random(100.0)
