    6 %%  Usage: start from command line with
    7 %%     ozc -x except.oz -o except.oz.exe
    8 %%     except.oz.exe 2000
    9 
   10 functor
   11 import System Application
   12 
   13 define
   14 
   15    LoCount
   16    HiCount
   17 
   18    proc {Blowup N}
   19       if N mod 2 == 0 then raise loException() end
   20       else raise hiException() end end
   21    end
   22 
   23    proc {LoFun N}
   24       try {Blowup N} catch
   25      loException then {Assign LoCount {Access LoCount}+1}
   26       end
   27    end
   28 
   29    proc {HiFun N}
   30       try {LoFun N} catch
   31      hiException then {Assign HiCount {Access HiCount}+1}
   32       end
   33    end
   34 
   35    proc {SomeFun N}
   36       if N > 0 then
   37      try {HiFun N} catch
   38         loException then {System.showInfo 'loException should not get
   39                           here'}
   40         [] hiException then {System.showInfo 'hiException should not get
   41                              here'}
   42      end
   43      {SomeFun N-1}
   44       end
   45    end
   46 in
   47    local Args N in
   48       [Args] = {Application.getArgs plain}
   49       N = {String.toInt Args}
   50 
   51       LoCount = {NewCell 0}
   52       HiCount = {NewCell 0}
   53       {SomeFun N}
   54 
   55       {System.printInfo "Exceptions: HI="}
   56       {System.printInfo {Access HiCount}}
   57       {System.printInfo " / LO="}
   58       {System.showInfo {Access LoCount}}
   59    end
   60    {Application.exit 0}
   61 end
