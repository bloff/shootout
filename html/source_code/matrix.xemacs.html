    5 (defvar size 30)
    6 
    7 (defun mkmatrix (rows cols)
    8   (let ((mx (make-vector rows 0))
    9     (count 1))
   10     (do ((i 0 (1+ i)))
   11     ((= i rows))
   12       (let ((row (make-vector cols 0)))
   13     (do ((j 0 (1+ j)))
   14         ((= j cols))
   15       (aset row j count)
   16       (setq count (+ count 1)))
   17     (aset mx i row)))
   18     mx))
   19 
   20 (defun num-cols (mx)
   21   (let ((row (aref mx 0)))
   22     (length row)))
   23 
   24 (defun num-rows (mx)
   25   (length mx))
   26 
   27 (defun mmult (rows cols m1 m2)
   28   (let ((m3 (make-vector rows 0)))
   29     (do ((i 0 (1+ i)))
   30     ((= i rows))
   31       (let ((m1i (aref m1 i))
   32         (row (make-vector cols 0)))
   33     (do ((j 0 (1+ j)))
   34         ((= j cols))
   35       (let ((val 0))
   36         (do ((k 0 (1+ k)))
   37         ((= k cols))
   38           (setq val (+ val (* (aref m1i k)
   39                   (aref (aref m2 k) j)))))
   40         (aset row j val)))
   41     (aset m3 i row)))
   42     m3))
   43 
   44 (defun matrix-print (m)
   45   (do ((i 0 (1+ i)))
   46       ((= i (num-rows m)))
   47     (let ((row (aref m i)))
   48       (do ((j 0 (1+ j)))
   49       ((= j (num-cols m)))
   50     (princ (aref row j))
   51     (if (< j (num-cols m))
   52         (princ " ")))
   53       (princ "\n"))))
   54 
   55 (defun main ()
   56   (let* ((arg (car command-line-args-left))
   57      (n (if arg (string-to-number arg) 1)))
   58     (let ((mm 0)
   59       (m1 (mkmatrix size size))
   60       (m2 (mkmatrix size size)))
   61       (dotimes (i n)
   62     (setq mm (mmult size size m1 m2)))
   63       (let ((r0 (aref mm 0))
   64         (r2 (aref mm 2))
   65         (r3 (aref mm 3))
   66         (r4 (aref mm 4)))
   67     (princ (format "%d %d %d %d\n"
   68                (aref r0 0) (aref r2 3)
   69                (aref r3 2) (aref r4 4)))))))
