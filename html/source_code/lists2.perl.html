    2 # $Id: lists2.perl.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    3 
    4 use strict;
    5 
    6 sub main {
    7     my $N = ($ARGV[0] < 1) ? 1 : int($ARGV[0]);
    8     my @A = (1..$N);
    9     my @B = ();
   10     while (@A) {
   11     push(@B, shift(@A));
   12     push(@B, pop(@A));
   13     }
   14     while (@B) {
   15     unshift(@A, pop(@B));
   16     unshift(@A, shift(@B));
   17     }
   18     print scalar(@A), "\n";
   19     print "@A[0..9]\n";
   20 }
   21 
   22 main();
