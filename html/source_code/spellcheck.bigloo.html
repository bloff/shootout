    6 (module hash
    7     (main main))
    8 
    9 (define (main args)
   10   (let* ((max (do ((max 1 (* max 2))) ((> max 10000) max)))
   11      (dict (make-hash-table max
   12                 (lambda (o) (string->0..2^x-1 o 14))
   13                 (lambda (x) x)
   14                 string=?
   15                 max)))
   16     (with-input-from-file "Usr.Dict.Words"
   17       (lambda ()
   18     (let loop ((line (read-line)))
   19       (cond ((not (eof-object? line))
   20          (put-hash! line dict)
   21          (loop (read-line)))))))
   22     (let loop ((word (read-line)))
   23       (cond ((not (eof-object? word))
   24          (cond ((not (get-hash word dict))
   25             (print word)))
   26          (loop (read-line)))))))
