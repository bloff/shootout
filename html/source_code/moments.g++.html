    8 #include <vector>
    9 #include <numeric>
   10 #include <iterator>
   11 #include <algorithm>
   12 #include <cstdio>
   13 #include <cstdlib>
   14 #include <cmath>
   15 
   16 using namespace std;
   17 
   18 template <class T>
   19 struct moments {
   20 public:
   21     template <class InputIterator>
   22     moments(InputIterator begin, InputIterator end)
   23         : median(0.0), mean(0.0), average_deviation(0.0),
   24           standard_deviation(0.0), variance(0.0),
   25           skew(0.0), kurtosis(0.0)
   26         {
   27             T sum = accumulate(begin, end, 0.0);
   28             size_t N = end - begin;
   29             mean = sum / N;
   30             for (InputIterator i = begin; i != end; ++i) {
   31                 T deviation = *i - mean;
   32                 average_deviation += fabs(deviation);
   33                 T temp = deviation * deviation;
   34                 variance += temp;
   35                 temp *= deviation;
   36                 skew += temp;
   37                 kurtosis += temp * deviation;
   38             }
   39             average_deviation /= N;
   40             variance /= (N - 1);
   41             standard_deviation = sqrt(variance);
   42 
   43             if (variance) {
   44                 skew /= (N * variance * standard_deviation);
   45                 kurtosis = kurtosis/(N * variance * variance) - 3.0;
   46             }
   47 
   48             InputIterator mid = begin+N/2;
   49             nth_element(begin, mid, end);
   50             if (N % 2 == 0) {
   51                 InputIterator next_biggest = max_element(begin,
   52 mid);
   53                 median = (*mid+*next_biggest)/2;
   54             }
   55             else
   56                 median = *mid;
   57         }
   58 
   59     T median;
   60     T mean;
   61     T average_deviation;
   62     T standard_deviation;
   63     T variance;
   64     T skew;
   65     T kurtosis;
   66 };
   67 
   68 int main() {
   69     vector<double> v;
   70     double d;
   71 
   72     while (scanf(" %lf", &d) == 1) v.push_back(d);
   73     moments<double> m(v.begin(), v.end());
   74 
   75     printf("n:                  %d\n", v.end() - v.begin());
   76     printf("median:             %f\n", m.median);
   77     printf("mean:               %f\n", m.mean);
   78     printf("average_deviation:  %f\n", m.average_deviation);
   79     printf("standard_deviation: %f\n", m.standard_deviation);
   80     printf("variance:           %f\n", m.variance);
   81     printf("skew:               %f\n", m.skew);
   82     printf("kurtosis:           %f\n", m.kurtosis);
   83 
   84     return 0;
   85 }
