   13 int test_lists()
   14 {
   15     mixed Li1, Li2, Li3;
   16     // create a list of integers from 1 to SIZE.
   17     Li1 = (string)indices(allocate(SIZE+1))[1..];
   18     // copy the list to Li2.
   19     Li2 = copy_value( Li1 );
   20     // remove each element from left side of Li2 and append to
   21     // the right side of Li3 (preserving order)
   22 
   23     Li3 = "";
   24 
   25     while( sizeof( Li2 ) )
   26     {
   27     Li3 += Li2[..0];
   28     Li2 = Li2[1..];
   29     }
   30     // Li2 is now empty.
   31     // Remove each element from right side of Li3 and append to right
   32     // side of Li2
   33     while( sizeof( Li3 ) )
   34     {
   35     Li2 += Li3[sizeof( Li3 )-1..];
   36     Li3 = Li3[..sizeof( Li3 )-2];
   37     }
   38     // Li2 is now reversed, and Li3 empty.
   39     // Reverse Li1 in place.
   40     Li1 = reverse( Li1 );
   41     if( Li1[0] != SIZE )
   42     return 0;
   43     // compare Li1 and Li2 for equality, and return the length of the list.
   44     if( equal( Li1, Li2 ) )
   45     return sizeof( Li1 );
   46     return 0;
   47 }
   48 
   49 void main(int argc, array argv)
   50 {
   51     int result, num = (int)argv[-1];
   52     if( num <=  0 )
   53     num = 1;
   54     while( num-- )
   55     result = test_lists();
   56     write("%d\n", result );
   57 }
