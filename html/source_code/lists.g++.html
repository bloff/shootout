    6 #include <iostream>
    7 #include <list>
    8 #include <numeric>
    9 #include <ext/numeric>
   10 
   11 using namespace std;
   12 
   13 const size_t SIZE = 10000;
   14 
   15 size_t test_lists() {
   16     std::list<size_t> li1(SIZE);
   17 
   18     __gnu_cxx::iota(li1.begin(), li1.end(), 1);
   19 
   20     std::list<size_t> li2(li1);
   21 
   22     std::list<size_t> li3;
   23 
   24     size_t N = li2.size();
   25     while (N--) {
   26         li3.push_back(li2.front());
   27         li2.pop_front();
   28     }
   29 
   30     N = li3.size();
   31     while (N--) {
   32         li2.push_back(li3.back());
   33         li3.pop_back();
   34     }
   35 
   36     li1.reverse();
   37 
   38     return (li1.front() == SIZE) && (li1 == li2) ? li1.size() : 0;
   39 }
   40 
   41 int main(int argc, char* argv[]) {
   42     size_t ITER = (argc == 2 ? (atoi(argv[1]) < 1 ? 1 : atoi(argv[1])): 1);
   43 
   44     size_t result = 0;
   45     while (ITER > 0) {
   46         result = test_lists();
   47         --ITER;
   48     }
   49 
   50     std::cout << result << std::endl;
   51 }
