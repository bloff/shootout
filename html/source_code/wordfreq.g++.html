    6 #include <cstdio>
    7 #include <cctype>
    8 #include <cstring>
    9 #include <ext/hash_map>
   10 #include <vector>
   11 #include <algorithm>

   13 namespace Sgi = ::__gnu_cxx;
   14 using namespace std;
   15 
   16 int const bufsize = 4096;
   17 int const wsize = 64;
   18 
   19 class word_reader
   20 {
   21     int ws;
   22     char buf[bufsize+1], *bptr, *word;
   23     FILE *input;
   24 
   25     bool fill();
   26 
   27 public:
   28     word_reader(FILE *i): ws(wsize), bptr(buf), word(new char[ws+1]),
   29                 input(i)
   30     {*bptr = *word = '\0';}
   31     int operator()(char const **);
   32 };
   33 
   34 inline bool word_reader::fill()
   35 {
   36     int nread = fread(buf, sizeof(char), bufsize, input);
   37     buf[nread] = '\0';
   38     bptr = buf;
   39     return nread > 0;
   40 }
   41 
   42 int word_reader::operator()(char const **w)
   43 {
   44     int len = 0;
   45     char c;
   46     while (*bptr || fill()) {
   47     if (isalpha(c = *bptr++)) {
   48         word[len] = tolower(c);
   49         if (++len == ws) {
   50         char *nword = new char[(ws *= 2)+1];
   51         memcpy(nword, word, len);
   52         delete[] word;
   53         word = nword;
   54         }
   55     }
   56     else if (len > 0) break;
   57     }
   58     *w = word;
   59     word[len] = '\0';
   60     return len;
   61 }
   62 
   63 typedef Sgi::hash_map<char const *, int> counter;
   64 typedef pair<char const *, int> hpair;
   65 
   66 namespace std
   67 {
   68     inline bool operator<(hpair const &lhs, hpair const &rhs)
   69     {
   70     return lhs.second != rhs.second ? lhs.second > rhs.second
   71         : strcmp(lhs.first, rhs.first) > 0;
   72     }
   73 
   74     template<> struct equal_to<char const *>
   75     {
   76     bool operator()(char const *s1, char const *s2) const
   77         {return strcmp(s1, s2) == 0;}
   78     };
   79 }
   80 
   81 int main()
   82 {
   83     int len;
   84     const char *w;
   85     counter hist;
   86     word_reader wr(stdin);
   87 
   88     while ((len = wr(&w)) > 0) {
   89     counter::iterator i = hist.find(w);
   90     if (i == hist.end()) hist[strcpy(new char[len+1], w)] = 1;
   91     else ++i->second;
   92     }
   93 
   94     vector<hpair> v(hist.begin(), hist.end());
   95     sort(v.begin(), v.end());
   96     for (size_t i = 0; i < v.size(); ++i)
   97     printf("%7d\t%s\n", v[i].second, v[i].first);
   98 
   99     return 0;
  100 }
