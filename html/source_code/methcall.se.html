   16 class METHCALL
   17 
   18 creation make
   19 
   20 feature -- Creation
   21 
   22    make is
   23       local
   24      index: INTEGER
   25      n: INTEGER
   26      toggle: TOGGLE
   27      value: BOOLEAN
   28      nth_toggle: NTH_TOGGLE
   29       do
   30      if argument_count = 0 then
   31         n := 1
   32      else
   33         n := argument(1).to_integer
   34      end
   35 
   36      value := True
   37      !!toggle.make(value)
   38      from index := 0 until index = n loop
   39         value := toggle.activate.value
   40         index := index + 1
   41      end
   42 
   43      if value then print("true%N") else print("false%N") end
   44         value := True
   45         !!nth_toggle.make(value, 3)
   46      from index := 0 until index = n loop
   47         value := nth_toggle.activate.value
   48         index := index + 1
   49      end
   50      if value then print("true%N") else print("false%N") end
   51 
   52       end -- make
   53 
   54 end
