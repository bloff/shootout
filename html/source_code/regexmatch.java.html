    4 import java.io.*;
    5 import java.util.*;
    6 import java.text.*;
    7 import java.lang.reflect.*;
    8 import com.oroinc.text.regex.*;
    9 
   10 public class regexmatch {
   11 
   12     public static void main(String args[])
   13     throws IOException, MalformedPatternException {
   14 
   15     int n = Integer.parseInt(args[0]);
   16     LinkedList lines = new LinkedList();
   17     try {
   18             BufferedReader in = new BufferedReader(new InputStreamReader(
   19                                 System.in));
   20         String line;
   21         while ((line = in.readLine()) != null) {
   22         lines.addLast(line);
   23         }
   24         in.close();
   25         } catch (IOException e) {
   26             System.err.println(e);
   27             return;
   28         }
   29 
   30     String regex =
   31         "(?: ^ | [^\\d\\(])        # must be preceeded by non-digit\n" +
   32         "(?:            # area code is 3 digits\n" +
   33         " \\((\\d\\d\\d)\\)        # match 1: with parens\n" +
   34         "|                # or\n" +
   35         " (\\d\\d\\d)        # match 2: without parens\n" +
   36         ")                # if match1 then match right paren\n" +
   37         "[ ]            # area code followed by one space\n" +
   38         "(\\d\\d\\d)        # match 3: prefix of 3 digits\n" +
   39         "[ -]            # separator is either space or dash\n" +
   40         "(\\d\\d\\d\\d)        # match 4: last 4 digits\n" +
   41         "(?:\\D|$)            # must be followed by a non-digit\n";
   42 
   43     PatternMatcher matcher = new Perl5Matcher();
   44     PatternCompiler compiler = new Perl5Compiler();
   45     Pattern pattern = null;
   46 
   47     try {
   48         pattern = compiler.compile(regex, Perl5Compiler.EXTENDED_MASK);
   49     } catch(MalformedPatternException e) {
   50         System.err.println("Bad pattern.");
   51         System.err.println(e.getMessage());
   52         System.exit(1);
   53     }
   54 
   55     for (int i = 0; i < n; i++) {
   56         LinkedList phones = new LinkedList();
   57         for (ListIterator li = lines.listIterator(); li.hasNext();) {
   58         String line = (String)li.next();
   59         if (matcher.contains(line, pattern)) {
   60             MatchResult match = matcher.getMatch();
   61             String num = "";
   62             if (match.group(1) != null) {
   63             num = "(" + match.group(1) + ") " + match.group(3) + "-" + match.group(
   64          4);
   65             } else {
   66             num = "(" + match.group(2) + ") " + match.group(3) + "-" + match.group(
   67          4);
   68             }
   69             phones.addLast(num);
   70         }
   71         }
   72         if (i == (n - 1)) {
   73         int count = 1;
   74         for (ListIterator li = phones.listIterator(); li.hasNext();) {
   75             String num = (String)li.next();
   76             System.out.println(count++ + ": " + num);
   77         }
   78         }
   79     }
   80     }
   81 }
