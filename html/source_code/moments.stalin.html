    5 (define (main args)
    6   (let* ((sum 0.0)
    7      (numlist (let loop ((line (read-line)) (numlist '()))
    8             (cond ((eof-object? line) numlist)
    9               (else
   10                (let ((num (string->real line)))
   11                  (set! sum (+ num sum))
   12                  (loop (read-line) (cons num numlist))))))))
   13     (let ((n (length numlist)))
   14       (let ((mean (/ sum n))
   15         (average_deviation 0.0)
   16         (standard_deviation 0.0)
   17         (variance 0.0)
   18         (skew 0.0)
   19         (kurtosis 0.0)
   20         (median 0.0)
   21         (deviation 0.0))
   22     (let loop ((nums numlist))
   23          (if (not (null? nums))
   24          (begin
   25            (set! deviation (- (car nums) mean))
   26            (set! average_deviation (+ average_deviation (abs deviation)))
   27            (set! variance (+ variance (expt deviation 2.0)))
   28            (set! skew (+ skew (expt deviation 3.0)))
   29            (set! kurtosis (+ kurtosis (expt deviation 4)))
   30            (loop (cdr nums)))))
   31 
   32     (set! average_deviation (/ average_deviation (fixnum->flonum n)))
   33     (set! variance (/ variance (- n 1)))
   34     (set! standard_deviation (sqrt variance))
   35 
   36     (cond ((> variance 0.0)
   37            (set! skew (/ skew (* n variance standard_deviation)))
   38            (set! kurtosis (- (/ kurtosis (* n variance variance))
   39                  3.0))))
   40 
   41     (set! numlist (sort numlist (lambda (x y) (<fl x y))))
   42 
   43     (let ((mid (/ n 2)))
   44       (if (zero? (modulo n 2))
   45           (set! median (/ (+ (car (list-tail numlist mid))
   46                  (car (list-tail numlist (- mid 1))))
   47                   2.0))
   48         (set! median (car (list-tail numlist mid)))))
   49 
   50 
   51     (set! standard_deviation (/ (round (* standard_deviation 1000000))
   52   1000000))
   53 
   54         (display "n:                  ~D~n" n)
   55     (display "median:             ~,6F~n" median)
   56     (display "mean:               ~,6F~n" mean)
   57     (display "average_deviation:  ~,6F~n" average_deviation)
   58     (display "standard_deviation: ~,6F~n" standard_deviation)
   59     (display "variance:           ~,6F~n" variance)
   60     (display "skew:               ~,6F~n" skew)
   61     (display "kurtosis:           ~,6F~n" kurtosis)))))
   62 
   63 (main (vector->list argv))
