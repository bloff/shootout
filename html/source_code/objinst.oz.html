    6 %% Uses local variables within the Object Instantiation
    7 %% loops. It might be quicker to create a Cell outside
    8 %% of the loop and use {Assign T {New Toggle init(true)}}
    9 %%
   10 %%  Usage: start from command line with
   11 %%     ozc -x objinst.oz -o objinst.oz.exe
   12 %%     objinst.oz.exe 1000000
   13 
   14 functor
   15 import System Application
   16 
   17 define
   18 
   19 class Toggle
   20    attr state: true
   21    meth state(V)
   22      V = @state
   23    end
   24    meth activate
   25       state <- {Not @state}
   26    end
   27    meth init(State)
   28       state <- State
   29    end
   30 end
   31 
   32 class NthToggle from Toggle
   33    attr trigger:0 count:0
   34    meth activate
   35       count <- @count + 1
   36       if @count >= @trigger then
   37      state <- {Not @state}
   38      count <- 0
   39       end
   40    end
   41    meth init(State Trigger)
   42       Toggle,init(State)
   43       trigger <- Trigger
   44       count <- 0
   45    end
   46 end
   47 
   48 proc {ShowBool X}
   49    if X then {System.showInfo "true"}
   50    else {System.showInfo "false"} end
   51 end
   52 
   53 in
   54    local Args N T1 T2 in
   55       [Args] = {Application.getArgs plain}
   56       N = {String.toInt Args}
   57 
   58       T1 = {New Toggle init(true)}
   59       for I in 1..5 do
   60          local V in {T1 activate}{T1 state(V)}{ShowBool V} end
   61       end
   62       {System.showInfo ""}
   63       for I in 1..N do
   64          local T in T = {New Toggle init(true)} end
   65       end
   66 
   67       T2 = {New NthToggle init(true 3)}
   68       for I in 1..8 do
   69          local V in {T2 activate}{T2 state(V)}{ShowBool V} end
   70       end
   71       {System.showInfo ""}
   72       for I in 1..N do
   73          local T in T = {New NthToggle init(true 3)} end
   74       end
   75    end
   76    {Application.exit 0}
   77 end
