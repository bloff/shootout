    5 -module(nestedloop).
    6 -export([main/1]).
    7 
    8 main([Arg]) ->
    9     Num = list_to_integer(atom_to_list(Arg)),
   10     io:fwrite("~w\n", [loopA(Num, Num, 0)]),
   11     halt(0).
   12 
   13 
   14 loopA(0, M, N) -> N;
   15 loopA(I, M, N) -> loopA(I - 1, M, loopB(M, M, N)).
   16 
   17 loopB(0, M, N) -> N;
   18 loopB(I, M, N) -> loopB(I - 1, M, loopC(M, M, N)).
   19 
   20 loopC(0, M, N) -> N;
   21 loopC(I, M, N) -> loopC(I - 1, M, loopD(M, M, N)).
   22 
   23 loopD(0, M, N) -> N;
   24 loopD(I, M, N) -> loopD(I - 1, M, loopE(M, M, N)).
   25 
   26 loopE(0, M, N) -> N;
   27 loopE(I, M, N) -> loopE(I - 1, M, loopF(M, N)).
   28 
   29 loopF(0, N) -> N;
   30 loopF(I, N) -> loopF(I - 1, 1 + N).
