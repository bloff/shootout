    6 int
    7 fib(int n) {
    8     if (n < 2) return(1);
    9     return( fib(n-2) + fib(n-1) );
   10 }
   11 
   12 void
   13 main(int argc, array(string) argv) {
   14     int n = max((int)argv[-1], 1);
   15     write("%d\n", fib(n));
   16 }
