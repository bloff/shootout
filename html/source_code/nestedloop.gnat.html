    5 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Command_Line;
    6 
    7 procedure NestedLoop is
    8    Count, M : Natural := 0;
    9    N        : Positive := 1;
   10 begin
   11 
   12    begin
   13       N := Positive'Value (Ada.Command_Line.Argument (1));
   14    exception
   15       when Constraint_Error => null;
   16    end;
   17    M := N - 1;
   18    for A in 0 .. M loop
   19       for B in 0 .. M loop
   20          for C in 0 .. M loop
   21             for D in 0 .. M loop
   22                for E in 0 .. M loop
   23                   for F in 0 .. M loop
   24                      Count := Count + 1;
   25                   end loop;
   26                end loop;
   27             end loop;
   28          end loop;
   29       end loop;
   30    end loop;
   31    Ada.Integer_Text_IO.Put (Item => Count, Width => 0);
   32    Ada.Text_IO.New_Line;
   33 end NestedLoop;
