    5 (defun main ()
    6   (let* ((arg (car command-line-args-left))
    7      (n (if arg (string-to-number arg) 1))
    8      (x 0))
    9     (dotimes (a n)
   10       (dotimes (b n)
   11     (dotimes (c n)
   12       (dotimes (d n)
   13         (dotimes (e n)
   14           (dotimes (f n)
   15         (setq x (+ x 1))))))))
   16     (princ (format "%d\n" x))))
