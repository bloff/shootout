1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: matrix.rep.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   (define size 30)
   16 
   17   (define (mkmatrix rows cols)
   18     (let ((mx (make-vector rows 0))
   19       (count 1))
   20       (do ((i 0 (1+ i)))
   21       ((= i rows))
   22     (let ((row (make-vector cols 0)))
   23       (do ((j 0 (1+ j)))
   24           ((= j cols))
   25         (aset row j count)
   26         (setq count (+ count 1)))
   27       (aset mx i row)))
   28       mx))
   29 
   30   (define (num-cols mx)
   31     (let ((row (aref mx 0)))
   32       (length row)))
   33 
   34   (define (num-rows mx)
   35     (length mx))
   36 
   37   (define (mmult rows cols m1 m2)
   38     (let ((m3 (make-vector rows 0)))
   39       (do ((i 0 (1+ i)))
   40       ((= i rows))
   41     (let ((m1i (aref m1 i))
   42           (row (make-vector cols 0)))
   43       (do ((j 0 (1+ j)))
   44           ((= j cols))
   45         (let ((val 0))
   46           (do ((k 0 (1+ k)))
   47           ((= k cols))
   48         (setq val (+ val (* (aref m1i k)
   49                     (aref (aref m2 k) j)))))
   50           (aset row j val)))
   51       (aset m3 i row)))
   52       m3))
   53 
   54   (define (matrix-print m)
   55     (do ((i 0 (1+ i)))
   56     ((= i (num-rows m)))
   57       (let ((row (aref m i)))
   58     (do ((j 0 (1+ j)))
   59         ((= j (num-cols m)))
   60       (princ (aref row j))
   61       (if (< j (num-cols m))
   62           (princ " ")))
   63     (princ "\n"))))
   64 
   65   (let* ((arg (get-command-line-option "--num" t))
   66      (n (if arg (read-from-string arg) 1)))
   67     (let ((mm 0)
   68       (m1 (mkmatrix size size))
   69       (m2 (mkmatrix size size)))
   70       (let loop ((iter n))
   71     (cond ((> iter 0)
   72            (setq mm (mmult size size m1 m2))
   73            (loop (- iter 1)))))
   74       (let ((r0 (aref mm 0))
   75         (r2 (aref mm 2))
   76         (r3 (aref mm 3))
   77         (r4 (aref mm 4)))
   78     (princ (aref r0 0))
   79     (princ " ")
   80     (princ (aref r2 3))
   81     (princ " ")
   82     (princ (aref r3 2))
   83     (princ " ")
   84     (princ (aref r4 4))
   85     (princ "\n")))))
