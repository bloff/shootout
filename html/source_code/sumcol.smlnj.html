    6 structure Test : sig
    7     val main : (string * string list) -> OS.Process.status
    8 end = struct
    9 
   10 fun sumlines sum =
   11   if TextIO.endOfStream TextIO.stdIn
   12   then (print (Int.toString sum); print "\n")
   13   else case (TextIO.inputLine TextIO.stdIn) of
   14       NONE => sumlines sum
   15     | SOME(str) =>
   16         (case (Int.fromString str) of
   17         NONE => sumlines sum
   18       | SOME(i) => sumlines (sum + i));
   19 
   20 fun main(name, args) = (sumlines 0; OS.Process.success);
   21 
   22 end
   23 
   24 val _ = SMLofNJ.exportFn("sumcol", Test.main);
