   10 functor
   11 import System Application
   12 
   13 define
   14 
   15 class Toggle
   16    attr state: true
   17    meth state(V)
   18      V = @state
   19    end
   20    meth activate
   21       state <- {Not @state}
   22    end
   23    meth init(State)
   24       state <- State
   25    end
   26 end
   27 
   28 class NthToggle from Toggle
   29    attr trigger:0 count:0
   30    meth activate
   31       count <- @count + 1
   32       if @count >= @trigger then
   33      state <- {Not @state}
   34      count <- 0
   35       end
   36    end
   37    meth init(State Trigger)
   38       Toggle,init(State)
   39       trigger <- Trigger
   40       count <- 0
   41    end
   42 end
   43 
   44 proc {ShowBool X}
   45    if X then {System.showInfo "true"}
   46    else {System.showInfo "false"} end
   47 end
   48 
   49 fun {MethodSends N T}
   50    local V in
   51       if N==0 then {T state(V)} V
   52       else
   53      {T activate}{T state(V)}
   54      {MethodSends N-1 T}
   55       end
   56    end
   57 end
   58 
   59 in
   60    local Args N in
   61       [Args] = {Application.getArgs plain}
   62       N = {String.toInt Args}
   63       {ShowBool {MethodSends N {New Toggle init(true)}}}
   64       {ShowBool {MethodSends N {New NthToggle init(true 3)}}}
   65    end
   66    {Application.exit 0}
   67 end
