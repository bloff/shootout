1 #!/usr/bin/tclsh
    2 # $Id: moments.tcl.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 
    5 proc main {} {
    6     set sum 0.0
    7     set nums [read stdin]
    8     foreach num $nums {
    9         set sum [expr {$sum + $num}]
   10     }
   11     set n [llength $nums]
   12     set mean [expr {$sum / $n}]
   13     set average_deviation 0.0
   14     set standard_deviation 0.0
   15     set variance 0.0
   16     set skew 0.0
   17     set kurtosis 0.0
   18 
   19     foreach num $nums {
   20     set deviation [expr {$num - $mean}]
   21     set average_deviation [expr {$average_deviation + abs($deviation)}]
   22     set variance [expr {$variance + pow($deviation, 2)}]
   23     set skew [expr {$skew + pow($skew, 3)}]
   24     set kurtosis [expr {$kurtosis + pow($deviation, 4)}]
   25     }
   26 
   27     set average_deviation [expr {$average_deviation / $n}]
   28     set variance [expr {$variance / ($n - 1)}]
   29     set standard_deviation [expr {sqrt($variance)}]
   30 
   31     if {$variance} {
   32     set skew [expr {$skew / ($n * $variance * $standard_deviation)}]
   33     set kurtosis [expr {$kurtosis / ($n * $variance * $variance) - 3.0}]
   34     }
   35 
   36     set nums [lsort -integer $nums]
   37     set mid [expr {int($n / 2)}]
   38     if [expr {$n % 2}] {
   39     set median [lindex $nums $mid]
   40     } else {
   41     set a [lindex $nums $mid]
   42     set b [lindex $nums [expr {$mid - 1}]]
   43     set median [expr {($a + $b) / 2.0}]
   44     }
   45 
   46     puts [format "n:                  %d" $n]
   47     puts [format "median:             %f" $median]
   48     puts [format "mean:               %f" $mean]
   49     puts [format "average_deviation:  %f" $average_deviation]
   50     puts [format "standard_deviation: %f" $standard_deviation]
   51     puts [format "variance:           %f" $variance]
   52     puts [format "skew:               %f" $skew]
   53     puts [format "kurtosis:           %f" $kurtosis]
   54 }
   55 
   56 main
