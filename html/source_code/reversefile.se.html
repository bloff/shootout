    6 class REVERSEFILE
    7 
    8 creation make
    9 
   10 feature
   11 
   12     make is
   13         local
   14             i: ITERATOR[STRING]
   15             ll_string: LINKED_LIST[STRING];
   16         do
   17             !!ll_string.make;
   18             from
   19                 io.read_line
   20             until io.end_of_input loop
   21                 ll_string.add_first(clone(io.last_string));
   22                 io.read_line;
   23             end;
   24             i := ll_string.get_new_iterator
   25             from
   26            i.start
   27         until i.is_off loop
   28                 io.put_string (i.item);
   29                 io.put_character('%N');
   30                 i.next;
   31             end;
   32         end;
   33 
   34 end -- class REVERSEFILE
