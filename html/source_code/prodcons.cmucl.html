    6 (defparameter *counter* 0)
    7 (defparameter *produced* 0)
    8 (defparameter *consumed* 0)
    9 (defparameter *data* 0)
   10 (defparameter *mutex* (mp:make-lock "Big Lock"))
   11 
   12 (defun main ()
   13   (let ((n (parse-integer (or (car (last extensions:*command-line-strings*)
   14    ) "1"))))
   15     (declare (optimize (speed 3) (safety 0))
   16          (fixnum n))
   17     (setf *counter* 0
   18       *produced* 0
   19       *consumed* 0
   20       *data* 0)
   21     (let ((producer (mp:make-process #'(lambda () (funcall #'producer n)) :
   22      name "Producer"))
   23       (consumer (mp:make-process #'(lambda () (funcall #'consumer n)) :name
   24     "Consumer")))
   25       (mp:process-wait "Wait on Producer" #'(lambda () (eq (mp:process-
   26        state producer) :killed)))
   27       (mp:process-wait "Wait on Consumer" #'(lambda () (eq (mp:process-
   28        state consumer) :killed)))
   29       (format t "~A ~A~%" *produced* *consumed*))))
   30 
   31 (defun producer (n)
   32   (declare (optimize (speed 3) (safety 0))
   33            (fixnum n))
   34   (loop :for i :of-type fixnum :from 1 :to n
   35         :do
   36         (mp:process-wait "Producer is waiting on Consumer" #'(lambda () (=
   37          *counter* 0)))
   38         (mp:with-lock-held (*mutex*)
   39           (setf *data* i
   40                     *counter* 1))
   41         (incf *produced*)))
   42 
   43 (defun consumer (n)
   44   (declare (optimize (speed 3) (safety 0))
   45            (fixnum n))
   46   (let ((i 0))
   47     (declare (fixnum i))
   48     (loop
   49      (mp:process-wait "Consumer is waiting on Producer" #'(lambda () (=
   50       *counter* 1)))
   51      (mp:with-lock-held (*mutex*)
   52        (setf i *data*
   53              *counter* 0))
   54      (incf *consumed*)
   55      (when (= i n)
   56        (return)))))
