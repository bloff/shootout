    5 class HASH2
    6 
    7 creation make
    8 
    9 feature -- Initialization
   10 
   11    make is
   12       local
   13      n, i, prev : INTEGER
   14      hash1 : DICTIONARY[INTEGER, STRING]
   15      hash2 : DICTIONARY[INTEGER, STRING]
   16      it: ITERATOR[STRING]
   17       do
   18      n := argument(1).to_integer
   19 
   20      !!hash1.with_capacity(10000);
   21      !!hash2.with_capacity(10000);
   22 
   23      from  i := 0
   24      until i = 10000
   25      loop
   26         hash1.put(i, "foo_" + i.to_string)
   27         i := i + 1
   28      end
   29 
   30      from  i := 0
   31      until i = n
   32      loop
   33         it := hash1.get_new_iterator_on_keys
   34         from  it.start
   35         until it.is_off
   36         loop
   37            if hash2.has(it.item) then
   38           prev := hash2.at(it.item)
   39            else
   40           prev := 0
   41            end
   42            hash2.put(prev + hash1.at(it.item), it.item)
   43            it.next;
   44         end
   45         i := i + 1
   46      end
   47 
   48          std_output.put_integer(hash1.at("foo_1"))
   49          std_output.put_string(" ")
   50          std_output.put_integer(hash1.at("foo_9999"))
   51          std_output.put_string(" ")
   52          std_output.put_integer(hash2.at("foo_1"))
   53          std_output.put_string(" ")
   54          std_output.put_integer(hash2.at("foo_9999"))
   55          std_output.put_character('%N')
   56       end
   57 
   58 end -- class HASH2
