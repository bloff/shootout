    6 module Main where
    7 
    8 import IArray
    9 import System
   10 import Data.Array.Unboxed
   11 
   12 make_x :: Int -> UArray Int Int
   13 make_x len = array (1,len) [(i, i) | i <- [1..len]]
   14 
   15 make_y0 :: Int -> UArray Int Int
   16 make_y0 len = array (1,len) [(i, 0) | i <- [1..len]]
   17 
   18 add_array :: Int -> UArray Int Int -> UArray Int Int -> UArray Int Int
   19 add_array 0 a b = b
   20 add_array len a b = array (1,len) [(i, ((a ! i) + (b ! i))) | i <- [1..len]
   21                     ]
   22 
   23 add_arrays_n :: Int -> Int -> UArray Int Int -> UArray Int Int -> UArray
   24 Int Int
   25 add_arrays_n 0 len a b = b
   26 add_arrays_n n len a b =
   27     add_arrays_n (n-1) len a (add_array len a b)
   28 
   29 ary :: Int -> IO ()
   30 ary len = do putStr (show (y ! 1)) ; putStr " " ; putStrLn (show (y ! len))
   31              where y = add_arrays_n 1000 len (make_x len) (make_y0 len)
   32 
   33 main = do
   34         ~[n] <- getArgs
   35         ary (read n::Int)
