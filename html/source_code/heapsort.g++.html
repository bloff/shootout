    5 #include <iostream>

    7 #include <math.h>

    9 using namespace std;
   10 
   11 #define IM 139968
   12 #define IA   3877
   13 #define IC  29573
   14 
   15 double
   16 gen_random(double max) {
   17     static long last = 42;
   18     return( max * (last = (last * IA + IC) % IM) / IM );
   19 }
   20 
   21 void
   22 heapsort(int n, double *ra) {
   23     int i, j;
   24     int ir = n;
   25     int l = (n >> 1) + 1;
   26     double rra;
   27 
   28     for (;;) {
   29     if (l > 1) {
   30         rra = ra[--l];
   31     } else {
   32         rra = ra[ir];
   33         ra[ir] = ra[1];
   34         if (--ir == 1) {
   35         ra[1] = rra;
   36         return;
   37         }
   38     }
   39     i = l;
   40     j = l << 1;
   41     while (j <= ir) {
   42         if (j < ir && ra[j] < ra[j+1]) { ++j; }
   43         if (rra < ra[j]) {
   44         ra[i] = ra[j];
   45         j += (i = j);
   46         } else {
   47         j = ir + 1;
   48         }
   49     }
   50     ra[i] = rra;
   51     }
   52 }
   53 
   54 int
   55 main(int argc, char *argv[]) {
   56     int N = ((argc == 2) ? atoi(argv[1]) : 1);
   57     double *ary;
   58     int i;
   59 
   60     /* create an array of N random doubles */
   61     ary = (double *)malloc((N+1) * sizeof(double));
   62     for (i=1; i<=N; i++) {
   63     ary[i] = gen_random(1);
   64     }
   65 
   66     heapsort(N, ary);
   67 
   68     printf("%.10f\n", ary[N]);
   69     free(ary);
   70     return(0);
   71 }
