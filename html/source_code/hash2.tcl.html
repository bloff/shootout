    6 proc main {} {
    7     global argv
    8     set n [lindex $argv 0]
    9     for {set i 0} {$i < 10000} {incr i} {
   10     set hash1(foo_$i) $i
   11     }
   12     for {set i $n} {$i > 0} {incr i -1} {
   13     foreach k [array names hash1] {
   14         if {[catch {set hash2($k) [expr {$hash1($k) + $hash2($k)}]}]} {
   15         set hash2($k) $hash1($k)
   16         }
   17     }
   18     }
   19     puts [join [list $hash1(foo_1) $hash1(foo_9999) $hash2(foo_1) $hash2(
   20                             foo_9999) ] " "]
   21 }
   22 
   23 main
