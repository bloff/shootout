    5 (defconst IM     139968)
    6 (defconst IA       3877)
    7 (defconst IC      29573)
    8 
    9 (defvar LAST 42)
   10 
   11 (defun gen_random (max)
   12   (/ (* max (setq LAST (% (+ (* LAST IA) IC) IM))) IM))
   13 
   14 (defun main ()
   15   (let* ((arg (car command-line-args-left))
   16      (num (if arg (string-to-number arg) 1)))
   17     (dotimes (i (- num 1))
   18       (gen_random 100.0))
   19     (princ (format "%.9f\n" (gen_random 100.0))))
   20   (kill-emacs 0))
