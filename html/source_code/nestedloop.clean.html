   17 module nestedloop 
   18 import StdEnv, ArgEnv, LanguageShootout 
   19  
   20 Start = toString (loopA n n 0) +++ "\n" 
   21    where 
   22        n = toInt (argv.[1]) 
   23        argv = getCommandLine 
   24                                                                       
   25 loopA n m x  
   26    | n > 0         = loopA (n-1) m (loopB m m x) 
   27    | otherwise     = x 
   28  
   29 loopB n m x 
   30    | n > 0         = loopB (n-1) m (loopC m m x) 
   31    | otherwise     = x 
   32  
   33 loopC n m x 
   34    | n > 0         = loopC (n-1) m (loopD m m x) 
   35    | otherwise     = x 
   36  
   37 loopD n m x 
   38    | n > 0         = loopD (n-1) m (loopE m m x) 
   39    | otherwise     = x 
   40  
   41 loopE n m x 
   42    | n > 0         = loopE (n-1) m (loopF m x) 
   43    | otherwise     = x 
   44  
   45 loopF n x 
   46    | n > 0         = loopF (n-1) (x+1) 
   47    | otherwise     = x
