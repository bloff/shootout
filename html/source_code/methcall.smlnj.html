    8 structure Test : sig
    9     val main : (string * string list) -> OS.Process.status
   10 end = struct
   11 
   12 fun for (start, stop, f) =
   13    let
   14       fun loop i =
   15      if i > stop
   16         then ()
   17      else (f i; loop (i + 1))
   18    in
   19       loop start
   20    end

   22 structure Toggle =
   23    struct
   24       datatype 'a t = T of {
   25                 state: 'a ref,
   26                 value: 'a t -> 'a,
   27                 activate: 'a t -> 'a t
   28                 }
   29 
   30       fun new state =
   31      T {state = ref state,
   32         value = fn T {state, ...} => !state,
   33         activate = fn this as T {state, ...} => (state := not(!state); this)}
   34 
   35       fun activate (this as T {activate, ...}) = activate this
   36       fun value (this as T {value, ...}) = value this
   37    end
   38 
   39 structure Ntoggle =
   40    struct
   41       datatype 'a t = T of {
   42                 state: 'a ref,
   43                 value: 'a t -> 'a,
   44                 activate: 'a t -> 'a t,
   45                 countMax: int,
   46                 counter: int ref
   47                 }
   48 
   49       fun new (state, countMax) =
   50      T {
   51         state = ref state,
   52         value = fn T {state, ...} => !state,
   53         activate = (fn this as T {state, counter, countMax, ...} =>
   54             let
   55                val newCounter = 1 + !counter
   56                val _ = counter := newCounter
   57                val _ =
   58                   if !counter >= countMax
   59                  then (state := not(!state);
   60                        counter := 0)
   61                   else ()
   62             in
   63                this
   64             end),
   65         countMax = countMax,
   66         counter = ref 0
   67         }
   68 
   69       fun activate (this as T {activate, ...}) = activate this
   70       fun value (this as T {value, ...}) = value this
   71    end
   72 
   73 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0
   74 fun printl [] = print "\n" | printl(h::t) = ( print h ; printl t )
   75 
   76 fun main (name, args) =
   77    let
   78       val n = atoi (hd (args @ ["1"]))
   79       val v = ref true
   80       val tog = Toggle.new true
   81       val _ = for (0, n - 1, fn _ => v := Toggle.value (Toggle.activate
   82               tog))
   83       val _ = print (if !v then "true\n" else "false\n")
   84       val _ = v := true
   85       val ntog = Ntoggle.new (!v, 3)
   86       val _ = for (0, n - 1, fn _ => v := Ntoggle.value (Ntoggle.activate
   87               ntog))
   88       val _ = print (if !v then "true\n" else "false\n")
   89    in
   90       OS.Process.success
   91    end
   92 end
