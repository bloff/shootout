    8 module fibo
    9 
   10 import StdEnv, ArgEnv
   11 
   12 fib :: !Int -> Int
   13 fib 0 = 1
   14 fib 1 = 1
   15 fib n = fib (n-2) + fib (n-1)
   16 
   17 Start
   18     | argc == 1
   19         = toString (fib n) +++ "\n"
   20     | otherwise
   21         = toString (fib 1) +++ "\n"
   22     where
   23         argc = size argv - 1
   24         argv = getCommandLine
   25         n = toInt (argv.[1])
