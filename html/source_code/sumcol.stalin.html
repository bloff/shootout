    5 (include "QobiScheme")
    6 
    7 (define (main args)
    8   (let ((sum 0))
    9     (let loop ((line (read-line)))
   10       (cond ((eof-object? line) #f)
   11         (else
   12          (set! sum (+ sum (string->number line)))
   13          (loop (read-line)))))
   14     (display sum) (newline)))
   15 
   16 (main (vector->list argv))
