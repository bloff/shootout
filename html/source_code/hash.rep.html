1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: hash.rep.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep
   14     rep.system
   15     rep.data.tables)
   16 
   17   (let* ((arg (get-command-line-option "--num" t))
   18      (n (if arg (read-from-string arg) 1))
   19      (last (- n 1))
   20      (c 0)
   21      (x (make-table string-hash eq-hash)))
   22     (do ((i 1 (+ i 1)))
   23     ((> i n))
   24       (table-set x (number->string i 16) i))
   25     (do ((i last (- i 1)))
   26     ((< i 0))
   27       (if (table-ref x (number->string i 10))
   28       (setq c (+ c 1))))
   29     (format standard-output "%d\n" c)))
   30 
   31
