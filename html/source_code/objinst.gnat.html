    8 with Ada.Command_Line, Ada.Characters.Handling, Text_IO, Ada.Tags;
    9 
   10 procedure ObjInst is    --  3.451
   11    pragma Suppress (Discriminant_Check);
   12    pragma Suppress (Access_Check);
   13    package CH renames Ada.Characters.Handling;
   14    use type Ada.Tags.Tag;
   15 
   16    package Toggles is
   17       type A_Rec is
   18          tagged record                 --  'tagged' allows fields to be
   19          added
   20             Value       : Boolean := True;
   21          end record;
   22 
   23       type B_Rec is new A_Rec with
   24          record
   25             Flip_Period : Positive := 1;
   26             Count       : Natural := 0;
   27          end record;
   28 
   29       type A_Recs_Family_Ptr is access all A_Rec'Class;
   30 
   31       function Activate (X : A_Recs_Family_Ptr) return A_Recs_Family_Ptr;
   32       pragma Inline (Activate);
   33    end Toggles;
   34 
   35    package body Toggles is
   36       function Activate (X : A_Recs_Family_Ptr) return A_Recs_Family_Ptr
   37                          is
   38          pragma Suppress (Range_Check);
   39          pragma Suppress (Tag_Check);
   40       begin                   --  X is a ptr: no "in out" parms in a
   41       function
   42          if X.all'Tag = A_Rec'Tag then                         --  A_Rec
   43                         case
   44             X.all.Value := not X.all.Value;
   45          else
   46             declare                       --  Make visible the 2 B_Rec
   47             fields
   48                pragma Suppress (Tag_Check);
   49                B        : B_Rec renames B_Rec (X.all);
   50             begin
   51                B.Count := B.Count + 1;
   52                if B.Count = B.Flip_Period then
   53                   B.Count := 0;
   54                   B.Value := not B.Value;
   55                end if;
   56             end;
   57          end if;
   58          return X;
   59       end Activate;
   60    end Toggles;
   61 
   62    A        : aliased Toggles.A_Rec;      --  "aliased", since X'Access is
   63    used
   64    B        : aliased Toggles.B_Rec;      --  "access all" above for
   65    aliased
   66    N        : Positive := 1;
   67    Res      : Boolean;
   68 begin
   69    begin
   70       N := Positive'Value (Ada.Command_Line.Argument (1));
   71    exception
   72       when Constraint_Error => null;
   73    end;
   74    A := Toggles.A_Rec'(Value => True);
   75    for K in 1 .. 5 loop
   76       Res := Toggles.Activate (A'Access).Value;
   77       Text_IO.Put_Line (CH.To_Lower (Boolean'Image (Res)));
   78    end loop;
   79    for Iter in 1 .. N loop
   80       A := Toggles.A_Rec'(Value => True);
   81    end loop;
   82    Text_IO.New_Line;
   83    B := Toggles.B_Rec'(Value => True, Flip_Period => 3, Count => 0);
   84    for K in 1 .. 8 loop
   85       Res := Toggles.Activate (B'Access).Value;
   86       Text_IO.Put_Line (CH.To_Lower (Boolean'Image (Res)));
   87    end loop;
   88    for Iter in 1 .. N loop
   89       B := Toggles.B_Rec'(Value => True, Flip_Period => 3, Count => 0);
   90    end loop;
   91 end ObjInst;
