   11 use strict;
   12 
   13 my $n = $ARGV[0] || 1;
   14 my %X = ();
   15 my $c = 0;
   16 
   17 for my $i (1..$n) {
   18     $X{sprintf('%x', $i)} = $i;
   19 }
   20 for my $i (reverse 1..$n) {
   21     ++$c if exists $X{$i};
   22 }
   23 print "$c\n";
