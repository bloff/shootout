   11 #define SIZE 10000

   13 // a simple Single Linked List
   14 // the head node is special, it's val is length of list
   15 typedef struct LL {
   16     int val;
   17     struct LL *next;    /* points to next or head (if at tail) */
   18 } LL;
   19 
   20 inline int list_length(LL *head) { return(head->val); }
   21 inline int list_empty(LL *head)  { return(list_length(head) == 0); }
   22 inline LL* list_first(LL *head)  { return(head->next); }
   23 
   24 void list_push_head(LL *head, LL *item) {
   25     LL *next = head->next;
   26     head->next = item;
   27     item->next = next;
   28     head->val++;
   29 }
   30 
   31 LL *list_pop_head(LL *head) {
   32     LL *next;
   33     if (list_empty(head)) return(NULL);
   34     next = head->next;
   35     head->next = next->next;
   36     head->val--;
   37     return(next);
   38 }
   39 
   40 LL *list_new_item(int v) {
   41     LL *l = (LL *)malloc(sizeof(LL));
   42     l->next = NULL;
   43     l->val = v;
   44     return(l);
   45 }
   46 
   47 inline LL* list_create()  { return(list_new_item(0)); }
   48 
   49 void list_free (LL* head) {
   50     LL *p, *tmp = head;
   51     if (head == NULL) return;
   52     if (head->next != NULL) {
   53     for (p=head->next; p->next != NULL; p=p->next) {
   54         free(tmp);
   55         tmp = p;
   56     }
   57     }
   58     free(tmp);
   59 }
   60 
   61 void list_append (LL* head_1, LL* head_2) {
   62     LL *p;
   63     for (p=head_1; p->next != NULL; p=p->next) ;
   64     p->next = head_2->next;
   65     head_2->next = NULL;
   66     head_1->val += head_2->val;
   67     head_2->val = 0;
   68 }
   69 
   70 int list_member(int item, LL* head) {
   71     LL* p = head->next;
   72     for (p=head; p != NULL; p=p->next) {
   73     if (p->val == item) return(1);
   74     }
   75     return(0);
   76 }
   77 
   78 void list_reverse (LL *head) {
   79     LL *prev, *p, *tmp, *first = list_first(head);
   80     if (first != NULL) {
   81     prev = NULL;
   82     p = first;
   83     while ((tmp = p->next) != NULL) {
   84         p->next = prev;
   85         prev = p;
   86         p = tmp;
   87     }
   88     p->next = prev;
   89     head->next = p;
   90     }
   91 }
   92 
   93 int list_last(LL *head) {
   94     LL* p;
   95     if (head == NULL) return(0);
   96     for (p=head; p->next != NULL; p=p->next) ;
   97     return(p->val);
   98 }

  100 int main(int argc, char *argv[]) {
  101     int i, mid, sum = 0, n = ((argc == 2) ? atoi(argv[1]) : 1);
  102     LL *p, *tmp, *A, *B;
  103 
  104     // create list (A) of integers from 1 through N
  105     A = list_create();
  106     for (i=n; i>0; i--) list_push_head(A, list_new_item(i));
  107 
  108     // create empty list B
  109     B = list_create();
  110 
  111     // move each individual item from A to B, in a loop, reversing order
  112     while (!list_empty(A)) list_push_head(B, list_pop_head(A));
  113 
  114     // print first 2 items in B
  115     printf("%d %d\n", B->next->val, B->next->next->val);
  116 
  117     // reverse B (can be done in place)
  118     list_reverse(B);
  119 
  120     // is 0 a member of B?
  121     printf("%s\n", list_member(0, B) ? "true" : "false" );
  122 
  123     // is N a member of B?
  124     printf("%s\n", list_member(n, B) ? "true" : "false" );
  125 
  126     // filter values from B to A that are less than N/2, preserving order
  127     mid = n/2;
  128     tmp = A; // tmp points to last item in list A
  129     for (p=list_first(B); p!=NULL; p=p->next) {
  130     if (p->val < mid) {
  131         tmp->next = list_new_item(p->val); // copy item
  132         tmp = tmp->next;
  133         A->val++;
  134     }
  135     }
  136 
  137     // print first ten items of A
  138     for (i=0, p=list_first(A); p != NULL && i<10; i++, p=p->next)
  139     printf("%d%s", p->val, (i<9) ? " " : "\n");
  140 
  141     // print sum of items in A that are less than 1000
  142     for (p=list_first(A); p!=NULL; p=p->next)
  143     if ((i = p->val) < 1000) sum += i;
  144     printf("%d\n", sum);
  145 
  146     // append B to end of A (can be done destructively)
  147     list_append(A, B);
  148 
  149     // print length and last element of A
  150     printf("%d %d\n", list_length(A), list_last(A));
  151 
  152     list_free(A);
  153     list_free(B);
  154     return 0;
  155 }
