   16 class SPELLCHECK
   17 
   18 creation make
   19 
   20 feature -- Creation
   21 
   22    make is
   23       do
   24      read_dictionary
   25      from
   26         std_input.read_line
   27      until std_input.end_of_input loop
   28         if dictionary.has(std_input.last_string) = False then
   29            print(std_input.last_string + "%N")
   30         end
   31         std_input.read_line
   32      end
   33       end -- make
   34 
   35 feature -- Queries
   36 
   37    dictionary: DICTIONARY[INTEGER, STRING]
   38 
   39 feature -- Commands
   40 
   41    read_dictionary is
   42       local
   43      file: STD_FILE_READ
   44      value: INTEGER
   45       do
   46      value := 1
   47      from
   48         !!dictionary.with_capacity(60000)
   49         !!file.connect_to("Usr.Dict.Words")
   50      until
   51         file.end_of_input
   52      loop
   53         file.read_line
   54         dictionary.add(value, file.last_string)
   55      end
   56      file.disconnect
   57       end -- read_dictionary
   58 
   59 end
