    5 with System, Ada.Command_Line, Ada.Text_IO;
    6 
    7 procedure Random is
    8    type Real is digits Positive'Max (15, System.Max_Digits);
    9    package Rio is new Ada.Text_IO.Float_IO (Num => Real);
   10 
   11    package Random_Real is
   12       function Gen_Random (Supr : Real) return Real;
   13       pragma Inline (Gen_Random);
   14    end Random_Real;
   15 
   16    package body Random_Real is
   17       IM          : constant Positive := 139968;
   18       IA          : constant Integer := 3877;
   19       IC          : constant Integer := 29573;
   20       Last        : Integer := 42;
   21 
   22       function Gen_Random (Supr : Real) return Real is
   23          pragma Suppress (Overflow_Check);
   24          pragma Suppress (Range_Check);
   25       begin
   26          Last := (Last * IA + IC) mod IM;
   27          return Supr * Real (Last) / Real (IM);
   28       end Gen_Random;
   29          --  Assume no overflow for "Natural ((IM - 1) * IA + IC)"
   30    end Random_Real;
   31 
   32    Result   : Real;
   33    N        : Natural := 0;
   34 begin
   35    begin
   36       N := Natural'Value (Ada.Command_Line.Argument (1));
   37    exception
   38       when Constraint_Error => null;
   39    end;
   40    for Iter in 1 .. N loop
   41       Result := Random_Real.Gen_Random (Supr => 100.0);
   42    end loop;
   43    Rio.Put (Result, Fore => 0, Aft => 9, Exp => 0);
   44    Ada.Text_IO.New_Line;
   45 end Random;
