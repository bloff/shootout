    6 \ read NUM from last command line argument
    7 0. argc @ 1- arg >number 2drop drop constant NUM
    8 
    9 1 constant *hi*
   10 2 constant *lo*
   11 
   12 variable lo
   13 variable hi
   14 
   15 : blowup ( n -- )
   16     1 and if
   17         *lo* throw
   18     else
   19         *hi* throw
   20     endif ;
   21 
   22 : lo-function ( n1 -- n2 )
   23     ['] blowup catch           ( n2 ball )
   24     dup *lo* <> tuck and throw ( n2 f )
   25     1+ lo +! ;
   26 
   27 : hi-function ( n1 -- n2 )
   28     ['] lo-function catch      ( n2 ball )
   29     dup *hi* <> tuck and throw ( n2 f )
   30     1+ hi +! ;
   31 
   32 : some-function ( n1 -- n2 )
   33     ['] hi-function catch abort" We shouldn't get here" ;
   34 
   35 : main ( -- )
   36     NUM 0 ?do
   37         i some-function drop
   38         loop
   39     ." Exceptions: HI=" hi ? ." / LO=" lo @ 1 u.r cr ;
   40 
   41 main bye
