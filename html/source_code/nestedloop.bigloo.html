    5 (module nestedloop
    6     (option (set! *genericity* #f))
    7     (main main))
    8 
    9 (define (main args)
   10   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   11      (x 0))
   12     (do ((a 0 (+ a 1)))
   13     ((= a n))
   14       (do ((b 0 (+ b 1)))
   15       ((= b n))
   16     (do ((c 0 (+ c 1)))
   17         ((= c n))
   18       (do ((d 0 (+ d 1)))
   19           ((= d n))
   20         (do ((e 0 (+ e 1)))
   21         ((= e n))
   22           (do ((f 0 (+ f 1)))
   23           ((= f n))
   24         (set! x (+ x 1))))))))
   25     (display x)
   26     (newline)))
