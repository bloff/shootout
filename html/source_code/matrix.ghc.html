    7 module Main(main) where
    8 import System(getArgs, exitWith, ExitCode(..))
    9 import Numeric(readDec)
   10 import List(transpose)
   11 main = do
   12        arg <- getArgs
   13        case arg of
   14             [number] -> putStrLn (disp m 0 0 ++ " " ++
   15                   disp m 2 3 ++ " " ++
   16                   disp m 3 2 ++ " " ++
   17                   disp m 4 4)
   18                   where
   19               disp m row col  = show (m!!row!!col)
   20               m = powmat (fst (head (readDec number)))
   21             _        -> exitWith (ExitFailure 1)
   22 
   23 size = 30
   24 
   25 -- ghc is able to optimize out enough invariants so that the
   26 -- traditional form this test runs in O(1)
   27 --
   28 -- this would have ghc trounce all other entrants, so to get
   29 -- closer to the general meaning of the test, we cascade
   30 -- matrix multiplications. this means the results are, by necessity,
   31 -- different, but involve just as many mmults. i.e for n=1 the results
   32 -- are identical.
   33 
   34 powmat n = power n (mmult . transpose $ mkmat size) (mkmat size)
   35     where
   36     power n f | n > 0     = f . (power (n-1) f)
   37               | otherwise = id
   38     mkmat x    = [[(y-1)*x+1..y*x]| y <- [1..x]]
   39     mmult a b  = [[dot row col 0 | col <- a]| row <- b]
   40            where
   41            dot :: [Int] -> [Int] -> Int -> Int
   42            dot (x:xs) (y:ys) z = dot xs ys (z + x*y)
   43            dot  _      _     z = z
   44
