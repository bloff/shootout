    6 (declaim (optimize (speed 3) (space 0) (safety 0) (debug 0) (compilation-
    7  speed 0)))
    8 (defun command-line-argument ()
    9   (parse-integer (or (car (last ext:*command-line-strings*)) "1")))
   10 
   11 (defconstant +digit+ "0123456789")
   12 
   13 (defconstant +digits-needed+
   14   '(10 100 1000 10000 100000 10000000 100000000 536870911))
   15 
   16 (defun fixnum-to-foo-string (n)
   17   (declare (fixnum n))
   18   (let* ((size (+ 4 (position-if (lambda (x) (> (the fixnum x) n))
   19                                  +digits-needed+)))
   20      (result (make-string (1+ size))))
   21     (replace result "foo_")
   22     (loop for i fixnum from size downto 4 with q fixnum = n and r fixnum =
   23      0
   24       do (multiple-value-setq (q r) (floor q 10))
   25          (setf (schar result i) (aref +digit+ r)))
   26     result))
   27 
   28 (defun main (&optional (n (command-line-argument)))
   29   (let ((hash1 (make-hash-table :test 'equal :size n))
   30         (hash2 (make-hash-table :test 'equal :size n)))
   31     (macrolet ((hash1 (i) `(gethash (fixnum-to-foo-string ,i) hash1))
   32                (hash2 (i) `(gethash (fixnum-to-foo-string ,i) hash2)))
   33       (loop for i fixnum below 10000 do (setf (hash1 i) i))
   34       (loop for i fixnum below n do
   35         (maphash (lambda (k v)
   36                    (declare (fixnum v))
   37                    (incf (the fixnum (gethash k hash2 0)) v))
   38                  hash1))
   39       (format t "~D ~D ~D ~D~%" (hash1 1) (hash1 9999) (hash2 1) (hash2
   40        9999)))))
