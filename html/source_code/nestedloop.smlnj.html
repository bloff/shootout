    6 structure Test : sig
    7     val main : (string * string list) -> OS.Process.status
    8 end = struct
    9 
   10 fun loopF 0 x = x
   11   | loopF n x = loopF (n-1) (x+1)
   12 
   13 fun loopE 0 m x = x
   14   | loopE n m x = loopE (n-1) m (loopF m x);
   15 
   16 fun loopD 0 m x = x
   17   | loopD n m x = loopD (n-1) m (loopE m m x);
   18 
   19 fun loopC 0 m x = x
   20   | loopC n m x = loopC (n-1) m (loopD m m x);
   21 
   22 fun loopB 0 m x = x
   23   | loopB n m x = loopB (n-1) m (loopC m m x);
   24 
   25 fun loopA 0 m x = x
   26   | loopA n m x = loopA (n-1) m (loopB m m x);
   27 
   28 
   29 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   30 
   31 fun main(name, args) =
   32   let
   33     val arg = hd(args @ ["1"]);
   34     val num = atoi arg;
   35     val result = loopA num num 0
   36   in
   37       print (Int.toString result); print "\n";
   38       OS.Process.success
   39   end
   40 end
   41 
   42 val _ = SMLofNJ.exportFn("nestedloop", Test.main);
