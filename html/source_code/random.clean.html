   12 module random 
   13 import StdEnv, ArgEnv, LanguageShootout 
   14  
   15 Start = toStringWith 9 (nthran n 100.0 42 0.0) +++ "\n" 
   16     where 
   17         n = toInt (argv.[1]) 
   18     argv = getCommandLine 
   19  
   20 im :== 139968 
   21 ia :== 3877 
   22 ic :== 29573 
   23 imd :== toReal im 
   24  
   25 nthran :: !Int Real Int !Real -> Real 
   26 nthran n max seed r 
   27    | n>0 = nthran (n-1) max newseed newran 
   28    = r 
   29    where 
   30       newseed = (seed * ia + ic) rem im 
   31       newran =  max * toReal newseed / imd
