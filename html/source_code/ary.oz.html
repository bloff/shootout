    6 functor
    7 import
    8   System
    9   Application
   10 define
   11  local Args N A1 A2 in
   12 
   13    {Application.getCmdArgs plain Args}
   14    if {List.length Args} \=  1 then
   15      N = 1
   16    else
   17      {String.toInt Args.1 N}
   18    end
   19 
   20    {NewArray 0 N 0 A1}
   21    {NewArray 0 N 0 A2}
   22 
   23    {For 0 (N - 1) 1
   24      proc {$ I} {Put A1 I (I + 1)} end }
   25 
   26    {For 0 999 1
   27       proc {$ I}
   28         {For (N - 1) 0 ~1
   29           proc {$ I} {Put A2 I ({Array.get A2 I} + {Get A1 I})} end}
   30       end}
   31 
   32 
   33    {System.showInfo {Get A2 0}#" "#{Get A2 (N - 1)}}
   34    {Application.exit 0}
   35  end
   36 end
