    8 #include <stdio.h>
    9 #include <unistd.h>
   10 
   11 #define CHAR int
   12 #define BSIZ 4096
   13 
   14 unsigned long ws[256];
   15 char buff[BSIZ];
   16 
   17 int main() {
   18 
   19     int w_cnt=0,l_cnt=0,b_cnt=0,cnt;
   20     unsigned long was_sp = 1;
   21     unsigned char *pp, *pe;
   22 
   23     /* Fill tables */
   24     for (cnt = 0; cnt < 256; cnt++) ws[cnt] = 0;
   25 
   26     /* also: ws['\r']=ws['\v']=ws['\f']= */
   27     ws[' ']=ws['\t']=ws['\n']=1;
   28     ws['\n']=65536 + 1;
   29 
   30     /* Main loop */
   31     while((cnt=read(0,buff,BSIZ))) {
   32         unsigned long tcnt = 0;
   33 
   34     b_cnt += cnt;
   35     pe = buff + cnt;
   36     pp = buff;
   37 
   38     while(pp < pe) {
   39         tcnt += ws[*pp] ^ was_sp;
   40         was_sp = ws[*pp] & 0xFFFF;
   41         pp ++;
   42     }
   43     w_cnt += tcnt  & 0xFFFF;
   44     l_cnt += tcnt >> 16;
   45     }
   46 
   47     w_cnt += (ws[ buff[b_cnt&(BSIZ-1)-1] ]^1)&1;
   48     w_cnt>>=1;
   49 
   50     printf("%d %d %d\n", l_cnt, w_cnt, b_cnt);
   51     return 0;
   52 }
