1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: sumcol.rep.html,v 1.1 2004-06-14 23:09:29 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   ;; eat the unused command line arg
   16   (get-command-line-option "--num" t)
   17   (let ((sum 0)
   18     (line))
   19     (while (setq line (read-line))
   20       (setq sum (+ sum (read-from-string line))))
   21     (format standard-output "%d\n" sum)))
