    6 import sys
    7 
    8 def main():
    9     x = 0
   10     iter = int(sys.argv[1])
   11     if iter < 1:
   12         iter = 1
   13 
   14     i_r = range(iter)
   15     for a in i_r:
   16         for b in i_r:
   17             for c in i_r:
   18                 for d in i_r:
   19                     for e in i_r:
   20                         for f in i_r:
   21                             x += 1
   22     print x
   23 
   24 main()
