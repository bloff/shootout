1 #!/usr/bin/tclsh
    2 # $Id: ackermann.tcl.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 
    5 set NUM [lindex $argv 0]
    6 if {$NUM < 1} {
    7     set NUM 1
    8 }
    9 
   10 proc ack {m n} {
   11     if {$m == 0} {
   12     return [expr {$n + 1}]
   13     } elseif {$n == 0} {
   14     return [ack [expr {$m - 1}] 1]
   15     } else {
   16     return [ack [expr {$m - 1}] [ack $m [expr {$n - 1}]]]
   17     }
   18 }
   19 
   20 set ack [ack 3 $NUM]
   21 puts "Ack(3,$NUM): $ack"
