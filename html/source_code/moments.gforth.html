    6 1024 constant max-line
    7 create line max-line 2 + allot
    8 
    9 : input-floats ( fid -- rsum )
   10     >r 0e begin
   11     line max-line r@ read-line throw
   12     while
   13     line swap >float 0= abort" float expected"
   14     fdup f, f+
   15     repeat
   16     rdrop drop ;
   17 
   18 : compute-loop ( a u sum -- kurtosis skew variance standard-dev avg-dev
   19                  mean )
   20     dup 0 d>f fdup { f: n } f/ { f: mean }
   21     0e fdup fdup fdup
   22     floats bounds do {  f: avg-deviation f: variance f: skew f: kurtosis }
   23     i f@ mean f- { f: deviation }
   24     deviation fabs avg-deviation f+ ( avg-deviation )
   25     deviation fdup f* fdup variance f+ ( avg-deviation dev^2 variance )
   26     fswap deviation f* fdup skew f+ ( avg-deviation variance dev^2 skew )
   27     fswap deviation f* kurtosis f+ ( avg-deviation variance skew kurtosis )
   28     float +loop
   29     frot n 1e f- f/ to variance
   30     frot to avg-deviation ( skew kurtosis )
   31     variance fsqrt { f: standard-deviation }
   32     variance f0<> if
   33     n variance fdup f* f* f/ 3e f-
   34     fswap n variance f*  standard-deviation f* f/ fswap
   35     endif
   36     fswap variance standard-deviation avg-deviation n f/ mean ;
   37 
   38 (field) float- -1 floats ,
   39 
   40 : partition { first last -- last-smaller first-larger }
   41     \ partition array addr1 u1 into all elements less than pivot and all
   42     \ others, addr1 u2 and addr3 u3 are the two partitions.
   43     \ lessthan-xt ( elemptr1 elemptr2 -- f ) compares the two elements
   44     first last + 1 rshift faligned f@ { f: pivot }
   45     first last begin ( i j )
   46     begin
   47         pivot dup f@ f< over first u> and
   48     while
   49         float-
   50     repeat
   51     swap begin ( j i )
   52         dup last u< over f@ pivot f< and
   53     while
   54         float+
   55     repeat
   56     2dup u>=
   57     while ( j i )
   58     dup f@ over f@ dup f! over f!
   59     float+ swap float-
   60     repeat ;
   61 
   62 : quantile ( first last quant-low quant-high -- ) recursive
   63     \ sorts the array [first,last] such that the contained part of
   64     \ [quant-low,quant-high] is the same as in the fully sorted array.
   65     { quant-low quant-high }
   66     begin { first last }
   67     first quant-high u< quant-low last u< and
   68     while
   69     first last partition ( last-smaller first-larger )
   70     last quant-low quant-high quantile
   71     first swap
   72     repeat ;
   73 
   74 : median { addr u -- rmedian }
   75     addr u 1- 2/ floats + addr u 2/ floats + ( quant-low quant-high )
   76     addr addr u 1- floats + 2over quantile
   77     f@ f@ f+ f2/ ;
   78 
   79 ( based on "f." word from gforth )
   80 : ff.  ( r -- )
   81   f$ dup >r 0<
   82   IF '0 emit ELSE scratch r@ min type r@ precision - zeros ENDIF
   83   '. emit
   84   r@ negate zeros
   85   scratch r> 0 max /string 0 max type ;
   86 
   87 create nums \ s" moments.input" r/o open-file throw input-floats
   88 stdin input-floats
   89 nums here over - float /
   90 ." n:                  " dup 0 .r cr
   91 compute-loop
   92 nums here over - float / median  9 set-precision
   93 ." median:             " ff. cr
   94 ." mean:               " ff. cr
   95 ." average_deviation:  " ff. cr
   96 ." standard_deviation: " ff. cr 11 set-precision
   97 ." variance:           " ff. cr  7 set-precision
   98 ." skew:               " ff. cr
   99 ." kurtosis:           " ff. cr
  100 bye
