    6 int Ack(int M, int N) {
    7   if (M == 0) return( N + 1 );
    8   if (N == 0) return( Ack(M - 1, 1) );
    9   return( Ack(M - 1, Ack(M, (N - 1))) );
   10 }
   11 
   12 int main(int argc, array(string) argv) {
   13   int N = (int)argv[-1] || 1;
   14   write("Ack(3,%d): %d\n", N, Ack(3, N));
   15 }
