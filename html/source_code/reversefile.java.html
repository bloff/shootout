    5 import java.io.*;
    6 
    7 public class reversefile {
    8     final static int BUFF_SIZE = 4096;
    9 
   10     public static void main( String[] args ) {
   11     InputStream in = System.in;
   12 
   13     byte[] buff = new byte[BUFF_SIZE];
   14     byte[] obuff = new byte[BUFF_SIZE];
   15 
   16     try {
   17         int pos = 0;
   18         for (int n ; (n = in.read(buff, pos, BUFF_SIZE)) > 0 ;) {
   19         pos += n;
   20         if ((pos + BUFF_SIZE) > buff.length) {
   21             byte[] new_buff = new byte[buff.length << 1];
   22             System.arraycopy(buff, 0, new_buff, 0, buff.length);
   23             buff = new_buff;
   24         }
   25         }
   26 
   27         int opos = 0;
   28 
   29         for (int p = --pos ; ;) {
   30         --p;
   31         if (p < 0 || buff[p] == '\n') {
   32             for( int bp = p ; ++bp <= pos ; ) {
   33             obuff[opos] = buff[bp];
   34             if (++opos >= BUFF_SIZE) {
   35                 System.out.write(obuff, 0, opos);
   36                 opos = 0;
   37             }
   38             }
   39             pos = p;
   40 
   41             if (p < 0)
   42             break;
   43         }
   44         }
   45 
   46         System.out.write(obuff, 0, opos);
   47     }
   48     catch(IOException ex) {
   49     }
   50     }
   51 }
