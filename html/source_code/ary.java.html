   10 import java.io.*;
   11 import java.util.*;
   12 
   13 public class ary {
   14     public static void main(String args[]) {
   15     int i, j, k, n = Integer.parseInt(args[0]);
   16     int x[] = new int[n];
   17     int y[] = new int[n];
   18 
   19     for (i = 0; i < n; i++)
   20         x[i] = i + 1;
   21     for (k = 0; k < 1000; k++ )
   22         for (j = n-1; j >= 0; j--)
   23         y[j] += x[j];
   24 
   25     System.out.println(y[0] + " " + y[n-1]);
   26     }
   27 }
