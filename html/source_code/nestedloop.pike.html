    6 void main(int argc, array(string) argv) {
    7   int n = (int)argv[-1];
    8   if (n < 1) n = 1;
    9   int x=0;
   10 
   11   for (int a; a<n; a++)
   12       for (int b; b<n; b++)
   13       for (int c; c<n; c++)
   14           for (int d; d<n; d++)
   15           for (int e; e<n; e++)
   16               for (int f; f<n; f++)
   17               x++;
   18 
   19   write("%d\n", x);
   20 }
