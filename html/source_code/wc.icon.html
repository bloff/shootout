    5 procedure main(argv)
    6     local nl, nw, nc, nonspaces
    7     nl := nw := nc := 0
    8     nonspaces := ~' \t'
    9     while line := read() do line ? {
   10     nl +:= 1
   11     nc +:= 1 + *line
   12     while tab(upto(nonspaces)) do {
   13         nw +:= 1
   14         tab(many(nonspaces))
   15     }
   16     }
   17     write(nl, " ", nw, " ", nc)
   18 end
