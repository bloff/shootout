    5 with Text_IO, Ada.Command_Line, Ada.Characters.Handling;
    6 
    7 procedure MethCall is
    8    package Toggles is
    9       type A_Rec is                          --  'tagged' implies hidden
   10       field,
   11          tagged record                       --   and can append new
   12          fields
   13             Value       : Boolean := True;
   14          end record;
   15 
   16       procedure Activate (A : in out A_Rec);
   17 
   18       type B_Rec is new A_Rec with                 --  Add more fields to
   19       A_Rec
   20          record
   21             Flip_Period : Positive := 1;
   22             Count       : Natural := 0;
   23          end record;
   24 
   25       procedure Activate (B : in out B_Rec);
   26       pragma Inline (Activate);
   27    end Toggles;
   28 
   29    package body Toggles is
   30       procedure Activate (A : in out A_Rec) is
   31       begin
   32          A.Value := not A.Value;
   33       end Activate;
   34 
   35       procedure Activate (B : in out B_Rec) is
   36       begin
   37          B.Count := B.Count + 1;
   38          if B.Count = B.Flip_Period then
   39             B.Count := 0;
   40             B.Value := not B.Value;
   41          end if;
   42       end Activate;
   43    end Toggles;
   44 
   45    X        : Toggles.A_Rec := (Value => True);
   46    Y        : Toggles.B_Rec := (Value => True, Count => 0, Flip_Period =>
   47                                3);
   48    Val      : Boolean;
   49    N        : Positive := 1;
   50 begin
   51    begin
   52       N := Positive'Value (Ada.Command_Line.Argument (1));
   53    exception
   54       when Constraint_Error => null;
   55    end;
   56    for Iter in 1 .. N loop
   57       Toggles.Activate (X);
   58       Val := X.Value;
   59    end loop;
   60    Text_IO.Put_Line (Ada.Characters.Handling.To_Lower (Boolean'Image (Val))
   61                      );
   62    for Iter in 1 .. N loop
   63       Toggles.Activate (Y);
   64       Val := Y.Value;
   65    end loop;
   66    Text_IO.Put_Line (Ada.Characters.Handling.To_Lower (Boolean'Image (Val))
   67                      );
   68 end MethCall;
