    7 let mkmatrix rows cols =
    8   let count = ref 1 and last_col = cols - 1
    9   and m = Array.make_matrix rows cols 0 in
   10   for i = 0 to rows - 1 do
   11     let mi = m.(i) in
   12     for j = 0 to last_col do mi.(j) <- !count; incr count done;
   13   done;
   14   m
   15 
   16 let rec inner_loop k v m1i m2 j =
   17   if k < 0 then v
   18   else inner_loop (k - 1) (v + m1i.(k) * m2.(k).(j)) m1i m2 j
   19 
   20 let mmult rows cols m1 m2 m3 =
   21   let last_col = cols - 1 and last_row = rows - 1 in
   22   for i = 0 to last_row do
   23     let m1i = m1.(i) and m3i = m3.(i) in
   24     for j = 0 to last_col do m3i.(j) <- inner_loop last_row 0 m1i m2 j
   25             done;
   26   done
   27 
   28 let size = 30
   29 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
   30         1
   31 let m1 = mkmatrix size size and m2 = mkmatrix size size
   32 let m3 = Array.make_matrix size size 0;;
   33 for i = 2 to n do mmult size size m1 m2 m3 done;
   34 mmult size size m1 m2 m3;
