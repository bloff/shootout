    7 set HI 0
    8 set LO 0
    9 
   10 proc some_function {num} {
   11     if {[catch {hi_function $num} result]} {
   12         puts stderr "We shouldn't get here ($result)"
   13     }
   14 }
   15 
   16 proc hi_function {num} {
   17     if {[set exc [catch {lo_function $num}]] == 11} {
   18         # handle
   19     incr ::HI
   20     } else {
   21         # rethrow
   22     return -code $exc
   23     }
   24 }
   25 
   26 proc lo_function {num} {
   27     if {[set exc [catch {blowup $num}]] == 10} {
   28         # handle
   29     incr ::LO
   30     } else {
   31         # rethrow
   32     return -code $exc
   33     }
   34 }
   35 
   36 proc blowup {num} {
   37     if {$num % 2} {
   38         #error "Lo_exception"
   39     return -code 10
   40     } else {
   41         #error "Hi_exception"
   42     return -code 11
   43     }
   44 }
   45 
   46 proc main {} {
   47     global argv HI LO
   48     set NUM [lindex $argv 0]
   49     if {$NUM < 1} {
   50         set NUM 1
   51     }
   52     incr NUM
   53     while {[incr NUM -1]} {
   54         some_function $NUM
   55     }
   56     puts "Exceptions: HI=$HI / LO=$LO"
   57 }
   58 
   59 main
