    6 void main(int argc, array(string) argv) {
    7     int n = (int)argv[-1];
    8     if (n < 1) n = 1;
    9 
   10     mapping(string:int) hash1 = ([]);
   11     mapping(string:int) hash2 = ([]);
   12     for (int i=0; i<10000; i++)
   13     hash1["foo_" + i] = i;
   14     for (int i=0; i<n; i++) {
   15     foreach (indices(hash1), string k) {
   16         hash2[k] += hash1[k];
   17     }
   18     }
   19     write("%d %d %d %d\n", hash1["foo_1"], hash1["foo_9999"],
   20       hash2["foo_1"], hash2["foo_9999"]);
   21 }
