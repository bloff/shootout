    5 (defun main ()
    6   (declare (optimize (speed 3) (debug 0) (safety 0)))
    7   (let* ((dict (make-hash-table :test 'equal :size 10000)))
    8 
    9     (with-open-file (dictfile "Usr.Dict.Words" :direction :input)
   10             (do ((line (read-line dictfile)
   11                    (read-line dictfile nil 'eof)))
   12             ((eq line 'eof))
   13               (setf (gethash line dict) t)))
   14 
   15     (do ((word (read-line *standard-input*)
   16            (read-line *standard-input* nil 'eof)))
   17     ((eq word 'eof))
   18       (if (not (gethash word dict))
   19       (write-line word)))))
