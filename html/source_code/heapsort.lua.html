    6 local IM = 139968
    7 local IA =   3877
    8 local IC =  29573
    9 
   10 local LAST = 42
   11 function gen_random(max)
   12     LAST = math.mod((LAST * IA + IC), IM)
   13     return( (max * LAST) / IM )
   14 end
   15 
   16 function heapsort(n, ra)
   17     local j, i, rra
   18     local l = math.floor(n/2) + 1
   19     local ir = n;
   20     while 1 do
   21     if l > 1 then
   22         l = l - 1
   23         rra = ra[l]
   24     else
   25         rra = ra[ir]
   26         ra[ir] = ra[1]
   27         ir = ir - 1
   28         if (ir == 1) then
   29         ra[1] = rra
   30         return
   31         end
   32     end
   33     i = l
   34     j = l * 2
   35     while j <= ir do
   36         if (j < ir) and (ra[j] < ra[j+1]) then
   37         j = j + 1
   38         end
   39         if rra < ra[j] then
   40         ra[i] = ra[j]
   41         i = j
   42         j = j + i
   43         else
   44         j = ir + 1
   45         end
   46     end
   47     ra[i] = rra
   48     end
   49 end
   50 
   51 local ary = {}
   52 local N = (tonumber((arg and arg[1])) or 1)
   53 
   54 for i=1, N do
   55     ary[i] = gen_random(1.0)
   56 end
   57 
   58 heapsort(N, ary)
   59 
   60 io.write(string.format("%0.10f\n", ary[N]))
