   10 let size = 10000
   11 
   12 let rec lect stack buf pos free =
   13   let nrd = input stdin buf pos free in
   14   if nrd = 0 then stack, buf, pos (* buf filled only until pos *)
   15   else if nrd = free then lect (buf :: stack) (String.create size) 0 size
   16   else lect stack buf (pos + nrd) (free - nrd)
   17 
   18 let output_buf (buf, len) = output stdout buf 0 len
   19 
   20 let rec rev_write tail stack buf len pos =
   21   if pos = 0 then
   22     match stack with
   23     | [] -> output stdout buf 0 len; List.iter output_buf tail
   24     | topbuf :: stack ->
   25         let toplen = String.length topbuf in
   26         rev_write ((buf, len) :: tail) stack topbuf toplen toplen
   27   else if buf.[pos - 1] = '\n' then
   28     begin
   29       output stdout buf pos (len - pos);
   30       List.iter output_buf tail;
   31       rev_write [] stack buf pos (pos - 1)
   32     end
   33   else rev_write tail stack buf len (pos - 1);;
   34 
   35 let stack, buf, length = lect [] (String.create size) 0 size;;
   36 rev_write [] stack buf length length
