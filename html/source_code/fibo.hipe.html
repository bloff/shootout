    5 -module(fibo).
    6 -export([main/1]).
    7 
    8 main() -> main(['1']).
    9 main([Arg]) ->
   10     Num = list_to_integer(atom_to_list(Arg)),
   11     io:fwrite("~w\n", [fib(Num)]),
   12     halt(0).
   13 
   14 fib(0) -> 1;
   15 fib(1) -> 1;
   16 fib(N) -> fib(N-2) + fib(N-1).
