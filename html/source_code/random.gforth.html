    6 0. argc @ 1- arg >number 2drop drop constant NUM
    7 
    8 : FIM 1e 139968e f/ POSTPONE FLITERAL ; immediate
    9 : IA 3877 POSTPONE LITERAL ; immediate
   10 : IC 29573 POSTPONE LITERAL ; immediate
   11 : IM 139968 POSTPONE LITERAL ; immediate
   12 42 value seed
   13 
   14 \ n = the uth power of n1
   15 : pow ( n1 u -- n )
   16   1 swap 0 u+do over * loop nip ;
   17 
   18 \ print float r with n fixed decimal points
   19 : ffix ( r n -- )
   20   dup 10 swap pow 0 d>f f* 0.5e0 f+ f>d
   21   <<# rot 0 ?do # loop [char] . hold #s #> type #>> ;
   22 
   23 : gen-random
   24     s" seed IA * IC + IM mod dup to seed " evaluate
   25     s" 0 d>f f* FIM f* " evaluate ; immediate
   26 
   27 : main
   28     10 SET-PRECISION
   29     0e NUM 0 do fdrop 100e gen-random loop 9 ffix cr ;
   30 
   31 main bye
