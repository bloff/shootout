    5 import sys, string, math, operator
    6 
    7 def main():
    8     sum = 0
    9     nums = []
   10     atof = string.atof
   11 
   12     nums = map(atof, sys.stdin.readlines())
   13     sum = reduce(operator.add, nums)
   14 
   15     n = len(nums)
   16     mean = sum/n
   17     average_deviation = 0
   18     standard_deviation = 0
   19     variance = 0
   20     skew = 0
   21     kurtosis = 0
   22 
   23     for num in nums:
   24         deviation = num - mean
   25         average_deviation += abs(deviation)
   26         variance += deviation**2;
   27         skew += deviation**3;
   28         kurtosis += deviation**4
   29 
   30     average_deviation /= n
   31     variance /= (n - 1)
   32     standard_deviation = math.sqrt(variance)
   33 
   34     if variance > 0.0:
   35         skew /= (n * variance * standard_deviation)
   36         kurtosis = kurtosis/(n * variance * variance) - 3.0
   37 
   38     nums.sort()
   39     mid = n / 2
   40 
   41     if (n % 2) == 0:
   42         median = (nums[mid] + nums[mid-1])/2
   43     else:
   44         median = nums[mid]
   45 
   46     print "n:                  %d" % n
   47     print "median:             %f" % median
   48     print "mean:               %f" % mean
   49     print "average_deviation:  %f" % average_deviation
   50     print "standard_deviation: %f" % standard_deviation
   51     print "variance:           %f" % variance
   52     print "skew:               %f" % skew
   53     print "kurtosis:           %f" % kurtosis
   54 
   55 main()
   56
