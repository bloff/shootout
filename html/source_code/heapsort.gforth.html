   14 : gen_random ( r -- r )
   15   IA SEED @ * IC + IM mod dup SEED ! s>d d>f
   16   f* [ IM s>d d>f ] fliteral f/ ;
   17 
   18 : heap-sort ( ra n -- )
   19     swap { ra }
   20     dup 2/ 1+ begin ( ir l )
   21     dup 1 > if  ( ir l )
   22         1- dup floats ra + f@ ( ir l rra )
   23     else
   24         over floats ra + dup >r f@ ( ir l rra R: ra+ir )
   25         1 floats ra + f@ r> f! ( ir l rra )
   26         swap 1- dup 1 = if ( l ir rra )
   27         1 floats ra + f!
   28         2drop exit
   29         endif
   30         swap endif ( ir l rra )
   31     { ir l } ( rra )
   32     l l 2* begin ( i j rra )
   33         dup ir <=
   34     while ( i j rra )
   35         dup ir < if
   36         dup floats ra + dup f@ float+ f@ f< if
   37             1+
   38         endif
   39         endif
   40         dup floats ra + f@ fover fover f< if ( i j rra ra[j] )
   41         over floats ra + f!
   42         nip dup 2*
   43         else
   44         fdrop drop ir 1+
   45         endif
   46     repeat
   47     drop floats ra + f!
   48     ir l
   49     again ;
   50 
   51 : my-f. ( r -- )
   52     \ prints r for 0.1<=r<1
   53     ." 0." pad 10 2dup represent 2drop drop type ;
   54 
   55 : main ( -- )
   56     NUM 1+ floats allocate throw ( ra )
   57     dup NUM floats bounds ?do
   58     1e gen_random i f!
   59     1 floats +loop
   60     ( ra ) dup NUM heap-sort
   61     NUM floats + f@ my-f. cr ;
   62 
   63 main bye
