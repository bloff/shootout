   10 let count = ref 0
   11 let data = ref 0
   12 let produced = ref 0
   13 let consumed = ref 0
   14 let m = Mutex.create ()
   15 let c = Condition.create ()
   16 
   17 let producer n =
   18   for i = 1 to n do
   19     Mutex.lock m;
   20     while !count = 1 do Condition.wait c m done;
   21     data := i;
   22     incr count;
   23     Condition.signal c;
   24     Mutex.unlock m;
   25     incr produced
   26   done
   27 
   28 let consumer n =
   29   let i = ref 0 in
   30   while !i <> n do
   31     Mutex.lock m;
   32     while !count = 0 do Condition.wait c m done;
   33     i := !data;
   34     decr count;
   35     Condition.signal c;
   36     Mutex.unlock m;
   37     incr consumed
   38   done
   39 
   40 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
   41         1
   42 let p = Thread.create producer n and c = Thread.create consumer n;;
   43 Thread.join p; Thread.join c;
   44 Printf.printf "%d %d\n" !produced !consumed
