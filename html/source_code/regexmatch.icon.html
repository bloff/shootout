    6 procedure main(argv)
    7     n := \argv[1] | 1
    8 
    9     lines := []
   10     while put(lines,read())
   11 
   12     counter := 0
   13 
   14     start := '(' ++ &digits        # Define what starts a phone number
   15 
   16     every i := 1 to n do {
   17         every line := !lines do {
   18             line ? {
   19                 tab(upto(start))   # Find a possible start...
   20                     (
   21 
   22                      ## Areacode
   23                      (( ="(" & integer(areacode := move(3)) & =")" ) |
   24                       (        integer(areacode := move(3)) )) &
   25                      =" " &                         # followed by a space
   26                      ## Exchange
   27                      integer(exchange := move(3)) & # followed by 3 digits
   28                      =(" " | "-") &                 # followed by space or
   29                       dash
   30                      ## Number
   31                      integer(num := move(4)) &      # followed by 4 digits
   32 
   33                      ( not any(&digits) | pos(0))   # followed by any non
   34                        digit
   35                                                     # or end of line
   36                      ) &
   37                          if i = n then              # Last time through??
   38                              write(counter +:= 1,   # then write the
   39                                    output
   40                                    ": (",areacode,") ",exchange,"-",num)
   41             }
   42         }
   43     }
   44 end
