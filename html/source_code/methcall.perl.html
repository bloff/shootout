    6 package Toggle;
    7 
    8 sub new {
    9     my($class, $start_state) = @_;
   10     bless( { Bool => $start_state }, $class );
   11 }
   12 
   13 sub value {
   14     (shift)->{Bool};
   15 }
   16 
   17 sub activate {
   18     my $self = shift;
   19     $self->{Bool} ^= 1;
   20     return($self);
   21 }
   22 
   23 
   24 package NthToggle;
   25 our @ISA = qw(Toggle);
   26 
   27 sub new {
   28     my($class, $start_state, $max_counter) = @_;
   29     my $self = $class->SUPER::new($start_state);
   30     $self->{CountMax} = $max_counter;
   31     $self->{Counter} = 0;
   32     return($self);
   33 }
   34 
   35 sub activate {
   36     my $self = shift;
   37     if (++$self->{Counter} >= $self->{CountMax}) {
   38     $self->{Bool} ^= 1;
   39     $self->{Counter} = 0;
   40     }
   41     return($self);
   42 }
   43 
   44 
   45 package main;
   46 
   47 sub main {
   48     my $NUM = $ARGV[0];
   49     $NUM = 1 if ($NUM < 1);
   50 
   51     my $val = 1;
   52     my $toggle = Toggle->new($val);
   53     for (1..$NUM) {
   54     $val = $toggle->activate->value;
   55     }
   56     print (($val) ? "true\n" : "false\n");
   57 
   58     $val = 1;
   59     my $ntoggle = NthToggle->new($val, 3);
   60     for (1..$NUM) {
   61     $val = $ntoggle->activate->value;
   62     }
   63     print (($val) ? "true\n" : "false\n");
   64 }
   65 
   66 main();
