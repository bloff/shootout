    5 
    6 n = Integer(ARGV.shift || 1)
    7 
    8 hash1 = {}
    9 for i in 0 .. 9999
   10     hash1["foo_" << i.to_s] = i
   11 end
   12 
   13 hash2 = Hash.new(0)
   14 n.times do
   15     for k in hash1.keys
   16     hash2[k] += hash1[k]
   17     end
   18 end
   19 
   20 printf "%d %d %d %d\n",
   21     hash1["foo_1"], hash1["foo_9999"], hash2["foo_1"], hash2["foo_9999"]
