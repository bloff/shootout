    5 (defvar HI 0)
    6 (defvar LO 0)
    7 
    8 (defun hi_ex (val)
    9   (cons 'hi_ex val))
   10 
   11 (defun lo_ex (val)
   12   (cons 'lo_ex val))
   13 
   14 (defun some_fun (num)
   15   (catch t
   16     (hi_fun num)))
   17 
   18 (defun hi_fun (num)
   19   (let ((val (catch 'Hi_Exception (lo_fun num))))
   20     (if (and (consp val)
   21          (eq (car val) 'hi_ex))
   22     (setq HI (1+ HI)))))
   23 
   24 (defun lo_fun (num)
   25   (let ((val (catch 'Lo_Exception (blowup num))))
   26     (if (and (consp val)
   27          (eq (car val) 'lo_ex))
   28     (setq LO (1+ LO)))))
   29 
   30 (defun blowup (num)
   31   (if (= 0 (mod num 2))
   32       (throw 'Hi_Exception (hi_ex num))
   33     (throw 'Lo_Exception (lo_ex num))))
   34 
   35 (defun main ()
   36   (let* ((arg (car command-line-args-left))
   37      (n (if arg (string-to-number arg) 1)))
   38     (while (> n 0)
   39       (setq n (1- n))
   40       (some_fun n)))
   41   (princ (format "Exceptions: HI=%d / LO=%d\n" HI LO)))
