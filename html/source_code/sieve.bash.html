1 #!/bin/bash
    2 # $Id: sieve.bash.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 # from Steve Fink
    5 
    6 NUM=${1:-1}
    7 
    8 while [ $NUM -gt 0 ]; do
    9     i=2
   10     while [ $i -le 8192 ]; do
   11         eval P$i=true
   12         i=$[ $i + 1 ]
   13     done
   14 
   15     count=0
   16     i=2
   17     while [ $i -le 8192 ]; do
   18         if eval \$P$i; then
   19             # remove all multiples of prime: i
   20             k=$[ $i + $i ]
   21             while [ $k -le 8192 ]; do
   22                 eval P$k=false
   23                 k=$[ $k + $i ]
   24             done
   25             count=$[ $count + 1 ]
   26         fi
   27         i=$[ $i + 1 ]
   28     done
   29 
   30     NUM=$[ $NUM - 1 ]
   31 done
   32 
   33 echo Count: $count
