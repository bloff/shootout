1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: prodcons.rep.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open    rep
   14     rep.system
   15     rep.threads
   16     rep.threads.mutex
   17     rep.threads.condition-variable)
   18 
   19   (declare (unsafe-for-call/cc))
   20 
   21   (define mutex (make-mutex))
   22   (define access (make-condition-variable))
   23   (define count 0)
   24   (define data -1)
   25   (define produced 0)
   26   (define consumed 0)
   27 
   28   (define (consumer n)
   29     (let ((i 0))
   30       (while (/= i n)
   31     (obtain-mutex mutex)
   32     (while (= count 0)
   33       (condition-variable-wait access mutex))
   34     (setq i data)
   35     (setq count 0)
   36     (setq consumed (1+ consumed))
   37     (condition-variable-signal access)
   38     (release-mutex mutex))))
   39 
   40   (define (producer n)
   41     (do ((i 1 (+ i 1)))
   42     ((> i n))
   43       (obtain-mutex mutex)
   44       (while (= count 1)
   45     (condition-variable-wait access mutex))
   46       (setq data i)
   47       (setq count 1)
   48       (setq produced (1+ produced))
   49       (condition-variable-signal access)
   50       (release-mutex mutex)))
   51 
   52   (let* ((arg (get-command-line-option "--num" t))
   53      (n (if arg (string->number arg) 1))
   54      (c (make-thread (lambda () (consumer n)))))
   55     (producer n)
   56     (thread-join c)
   57     (format standard-output "%d %d\n" produced consumed)))
