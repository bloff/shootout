    4 
    5 -module(regexmatch).
    6 -export([main/0, main/1]).
    7 
    8 %% get the program argument, which is how many test iterations to run
    9 main() -> main(['1']).
   10 main([Arg]) ->
   11     Num = list_to_integer(atom_to_list(Arg)),
   12     {ok, Re} = regexp:parse(
   13          "(^|[^0-9\\(])"        % preceeding non-digit or bol
   14          "("                % area code
   15          "\\([0-9][0-9][0-9]\\)"    % is either 3 digits in parens
   16          "|"                % or
   17          "[0-9][0-9][0-9]"        % just 3 digits
   18          ")"                % end of area code
   19          " "                % area code is followed by one space
   20          "[0-9][0-9][0-9]"        % exchange is 3 digits
   21          "[ -]"                % separator is either space or dash
   22          "[0-9][0-9][0-9][0-9]"        % last 4 digits
   23          "($|[^0-9])"            % must be followed by a non-digit
   24         ),
   25     Plist = readlines(),
   26     test(Num, Re, Plist),
   27     halt(0).
   28 
   29 
   30 test(1, Regexp, Plist) ->
   31     % display output on last iteration
   32     Nums = match_phones(Regexp, Plist),
   33     print_phones(1, Nums),
   34     true;
   35 test(N, Regexp, Plist) ->
   36     match_phones(Regexp, Plist),
   37     test(N-1, Regexp, Plist).
   38 
   39 
   40 print_phones(Count, [H|T]) ->
   41     [A,E,N] = H,
   42     % A,E,N is a list of the matching sub-expressions, which are:
   43     % Areacode (3 digits), Exchange (3 digits), Number (4 digits)
   44     io:fwrite("~w: (~s) ~s-~s~n", [Count, A,E,N]),
   45     print_phones(Count+1, T);
   46 print_phones(_, []) ->
   47     true.
   48 
   49 
   50 match_phones(Regexp, List) ->
   51     mapfilter(
   52       fun(String) ->
   53           case regexp:matches(String, Regexp) of
   54           {match, []}      -> false;
   55           {match, Matches} -> parse_phone(String, Matches);
   56           _                -> false
   57           end
   58       end,
   59       List).
   60 
   61 
   62 parse_phone(Str, [H|T]) ->
   63     {Start, Len} = H,
   64     % Numstr is something that looks like a complete phone #
   65     Numstr = string:substr(Str, Start, Len),
   66     case regexp:matches(Numstr, "[0-9][0-9][0-9][0-9]*") of
   67     {match, []}      -> false;
   68     {match, Matches} ->
   69         lists:map(fun({Offset, Length}) ->
   70                   string:substr(Numstr, Offset, Length) end,
   71               Matches);
   72     _                -> false
   73     end;
   74 parse_phone(Str, []) -> [].
   75 
   76 
   77 mapfilter(Fun, [H|T]) ->
   78     case Fun(H) of
   79     false -> mapfilter(Fun, T);
   80     New   -> [New | mapfilter(Fun, T)]
   81     end;
   82 mapfilter(_, []) -> [].
   83 
   84 
   85 readlines() ->
   86     Port = open_port({fd, 0, 1}, [eof, {line, 512}]),
   87     readlines_from_stream([], Port).
   88 
   89 readlines_from_stream(Lines, Port) ->
   90     receive
   91     {Port, eof} ->
   92         lists:reverse(Lines);
   93     {Port, {_, {_, Line}}} ->
   94         readlines_from_stream([Line|Lines], Port)
   95     end.
