   48 // To do this in the "same way" we need to define a Deque 
   49 // Let's only define the functions we need for this test 
   50  
   51 :: Deque a = Dq !(!Int,[a],[a]) 
   52  
   53 emptyDq = Dq (0,[],[]) 
   54  
   55 copy (Dq (n,xs,ys)) = Dq (n,xs,ys) // copy would be implicit  
   56  
   57 addLast a (Dq (n,xs,ys)) = Dq (balance (n+1,xs,[a:ys])) 
   58  
   59 popFirst (Dq (n,[x:xs],ys)) = (x, (Dq (balance (n-1,xs,ys)))) 
   60 popFirst (Dq (_,[],[y])) = (y, emptyDq)   
   61  
   62 popLast (Dq (n,xs,[y:ys])) = (y, (Dq (balance (n-1,xs,ys)))) 
   63 popLast (Dq (_,[x],[])) = (x, emptyDq) 
   64  
   65 first (Dq (_,[x:_],_)) = x 
   66 first (Dq (_,[],[y])) = y 
   67  
   68 reverseDq (Dq (n,xs,ys)) = (Dq (n,ys,xs)) 
   69  
   70 equal (Dq (n1,x1,y1)) (Dq (n2,x2,y2)) =  
   71    n1==n2 && equal_ x1 (reverse y1) x2 (reverse y2) 
   72       where       
   73       equal_ [] [] [] [] = True   
   74       equal_ [] y1 x2 y2 
   75          | y1<>[] = equal_ y1 [] x2 y2 
   76          = False 
   77       equal_ x1 y1 [] y2  
   78           | y2<>[] = equal_ x1 y1 y2 [] 
   79           = False 
   80       equal_ [a:as] y1 [b:bs] y2 
   81          | a==b = equal_ as y1 bs y2 
   82          = False

   84 //   Instances of overloaded functions: 
   85 instance length Deque 
   86    where length (Dq (n,_,_)) = n

   88 // balance would be private to a Deque module 
   89 balance (n,[],ys) 
   90    | n <= 1   = (n,[],ys) 
   91    = (n, reverse f, r) 
   92       where (r,f) = splitAt (n/2) ys 
   93 balance (n,xs,[]) 
   94    | n <= 1   = (n,xs,[]) 
   95    = (n, f, reverse r) 
   96       where (f,r) = splitAt (n/2) xs
