1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: nestedloop.rep.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   (let* ((arg (get-command-line-option "--num" t))
   16      (n (if arg (read-from-string arg) 1))
   17      (x 0))
   18     (do ((a 0 (+ a 1)))
   19     ((= a n))
   20       (do ((b 0 (+ b 1)))
   21       ((= b n))
   22     (do ((c 0 (+ c 1)))
   23         ((= c n))
   24       (do ((d 0 (+ d 1)))
   25           ((= d n))
   26         (do ((e 0 (+ e 1)))
   27         ((= e n))
   28           (do ((f 0 (+ f 1)))
   29           ((= f n))
   30         (setq x (+ x 1))))))))
   31     (format standard-output "%d\n" x)))
