1 #!/usr/bin/mzscheme -r
    2 ;
    3 ;  Faster, more idiomatic Scheme by Neil Van Dyke
    4 ;
    5 (define (main iport)
    6   (apply printf "~s ~s ~s\n"
    7          (let wc ((i #f) (lines 0) (words 0) (chars 0))
    8        (let ((x (read-char iport)))
    9          (if (eof-object? x)
   10              (list lines words chars)
   11          (case x
   12            ((#\newline)     (wc #f (add1 lines) words (add1 chars)))
   13            ((#\space #\tab) (wc #f       lines  words (add1 chars)))
   14            (else
   15              (wc #t lines (if i words (add1 words)) (add1 chars)))))))))
   16 
   17 (main (current-input-port))
