    6 import sys
    7 
    8 IM = 139968
    9 IA = 3877
   10 IC = 29573
   11 
   12 LAST = 42
   13 def gen_random(max):
   14     global LAST
   15     LAST = (LAST * IA + IC) % IM
   16     return( (max * LAST) / IM )
   17 
   18 def main():
   19     N = int(sys.argv[1])
   20     if N < 1:
   21         N = 1
   22     gr = gen_random
   23     for i in xrange(1,N):
   24         gr(100.0)
   25     print "%.9f" % gr(100.0)
   26 
   27 main()
