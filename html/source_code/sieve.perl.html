    2 # $Id: sieve.perl.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $

    4 # with help from Tony Bowden
    5 
    6 use strict;
    7 use integer;
    8 
    9 my $NUM = ($ARGV[0] < 1) ? 1 : $ARGV[0];
   10 my $count;
   11 my @flags = ();
   12 while ($NUM--) {
   13     $count = 0;
   14     my @flags = (0 .. 8192);
   15     for my $i (2 .. 8192 ) {
   16     next unless defined $flags[$i];
   17     # remove all multiples of prime: i
   18     my $k = $i;
   19     undef $flags[$k] while (($k+=$i) < 8193);
   20     $count++;
   21     }
   22 }
   23 print "Count: $count\n";
