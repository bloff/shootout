1 #!/usr/bin/guile \
    2 -e main -s
    3 !#
    4 
    5 ;;; $Id: lists.guile.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 ;;; from Brad Knotwell
    8 
    9 (use-modules (ice-9 format))
   10 
   11 (define SIZE 10000)
   12 (define Li1 (cdr (iota (+ SIZE 1))))
   13 (define Li2 (list-copy Li1))
   14 (define Li3 '())
   15 
   16 ;;;  note the reverses
   17 ;;;  AFAIK, guile doesn't have a primitive for grabbing elts from the
   18 ;;;  end of a list.  no scheme programmer would do this anyway.  they'd
   19 ;;;  reverse the list in place
   20 (define (test-lists)
   21   (begin
   22     (do ((elt (car Li2) (car Li2)))
   23     ((eq? (cdr Li2) '()) (begin  (set! Li3 (cons elt Li3))
   24                      (set! Li2 '())
   25                      (set! Li3 (reverse! Li3))))
   26       (begin (set! Li3 (cons elt Li3))
   27          (set! Li2 (cdr Li2))))
   28     (set! Li3 (reverse! Li3))
   29     (do ((elt (car Li3) (car Li3)))
   30     ((eq? (cdr Li3) '()) (begin (set! Li2 (cons elt Li2))
   31                     (set! Li3 '())
   32                     (set! Li2 (reverse! Li2))))
   33       (begin (set! Li2 (cons elt Li2))
   34          (set! Li3 (cdr Li3))))
   35     (set! Li1 (reverse! Li1))
   36     (if (and (= (car Li1) SIZE) (every-2? = Li1 Li2)) (length Li1) 0)))
   37 
   38 (define every-2? (lambda (test l1 l2)
   39   (or (null? l1)
   40       (and (test (car l1) (car l2))
   41            (every-2? test (cdr l1) (cdr l2))))))
   42 
   43 
   44 (define (main args)
   45   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   46     (do ((i n (set! n (1- n))))
   47     ((= n 1) (write-line (test-lists)))
   48       (test-lists))))
