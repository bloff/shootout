    5 $define IM 139968
    6 $define IA 3877
    7 $define IC 29573
    8 
    9 procedure gen_random(max)
   10     static LAST; initial LAST := 42;
   11     repeat { suspend((max * (LAST := (LAST * IA + IC) % IM)) / IM) }
   12 end
   13 
   14 procedure main(argv)
   15     n := argv[1] | 1;
   16     every i := 2 to n do gen_random(100.0)
   17     # Icon has fixed number of output decimal points to 9, bogus!
   18     write(gen_random(100.0))
   19 end
