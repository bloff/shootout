   13 module sieve 
   14 import StdEnv, LanguageShootout 
   15  
   16 Start = "Count: " +++ toString (sieve primes lBound 0 argi) +++ "\n" 
   17    where primes = {True \\ i<-[0..uBound]} 
   18  
   19 lBound :== 2 
   20 uBound :== 8192 
   21  
   22 sieve :: !*{#.Bool} !Int !Int !Int -> Int 
   23 sieve primes=:{[i]=primei} i count rep 
   24    | i>uBound  
   25       | rep==0   = count  
   26                  = sieve {primes&[j]=True \\ j<-[0..uBound]} lBound 0 (rep-
   27                    1)    
   28                        
   29    | primei      = sieve (loop primes i i) (i+1) (count+1) rep  
   30                  = sieve primes (i+1) count rep 
   31  
   32    where 
   33    loop primes i ii  
   34       | ii<=uBound  = loop {primes&[ii]=False} i (ii+i) 
   35                     = primes
