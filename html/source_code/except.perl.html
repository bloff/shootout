1 #!/usr/bin/perl
    2 # $Id: except.perl.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 
    5 use integer;
    6 
    7 my $HI = 0;
    8 my $LO = 0;
    9 my $NUM = $ARGV[0];
   10 $NUM = 1 if ($NUM < 1);
   11 
   12 package Lo_Exception;
   13 
   14 sub new {
   15     bless({Val => shift}, __PACKAGE__);
   16 }
   17 
   18 package Hi_Exception;
   19 
   20 sub new {
   21     bless({Val => shift}, __PACKAGE__);
   22 }
   23 
   24 package main;
   25 
   26 sub some_function {
   27     my $num = shift;
   28     eval {
   29     &hi_function($num);
   30     };
   31     if ($@) {
   32     die "We shouldn't get here ($@)";
   33     }
   34 }
   35 
   36 sub hi_function {
   37     my $num = shift;
   38     eval {
   39     &lo_function($num);
   40     };
   41     if (ref($@) eq "Hi_Exception") {
   42     $HI++;        # handle
   43     } elsif ($@) {
   44     die $@;        # rethrow
   45     }
   46 }
   47 
   48 sub lo_function {
   49     my $num = shift;
   50     eval {
   51     &blowup($num);
   52     };
   53     if (ref($@) eq "Lo_Exception") {
   54     $LO++;        # handle
   55     } elsif ($@) {
   56     die $@;        # rethrow
   57     }
   58 }
   59 
   60 sub blowup {
   61     my $num = shift;
   62     if ($num % 2) {
   63     die Lo_Exception->new(Num => $num);
   64     } else {
   65     die Hi_Exception->new(Num => $num);
   66     }
   67 }
   68 
   69 $NUM = $ARGV[0];
   70 while ($NUM--) {
   71     &some_function($NUM);
   72 }
   73 print "Exceptions: HI=$HI / LO=$LO\n";
