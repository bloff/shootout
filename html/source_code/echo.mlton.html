    7 exception Error of string
    8 
    9 open MLton.Socket
   10 
   11 val data = "Hello there sailor\n"
   12 
   13 val num = case CommandLine.arguments() of
   14     nil => 1
   15   | n::_ => valOf (Int.fromString n)
   16 
   17 val (port, listener) = listen ()
   18     handle _ =>
   19         raise Error ("Can't listen...\n")
   20 
   21 fun server () =
   22     let val (_, _, ins, outs) = accept listener
   23         fun s b =
   24             case TextIO.inputLine ins of
   25             NONE => let in
   26                   Posix.Process.wait ();
   27               print (concat ["server processed ",
   28                              Int.toString b,
   29                      " bytes\n"])
   30             end
   31           | SOME(instr) =>
   32                   case instr of
   33                   "" => let in
   34                       Posix.Process.wait ();
   35                                   print (concat ["server processed ",
   36                                  Int.toString b,
   37                              " bytes\n"])
   38                                 end
   39                          | i =>  let in
   40                                    TextIO.output(outs, i);
   41                        TextIO.flushOut outs;
   42                                    s (b + 19)
   43                                  end
   44     in s 0
   45     end
   46 
   47 fun client () =
   48     let
   49         val (ins, outs) = connect ("127.0.0.1", port)
   50         fun c 0 = let in
   51                       TextIO.closeOut outs;
   52                       TextIO.closeIn ins
   53                   end
   54           | c n = let in
   55                       TextIO.output(outs, data);
   56               TextIO.flushOut outs;
   57                       case TextIO.inputLine ins of
   58                   NONE => raise Error "Didn't receive the same data"
   59             | SOME(instr) =>
   60                   (case instr of
   61                       data => let in
   62                             c (n - 1)
   63                       end
   64                       | _  => raise Error "Didn't receive the same data")
   65            end
   66     in
   67         c num
   68     end
   69 
   70 val _ = case Posix.Process.fork () of
   71     SOME pid => server ()
   72   | NONE => client ()
