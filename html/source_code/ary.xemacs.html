    7      (n (if arg (string-to-number arg) 1))
    8      (x (make-vector n 0))
    9      (y (make-vector n 0))
   10      (last (- n 1)))
   11     (dotimes (i n)
   12       (aset x i (+ i 1)))
   13     (dotimes (k 1000)
   14       (do ((i last (- i 1)))
   15       ((< i 0))
   16     (aset y i (+ (aref x i) (aref y i)))))
