    9 # modified to pass rows and cols, and avoid matrix size checks
   10 # and added one optimization to reduce subscripted references in
   11 # inner loop.
   12 
   13 import sys
   14 
   15 size = 30
   16 
   17 def mkmatrix(rows, cols):
   18     count = 1
   19     mx = [ None ] * rows
   20     for i in range(rows):
   21         mx[i] = [0] * cols
   22         for j in range(cols):
   23             mx[i][j] = count
   24             count += 1
   25     return mx
   26 
   27 def mmult(rows, cols, m1, m2):
   28     m3 = [ None ] * rows
   29     for i in range( rows ):
   30         m3[i] = [0] * cols
   31         for j in range( cols ):
   32             val = 0
   33             for k in range( cols ):
   34                 val += m1[i][k] * m2[k][j]
   35             m3[i][j] = val
   36     return m3
   37 
   38 def mxprint(m):
   39     for i in range(size):
   40         for j in range(size):
   41             print m[i][j],
   42         print ""
   43 
   44 def main():
   45     iter = int(sys.argv[1])
   46     m1 = mkmatrix(size, size)
   47     m2 = mkmatrix(size, size)
   48     for i in xrange(iter):
   49         mm = mmult(size, size, m1, m2)
   50     print mm[0][0], mm[2][3], mm[3][2], mm[4][4]
   51 
   52 main()
