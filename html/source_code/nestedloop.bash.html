1 #!/bin/bash
    2 # $Id: nestedloop.bash.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 
    5 let iter=1
    6 [ ${#1} -gt 0 -a $1 -gt 0 ] && iter=$1
    7 
    8 x=0
    9 a=$iter
   10 while [ $a -gt 0 ]; do a=$[$a - 1]
   11     b=$iter
   12     while [ $b -gt 0 ]; do b=$[$b - 1]
   13     c=$iter
   14         while [ $c -gt 0 ]; do c=$[$c - 1]
   15         d=$iter
   16             while [ $d -gt 0 ]; do d=$[$d - 1]
   17         e=$iter
   18                 while [ $e -gt 0 ]; do e=$[$e - 1]
   19             f=$iter
   20                     while [ $f -gt 0 ]; do f=$[$f - 1]
   21                         x=$[$x + 1]
   22                     done
   23                 done
   24             done
   25         done
   26     done
   27 done
   28 
   29 echo $x
