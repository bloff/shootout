1 #!/usr/bin/guile \
    2 -e main -s
    3 !#
    4 
    5 ;;; $Id: moments.guile.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 ;;; from Brad Knotwell
    8 
    9 (use-modules (ice-9 format))
   10 
   11 (define sum 0)
   12 (define nums '())
   13 (define (compute-all mean n continuation)
   14   (let ((average-deviation 0) (standard-deviation 0) (variance 0) (skew 0)
   15         (kurtosis 0) (mid 0) (median 0) (deviation 0) (tmp-lst nums))
   16     (do ((num (car tmp-lst) (if (eq? tmp-lst '()) '() (car tmp-lst))))
   17     ((eq? num '()) (begin (set! variance (/ variance (1- n)))
   18                   (set! standard-deviation (sqrt variance))
   19                   (if (> variance 0.0)
   20                   (begin (set! skew (/ skew (* n variance standard-deviation)))
   21                      (set! kurtosis (- (/ kurtosis (* n variance variance)) 3))))
   22                   (set! nums (sort-list nums >))
   23                   (set! mid (/ n 2))
   24                   (set! median (if (= (remainder n 2) 0)
   25                            (/ (+ (list-ref nums mid)
   26                              (list-ref nums (1- mid)))
   27                           2)
   28                            (list-ref nums mid)))
   29                   (continuation n median mean
   30                         (/ average-deviation n)
   31                         standard-deviation variance
   32                         skew kurtosis)))
   33       (let ((deviation (- num mean)))
   34     (begin (set! tmp-lst (cdr tmp-lst))
   35            (set! average-deviation (+ average-deviation (abs deviation)))
   36            (set! variance (+ variance (expt deviation 2)))
   37            (set! skew (+ skew (expt deviation 3)))
   38            (set! kurtosis (+ kurtosis (expt deviation 4))))))))
   39 
   40 (define output-format
   41 "n:                  ~D
   42 median:             ~,6F
   43 mean:               ~,6F
   44 average_deviation:  ~,6F
   45 standard_deviation: ~,6F
   46 variance:           ~,6F
   47 skew:               ~,6F
   48 kurtosis:           ~,6F
   49 ")
   50 
   51 (define (main args)
   52   (do ((line (read-line) (read-line)))
   53       ((eof-object? line)
   54        (compute-all (/ sum (length nums)) (length nums)
   55             (lambda (x . y) (display (apply format (cons output-format (cons x y)
   56        ))))))
   57     (let ((num (string->number line)))
   58       (begin (set! nums (cons num nums)) (set! sum (+ sum num))))))
