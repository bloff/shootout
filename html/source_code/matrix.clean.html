   17 SIZE :== 30 
   18  
   19 Start = resultString (mmultLoop n SIZE SIZE a b c) 
   20    where 
   21       a = initm (mkm SIZE SIZE) 0 1 SIZE SIZE 
   22       b = initm (mkm SIZE SIZE) 0 1 SIZE SIZE       
   23       c = mkm SIZE SIZE 
   24       n = toInt ( argv.[1] ) 
   25       argv = getCommandLine 
   26        
   27       mkm :: Int Int -> {#.{#Int}}    
   28       mkm nrows ncols = {createArray ncols 0 \\ i<-[0..nrows-1]}  
   29  
   30       initm m i count nrows ncols 
   31          | i<nrows = initm {m&[i,j]=count+j \\ j<-[0..ncols-1]}  
   32             (i+1) (count+ncols) nrows ncols  
   33          = m 
   34  
   35       resultString a = 
   36          toString(a.[0,0]) +++ " " +++ 
   37          toString(a.[2,3]) +++ " " +++ 
   38          toString(a.[3,2]) +++ " " +++ 
   39          toString(a.[4,4]) +++ "\n" 
   40       
   41     
   42 mmultLoop n nrows ncols a b c 
   43    | n==0 = c 
   44    = mmultLoop (n-1) nrows ncols a b  
   45       (mmult nrows ncols a (transpose ncols nrows b) c) 
   46        
   47  
   48 mmult nrows ncols a bt c 
   49    = {c&[i,j]= rmult 0 ncols 0 a.[i] bt.[j] \\ i<-[0..nrows-1], j<-[0..
   50      nrows-1]} 
   51    where 
   52       rmult :: !Int !.Int !Int !{#Int} !{#Int} -> Int 
   53       rmult k n v a b 
   54          | k<n = rmult (k+1) n (v + a.[k]*b.[k]) a b 
   55          = v 
   56  
   57 transpose :: !Int !Int !{#.{#Int}} -> {#.{#Int}} 
   58 transpose nrows ncols m = { {m.[i,j] \\ i<-[0..nrows-1]} \\ j<-[0..ncols-1]
   59                           }
