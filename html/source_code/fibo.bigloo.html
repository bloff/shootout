    5 (module fibo
    6     (option (set! *genericity* #f))
    7     (main main))
    8 
    9 (define (main args)
   10   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   11     (display (fib n))
   12     (newline)))
   13 
   14 (define (fib n)
   15   (cond ((< n 2) 1)
   16     (else (+ (fib (- n 2)) (fib (- n 1))))))
