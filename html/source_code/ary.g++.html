    5 #include <iostream>
    6 #include <vector>
    7 
    8 using namespace std;
    9 
   10 int main(int argc, char *argv[]) {
   11     int i, k, n = ((argc == 2) ? atoi(argv[1]) : 1);
   12     typedef vector<int> ARY;
   13     ARY x(n);
   14     ARY y(n);
   15 
   16     for (i=0; i<n; i++) {
   17     x[i] = i + 1;
   18     }
   19     for (k=0; k<1000; k++) {
   20     for (int i = n - 1; i >= 0; --i) {
   21         y[i] += x[i];
   22     }
   23     }
   24 
   25     cout << y[0] << " " << y.back() << endl;
   26 }
