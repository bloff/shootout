    7 #include <stdio.h>
    8 #include <stdlib.h>
    9 #include <string.h>
   10 #include <unistd.h>
   11 
   12 #define MAXREAD 4096
   13 
   14 int main(int argc, char *argv[]) {
   15     int nread, len = 0, size = (4 * MAXREAD);
   16     char *cp, *buf = malloc(size + 1);
   17 
   18     while((nread = read(0,(buf+len),MAXREAD)) > 0) {
   19         len += nread;
   20         if(MAXREAD > (size - len)) {
   21             size <<= 1;
   22             if((buf = realloc(buf,size+1)) == NULL)
   23                 return(fprintf(stderr,"realloc failed\n"),EXIT_FAILURE);
   24         }
   25     }
   26 
   27     if(nread == -1) return(fprintf(stderr,"read\n"),EXIT_FAILURE);
   28 
   29     for (cp = buf+len-1; cp != buf; --cp,nread++)
   30     if ('\n' == *cp) {
   31             fwrite(cp+1,nread,1,stdout);
   32             nread = 0;
   33     }
   34 
   35     fwrite(cp,nread+1,1,stdout);
   36     free(buf);
   37     return(EXIT_SUCCESS);
   38 }
