    6 #include <stdio.h>
    7 #include <stdlib.h>
    8 #include <unistd.h>
    9 #include "../../Include/simple_hash.h"
   10 
   11 int main(int argc, char *argv[]) {
   12     int i, c=0, n = ((argc == 2) ? atoi(argv[1]) : 1);
   13     char buf[32];
   14 
   15     struct ht_ht *ht = ht_create(n);
   16 
   17     for (i=1; i<=n; i++) {
   18     sprintf(buf, "%x", i);
   19     (ht_find_new(ht, buf))->val = i;
   20     }
   21 
   22     for (i=n; i>0; i--) {
   23     sprintf(buf, "%d", i);
   24     if (ht_find(ht, buf)) c++;
   25     }
   26 
   27     ht_destroy(ht);
   28 
   29     printf("%d\n", c);
   30     return(0);
   31 }
