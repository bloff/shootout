    6 def main (n)
    7     # create empty list B
    8     b = Array.new
    9 
   10     # create list (A) of integers from 1 through N
   11     a = (1..n).to_a
   12 
   13     # move each individual item from A to B, in a loop, reversing order
   14     while (not a.empty?)
   15     b.push(a.pop)
   16     end
   17 
   18     # print first 2 elements of B
   19     puts "#{b[0]} #{b[1]}"
   20 
   21     # reverse B (can be done in place)
   22     b.reverse!
   23 
   24     # is 0 a member of B?
   25     puts (b.member?(0)) ? "true" : "false"
   26 
   27     # is N a member of B?
   28     puts (b.member?(n)) ? "true" : "false"
   29 
   30     # filter values from B to A that are less than N/2, preserving order
   31     mid = n/2;
   32     a = b.find_all {|i| i < mid}
   33 
   34     # print first ten items of A
   35     puts a[0..9].join(" ")
   36 
   37     # print sum of items in A that are less than 1000
   38     sum = 0
   39     for i in a
   40     if i < 1000 then sum += i end
   41     end
   42     puts sum
   43 
   44     # append B to end of A
   45     a.concat(b)
   46 
   47     # print length and last element of A
   48     puts "#{a.length} #{a[-1]}"
   49 end
   50 
   51 main(Integer(ARGV.shift || 1))
