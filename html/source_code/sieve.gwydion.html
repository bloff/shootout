    5 define constant <int-vector> = limited(<simple-vector>, of: <integer>);
    6 
    7 define function sieve(limit :: <integer>) => result :: <integer>;
    8   let vec :: <int-vector> = make(<int-vector>, size: limit + 1, fill: 1);
    9 
   10   // We know 1 isn't prime.
   11   vec[0] := 0;
   12   let prime-count = 0;
   13 
   14   // Sieve for the rest.
   15   for (i from 1 below limit)
   16     // We found a new prime!
   17     if (vec[i] ~= 0)
   18       prime-count := prime-count + 1;
   19       let prime = i + 1;
   20       // Eliminate our multiples.
   21       for (j from (i + prime) below limit by prime)
   22     vec[j] := 0;
   23       end for;
   24     end if;
   25   end for;
   26 
   27   // Print our findings.
   28   prime-count;
   29 end function sieve;
   30 
   31 begin
   32   let arg = application-arguments()[0].string-to-integer;
   33   let count = 0;
   34   for (num from 0 below arg)
   35       count := sieve(8192);
   36   end for;
   37   format-out("Count: %d\n", count);
   38 end
