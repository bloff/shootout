    8 proc newClient {sock addr port} {
    9     fconfigure $sock -buffering line -encoding unicode
   10     set r [gets $sock]
   11     set rLength 0
   12     while {![eof $sock]} {
   13     # Extra increase because [gets] doesn't return \n
   14     incr rLength [expr {[string length $r] + 1}]
   15     puts $sock $r
   16     set r [gets $sock]
   17     }
   18     puts "server processed $rLength bytes"
   19     exit
   20 }
   21 
   22 
   23 proc runClient {n addr port} {
   24     set sock [socket $addr $port]
   25     fconfigure $sock -buffering line -encoding unicode
   26     set msg "Hello there sailor"
   27 
   28     incr n
   29     while {[incr n -1]} {
   30     puts $sock $msg
   31     if {[string equal [gets $sock] $msg]} continue
   32     error "Received different message: $r."
   33     }
   34 }
   35 
   36 
   37 set n [lindex $argv 0]
   38 
   39 if {[llength $argv] < 2} {
   40     socket -server newClient -myaddr localhost 10000
   41     #exec tclsh [info script] $n client &
   42     vwait forever
   43 } else {
   44     runClient $n localhost 10000
   45 }
