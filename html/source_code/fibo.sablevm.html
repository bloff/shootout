    4 public class fibo {
    5     public static void main(String args[]) {
    6     int N = Integer.parseInt(args[0]);
    7     System.out.println(fib(N));
    8     }
    9     public static int fib(int n) {
   10     if (n < 2) return(1);
   11     return( fib(n-2) + fib(n-1) );
   12     }
   13 }
