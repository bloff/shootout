    3 // Producer-Consumer Example by Bill Lear
    4 // Adapted from http://java.sun.com/docs/books/tutorial/essential/threads
    5 
    6 public class prodcons {
    7     private class CubbyHole {
    8         private int m_contents;
    9         private boolean m_available = false;
   10 
   11         public synchronized int get() {
   12             while (m_available == false) {
   13                 try {
   14                     wait();
   15                 } catch (InterruptedException e) { }
   16             }
   17             m_available = false;
   18             notifyAll();
   19             return m_contents;
   20         }
   21 
   22         public synchronized void put(int value) {
   23             while (m_available == true) {
   24                 try {
   25                     wait();
   26                 } catch (InterruptedException e) { }
   27             }
   28             m_contents = value;
   29             m_available = true;
   30             notifyAll();
   31         }
   32     }
   33 
   34     private class Producer extends Thread {
   35         private CubbyHole m_cubbyhole;
   36         private int m_count;
   37 
   38         public Producer(CubbyHole c, int count) {
   39             m_cubbyhole = c;
   40             m_count = count;
   41         }
   42 
   43         public void run() {
   44             for (int i = 0; i < m_count; i++) {
   45                 m_cubbyhole.put(i);
   46                 ++m_produced;
   47             }
   48         }
   49     }
   50 
   51     private class Consumer extends Thread {
   52         private CubbyHole m_cubbyhole;
   53         private int m_count;
   54 
   55         public Consumer(CubbyHole c, int count) {
   56             m_cubbyhole = c;
   57             m_count = count;
   58         }
   59 
   60         public void run() {
   61             int value = 0;
   62             for (int i = 0; i < m_count; i++) {
   63                 value = m_cubbyhole.get();
   64                 ++m_consumed;
   65             }
   66         }
   67     }
   68 
   69     public void run() {
   70         m_producer.start();
   71         m_consumer.start();
   72         try { m_producer.join(); } catch (InterruptedException e) { }
   73         try { m_consumer.join(); } catch (InterruptedException e) { }
   74         System.out.println(m_produced + " " + m_consumed);
   75     }
   76 
   77     public prodcons(int count) {
   78         CubbyHole m_cubbyhole = new CubbyHole();
   79         m_producer = new Producer(m_cubbyhole, count);
   80         m_consumer = new Consumer(m_cubbyhole, count);
   81     }
   82 
   83     public static void main(String[] args) {
   84         int count = 1;
   85         try { count = Integer.parseInt(args[0]); } catch (Exception e) { }
   86         new prodcons(count).run();
   87     }
   88 
   89     private Producer m_producer;
   90     private Consumer m_consumer;
   91     private int m_produced = 0;
   92     private int m_consumed = 0;
   93 }
