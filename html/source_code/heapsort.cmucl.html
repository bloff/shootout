    6 (defconstant IM     139968)
    7 (defconstant IA       3877)
    8 (defconstant IC      29573)
    9 
   10 (defvar LAST 42)
   11 
   12 (defun gen_random (max)
   13   (declare (optimize (speed 3) (debug 0) (safety 0)))
   14   (declare (type (signed-byte 32) IM IA IC LAST))
   15   (declare (double-float max))
   16   (setq LAST (mod (+ (* LAST IA) IC) IM))
   17   (/ (* max LAST) IM))
   18 
   19 (defun heapsort (n ra)
   20   (declare (optimize (speed 3) (debug 0) (safety 0)))
   21   (let ((ir n)
   22     (l (+ (ash n -1) 1))
   23     (i 0)
   24     (j 0)
   25     (rra 0.0d0))
   26     (declare (type (simple-array double-float (*)) ra))
   27     (declare (fixnum ir l i j))
   28     (declare (double-float rra))
   29     (block here
   30       (loop
   31     (cond ((> l 1)
   32            (setq rra (aref ra (setq l (- l 1)))))
   33           (t
   34            (setq rra (aref ra ir))
   35            (setf (aref ra ir) (aref ra 1))
   36            (setq ir (- ir 1))
   37            (if (= ir 1)
   38            (progn
   39              (setf (aref ra 1) rra)
   40              (return-from here nil)))))
   41     (setq i l)
   42     (setq j (ash l 1))
   43     (do ()
   44         ((> j ir))
   45       (cond ((and (< j ir) (< (aref ra j) (aref ra (+ j 1))))
   46          (setq j (+ j 1))))
   47       (cond ((< rra (aref ra j))
   48          (setf (aref ra i) (aref ra j))
   49          (setq j (+ j (the fixnum (setq i j)))))
   50         (t
   51          (setq j (+ ir 1)))))
   52     (setf (aref ra i) rra)))))
   53 
   54 (defun main ()
   55   (declare (optimize (speed 3) (debug 0) (safety 0)))
   56   (let* ((n (parse-integer (or (car (last extensions:*command-line-
   57    strings*)) "1")))
   58      (ary (make-array (1+ n) :element-type 'double-float)))
   59     (declare (fixnum n))
   60     (loop for i fixnum from 0 below n do
   61       (setf (aref ary i) (gen_random 1.0d0)))
   62     (heapsort n ary)
   63     (format t "~,10F~%" (aref ary n))))
