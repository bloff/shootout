    5 procedure main(argv)
    6     # local n, i, k, x, y, last
    7     n := argv[1] | 1
    8     x := list(n,0)
    9     y := list(n,0)
   10     every i := 1 to n do {
   11     x[i] := i
   12     }
   13     every k := 0 to 999 do {
   14     every i := n to 1 by -1 do {
   15         y[i] +:= x[i]
   16     }
   17     }
   18     write(y[1], " ", y[n])
   19 end
