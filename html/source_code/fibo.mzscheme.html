1 #!/usr/bin/mzscheme -r
    2 ; Recursive definition, per benchmark requirements (not iterative!)
    3 (define (fib n)
    4   (cond ((< n 2) 1)
    5     (else (+ (fib (- n 2)) (fib (- n 1))))))
    6 
    7 (define (main args)
    8   (let ((n (if (= (vector-length args) 0)
    9                1
   10                (string->number (vector-ref args 0)))))
   11     (display (fib n))
   12     (newline)))
   13 
   14 (main (current-command-line-arguments))
   15
