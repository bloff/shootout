    5 %%% strcat.erl: 22 Jul 2001 ms@mcdev.com.au (Martin Sandiford)
    6 %%%
    7 %%% This is as close to the shootout specs for string concatenation as
    8 %%% I can get it.  There is a small cheat that appends shorter strings
    9 %%% to longer strings, rather than the other way around.  The results
   10 %%% are the same.
   11 %%%
   12 %%% String handling is not really one of Erlang's strengths.
   13 %%% From the Erlang FAQ at http://www.erlang.org/faq/x304.html
   14 %%% question 5.2:
   15 %%%
   16 %%%    String handling in Erlang is less efficient than it could
   17 %%%    be. Each character consumes 8 bytes of memory (a 32 bit integer
   18 %%%    and a 32 bit pointer) and access to the nth. element takes O(n)
   19 %%%    time. The net result is that many typical string handling
   20 %%%    operations are significantly slower in Erlang than in (say) C or
   21 %%%    even Java.
   22 %%%
   23 %%%
   24 %%% Usage: start from command line with:
   25 %%%    erlc strcat.erl # to compile
   26 %%%    erl -noinput -s strcat main 40000
   27 
   28 -module(strcat).
   29 -export([main/1]).
   30 
   31 main([Arg]) ->
   32     Num = list_to_integer(atom_to_list(Arg)),
   33     io:fwrite("~w\n", [length(make_string("hello\n", Num))]),
   34     halt(0).
   35 
   36 make_string(T, N) -> make_string("", T, N).
   37 
   38 make_string(S, _, 0) -> S;
   39 make_string(S, T, N) -> make_string(T ++ S, T, N-1).
