    5 (module hash
    6     (option (set! *genericity* #f))
    7     (main main))
    8 
    9 (define (main args)
   10   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   11      (last (- n 1))
   12      (max (do ((max 1 (* max 2))) ((> max n) max)))
   13      (x (make-hash-table max
   14                  (lambda (o) (string->0..2^x-1 o 14))
   15                  (lambda (c) (car c))
   16                  string=?
   17                  max))
   18      (c 0))
   19     (do ((i 1 (+ i 1)))
   20     ((> i n))
   21       (put-hash! (cons (number->string i 16) i) x))
   22     (do ((i last (- i 1)))
   23     ((< i 0))
   24       (if (get-hash (number->string i 10) x)
   25       (set! c (+ c 1))))
   26     (display c)
   27     (newline)))
