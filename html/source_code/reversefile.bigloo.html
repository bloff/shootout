    5 (module reversefile
    6     (main main))
    7 
    8 (define (main args)
    9   (reverse-input))
   10 
   11 (define (reverse-input)
   12   (let ((line (read-line)))
   13     (cond ((eof-object? line) #f)
   14       (else
   15        (reverse-input)
   16        (display line)
   17        (newline)))))
