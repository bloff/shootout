    6 fun strcat 1 s0 s = s
    7   | strcat n s0 s = strcat (n-1) s0 (s0 ^ s);
    8 
    9 
   10 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   11 
   12 fun main(name, args) =
   13   let
   14     val arg = hd(args @ ["1"]);
   15     val num = atoi arg;
   16     val str = "hello\n";
   17     val result = String.size (strcat num str str);
   18   in (
   19       print (Int.toString result); print "\n";
   20       OS.Process.success
   21   ) end;
