    3 define function ackermann(M :: <integer>, N :: <integer>)
    4  => result :: <integer>;
    5   case
    6     M = 0     => N + 1;
    7     N = 0     => ackermann(M - 1, 1);
    8     otherwise => ackermann (M - 1, ackermann(M, N - 1));
    9   end case;
   10 end function ackermann;
