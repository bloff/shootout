1 #!/usr/bin/perl
    2 # $Id: heapsort.perl.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 # Matt Harris suggested passing the array via typeglob
    5 
    6 use strict;
    7 
    8 use constant IM => 139968;
    9 use constant IA =>   3877;
   10 use constant IC =>  29573;
   11 
   12 use vars qw(@ra);
   13 
   14 my $LAST = 42;
   15 sub gen_random { ($_[0] * ($LAST = ($LAST * IA + IC) % IM)) / IM }
   16 
   17 sub heapsort ($\@) {
   18     my $n = shift;
   19     # use typeglob ra to refer to array.
   20     local *ra = shift;
   21 
   22     my($rra, $i, $j);
   23 
   24     my $l = ($n >> 1) + 1;
   25     my $ir = $n;
   26     while (1) {
   27     if ($l > 1) {
   28         $rra = $ra[--$l];
   29     } else {
   30         $rra = $ra[$ir];
   31         $ra[$ir] = $ra[1];
   32         if (--$ir == 1) {
   33         $ra[1] = $rra;
   34         return;
   35         }
   36     }
   37     $i = $l;
   38     $j = $l << 1;
   39     while ($j <= $ir) {
   40         $j++ if (($j < $ir) && ($ra[$j] < $ra[$j+1]));
   41         if ($rra < $ra[$j]) {
   42         $ra[$i] = $ra[$j];
   43         $j += ($i = $j);
   44         } else {
   45         $j = $ir + 1;
   46         }
   47     }
   48     $ra[$i] = $rra;
   49     }
   50 }
   51 
   52 
   53 my $N = $ARGV[0];
   54 $N = 1 if ($N < 1);
   55 
   56 # create an array of N random doubles
   57 my @ary = ();
   58 for (my $i=1; $i<=$N; $i++) {
   59     $ary[$i] = gen_random(1.0);
   60 }
   61 
   62 heapsort($N, @ary);
   63 
   64 printf("%.10f\n", $ary[-1]);
   65
