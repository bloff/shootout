    6 void main(int argc, array(string) argv)
    7 {
    8     int n = (int)argv[-1];
    9     if (n < 1) n = 1;
   10 
   11     array(int) x = allocate(n);
   12     array(int) y = allocate(n);
   13 
   14     for (int i; i<n; i++) {
   15     x[i] = i + 1;
   16     }
   17     for (int k; k<1000; k++) {
   18     for (int i=n-1; i>=0; i--) {
   19         y[i] += x[i];
   20     }
   21     }
   22     write("%d %d\n", y[0], y[-1]);
   23 }
