    4 BEGIN {
    5     delete ARGV;
    6     FS = "[^A-Za-z][^A-Za-z]*";
    7 }
    8 {
    9     for (i=1; i<=NF; i++) {
   10     freq[tolower($(i))]++;
   11     }
   12 }
   13 END {
   14     # gawk doesn't have a builtin sort routine
   15     # so we have to pipe through the shell sort program
   16     sort = "sort -nr"
   17     for (word in freq) {
   18     if (word) {
   19         printf "%7d\t%s\n", freq[word], word | sort
   20     }
   21     }
   22     close(sort)
   23 }
