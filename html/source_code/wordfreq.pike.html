1 #!/usr/bin/pike
    2 // -*- mode: pike -*-
    3 // $Id: wordfreq.pike.html,v 1.1 2004-06-14 23:09:29 bfulgham Exp $
    4 // http://www.bagley.org/~doug/shootout/
    5 // from: Fredrik Noring
    6 
    7 void main()
    8 {
    9     mapping(string:int) dictionary = ([]);
   10     string buffer = "";
   11 
   12     array(string) f = filter(map(enumerate(128),
   13                    lambda(int i)
   14                    {
   15                        return !('A' <= i && i <= 'Z' ||
   16                         'a' <= i && i <= 'z' ||
   17                         i == ' ') &&
   18                           sprintf("%c", i);
   19                    }), `!=, 0);
   20     array(string) t = allocate(sizeof(f), " ");
   21 
   22     for(;;)
   23     {
   24       string data =
   25           buffer + replace(lower_case(Stdio.stdin.read(4096)), f, t);
   26 
   27       if(!sizeof(data))
   28           break;
   29 
   30       array(string) words = data/" ";
   31 
   32       if(1 < sizeof(words) && sizeof(words[-1]))
   33           buffer = words[-1],
   34            words = words[..sizeof(words)-2];
   35       else
   36           buffer = "";
   37 
   38       foreach(words, string word)
   39           dictionary[word]++;
   40     }
   41 
   42     m_delete(dictionary, "");
   43 
   44     mapping(int:array(string)) revdictionary = ([]);
   45     array(string) words = indices(dictionary);
   46     array(int) freqs = values(dictionary);
   47 
   48     for(int i = 0; i < sizeof(dictionary); i++)
   49       revdictionary[freqs[i]] += ({ words[i] });
   50 
   51     freqs = sort(indices(revdictionary));
   52     for(int i = sizeof(freqs)-1; 0 <= i; i--)
   53     {
   54       int freq = freqs[i];
   55       words = sort(revdictionary[freq]);
   56 
   57       for(int j = sizeof(words)-1; 0 <= j; j--)
   58           write("%7d\t%s\n", freq, words[j]);
   59     }
   60 }
