    5 module Main where
    6 
    7 import System(getArgs, exitWith, ExitCode(..))
    8 import Numeric(readDec)
    9 
   10 main = do
   11         arg <- getArgs
   12         case arg of
   13          [number] -> let n = fst (head (readDec number)) in
   14                      putStrLn (show (loopA n n 0))
   15          _      -> exitWith (ExitFailure 1)
   16 
   17 loopA :: Int -> Int -> Int -> Int
   18 loopA n m x
   19    | n > 0         = loopA (n-1) m (loopB m m x)
   20    | otherwise     = x
   21 
   22 loopB :: Int -> Int -> Int -> Int
   23 loopB n m x
   24    | n > 0         = loopB (n-1) m (loopC m m x)
   25    | otherwise     = x
   26 
   27 loopC :: Int -> Int -> Int -> Int
   28 loopC n m x
   29    | n > 0         = loopC (n-1) m (loopD m m x)
   30    | otherwise     = x
   31 
   32 loopD :: Int -> Int -> Int -> Int
   33 loopD n m x
   34    | n > 0         = loopD (n-1) m (loopE m m x)
   35    | otherwise     = x
   36 
   37 loopE :: Int -> Int -> Int -> Int
   38 loopE n m x
   39    | n > 0         = loopE (n-1) m (loopF m x)
   40    | otherwise     = x
   41 
   42 loopF :: Int -> Int -> Int
   43 loopF n x
   44    | n > 0         = loopF (n-1) (x+1)
   45    | otherwise     = x
