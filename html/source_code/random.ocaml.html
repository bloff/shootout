    7 let im = 139968 and ia = 3877 and ic = 29573 and last_ref = ref 42
    8 
    9 let gen_random max =
   10   let new_last = (!last_ref * ia + ic) mod im in
   11   last_ref := new_last;
   12   max *. float_of_int new_last /. float im
   13 
   14 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
   15         1
   16 let rec loop i = let r = gen_random 100.0 in if i > 1 then loop (i - 1)
   17                  else r;;
   18 Printf.printf "%.9f\n" (loop n)
