    6 #  first create a list (L1) of integers from 1 through SIZE (SIZE is
    7 #  currently defined as 10000).
    8 
    9 #  copy L1 to L2 (can use any builtin list copy function, if
   10 #  available)
   11 
   12 #  remove each individual item from left side (head) of L2 and append
   13 #  to right side (tail) of L3 (preserving order). (L2 should be
   14 #  emptied by one item at a time as that item is appended to L3).
   15 
   16 #  remove each individual item from right side (tail) of L3 and append
   17 #  to right side (tail) of L2 (reversing list). (L3 should be emptied
   18 #  by one item at a time as that item is appended to L2).
   19 
   20 #  reverse L1 (preferably in place) (can use any builtin function for
   21 #  this, if available).
   22 
   23 #  check that first item of L1 is now == SIZE.
   24 
   25 #  and compare L1 and L2 for equality and return length of L1 (which
   26 #  should be equal to SIZE).
   27 
   28 procedure main(argv)
   29     n := argv[1] | 1
   30     every i := 1 to n do count := test_lists()
   31     write(count)
   32 end
   33 
   34 procedure test_lists(args)
   35     SIZE := 10000
   36 
   37     L1 := list()
   38     every put(L1,1 to SIZE)
   39 
   40     L2 := copy(L1)
   41 
   42     L3 := list()
   43     while put(L3,get(L2))
   44 
   45     while push(L2,pop(L3))
   46 
   47     reverseL(L1)
   48 
   49     if L1[1] ~= SIZE then write("Failed test")
   50 
   51     every i := 1 to SIZE do
   52         if L1[i] ~= L2[i] then write("L1 ~= L2")
   53 
   54     return(*L1)
   55 end
   56 
   57 # Reverse a list in place
   58 procedure reverseL(l)
   59     ender := *l + 1
   60     middle := integer(*l / 2)
   61     every i := 1 to middle do
   62         l[i] :=: l[ender - i]
   63 end
