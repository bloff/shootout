    8 (use-modules (ice-9 format))
    9 (define DATA "Hello there sailor\n")
   10 (define bufferSize (string-length DATA))
   11 
   12 (define (echo-client n port-number)
   13   (let ((new-sock (socket AF_INET SOCK_STREAM 0))
   14     (buf (make-string bufferSize)))
   15     (begin (connect new-sock
   16             AF_INET
   17             INADDR_LOOPBACK
   18             port-number)
   19        (do ((i 0 (1+ i)))
   20            ((= i n) (close new-sock))
   21          (begin
   22                  (send new-sock DATA)
   23          (recv! new-sock buf)
   24                  (if (not (string=? buf DATA)) (throw 'badData)))))))
   25 
   26 (define (echo-server n)
   27   (let ((sock (socket AF_INET SOCK_STREAM 0)))
   28     (begin (setsockopt sock SOL_SOCKET SO_REUSEADDR 1)
   29        (bind sock AF_INET INADDR_LOOPBACK 0)
   30        (listen sock 2)
   31        (let ((pid (primitive-fork)))
   32          (if (= pid 0)
   33          (echo-client n (array-ref (getsockname sock) 2))
   34          (let ((new-sock (car (accept sock)))
   35                (buf (make-string bufferSize))
   36                (num-read 0))
   37            (do ((i (recv! new-sock buf) (recv! new-sock buf)))
   38                ((= 0 i) (begin (waitpid pid WNOHANG)
   39                        (display (format "server processed ~D bytes\n" num-read))))
   40              (send new-sock buf) (set! num-read (+ num-read i)))))))))
   41 
   42 (echo-server (or (and (= (length args) 2) (string->number (cadr args))) 1))
   43
