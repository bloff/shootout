1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: ary.rep.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   (let* ((arg (get-command-line-option "--num" t))
   16      (n (if arg (read-from-string arg) 1))
   17      (x (make-vector n 0))
   18      (y (make-vector n 0))
   19      (last (- n 1)))
   20     (do ((i 0 (+ i 1)))
   21     ((= i n))
   22       (aset x i (+ i 1)))
   23     (do ((k 0 (+ k 1)))
   24     ((= k 1000))
   25       (do ((i last (- i 1)))
   26       ((< i 0))
   27     (aset y i (+ (aref x i) (aref y i)))))
   28     (format standard-output "%d %d\n" (aref y 0) (aref y last))))
