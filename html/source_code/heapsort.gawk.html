    4 function gen_random(n) { return( (n * (LAST = (LAST * IA + IC) % IM)) / IM
    5                     ); }
    6 
    7 function heapsort (n, ra) {
    8     l = int(0.5+n/2) + 1
    9     ir = n;
   10     for (;;) {
   11         if (l > 1) {
   12             rra = ra[--l];
   13         } else {
   14             rra = ra[ir];
   15             ra[ir] = ra[1];
   16             if (--ir == 1) {
   17                 ra[1] = rra;
   18                 return;
   19             }
   20         }
   21         i = l;
   22         j = l * 2;
   23         while (j <= ir) {
   24             if (j < ir && ra[j] < ra[j+1]) { ++j; }
   25             if (rra < ra[j]) {
   26                 ra[i] = ra[j];
   27                 j += (i = j);
   28             } else {
   29                 j = ir + 1;
   30             }
   31         }
   32         ra[i] = rra;
   33     }
   34 }

   38     IA = 3877;

   42     n = (ARGV[1] < 1) ? 1 : ARGV[1];
   43     ary[0] = 0;
   44     for (i=1; i<=n; i++) {
   45     ary[i] = gen_random(1.0);
   46     }

   48     heapsort(n, ary);
