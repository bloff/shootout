    5 (module regexmatch
    6     (main main))
    7 
    8 (define match-phone
    9   (let ((last-char-is-non-digit #t)
   10     (found #f)
   11     (parts '()))
   12     (regular-grammar ((non-digit (out ("09") #\(  ))
   13               (area-code (= 3 digit))
   14               (exchange (= 3 digit))
   15               (separator (in (" -")))
   16               (number (= 4 digit))
   17 
   18               ;; parts of phone number begins with a 3-digit area code
   19               (phone-parts (: (or (: "(" (submatch area-code) ")")
   20                       (submatch area-code))
   21                       ;; the area code is followed by one space
   22                       space
   23                       ;; which is followed by the 3 digits of the exchange
   24                       (submatch exchange)
   25                       ;; the exchange is followed by a space or hyphen [ -]
   26                       separator
   27                       ;; which is followed by the last 4 digits: DDDD
   28                       (submatch number))))
   29 
   30              ;; a telephone number must be preceeded by a non-digit (or bol).
   31              ;; and followed by a non-digit (or eol).
   32 
   33              ;; scanned a non-digit char
   34              ((: non-digit)
   35               (set! last-char-is-non-digit #t)
   36               (if found (begin (set! found #f) parts) (ignore)))
   37              ;; scanned a digit char
   38              ((: digit)
   39               (set! last-char-is-non-digit #f)
   40               (if found (begin (set! found #f) '()) (ignore)))
   41              ((when last-char-is-non-digit (: phone-parts))
   42               (set! found #t)
   43               (set! parts (list (the-submatch 1)
   44                     (the-submatch 2)
   45                     (the-submatch 3)
   46                     (the-submatch 4)))
   47               (ignore))
   48              (else
   49               ;; possibly get here on eol
   50               (if found (begin (set! found #f) parts) '())))))
   51 
   52 
   53 (define (main args)
   54   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   55     (phonelines '())
   56     (count 0))
   57     (let loop ((line (read-line)))
   58       (cond ((eof-object? line) #f)
   59         (else
   60          (set! phonelines (append phonelines (list line)))
   61          (loop (read-line)))))
   62 
   63     (do ((i N (- i 1)))
   64     ((= i 0))
   65       (let ((count 0))
   66     (for-each (lambda (phone)
   67             (let* ((port (open-input-string phone))
   68                (match-parts (read/rp match-phone port)))
   69               (if (and (not (null? match-parts))
   70                    (= 1 i))
   71               (begin
   72                 (let* ((area
   73                     (if (string=? (car match-parts) "")
   74                     (cadr match-parts)
   75                     (car match-parts)))
   76                    (exch (caddr match-parts))
   77                    (numb (cadddr match-parts))
   78                    (num (string-append "(" area ") " exch "-" numb)))
   79                   (set! count (+ count 1))
   80                   (print count ": " num))))))
   81           phonelines)))))
