    7 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
    8         1
    9 let rec loopF x = function 0 -> x | i -> loopF (x+1) (i-1)
   10 let rec loopE x = function 0 -> x | i -> loopE (loopF x n) (i-1)
   11 let rec loopD x = function 0 -> x | i -> loopD (loopE x n) (i-1)
   12 let rec loopC x = function 0 -> x | i -> loopC (loopD x n) (i-1)
   13 let rec loopB x = function 0 -> x | i -> loopB (loopC x n) (i-1)
   14 let rec loopA x = function 0 -> x | i -> loopA (loopB x n) (i-1);;
   15 Printf.printf "%d\n" (loopA 0 n)
