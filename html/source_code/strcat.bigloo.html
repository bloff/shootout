    6 (module strcat
    7     (option (set! *genericity* #f))
    8     (main main))
    9 
   10 (define hello #"hello\n")
   11 
   12 (define (main args)
   13   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   14      (i 0)
   15      (buflen 32)
   16      (strbuf (make-string buflen))
   17      (stufflen (string-length hello))
   18      (stuff hello)
   19      (stuffstart 0))
   20 
   21     (do ((i 0 (+ i 1)))
   22     ((= i n))
   23       (if (>= (+fx stufflen stuffstart) buflen)
   24       (let* ((nbuflen (*fx 2 buflen))
   25          (nstrbuf (make-string nbuflen)))
   26         (blit-string! strbuf 0 nstrbuf 0 buflen)
   27         (set! buflen nbuflen)
   28         (set! strbuf nstrbuf)))
   29       (blit-string! stuff 0 strbuf stuffstart stufflen)
   30       (set! stuffstart (+fx stuffstart stufflen)))
   31 
   32     (set! strbuf (substring strbuf 0 stuffstart))
   33 
   34     (display (string-length strbuf))
   35     (newline)))
