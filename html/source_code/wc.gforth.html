    5 variable  nn       0       nn !       \ number of newlines
    6 variable  nw       0       nw !       \ number of words
    7 variable  nc       0       nc !       \ number of chars
    8 variable  in_word  0       in_word !  \ flag: "in word"
    9 
   10 10  constant  nl_ch
   11 9   constant  tab_ch
   12 32  constant  space_ch
   13 
   14 4096 constant MAXREAD
   15 create buff MAXREAD allot
   16 
   17 \ scan the buffer and count lines, words, chars
   18 : scanbuff ( n -- )
   19     dup nc +!                 \ update nc with amount of chars in buffer
   20     buff + buff           \ from start of buff to buff + n
   21     do
   22     i c@
   23     case
   24         nl_ch    of  0 in_word !  1 nn +!  endof
   25         tab_ch   of  0 in_word !  endof
   26         space_ch of  0 in_word !  endof
   27         \ otherwise:
   28         in_word @ 0=
   29         if
   30         1 in_word !
   31         1 nw +!
   32         endif
   33     endcase
   34     loop ;
   35 
   36 : wc ( -- )
   37     buff
   38     begin
   39         buff MAXREAD stdin read-file throw dup
   40     while
   41     scanbuff
   42     repeat ;
   43 
   44 wc nn @ . nw @ . nc @ 1 u.r cr
   45 
   46 bye \ th-th-that's all folks!
