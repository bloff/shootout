   14 #define QUICKIE_STRCMP(a, b)  (*(a) != *(b) ? *(a) - *(b) : strcmp((a), (b)
   15                        ))
   16 
   17 typedef int (*comparator)(const void *, const void *);
   18 
   19 int cmp_hash(struct ht_node **a, struct ht_node **b) {
   20     int val = (*b)->val - (*a)->val;
   21     return((val == 0) ? QUICKIE_STRCMP((*b)->key, (*a)->key) : val);
   22 }

   24 int main() {
   25     int readbufsize = 4096;
   26     int wordbufsize=16;
   27     char *readbuf = (char *)malloc(readbufsize + 1);
   28     char *wordbuf = (char *)malloc(wordbufsize + 1);
   29     int i = 0;
   30     struct ht_ht *ht = ht_create(2048);
   31     struct ht_node **sort_array, **sort_tmp, *node;
   32     /*new code*/
   33     int nread =0;
   34     int wordlen=0;
   35     readbuf[0]=0;
   36     while (readbuf[i] > 0||(nread = fread(readbuf, sizeof(char),
   37            readbufsize, stdin),readbuf[nread] = '\0',i=0,nread > 0) ) {
   38     if (isalpha(readbuf[i])){
   39         wordbuf[wordlen++] = tolower(readbuf[i]);
   40         if (wordlen == wordbufsize) {
   41         wordbufsize *= 2;
   42         wordbuf = realloc(wordbuf, wordbufsize + 1);
   43         }
   44     }
   45     else{
   46         if (wordlen > 0) {
   47         wordbuf[wordlen] = '\0';
   48         ++(ht_find_new(ht, wordbuf)->val);
   49         wordlen = 0;
   50         }
   51     }
   52     i++;
   53     }
   54     free(readbuf);
   55     free(wordbuf);
   56     sort_array = sort_tmp =
   57     malloc(sizeof(struct ht_node *) * ht_count(ht));
   58 
   59     for (node=ht_first(ht); (*sort_tmp++ = node) != 0; node=ht_next(ht)) ;
   60 
   61     qsort(sort_array, ht_count(ht), sizeof(struct ht_node *),
   62       (comparator)cmp_hash);
   63 
   64     for (i=0; i<ht_count(ht); i++)
   65     printf("%7d\t%s\n", ht_val(sort_array[i]), ht_key(sort_array[i]));
   66 
   67     ht_destroy(ht);
   68     return(0);
   69 }
