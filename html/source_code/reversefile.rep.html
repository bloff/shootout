1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: reversefile.rep.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   ;; eat the unused command line arg
   16   (get-command-line-option "--num" t)
   17 
   18   (let ((lines '()))
   19     (let loop ((line (read-line)))
   20      (cond ((not line) nil)
   21            (t
   22         (setq lines (cons line lines))
   23         (loop (read-line)))))
   24     (let loop ((lines lines))
   25      (cond ((not lines) nil)
   26            (t
   27         (princ (car lines))
   28         (loop (cdr lines)))))))
