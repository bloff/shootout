1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: random.rep.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   (define IM     139968)
   16   (define IA       3877)
   17   (define IC      29573)
   18 
   19   (define LAST 42)
   20   (define (gen_random max)
   21     (/ (* max (setq LAST (modulo (+ (* LAST IA) IC) IM))) IM))
   22 
   23   (let* ((arg (get-command-line-option "--num" t))
   24      (n (if arg (read-from-string arg) 1)))
   25     (let loop ((iter n))
   26      (if (> iter 1)
   27          (progn
   28            (gen_random 100.0)
   29            (loop (- iter 1)))))
   30     (format standard-output "%^11d\n"
   31         (/ (round (* (gen_random 100.0) 1000000000)) 1000000000))))
