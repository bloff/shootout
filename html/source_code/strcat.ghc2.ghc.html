    5 module Main where
    6 import System(getArgs, getProgName, exitWith, ExitCode(..))
    7 import Numeric(readDec)
    8 
    9 nAppend :: Int -> [Char] -> [Char] -> [Char]
   10 nAppend 0 s1 _  = s1
   11 nAppend n s1 s2 = nAppend (n - 1) (s1 ++ s2) s2
   12 
   13 lengthNHellos :: Int -> Int
   14 lengthNHellos n = length (nAppend n "" "hello\n")
   15 
   16 main = do
   17        arg <- getArgs
   18        case arg of
   19          [number] -> let n = fst (head (readDec number)) in
   20                      putStrLn (show (lengthNHellos n))
   21          _        -> do
   22                      putStr "Usage: "
   23                      getProgName >>= putStr
   24                      putStrLn " number"
   25                      exitWith (ExitFailure 1)
