    5 local text = io.read("*a")
    6 
    7 -- make sure text does not start with a number
    8 text = "\n" .. text
    9 
   10 -- pattern is: not a digit, optional (, 3 digits, optional ),
   11 -- space, 3 digits, space or hyphen, 4 digits, not a digit
   12 local pattern = "%D(%(?)(%d%d%d)(%)?) (%d%d%d)[- ](%d%d%d%d)%f[%D]"
   13 
   14 local N = tonumber((arg and arg[1])) or 1
   15 local count = 0
   16 for i=N,1,-1 do
   17   for open,area,close,exch,digits in string.gfind(text, pattern) do
   18       if (open == '(') == (close == ')') then
   19         local tel = "("..area..") "..exch.."-"..digits
   20         if i == 1 then
   21           count = count+1
   22           io.write(count, ": ", tel, "\n")
   23         end
   24       end
   25     end
   26 end
