    6 (module method
    7     (main main)
    8     (static (class toggle
    9                (state::bool (default #t)))
   10         (class nthtoggle::toggle
   11                (count-max::int (default 0))
   12                (counter::int (default 0)))))
   13 
   14 (define-generic (value o::toggle))
   15 
   16 (define (print-bool b)
   17   (display (if b #"true\n" #"false\n")))
   18 
   19 (define-method (value o::toggle)
   20   (with-access::toggle o (state)
   21                state))
   22 
   23 (define-generic (activate o::toggle))
   24 
   25 (define-method (activate o::toggle)
   26   (with-access::toggle o (state)
   27                (set! state (not state))
   28                o))
   29 
   30 (define-method (activate o::nthtoggle)
   31   (with-access::nthtoggle o (counter count-max state)
   32               (set! counter (+fx 1 counter))
   33               (if (>=fx counter count-max)
   34                   (begin
   35                 (set! state (not state))
   36                 (set! counter 0)))
   37               o))
   38 
   39 (define (main args)
   40   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   41     (let* ((val #t)
   42        (toggle (instantiate::toggle
   43             (state val))))
   44       (let loop ((i 0))
   45     (if (<fx i n)
   46         (begin
   47           (set! val (value (activate toggle)))
   48           (loop (+fx i 1)))))
   49       (print-bool val)
   50       (set! val #t)
   51       (let ((ntoggle (instantiate::nthtoggle
   52               (state #t)
   53               (count-max 3))))
   54     (let loop ((i 0))
   55       (if (<fx i n)
   56           (begin
   57         (set! val (value (activate ntoggle)))
   58         (loop (+fx i 1)))))
   59     (print-bool val)))))
