    7 structure Test : sig
    8     val main : (string * string list) -> OS.Process.status
    9 end = struct
   10 
   11 val bufsize = 4096
   12 val rdbufsize = 4096
   13 
   14 val stdout = Posix.FileSys.wordToFD 0w1
   15 val stdin = Posix.FileSys.wordToFD 0w0
   16 
   17 datatype block = END
   18                | MORE of int ref * Word8Array.array * block
   19 
   20 val buff = Unsafe.Word8Array.create rdbufsize
   21 
   22 fun out END = ()
   23   | out (MORE (ir as ref i, a, next)) =
   24   let in
   25     Posix.IO.writeArr (stdout, Word8ArraySlice.slice(a, i, NONE));
   26     out next
   27   end
   28 
   29 fun rd (start, len, count, b) =
   30   if (start + len) >= count then
   31     (* done with this block.
   32        Copy from start to the end of the array into
   33        buff, then return the starting index into buff. *)
   34     let in
   35       Word8ArraySlice.copy {src=Word8ArraySlice.slice(buff, start, SOME(
   36                             len)),
   37                             dst=buff, di=0};
   38       (b, len)
   39     end
   40   else
   41     if Unsafe.Word8Array.sub(buff, start + len) = 0w10 then
   42       (* found newline *)
   43       case b of MORE(ir as ref i, a, _) =>
   44         if i > len then
   45             let in
   46               (* enough room *)
   47               Word8ArraySlice.copy {src=Word8ArraySlice.slice(buff, start,
   48                                     SOME(len+1)),
   49                                 dst=a, di=(i - len - 1)};
   50               ir := i - (len + 1);
   51 
   52               rd(start + len + 1, 0, count, b)
   53             end
   54           else (* not enough room *)
   55             let
   56               (* going to need a new buffer *)
   57               val na = Unsafe.Word8Array.create bufsize
   58               val l = (len + 1) - i
   59             in
   60               (* put the tail in whatever room is left *)
   61               Word8ArraySlice.copy {src=Word8ArraySlice.slice(buff, (start
   62                                     + len + 1) - i, SOME(i)),
   63                                 dst=a, di=0 };
   64 
   65               (* put the head in a new buffer *)
   66               Word8ArraySlice.copy {src=Word8ArraySlice.slice(buff, start,
   67                                     SOME(l)),
   68                                 dst=na, di=bufsize - l };
   69               ir := 0;
   70               rd(start + len + 1, 0, count, MORE(ref (bufsize - l), na, b))
   71             end
   72     else rd (start, len + 1, count, b)
   73 
   74 fun loop (b, s) =
   75   let
   76     val count = Posix.IO.readArr (stdin, Word8ArraySlice.slice(buff, s,
   77                 SOME(rdbufsize-s)))
   78     val (bb, bs) = rd (0, s, count + s, b)
   79   in
   80     case count of
   81       0 => out bb
   82     | _ => loop (bb, bs)
   83   end
   84 
   85 fun main(name, args) =
   86     ( loop (MORE(ref bufsize, Unsafe.Word8Array.create bufsize, END), 0);
   87       OS.Process.success);
   88 
   89 end
   90 
   91 val _ = SMLofNJ.exportFn("reversefile", Test.main);
