    5 (module matrix
    6     (option (set! *genericity* #f))
    7     (main main))
    8 
    9 (define size 30)
   10 
   11 (define (1+ x) (+ x 1))
   12 
   13 (define (mkmatrix rows cols)
   14   (let ((mx (make-vector rows 0))
   15     (count 1))
   16     (do ((i 0 (1+ i)))
   17     ((= i rows))
   18       (let ((row (make-vector cols 0)))
   19     (do ((j 0 (1+ j)))
   20         ((= j cols))
   21       (vector-set! row j count)
   22       (set! count (+ count 1)))
   23     (vector-set! mx i row)))
   24     mx))
   25 
   26 (define (num-cols mx)
   27   (let ((row (vector-ref mx 0)))
   28     (vector-length row)))
   29 
   30 (define (num-rows mx)
   31   (vector-length mx))
   32 
   33 (define (mmult rows cols m1 m2)
   34   (let ((m3 (make-vector rows 0)))
   35     (do ((i 0 (1+ i)))
   36     ((= i rows))
   37       (let ((m1i (vector-ref m1 i))
   38         (row (make-vector cols 0)))
   39     (do ((j 0 (1+ j)))
   40         ((= j cols))
   41       (let ((val 0))
   42         (do ((k 0 (1+ k)))
   43         ((= k cols))
   44           (set! val (+ val (* (vector-ref m1i k)
   45                   (vector-ref (vector-ref m2 k) j)))))
   46         (vector-set! row j val)))
   47     (vector-set! m3 i row)))
   48     m3))
   49 
   50 (define (matrix-print m)
   51   (do ((i 0 (1+ i)))
   52       ((= i (num-rows m)))
   53     (let ((row (vector-ref m i)))
   54       (do ((j 0 (1+ j)))
   55       ((= j (num-cols m)))
   56     (display (vector-ref row j))
   57     (if (< j (num-cols m))
   58         (display " ")))
   59       (newline))))
   60 
   61 (define (print-list . items) (for-each display items) (newline))
   62 
   63 (define (main args)
   64   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   65     (let ((mm 0)
   66       (m1 (mkmatrix size size))
   67       (m2 (mkmatrix size size)))
   68       (let loop ((iter n))
   69     (cond ((> iter 0)
   70            (set! mm (mmult size size m1 m2))
   71            (loop (- iter 1)))))
   72       (let ((r0 (vector-ref mm 0))
   73         (r2 (vector-ref mm 2))
   74         (r3 (vector-ref mm 3))
   75         (r4 (vector-ref mm 4)))
   76     (print-list (vector-ref r0 0) " " (vector-ref r2 3) " "
   77             (vector-ref r3 2) " " (vector-ref r4 4))))))
