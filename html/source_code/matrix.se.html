   16 class MATRIX
   17 
   18 creation make
   19 
   20 feature -- Creation
   21 
   22    make is
   23       local
   24      index, count: INTEGER
   25      m1, m2: like matrix
   26       do
   27      from
   28         if argument_count < 1 then
   29            count := 1
   30         else
   31            count := argument(1).to_integer
   32         end
   33         index := 0
   34         m1 := new_matrix(30, 30)
   35         m2 := new_matrix(30, 30)
   36         !!matrix.make(0, 29, 0, 29)
   37      until
   38         index = count
   39      loop
   40         mmult(30, 30, m1, m2)
   41         index := index + 1
   42      end -- from
   43      print(matrix.item(0, 0).to_string + " " + matrix.item(2, 3).to_string +
   44         " " +
   45            matrix.item(3, 2).to_string + " " + matrix.item(4, 4).to_string +
   46                     "%N")
   47       end -- make
   48 
   49 feature -- Queries
   50 
   51    matrix: ARRAY2[INTEGER]
   52 
   53    new_matrix(rows, columns: INTEGER): like matrix is
   54      -- Create and populate a new matrix.
   55       local
   56      i, j, count: INTEGER
   57       do
   58      !!Result.make(0, rows - 1, 0, columns - 1)
   59      from
   60         count := 1
   61         i := 0
   62      until i = rows loop
   63         from j := 0 until j = columns loop
   64            Result.put(count, i, j)
   65            count := count + 1
   66            j := j + 1
   67         end
   68         i := i + 1
   69      end
   70       end -- new_matrix
   71 
   72 feature -- Commands
   73 
   74    zero_matrix(rows, columns: INTEGER; a_matrix: like matrix) is
   75      -- Clear a matrix
   76       do
   77      matrix.make(0, rows - 1, 0, columns - 1)
   78       end -- zero_matrix
   79 
   80    mmult(rows, columns: INTEGER; first, second: like matrix) is
   81      -- Multiply two matrices.
   82       local
   83      i, j, k, val: INTEGER
   84       do
   85      zero_matrix(rows, columns, matrix)
   86      from i := 0 until i = rows loop
   87         from j := 0 until j = columns loop
   88            val := 0
   89            from k := 0 until k = columns loop
   90           val := val + first.item(i, k) * second.item(k, j)
   91           k := k + 1
   92            end
   93            matrix.put(val, i, j)
   94            j := j + 1
   95         end
   96         i := i + 1
   97      end -- from
   98       end -- mmult
   99 
  100 end
