    6 %%  Usage: start from command line with
    7 %%     ozc -x hash2.oz -o hash2.oz.exe
    8 %%     hash2.oz.exe 150
    9 
   10 functor
   11 import System Application
   12 
   13 define
   14 
   15 proc {InitHash H}
   16    for I in 0..9999 do
   17       {Dictionary.put H
   18          {String.toAtom {List.append "foo_" {IntToString I}}} I}
   19    end
   20 end
   21 
   22 proc {AddValues L H}
   23    local Key Value Tail V in
   24       if L \=  nil then
   25          (Key#Value|Tail) = L
   26          {Dictionary.condGet H Key nil V}
   27          if V == nil then
   28             {Dictionary.put H Key Value}
   29          else
   30             {Dictionary.put H Key Value+V}
   31          end
   32          {AddValues Tail H}
   33       end
   34    end
   35 end
   36 
   37 in
   38    local Args N H1 H2 in
   39       [Args] = {Application.getArgs plain}
   40       N = {String.toInt Args}
   41 
   42       {NewDictionary H1}
   43       {NewDictionary H2}
   44 
   45       {InitHash H1}
   46       for I in 1..N do
   47          {AddValues {Dictionary.entries H1} H2} end
   48 
   49       {System.printInfo {Dictionary.get H1 'foo_1'}}
   50       {System.printInfo ' '}
   51       {System.printInfo {Dictionary.get H1 'foo_9999'}}
   52       {System.printInfo ' '}
   53       {System.printInfo {Dictionary.get H2 'foo_1'}}
   54       {System.printInfo ' '}
   55       {System.printInfo {Dictionary.get H2 'foo_9999'}}
   56       {System.printInfo "\n"}
   57    end
   58    {Application.exit 0}
   59 end
