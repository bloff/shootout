    5 (module except
    6     (main main))
    7 
    8 (define HI 0)
    9 (define LO 0)
   10 
   11 (define (some_fun n)
   12   (try
   13    (hi_fun n)
   14    (lambda (escape proc msg obj)
   15      (print "should not get here " proc ":" msg " -- " obj)
   16      (escape #f))))
   17 
   18 (define (hi_fun n)
   19   (try
   20    (lo_fun n)
   21    (lambda (escape proc msg obj)
   22      (cond ((not (equal? obj 'Hi))
   23         (error proc msg obj)) ; rethrow
   24        (else
   25         (set! HI (+ HI 1))
   26         (escape #t))))))
   27 
   28 (define (lo_fun n)
   29   (try
   30    (blowup n)
   31    (lambda (escape proc msg obj)
   32      (cond ((not (equal? obj 'Lo))
   33         (error proc msg obj)) ; rethrow
   34        (else
   35         (set! LO (+ LO 1))
   36         (escape #t))))))
   37 
   38 (define (blowup n)
   39   (if (= 0 (modulo n 2))
   40       (error '() "Hi Exception" 'Hi)
   41       (error '() "Lo Exception" 'Lo)))
   42 
   43 (define (main args)
   44   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   45     (do ((i 0 (+ i 1)))
   46     ((= i n))
   47       (some_fun i)))
   48   (display* "Exceptions: HI=" HI " / LO=" LO)
   49   (newline))
