    6 #include <iostream>
    7 #include <cstdlib>
    8 #include <cstdio>
    9 
   10 using namespace std;
   11 
   12 size_t HI = 0;
   13 size_t LO = 0;
   14 
   15 class Hi_exception {
   16 public:
   17     explicit Hi_exception(size_t _n) : n(_n) {}
   18     const char* what() { sprintf(N, "%d", n); return N; }
   19 private:
   20     size_t n; char N[8];
   21 };
   22 
   23 class Lo_exception {
   24 public:
   25     explicit Lo_exception(size_t _n) : n(_n) {}
   26     const char* what() { sprintf(N, "%d", n); return N; }
   27 private:
   28     size_t n; char N[8];
   29 };
   30 
   31 void blowup(size_t num) {
   32     if (num % 2) {
   33         throw Lo_exception(num);
   34     }
   35     throw Hi_exception(num);
   36 }
   37 
   38 void lo_function(size_t num) {
   39     try {
   40         blowup(num);
   41     } catch(const Lo_exception& ex) {
   42         ++LO;
   43     }
   44 }
   45 
   46 void hi_function(size_t num) {
   47     try {
   48         lo_function(num);
   49     } catch(const Hi_exception& ex) {
   50         ++HI;
   51     }
   52 }
   53 
   54 void some_function(size_t num) {
   55     try {
   56         hi_function(num);
   57     } catch (...) {
   58         cerr << "We shouldn't get here\n"; exit(1);
   59     }
   60 }
   61 
   62 int
   63 main(int argc, char* argv[]) {
   64     size_t NUM = (argc == 2 ? (atoi(argv[1]) < 1 ? 1 : atoi(argv[1])): 1);
   65     while (NUM--) {
   66         some_function(NUM);
   67     }
   68     cout << "Exceptions: HI=" << HI << " / " << "LO=" << LO << endl;
   69 }
