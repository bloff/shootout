    4 import java.text.*;
    5 
    6 public class random {
    7 
    8     public static final long IM = 139968;
    9     public static final long IA = 3877;
   10     public static final long IC = 29573;
   11 
   12     public static void main(String args[]) {
   13     int N = Integer.parseInt(args[0]) - 1;
   14     NumberFormat nf = NumberFormat.getInstance();
   15     nf.setMaximumFractionDigits(9);
   16     nf.setMinimumFractionDigits(9);
   17     nf.setGroupingUsed(false);
   18 
   19     while (N-- > 0) {
   20         gen_random(100);
   21     }
   22     System.out.println(nf.format(gen_random(100)));
   23     }
   24 
   25     public static long last = 42;
   26     public static double gen_random(double max) {
   27     return( max * (last = (last * IA + IC) % IM) / IM );
   28     }
   29 }
