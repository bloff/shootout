    8 module ackermann
    9 
   10 import StdEnv, ArgEnv
   11 
   12 Ack :: !Int !Int -> Int
   13 Ack 0 n = n+1
   14 Ack m 0 = Ack (m-1) 1
   15 Ack m n = Ack (m-1) (Ack m (n-1))
   16 
   17 Start
   18     | argc == 1
   19         = "Ack(3," +++ toString n +++ "): " +++ toString (Ack 3 n) +++ "\n"
   20     | otherwise
   21         = "Ack(3,1): " +++ toString (Ack 3 1) +++ "\n"
   22     where
   23         argc = size argv - 1
   24         argv = getCommandLine
   25         n = toInt (argv.[1])
