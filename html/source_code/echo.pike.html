    7 #define DATA "Hello there sailor\n"
    8 
    9 void echo_server(Stdio.Port p, int n) {
   10     Stdio.File f = p->accept();
   11     int tbytes;
   12     string q;
   13     while( (q = f->read( 8192,1  )) && strlen( q ) ) {
   14     tbytes += strlen(q);
   15     f->write( q );
   16     }
   17     write( "server processed %d bytes\n", tbytes );
   18 }
   19 
   20 void echo_client(int p, int n) {
   21     int i;
   22     Stdio.File f = Stdio.File();
   23 
   24     f->connect( "localhost", p );
   25     int s = strlen(DATA);
   26     for (i=0; i<n; i++) {
   27     f->write( DATA );
   28     if(  f->read( s ) != DATA ) {
   29         werror( "Transfer error at repetition "+i+"\n");
   30         _exit( 1 );
   31     }
   32     }
   33     f->close();
   34     _exit( 0 );
   35 }
   36 
   37 /* Fork is not really available in a threaded pike. Thus this hack. It
   38  * assumes the pike binary can be found in your path, and that you have
   39  * a /usr/bin/env
   40  */
   41 void start_client( int p, int n )
   42 {
   43     Process.create_process( ({ "/usr/bin/env", "pike", __FILE__,
   44                    (string)p, (string)n }) );
   45 }
   46 
   47 void main(int argc, array argv)
   48 {
   49     if( argc < 3 )
   50     {
   51     int n = max((int)argv[-1],1);
   52     Stdio.Port p = Stdio.Port( 0 );
   53     int pno = (int)((p->query_address( )/" ")[1]);
   54     start_client( pno, n );
   55     echo_server( p, n );
   56     } else {
   57     echo_client( (int)argv[1], (int)argv[2] );
   58     }
   59     sleep(1);
   60 }
