    7 #include <iostream>
    8 #include <zopyra/regx>
    9 
   10 using namespace std;
   11 
   12 typedef pair<const char*, const char*> span;
   13 
   14 int main(int ac, char* av[]) {
   15     zopyra::regx re(
   16     "(?x)              # set extended flag for embedded comment fun\n"
   17         "(?:^|[^\\d(])    # must be preceded by non-digit\n"
   18         "([(])?           # match 1: possible initial left paren\n"
   19         "(\\d{3})         # match 2: area code is 3 digits\n"
   20         "(?(1)[)])        # if match1 then match right paren\n"
   21         "[ ]              # area code followed by one space\n"
   22         "(\\d{3})         # match 3: prefix of 3 digits\n"
   23         "[- ]             # separator is either space or dash\n"
   24         "(\\d{4})         # match 4: last 4 digits\n"
   25         "(?:\\D|\\b)      # followed by non-digit or break\n"
   26     );
   27 
   28     string line;
   29     vector<span> lines;
   30     while (getline(cin, line)) {
   31         char* phone = new char[line.size()];
   32         copy(line.begin(), line.end(), phone);
   33         lines.push_back(span(phone, phone + line.size()));
   34     }
   35 
   36     size_t ITER = (ac == 2 ? (atoi(av[1]) < 1 ? 1 : atoi(av[1])): 1);
   37 
   38     char num[13];
   39     num[0] = '(';
   40     num[4] = ')';
   41     num[5] = ' ';
   42     num[9] = '-';
   43     size_t count = 0;
   44     while (ITER--) {
   45         vector<span>::iterator end = lines.end();
   46         for (vector<span>::iterator i = lines.begin(); i != end; ++i) {
   47             zopyra::regx::iterator p = re.find(i->first, i->second);
   48             if (p++ != re.end()) {
   49                 char* num_p = &num[1];
   50                 ++p;
   51                 copy(p->first, p->second, num_p);
   52 
   53                 num_p = &num[6];
   54                 ++p;
   55                 copy(p->first, p->second, num_p);
   56 
   57                 num_p = &num[10];
   58                 ++p;
   59                 copy(p->first, p->second, num_p);
   60 
   61                 if (!ITER) {
   62                     cout << ++count << ": ";
   63                     copy(num, num + 14, ostream_iterator<char>(cout));
   64                     cout << '\n';
   65                 }
   66             }
   67         }
   68     }
   69 }
