    5 import System
    6 import Exception
    7 import IOExts
    8 
    9 blowup n | even n    = throw (ErrorCall "H")
   10      | otherwise = throw (ErrorCall "L")
   11 
   12 lo_function lo n =
   13   Exception.catch (blowup n)
   14     (\ex -> case ex of
   15         ErrorCall "L" -> do nlo <- readIORef lo
   16                     writeIORef lo (nlo + 1)
   17         _ -> throw ex
   18     )
   19 
   20 hi_function hi lo n =
   21   Exception.catch (lo_function lo n)
   22     (\ex -> case ex of
   23         ErrorCall "H" -> do nhi <- readIORef hi
   24                     writeIORef hi (nhi + 1)
   25         _ -> throw ex
   26     )
   27 
   28 some_function hi lo n = hi_function hi lo n
   29 
   30 main = do
   31   [arg] <- getArgs
   32   let n = read arg :: Int
   33   hi <- newIORef (0 :: Int)
   34   lo <- newIORef (0 :: Int)
   35   mapM (some_function hi lo) [n,n-1..1]
   36   nhi <- readIORef hi
   37   nlo <- readIORef lo
   38   putStrLn ("Exceptions: HI=" ++ show nhi ++ " / LO=" ++ show nlo)
