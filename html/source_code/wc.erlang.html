    5 -module(wc).
    6 -export([main/0, main/1]).
    7 
    8 % TBD - this program should not assume lines are less than 10000
    9 characters long
   10 
   11 main() -> main(['1']).
   12 main(Args) ->
   13     Port = open_port({fd, 0, 1}, [eof, {line, 10000}]),
   14     wc(Port, 0, 0, 0),
   15     halt(0).
   16 
   17 wc(Port, NL, NW, NC) ->
   18     receive
   19     {Port, {_, {_, Line}}} ->
   20         wc(Port, NL + 1, NW + cw(Line, out, 0), NC + length(Line) + 1);
   21     {Port, eof} ->
   22         io:format("~w ~w ~w~n", [NL, NW, NC])
   23     end.
   24 
   25 % count words in a line (list)
   26 cw([],        _, Count) -> Count;
   27 cw([$\ |T],   _, Count) -> cw(T, out, Count);
   28 cw([$\t|T],   _, Count) -> cw(T, out, Count);
   29 cw([_|T],   out, Count) -> cw(T, in, 1 + Count);
   30 cw([_|T],    in, Count) -> cw(T, in, Count).
