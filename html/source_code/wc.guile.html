    8 (define (wc iport)
    9   (let ((chars 0) (words 0) (lines 0) (inword 0) (x #f))
   10     (let loop ()
   11       (set! x (read-char iport))
   12       (if (eof-object? x)
   13       (begin (display lines) (display " ") (display words)
   14          (display " ") (display chars) (newline))
   15       (begin
   16         (set! chars (+ 1 chars))
   17         (if (not (char-whitespace? x))
   18         (set! inword 1)
   19         (begin
   20           (set! words (+ words inword))
   21           (set! inword 0)
   22           (if (char=? x #\newline)
   23               (set! lines (+ 1 lines)))))
   24         (loop))))))
   25 
   26 (wc (current-input-port))
