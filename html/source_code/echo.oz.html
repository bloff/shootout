    6 %%  Usage: start from command line with
    7 %%     ozc -x echo.oz -o echo.oz.exe
    8 %%     echo.oz.exe 100000
    9 
   10 functor
   11 import
   12    System
   13    Application
   14    Open
   15    OS
   16 
   17 define
   18    Data = "Hello there sailor\n"
   19 
   20 proc {ServerThread Sock SPort Bytes}
   21    Sock = {New Open.socket server(port:SPort)}
   22    {ServerLoop Sock 0 Bytes}
   23    {Sock shutDown(how: [receive send])}{Sock close}
   24 end
   25 
   26 proc {ServerLoop Sock Sum Bytes}
   27    local Message NewSum DR DW CR ST in
   28 
   29       %% low-level Selects seem ~6% faster total
   30       {Sock getDesc(DR DW)}{OS.readSelect DR}
   31       {OS.read DR 1024 Message nil CR}
   32       %% {Sock read(list: Message)} %% normal read
   33 
   34       if Message == nil then %% connection has been closed
   35          Bytes = Sum
   36       else
   37          NewSum = {Length Message} + Sum
   38 
   39          {OS.writeSelect DW}{OS.write DW Message ST}
   40          %% {Sock write(vs: Message)} %% normal write
   41 
   42          {ServerLoop Sock NewSum Bytes}
   43       end
   44    end
   45 end
   46 
   47 
   48 proc {ClientThread SPort N}
   49    local Sock in
   50       Sock = {New Open.socket client(port:SPort)}
   51       {ClientLoop Sock N}
   52       {Sock shutDown(how: [receive send])}{Sock close}
   53    end
   54 end
   55 
   56 proc {ClientLoop Sock N}
   57    local Message DR DW CR ST in
   58       if N > 0 then
   59 
   60          {Sock getDesc(DR DW)}
   61          {OS.writeSelect DW}{OS.write DW Data ST}
   62              {OS.readSelect DR}{OS.read DR 1024 Message nil CR}
   63 
   64          %% {Sock write(vs: Data)}     %% normal write
   65              %% {Sock read(list: Message)} %% normal read
   66 
   67          if Message == Data then {ClientLoop Sock N-1} end
   68       end
   69    end
   70 end
   71 
   72 in
   73    local Args A1 A2 A3 Socket SPort Bytes ArgList Pid in
   74       Args = {Application.getArgs plain}
   75 
   76       if {Length Args} == 1 then
   77          %% We are the server process
   78 
   79          A3|nil = Args
   80 
   81          thread {ServerThread Socket SPort Bytes} end
   82 
   83          %% Prepare to fork an OS process for the client
   84          %%    automatically close cmd.exe
   85          %%    start echo.oz.exe
   86          %%    pass a flag indicating the client process
   87          %%    pass the open server port SPort
   88          %%    pass the number of times the client should send the data
   89 
   90          ArgList = ["/C" "echo.oz" "client" {IntToString SPort} A3]
   91          Pid = {New Open.pipe init(cmd: "cmd.exe" args: ArgList)}
   92 
   93          %% Synchronize with server thread completion and indirectly with
   94          %% the client process. Wait here until the dataflow variable
   95          Bytes
   96          %% has been given a value in the server thread. That happens
   97          after
   98          %% the client process closes the socket connection, when the
   99          client
  100          %% process has sent all it's data and received all the replies.
  101 
  102          {System.showInfo 'server processed '#{IntToString Bytes}#' bytes'}
  103 
  104       elseif {Length Args} == 3 then
  105          %% We are the client process
  106 
  107          %% Take the flag, server port, times-to-repeat from the args
  108          %% and use the main thread for the client
  109          A1|A2|A3|nil = Args
  110          if A1 == "client" then
  111             {ClientThread {StringToInt A2} {StringToInt A3}}
  112          end
  113       end
  114    end
  115    {Application.exit 0}
  116 end
