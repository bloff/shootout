    7 main = do
    8        ~[number] <- getArgs
    9        let n = read number in
   10        putStrLn ("Ack(3," ++ (show n) ++ "): " ++ (show (ack 3 n)))
   11 
   12 ack :: Int -> Int -> Int
   13 ack 0 n = n+1
   14 ack m 0 = ack (m-1) 1
   15 ack m n = ack (m-1) (ack m (n-1));
