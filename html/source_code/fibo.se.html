    5 class FIBO
    6 
    7 creation make
    8 
    9 feature
   10 
   11    make is
   12 
   13       local
   14      n: INTEGER;
   15       do
   16      if argument_count = 1 then
   17         n := argument(1).to_integer
   18      else
   19         n := 1
   20      end
   21          std_output.put_integer(fib(n))
   22          std_output.put_new_line
   23       end;
   24 
   25 feature {NONE}
   26 
   27    fib(n:INTEGER):INTEGER is
   28       do
   29      if n < 2 then
   30         Result := 1
   31      else
   32         Result := fib(n-2) + fib(n-1)
   33      end
   34       end;
   35 end
