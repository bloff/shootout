    9 %% get the program argument, which is how many test iterations to run
   10 %% for this test, we ignore this parameter
   11 main() -> main(['1']).
   12 main(Args) ->
   13     spawn(?MODULE,init,[]),
   14     init:stop().
   15 
   16 init() ->
   17     Port = open_port({fd, 0, 1}, [eof, {line, 32}]),
   18     io:format("~w~n",[sumcol(Port)]).
   19 
   20 sumcol(Port) ->
   21     sumcol(Port, 0).
   22 sumcol(Port, Sum) ->
   23     receive
   24     {Port, {_, {_, Line}}} ->
   25         sumcol(Port, Sum + list_to_integer(Line));
   26     {Port, eof} ->
   27         Sum
   28     end.
