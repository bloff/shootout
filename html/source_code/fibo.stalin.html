    5 (define (string->exact-integer string)
    6  (let ((n (string->number string)))
    7   (if (integer? n)
    8       (if (exact? n)
    9       n
   10       (panic "STRING->EXACT-INTEGER yields non-EXACT-INTEGER"))
   11       (panic "STRING->EXACT-INTEGER yields non-EXACT-INTEGER"))))

   13 (define (main args)
   14   (let ((n (or (and (= (length args) 2) (string->exact-integer (cadr args))
   15    ) 1)))
   16     (display (fib n))
   17     (newline)))
   18 
   19 (define (fib n)
   20   (cond ((< n 2) 1)
   21     (else (+ (fib (- n 2)) (fib (- n 1))))))
