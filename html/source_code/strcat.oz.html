   10 functor
   11 import System Application

   13 define

   15 fun {RepeatAppend N S}
   16    if N == 0 then S else
   17       {RepeatAppend N-1 {Append "hello\n" S}} end
   18 end
   19 
   20 in
   21    local Args N in
   22       [Args] = {Application.getArgs plain}
   23       N = {String.toInt Args}
   24 
   25       {System.showInfo {Length {RepeatAppend N nil}} }
   26    end
   27    {Application.exit 0}
