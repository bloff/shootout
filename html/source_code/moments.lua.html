    5 local nums = {}
    6 local n = 0
    7 local sum = 0
    8 while 1 do
    9   local line = io.read()
   10   if line == nil then break end
   11   line = line+0        -- convert line to number
   12   sum = sum + line
   13   n = n + 1
   14   nums[n] = line
   15 end
   16 
   17 local mean = sum/n
   18 
   19 local average_deviation, variance, skew, kurtosis = 0, 0, 0, 0
   20 
   21 for i = 1, n do
   22   local deviation = nums[i] - mean
   23   average_deviation = average_deviation + math.abs(deviation)
   24   variance = variance + deviation^2
   25   skew = skew + deviation^3
   26   kurtosis = kurtosis + deviation^4
   27 end
   28 
   29 average_deviation = average_deviation/n
   30 variance = variance/(n-1)
   31 local standard_deviation = math.sqrt(variance)
   32 if variance ~= 0 then
   33   skew = skew / (n * variance * standard_deviation)
   34   kurtosis = kurtosis/(n * variance * variance) - 3.0
   35 end
   36 
   37 table.sort(nums)
   38 local mid = math.floor(n/2)
   39 local median
   40 if math.mod(n,2) == 1 then
   41   median = nums[mid+1]
   42 else
   43   median = (nums[mid] + nums[mid+1])/2
   44 end
   45 
   46 io.write(string.format("n:                  %d\n", n))
   47 io.write(string.format("median:             %f\n", median))
   48 io.write(string.format("mean:               %f\n", mean))
   49 io.write(string.format("average_deviation:  %f\n", average_deviation))
   50 io.write(string.format("standard_deviation: %f\n", standard_deviation))
   51 io.write(string.format("variance:           %f\n", variance))
   52 io.write(string.format("skew:               %f\n", skew))
   53 io.write(string.format("kurtosis:           %f\n", kurtosis))
