    6 %%  Usage: start from command line with
    7 %%     ozc -x matrix.oz -o matrix.oz.exe
    8 %%     matrix.oz.exe 300
    9 
   10 functor
   11 import System Application
   12 
   13 define
   14 
   15    proc {MakeMatrix Rows Cols M}
   16       local Count in
   17      {NewArray 0 Rows-1 0 M}
   18      Count ={NewCell 0}
   19      for I in 0..Rows-1 do
   20         local R in
   21            {NewArray 0 Cols-1 0 R}
   22            {Put M I R}
   23            for J in 0..Cols-1 do
   24           {Assign Count {Access Count}+1}
   25           {Put R J {Access Count}}
   26            end
   27         end
   28      end
   29       end
   30    end
   31 
   32 
   33    proc {MMult M1 M2 MM}
   34       local S1 N1 Prod S2 N2 in
   35      S1 = {Array.low M1}
   36      N1 = {Array.high M1}
   37      S2 = {Array.low {Get M1 S1}}
   38      N2 = {Array.high {Get M1 S1}}
   39      Prod = {NewCell 0}
   40      {NewArray S1 N1 0 MM}
   41      for I in S1..N1 do
   42         local R in
   43            {NewArray S1 N1 0 R}
   44            {Put MM I R}
   45            for J in S1..N1 do
   46           {Assign Prod 0}
   47           for K in S2..N2 do
   48              {Assign Prod {Get {Get M1 I} K}*
   49               {Get {Get M2 K} J}+{Access Prod}}
   50           end
   51           {Put R J {Access Prod}}
   52            end
   53         end
   54      end
   55       end
   56    end
   57 
   58    proc {RepeatMMult N M1 M2 MM}
   59       local T in
   60      if N > 1 then
   61         {MMult M1 M2 T}
   62         {RepeatMMult N-1 M1 M2 MM}
   63      else {MMult M1 M2 T} MM = T end
   64       end
   65    end
   66 
   67 in
   68    local Args Repeat N M1 M2 MM in
   69       [Args] = {Application.getArgs plain}
   70       Repeat = {String.toInt Args}
   71       N = 30
   72       {MakeMatrix N N M1}
   73       {MakeMatrix N N M2}
   74       {RepeatMMult Repeat M1 M2 MM}
   75 
   76       {System.printInfo {Get {Get MM 0} 0}}{System.printInfo ' '}
   77       {System.printInfo {Get {Get MM 2} 3}}{System.printInfo ' '} % get
   78        col 3 out of row 2
   79       {System.printInfo {Get {Get MM 3} 2}}{System.printInfo ' '} % get
   80        col 2 out of row 3
   81       {System.showInfo {Get {Get MM 4} 4} }
   82    end
   83    {Application.exit 0}
   84 end
