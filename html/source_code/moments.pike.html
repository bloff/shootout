    7 class Moments
    8 {
    9     int N;
   10     float median;
   11     float mean;
   12     float average_deviation;
   13     float standard_deviation;
   14     float variance;
   15     float skew;
   16     float kurtosis;
   17 
   18     void create(array(float) v)
   19     {
   20         float sum = `+(@v);
   21         N = sizeof(v);
   22         mean = sum / N;
   23 
   24         foreach(v, float i)
   25         {
   26             float deviation = i - mean;
   27             average_deviation += abs(deviation);
   28             variance += pow(deviation, 2);
   29             skew += pow(deviation, 3);
   30             kurtosis += pow(deviation, 4);
   31         }
   32 
   33         average_deviation /= N;
   34         variance /= (N - 1);
   35         standard_deviation = sqrt(variance);
   36 
   37         if (variance)
   38         {
   39             skew /= (N * variance * standard_deviation);
   40             kurtosis = kurtosis/(N * variance * variance) - 3.0;
   41         }
   42 
   43         sort(v);
   44         int mid = N/2;
   45         median = N % 2 ? v[mid] : (v[mid] + v[mid-1])/2;
   46     }
   47 };
   48 
   49 int main()
   50 {
   51     array input = Stdio.stdin.read()/"\n";
   52     Moments m=Moments( (array(float)) input[..sizeof(input)-2] );
   53 
   54     write("n:                  %d\n", m->N);
   55     write("median:             %.6f\n", m->median);
   56     write("mean:               %.6f\n", m->mean);
   57     write("average_deviation:  %.6f\n", m->average_deviation);
   58     write("standard_deviation: %.6f\n", m->standard_deviation);
   59     write("variance:           %.6f\n", m->variance);
   60     write("skew:               %.6f\n", m->skew);
   61     write("kurtosis:           %.6f\n", m->kurtosis);
   62 }
