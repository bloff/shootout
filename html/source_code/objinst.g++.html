    5 #include <stdlib.h>
    6 #include <iostream>
    7 
    8 using namespace std;
    9 
   10 class Toggle {
   11 public:
   12     Toggle(bool start_state) : state(start_state) { }
   13     virtual ~Toggle() {  }
   14     bool value() {
   15     return(state);
   16     }
   17     virtual Toggle& activate() {
   18     state = !state;
   19     return(*this);
   20     }
   21     bool state;
   22 };
   23 
   24 class NthToggle : public Toggle {
   25 public:
   26     NthToggle(bool start_state, int max_counter) :
   27     Toggle(start_state), count_max(max_counter), counter(0) {
   28     }
   29     Toggle& activate() {
   30     if (++this->counter >= this->count_max) {
   31         state = !state;
   32         counter = 0;
   33     }
   34     return(*this);
   35     }
   36 private:
   37     int count_max;
   38     int counter;
   39 };
   40 
   41 int
   42 main(int argc, char *argv[]) {
   43     int n = ((argc == 2) ? atoi(argv[1]) : 1);
   44 
   45     Toggle *toggle1 = new Toggle(true);
   46     for (int i=0; i<5; i++) {
   47     cout << ((toggle1->activate().value()) ? "true" : "false") << endl;
   48     }
   49     delete toggle1;
   50     for (int i=0; i<n; i++) {
   51     Toggle *toggle = new Toggle(true);
   52     delete toggle;
   53     }
   54 
   55     cout << endl;
   56 
   57     NthToggle *ntoggle1 = new NthToggle(true, 3);
   58     for (int i=0; i<8; i++) {
   59     cout << ((ntoggle1->activate().value()) ? "true" : "false") << endl;
   60     }
   61     delete ntoggle1;
   62     for (int i=0; i<n; i++) {
   63     NthToggle *ntoggle = new NthToggle(true, 3);
   64     delete ntoggle;
   65     }
   66     return 0;
   67 }
