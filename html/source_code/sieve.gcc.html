    6 #include <stdio.h>
    7 #include <stdlib.h>
    8 
    9 int
   10 main(int argc, char *argv[]) {
   11     int NUM = ((argc == 2) ? atoi(argv[1]) : 1);
   12     static char flags[8192 + 1];
   13     long i, k;
   14     int count = 0;
   15 
   16     while (NUM--) {
   17     count = 0;
   18     for (i=2; i <= 8192; i++) {
   19         flags[i] = 1;
   20     }
   21     for (i=2; i <= 8192; i++) {
   22         if (flags[i]) {
   23         // remove all multiples of prime: i
   24         for (k=i+i; k <= 8192; k+=i) {
   25             flags[k] = 0;
   26         }
   27         count++;
   28         }
   29     }
   30     }
   31     printf("Count: %d\n", count);
   32     return(0);
   33 }
