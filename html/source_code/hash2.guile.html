    5 ;;; $Id: hash2.guile.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $

    7 
    8 (use-modules (ice-9 format))
    9 
   10 (define (main args)
   11   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   12      (hash1 (make-hash-table 10000))
   13      (hash2 (make-hash-table 10000)))
   14     (do ((i 0 (+ i 1)))
   15     ((= i 10000))
   16       (hash-set! hash1 (string-append "foo_" (number->string i 10)) i))
   17     (do ((i 0 (+ i 1)))
   18     ((= i n))
   19       (hash-fold (lambda (key val init)
   20            (hash-set! hash2 key (+ (hash-ref hash2 key 0)
   21                        (hash-ref hash1 key))))
   22          '() hash1))
   23     (display (format "~D ~D ~D ~D\n"
   24              (hash-ref hash1 "foo_1")
   25              (hash-ref hash1 "foo_9999")
   26              (hash-ref hash2 "foo_1")
   27              (hash-ref hash2 "foo_9999")))))
