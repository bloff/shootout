    5 module Main where
    6 
    7 -- compile with:  ghc -O -o wc -package lang wc.hs
    8 
    9 import IO
   10 import IOExts
   11 import PackedString
   12 
   13 main = do
   14          -- set buffer to 4k
   15          hSetBuffering stdin (BlockBuffering (Just 4096))
   16          -- these are mutable variables
   17          nL <- newIORef 0
   18          nW <- newIORef 0
   19      nC <- newIORef 0
   20          (nl,nw,nc) <- countAll nL nW nC
   21      putStrLn ((show nl)++" "++(show nw)++" "++(show nc))
   22 
   23 countAll :: IORef Int -> IORef Int -> IORef Int -> IO (Int,Int,Int)
   24 countAll nL nW nC = do
   25          end <- hIsEOF stdin
   26          nl <- readIORef nL
   27         nw <- readIORef nW
   28          nc <- readIORef nC
   29          if (not end)
   30             then (do
   31               inStr <- hGetLine stdin
   32               -- using a packed string is a small speed win
   33               let str = packString inStr
   34               -- using IORefs makes it easy to force strict
   35               -- evaluation - how to easily do this without
   36               -- IORefs?
   37               writeIORef nL $! (nl + 1)
   38               writeIORef nW $! (nw + (length (wordsPS str)))
   39               writeIORef nC $! (nc + 1 + (lengthPS str))
   40               countAll nL nW nC)
   41             else  return (nl,nw,nc)
