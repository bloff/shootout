   10 (define (main args)
   11   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   12     (dict (make-hash-table 10000)))
   13     (with-input-from-file "Usr.Dict.Words"
   14       (lambda ()
   15     (let loop ((line (read-line)))
   16       (cond ((eof-object? line) #f)
   17         (else
   18          (hash-set! dict line #t)
   19          (loop (read-line)))))))
   20     (let loop ((word (read-line)))
   21       (cond ((eof-object? word) #f)
   22         (else
   23          (if (not (hash-ref dict word))
   24          (display (format "~A\n" word)))
   25          (loop (read-line)))))))
