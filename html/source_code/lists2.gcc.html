    6 #include <stdio.h>
    7 #include <stdlib.h>
    8 #include <string.h>
    9 #include <unistd.h>

   11 // a simple Double Linked List
   12 // the head node is special, it's val is length of list
   13 typedef struct DLL {
   14     int val;
   15     struct DLL *next;    /* points to next or head (if at tail) */
   16     struct DLL *prev;    /* points to prev or tail (if at head) */
   17 } DLL;
   18 
   19 inline int list_length(DLL *head) { return(head->val); }
   20 inline int list_empty(DLL *head) { return(list_length(head) == 0); }
   21 inline DLL *list_first(DLL *head) { return(head->next); }
   22 inline DLL *list_last(DLL *head) { return(head->prev); }
   23 
   24 void list_push_tail(DLL *head, DLL *item) {
   25     DLL *tail = head->prev;
   26     tail->next = item;
   27     item->next = head;
   28     head->prev = item;
   29     item->prev = tail;
   30     head->val++;
   31 }
   32 
   33 DLL *list_pop_tail(DLL *head) {
   34     DLL *prev, *tail;
   35     if (list_empty(head)) return(NULL);
   36     tail = head->prev;
   37     prev = tail->prev;
   38     prev->next = head;
   39     head->prev = prev;
   40     head->val--;
   41     return(tail);
   42 }
   43 
   44 void list_push_head(DLL *head, DLL *item) {
   45     DLL *next = head->next;
   46     head->next = item;
   47     next->prev = item;
   48     item->next = next;
   49     item->prev = head;
   50     head->val++;
   51 }
   52 
   53 DLL *list_pop_head(DLL *head) {
   54     DLL *next;
   55     if (list_empty(head)) return(NULL);
   56     next = head->next;
   57     head->next = next->next;
   58     next->next->prev = head;
   59     head->val--;
   60     return(next);
   61 }
   62 
   63 DLL *list_new_item(int v) {
   64     DLL *l = (DLL *)malloc(sizeof(DLL));
   65     l->next = l;
   66     l->prev = l;
   67     l->val = v;
   68     return(l);
   69 }
   70 
   71 inline DLL* list_create()  { return(list_new_item(0)); }
   72 
   73 void dump_list(char *msg, DLL *x) {
   74     DLL *xp = NULL, *first = list_first(x);
   75     int i = 0, len = list_length(x);
   76     fputs(msg, stdout);
   77     printf("length: %d\n", len);
   78     if (len > 0) {
   79     for (xp=x->next; xp->next != first; xp=xp->next) {
   80         printf("i:%3d  v:%3d  n:%3d  p:%3d\n", ++i,
   81            xp->val, xp->next->val, xp->prev->val);
   82     }
   83     printf("[val of next of tail is:  %d]\n\n", xp->next->val);
   84     }
   85     // printf("[last entry points to list head]\n");
   86 }
   87 
   88 void print_list(char *msg, DLL *x) {
   89     DLL *xp = NULL, *first = list_first(x);
   90     int i = 0;
   91     fputs(msg, stdout);
   92     if (list_length(x) > 0) {
   93     for (xp=x->next; xp->next != first; xp=xp->next) {
   94         printf("%d ", xp->val);
   95     }
   96     }
   97     printf("\n");
   98 }

  100 int main(int argc, char *argv[]) {
  101     int i, n = ((argc == 2) ? atoi(argv[1]) : 1);
  102     DLL *p, *A, *B;
  103 
  104     // create a list of integers (li1) from 1 to N
  105     A = list_create();
  106     for (i=n; i>0; i--) {
  107     list_push_head(A, list_new_item(i));
  108     }
  109     B = list_create();
  110 
  111     while (!list_empty(A)) {
  112     list_push_tail(B, list_pop_head(A));
  113     list_push_tail(B, list_pop_tail(A));
  114     }
  115     while (!list_empty(B)) {
  116     list_push_head(A, list_pop_tail(B));
  117     list_push_head(A, list_pop_head(B));
  118     }
  119     printf("%d\n", list_length(A));
  120     for (i=0, p=list_first(A); p != list_last(A) && i<10; i++, p=p->next) {
  121     printf("%d%s", p->val, (i<9) ? " " : "\n");
  122     }
  123     return 0;
  124 }
