   17 class MOMENTS
   18 
   19 creation make
   20 
   21 feature -- Creation
   22 
   23    make is
   24       local
   25      index: INTEGER
   26      number: INTEGER
   27      mid: INTEGER
   28      deviation: DOUBLE
   29      sorter: COLLECTION_SORTER[INTEGER]
   30       do
   31      !!values.make(0,499)
   32      read_values
   33      mean := sum / values.count
   34 
   35      from index := values.lower until index > values.upper loop
   36         number := values @ index
   37         deviation := number - mean
   38         average_deviation := average_deviation + deviation.abs
   39         variance := variance + deviation ^ 2
   40         skew := skew + deviation ^ 3
   41         kurtosis := kurtosis + deviation ^ 4
   42         index := index + 1
   43      end
   44 
   45      average_deviation := average_deviation / values.count
   46      variance := variance / (values.count - 1)
   47      standard_deviation := variance.sqrt --math.sqrt(variance)
   48      if variance > 0.0 then
   49         skew := skew / (values.count * variance * standard_deviation)
   50         kurtosis := kurtosis / (values.count * variance * variance) - 3.0
   51      end
   52      sorter.sort(values)
   53      mid := values.count // 2
   54      if (values.count \\ 2) = 0 then
   55         median := (values.item(mid) + values.item(mid - 1)) / 2
   56      else
   57         median := values @ mid
   58      end
   59 
   60      print ("n:                  " + values.count.to_string + "%N")
   61      print ("median:             " + median.to_string + "%N")
   62      print ("mean:               " + mean.to_string + "%N")
   63      print ("average_deviation:  " + average_deviation.to_string + "%N")
   64      print ("standard_deviation: " + standard_deviation.to_string + "%N")
   65      print ("variance:           " + variance.to_string + "%N")
   66      print ("skew:               " + skew.to_string + "%N")
   67      print ("kurtosis:           " + kurtosis.to_string + "%N")
   68       end -- make
   69 
   70 feature -- Queries
   71 
   72    sum: INTEGER
   73 
   74    mean,
   75    median,
   76    average_deviation,
   77    standard_deviation,
   78    variance,
   79    skew,
   80    kurtosis: DOUBLE
   81 
   82    values: ARRAY[INTEGER]
   83      -- Values read from stdin
   84 
   85    read_values is
   86       local
   87      value: INTEGER
   88      index: INTEGER
   89       do
   90 
   91      from
   92         index := values.lower
   93         std_input.read_line
   94      until std_input.end_of_input loop
   95         value := std_input.last_string.to_integer
   96         values.force(value, index)
   97         std_input.read_line
   98         sum := sum + value
   99         index := index + 1
  100      end
  101 
  102       end -- get_stdin
  103 
  104 end
