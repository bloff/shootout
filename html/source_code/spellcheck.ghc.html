    7 import Data.Set
    8 
    9 main = do
   10     d <- readFile "Usr.Dict.Words"
   11     let words = mkSet (lines d)
   12     let misspelled x = not $ x `elementOf` words
   13     interact $ unlines . filter misspelled . lines
