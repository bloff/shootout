1 #!/usr/bin/guile \
    2 -e main -s
    3 !#
    4 
    5 ;;; $Id: lists1.guile.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 
    8 (define (main args)
    9   (let* ((N (or (and (= (length args) 2) (string->number (cadr args))) 1))
   10      ;; create list (A) of integers from 1 through N
   11      (A (cdr (iota (+ N 1))))
   12      ;; create empty list B
   13      (B '())
   14      (mid (/ N 2))
   15      (sum 0))
   16 
   17     ;; move each individual item from A to B, in a loop, reversing order
   18     (while (not (null? A))
   19        (set! B (cons (car A) B))
   20        (set! A (cdr A)))
   21 
   22     ;; print first 2 elements of B
   23     (display (car B)) (display " ") (display (cadr B)) (newline)
   24 
   25     ;; reverse B (can be done in place)
   26     (set! B (reverse! B))
   27 
   28     ;; is 0 a member of B?
   29     (display (if (member 0 B) "true" "false")) (newline)
   30 
   31     ;; is N a member of B?
   32     (display (if (member N B) "true" "false")) (newline)
   33 
   34     ;; filter values from B to A that are less than N/2, preserving order
   35     (for-each (lambda (x) (if (< x mid) (set! A (cons x A)))) B)
   36     (set! A (reverse! A))
   37 
   38     ;; print first ten items of A
   39     (do ((i 0 (+ i 1))
   40      (x A (cdr x)))
   41     ((or (= i 10) (null? x)))
   42       (display (car x))
   43       (if (< i 9) (display " ") (newline)))
   44 
   45     ;; print sum of items in A that are less than 1000
   46     (for-each (lambda (x) (if (< x 1000) (set! sum (+ x sum)))) A)
   47     (display sum) (newline)
   48 
   49     ;; append B to end of A
   50     (set-cdr! (last-pair A) B)
   51 
   52     ;; print length of A
   53     (display (length A)) (display " ")
   54     (display (car (last-pair A))) (newline)))
