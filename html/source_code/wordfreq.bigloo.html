    6 (module hash
    7     (main main)
    8     (extern (macro printf::int (string int string) "printf")))
    9 
   10 
   11 (define (main args)
   12   (let* ((max (do ((max 1 (* max 2))) ((> max 75000) max)))
   13      (count (make-hash-table max
   14                   (lambda (o) (string->0..2^x-1 o 16))
   15                   (lambda (c) (car c))
   16                   string=?
   17                   max)))
   18 
   19     (read/rp (regular-grammar ((word (+ (in ("azAZ"))))
   20                    (not-word (+ (out ("azAZ")))))
   21                   ((: word)
   22                    (let* ((key (string-downcase! (the-string)))
   23                       (old-cell (get-hash key count)))
   24                                  (cond (old-cell
   25                     (set-cdr! old-cell (+ (cdr old-cell) 1)))
   26                                        (else
   27                           (put-hash! (cons key 1) count))))
   28                    (ignore))
   29                   ((: not-word)
   30                    (ignore))
   31                   (else
   32                    (the-failure)))
   33          (current-input-port))
   34 
   35     (let ((alist '()))
   36       (for-each-hash (lambda (cell)
   37                (let ((key (car cell))
   38                  (val (cdr cell)))
   39              (set! alist (cons (cons key val) alist))))
   40              count)
   41 
   42       (for-each (lambda (cell)
   43           (printf #"%7d\t%s\n" (cdr cell) (car cell)))
   44         (sort alist (lambda (x y)
   45                   (let ((xval (cdr x))
   46                     (yval (cdr y)))
   47                   (if (= xval yval)
   48                   (string<? (car y) (car x))
   49                   (< yval xval)))))))))
