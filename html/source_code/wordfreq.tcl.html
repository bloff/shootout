    6 proc main {} {
    7     while {1} {
    8     set data [read stdin 4096]
    9     if {[string equal $data {}]} {break}
   10     if {[gets stdin extra] >= 0} {
   11         append data $extra
   12     }
   13     regsub -all  {[^[:alpha:]]+} $data { } line
   14     foreach word [string tolower $line] {
   15         if {[catch {incr count($word)}]} {
   16         set count($word) 1
   17         }
   18     }
   19     }
   20     foreach {word cnt}  [array get count] {
   21     lappend lines [format "%7d\t%s" $cnt $word]
   22     }
   23     puts [join [lsort -decreasing $lines] "\n"]
   24 }
   25 
   26 main
