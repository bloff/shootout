    5 
    6 n = Integer(ARGV.shift || 1)
    7 
    8 size = 30
    9 
   10 def mkmatrix(rows, cols)
   11     count = 1
   12     mx = Array.new(rows)
   13     for i in 0 .. (rows - 1)
   14     row = Array.new(cols, 0)
   15     for j in 0 .. (cols - 1)
   16         row[j] = count
   17         count += 1
   18     end
   19     mx[i] = row
   20     end
   21     mx
   22 end
   23 
   24 def mmult(rows, cols, m1, m2)
   25     m3 = Array.new(rows)
   26     for i in 0 .. (rows - 1)
   27     row = Array.new(cols, 0)
   28     for j in 0 .. (cols - 1)
   29         val = 0
   30         for k in 0 .. (cols - 1)
   31         val += m1.at(i).at(k) * m2.at(k).at(j)
   32         end
   33         row[j] = val
   34     end
   35     m3[i] = row
   36     end
   37     m3
   38 end
   39 
   40 m1 = mkmatrix(size, size)
   41 m2 = mkmatrix(size, size)
   42 mm = Array.new
   43 n.times do
   44     mm = mmult(size, size, m1, m2)
   45 end
   46 puts "#{mm[0][0]} #{mm[2][3]} #{mm[3][2]} #{mm[4][4]}"
