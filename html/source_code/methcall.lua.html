    5 --------------------------------------------------------------
    6 -- Toggle class
    7 --------------------------------------------------------------
    8 
    9 Toggle = {}
   10 
   11 function Toggle:value ()
   12   return self.state
   13 end
   14 
   15 function Toggle:activate ()
   16   self.state = not self.state
   17   return self
   18 end
   19 
   20 function Toggle:new (start_state)
   21   local o = {state = start_state}
   22   self.__index =self
   23   setmetatable(o, self)
   24   return o
   25 end
   26 
   27 
   28 --------------------------------------------------------------
   29 -- NthToggle class
   30 --------------------------------------------------------------
   31 
   32 NthToggle = Toggle:new()
   33 
   34 function NthToggle:activate ()
   35   self.counter = self.counter + 1
   36   if self.counter >= self.count_max then
   37     self.state = not self.state
   38     self.counter = 0
   39   end
   40   return self
   41 end
   42 
   43 function NthToggle:new (start_state, max_counter)
   44   local o = Toggle.new(self, start_state)
   45   o.count_max = max_counter
   46   o.counter = 0
   47   return o
   48 end

   51 -----------------------------------------------------------
   52 -- main
   53 -----------------------------------------------------------
   54 
   55 function main ()
   56   local N = tonumber((arg and arg[1])) or 1
   57 
   58   local val = 1
   59   local toggle = Toggle:new(val)
   60   for i=1,N do
   61     val = toggle:activate():value()
   62   end
   63   print(val and "true" or "false")
   64 
   65   val = 1
   66   local ntoggle = NthToggle:new(val, 3)
   67   for i=1,N do
   68     val = ntoggle:activate():value()
   69   end
   70   print(val and "true" or "false")
   71 end
