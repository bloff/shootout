    4 module Main(main) where
    5 import System(getArgs)
    6 import Numeric(showFFloat)
    7 
    8 main = do
    9          ~[n] <- getArgs
   10          putStrLn (showFFloat (Just 9) (randloop (read n::Int) 42 0.0 100.
   11                    0) "")
   12      return 1
   13 
   14 randloop :: Int -> Int -> Double -> Double -> Double
   15 randloop 0 seed r max = r
   16 randloop n seed r max = randloop (n-1) newseed newrand max
   17     where normalize x max = (fromIntegral x) * (max / imd)
   18           newseed         = (seed * ia + ic) `mod` im
   19           newrand         = normalize newseed max
   20       im              = 139968
   21       imd             = fromIntegral im
   22       ia              = 3877
   23           ic              = 29573
