    6 structure Test : sig
    7     val main : (string * string list) -> OS.Process.status
    8 end = struct
    9 
   10 open HashTable;
   11 
   12 fun hashtest n =
   13     let
   14     val hx = mkTable (HashString.hashString, op =) (n, (Fail "not found"))
   15     fun doinserts i n =
   16         if i < n then (
   17         insert hx ((Int.fmt StringCvt.HEX i), i);
   18         doinserts (i+1) n
   19         ) else ()
   20     fun dolookups i c =
   21         if i > 0 then
   22         case find hx (Int.toString i) of
   23             SOME key => dolookups (i-1) (c+1)
   24           | _ => dolookups (i-1) c
   25         else c
   26     in (
   27     doinserts 0 n;
   28     dolookups n 0
   29     ) end;
   30 
   31 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   32 
   33 fun main(name, args) =
   34     let
   35     val arg = hd(args @ ["1"])
   36     val num = atoi arg
   37     val result = hashtest num
   38     in
   39     print (Int.toString result) ; print "\n" ;
   40     OS.Process.success
   41     end
   42 end
   43 
   44 val _ = SMLofNJ.exportFn("hash", Test.main);
