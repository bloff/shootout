    6 \ if x = 0         A(x,y) = y + 1
    7 \ if y = 0        A(x,y) = A(x-1,1)
    8 \ A(x,y) = A(x-1, A(x, y-1))
    9 
   10 \ read NUM from last command line argument
   11 0. argc @ 1- arg >number 2drop drop constant NUM
   12 
   13 : -ack ( y x -- n )
   14    dup 0= if
   15      drop 1+
   16    else
   17      over if
   18        swap 1- over recurse swap 1- recurse
   19      else
   20        1 -1 d+ recurse
   21      then
   22    then ;
   23 
   24 \ run ack(3, NUM) and print result from stack
   25 ." Ack(3," NUM 1 u.r ." ): " NUM 3 -ack 1 u.r cr
