1 module: wordfreq
    2 use-libraries: common-dylan, io, table-extensions, string-extensions
    3 use-modules: common-dylan, standard-io, streams, format-out, table-
    4 extensions, character-type
    5 
    6 begin
    7   let words = make(<case-insensitive-string-table>);
    8   while(~stream-at-end?(*standard-input*))
    9     do(method(x)
   10            unless(x = "")
   11          let rx = make(<string>, size: x.size);
   12          let j = 0;
   13          for (i from 0 below x.size)
   14             if (alphabetic?(x[i]) )
   15             rx[j] := x[i];
   16             j := j + 1;
   17         end if;
   18          end for;
   19              let count = element(words, rx, default: 0);
   20              words[rx] := count + 1;
   21            end unless;
   22        end method, split(read-line(*standard-input*), ' '));
   23   end while;
   24   let results = make(<stretchy-vector>);
   25   for(w keyed-by k in words)
   26     add!(results, pair(k, w));
   27   end for;
   28   do(method(x)
   29          format-out("%= %s\n", x.tail, as-lowercase(x.head))
   30      end method, sort(results, test: method(x, y)
   31                                          if(x.tail == y.tail)
   32                                            x.head > y.head
   33                                          else
   34                                            x.tail > y.tail
   35                                          end if;
   36                                      end method));
   37 end;
