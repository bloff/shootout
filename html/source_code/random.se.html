   17 class RANDOM
   18 
   19 creation make
   20 
   21 feature -- Creation
   22 
   23    make is
   24       local
   25      random: RANDOMNUMBER
   26      n: INTEGER
   27      answer: DOUBLE
   28      index: INTEGER
   29       do
   30      if argument_count = 0 then
   31         n := 1
   32      else
   33         n := argument(1).to_integer
   34      end
   35      from
   36         index := n
   37         !!random.make
   38      until index = 0 loop
   39         answer := random.next(100.0)
   40         index := index - 1
   41      end
   42      io.put_string(answer.to_string_format(9))
   43      io.put_new_line
   44       end
   45 
   46 end
