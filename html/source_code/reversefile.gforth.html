    8 variable size    2000000                  size !
    9 variable sbuf    size @ allocate throw    sbuf !
   10 
   11 10   constant  nl_ch
   12 4096 constant  MAXREAD
   13 
   14 : add_terminal_newline ( addr -- addr )
   15     dup c@ nl_ch <>
   16     if
   17     dup nl_ch swap c!
   18     1 +
   19     endif ;
   20 
   21 : reversefile ( -- )
   22      nl_ch sbuf @ c!
   23     sbuf @ 1 +
   24     dup dup
   25     begin
   26         MAXREAD stdin read-file throw dup
   27     while
   28     \ add number of bytes read to current buffer position
   29     + dup
   30     \ now stack has start-of-buffer end-of-buffer addresses
   31     repeat
   32     drop
   33     \ stack: start-of-buffer end-of-buffer
   34 
   35     \ if input didn't end in a newline, then add one
   36     add_terminal_newline
   37 
   38     \ adjust end pointer
   39     2 -
   40 
   41     \ adjust start pointer
   42     swap 2 - swap
   43 
   44     \ now scan the buffer backwards, printing out the lines
   45     tuck
   46     -do
   47     \ stack: pointer to end of buffer
   48     i c@ nl_ch =
   49     if
   50         dup i 1 + swap i -
   51         stdout write-file throw
   52         drop
   53         i
   54     endif
   55     1 -loop
   56     ;
   57 
   58 reversefile
