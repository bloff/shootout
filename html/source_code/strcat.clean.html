   17 Start =  toString (sbLength (appendLoop argi sbNew)) +++ "\n" 
   18  
   19 appendLoop n sb 
   20    | n>0   = appendLoop (n-1) (sbAppend sb "hello\n") 
   21    = sb

   25 // To do this in the "same way" we need to define a StringBuffer 
   26   
   27 :: StringBuffer = Sb !Int !.String 
   28  
   29 sbNew = Sb 0 (createArray 32 ' ') 
   30  
   31 sbAppend (Sb n b) s 
   32    = if (minsize > size b) 
   33       (Sb minsize (copyTo s n (resize b minsize) )) 
   34       (Sb minsize (copyTo s n b)) 
   35    where  
   36       minsize = n + size s 
   37        
   38       copyTo s n a = {a & [n+i] = s.[i] \\ i <- [0..size s - 1]} 
   39           
   40       resize b min 
   41          = if (newsize < min)  
   42             (copyTo b 0 (createArray min ' ')) 
   43             (copyTo b 0 (createArray newsize ' '))    
   44          where newsize = size b * 2 + 5    
   45        
   46 sbLength (Sb n b) = n
