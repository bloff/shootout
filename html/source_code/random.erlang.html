    5 -module(random).
    6 -export([main/1]).
    7 
    8 main() -> main(['1']).
    9 main([Arg]) ->
   10     N = list_to_integer(atom_to_list(Arg)),
   11     io:fwrite("~.9f\n", [rand(N, 42, 0.0, 100.0)]),
   12     halt(0).
   13 
   14 
   15 -define(IM, 139968).
   16 -define(IA, 3877).
   17 -define(IC, 29573).
   18 
   19 
   20 rand(0, _, Rand, _) -> Rand;
   21 rand(N, Seed, Rand, Max) ->
   22     NewSeed = (Seed * ?IA + ?IC) rem ?IM,
   23     NewRand = Max * NewSeed / ?IM,
   24     rand(N-1, NewSeed, NewRand, Max).
