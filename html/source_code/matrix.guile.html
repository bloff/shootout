    2 -e main -s
    3 !#
    4 
    5 ;;; $Id: matrix.guile.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 ;;; with help from Brad Knotwell
    8 
    9 (define size 30)
   10 
   11 (define (mkmatrix rows cols)
   12   (define count 1)
   13   (define (set-row cols)
   14     (let ((row (make-vector cols 0)))
   15       (do ((i 0 (1+ i)))
   16       ((= i cols) row)
   17     (begin (vector-set! row i count) (set! count (1+ count))))))
   18   (let ((mx (make-vector rows cols)))
   19     (begin (array-map-in-order! mx set-row mx) mx)))
   20 
   21 (define (mmult rows cols m1 m2)
   22   (let ((m3 (make-vector rows 0)))
   23     (do ((i 0 (+ i 1)))
   24     ((= i rows))
   25       (let ((m1i (vector-ref m1 i))
   26         (row (make-vector cols 0)))
   27     (do ((j 0 (+ j 1)))
   28         ((= j cols))
   29       (let ((val 0))
   30         (do ((k 0 (+ k 1)))
   31         ((= k cols))
   32           (set! val (+ val (* (vector-ref m1i k)
   33                   (vector-ref (vector-ref m2 k) j)))))
   34         (vector-set! row j val)))
   35     (vector-set! m3 i row)))
   36     m3))
   37 
   38 (define (main args)
   39   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   40     (m1 (mkmatrix size size))
   41     (m2 (mkmatrix size size))
   42     (mm 0))
   43     (do ((i 0 (1+ i)))
   44     ((= i n) (begin
   45            (display (vector-ref (vector-ref mm 0) 0)) (display " ")
   46            (display (vector-ref (vector-ref mm 2) 3)) (display " ")
   47            (display (vector-ref (vector-ref mm 3) 2)) (display " ")
   48            (display (vector-ref (vector-ref mm 4) 4)) (newline)))
   49     (set! mm (mmult size size m1 m2)))))
