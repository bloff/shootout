1 #!/usr/bin/ruby
    2 # -*- mode: ruby -*-
    3 # $Id: hash.ruby.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
    4 # http://www.bagley.org/~doug/shootout/
    5 # with help from Aristarkh A Zagorodnikov
    6 
    7 n = (ARGV.shift || 1).to_i
    8 
    9 hash = {}
   10 for i in 1..n
   11     hash['%x' % i] = 1
   12 end
   13 
   14 c = 0
   15 n.downto 1 do |i|
   16     c += 1 if hash.has_key? i.to_s
   17 end
   18 
   19 puts c
