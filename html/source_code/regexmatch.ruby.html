1 #!/usr/bin/ruby
    2 # -*- mode: ruby -*-
    3 # $Id: regexmatch.ruby.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
    4 # http://www.bagley.org/~doug/shootout/
    5 
    6 re = Regexp.new(
    7     '(?:^|[^\d\(])' +            # must be preceeded by non-digit
    8     '(?:\((\d\d\d)\)|(\d\d\d))' +    # match 1 or 2: area code is 3 digits
    9     '[ ]' +                # area code followed by one space
   10     '(\d\d\d)' +            # match 3: prefix of 3 digits
   11     '[ -]' +                # separator is either space or dash
   12     '(\d\d\d\d)' +            # match 4: last 4 digits
   13     '\D'                # must be followed by a non-digit
   14 )
   15 
   16 NUM = Integer(ARGV[0] || 1)
   17 
   18 phones = STDIN.readlines
   19 
   20 count = m = line = iter = 0
   21 for iter in 1..NUM
   22     for line in phones
   23     if m = re.match(line)
   24         num = '(' + (m[1] || m[2]) + ') ' + m[3] + '-' + m[4];
   25         if iter == NUM
   26         count += 1
   27         puts "#{count}: #{num}"
   28         end
   29     end
   30     end
   31 end
