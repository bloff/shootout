   10 -module(moments).  
   11 -export([main/1, init/0]).  
   12 -import(lists, [sort/1, foldl/3, nth/2, sum/1, map/2, tokens/2]). 
   13  
   14  
   15 accumulate([], Mean, Ad, Av, As, Ak) -> 
   16     {Ad, Av, As, Ak}; 
   17 accumulate([H|T], Mean, Ad, Av, As, Ak) -> 
   18     D = H - Mean, 
   19     D2 = D * D, 
   20     accumulate(T, Mean, Ad + abs(D), Av + D2, As + (D2 * D), Ak + (D2 * D2)
   21                ). 
   22  
   23  
   24 median(L, N) -> medianS(sort(L), N). 
   25  
   26  
   27 medianS(L, N) -> 
   28     Mid = N div 2, 
   29     case N rem 2 of 
   30         0 -> (nth(Mid, L) + nth(Mid + 1, L)) / 2; 
   31         1 -> nth(Mid, L) 
   32     end. 
   33  
   34  
   35 skew(N, As, V, SD) when V > 0.0 -> As / (N * V * SD); 
   36 skew(N, As, V, SD) -> 0. 
   37  
   38  
   39 kurtosis(N, Ak, V) when V > 0.0 -> Ak / (N * V * V) - 3; 
   40 kurtosis(N, Ak, V) -> 0. 
   41  
   42  
   43 stats(L) ->  
   44     N = length(L), 
   45     io:fwrite("~s~w~n", ["n:                  ", N]), 
   46     io:fwrite("~s~f~n", ["median:             ", median(L, N)]), 
   47  
   48     Mean = sum(L) / N, 
   49     io:fwrite("~s~f~n", ["mean:               ", Mean]), 
   50  
   51     {Ad, Av, As, Ak} = accumulate(L, Mean, 0.0, 0.0, 0.0, 0.0), 
   52     io:fwrite("~s~f~n", ["average_deviation:  ", Ad / N]), 
   53  
   54     Variance = Av / (N - 1), 
   55     Standard_deviation = math:sqrt(Variance), 
   56     io:fwrite("~s~f~n", ["standard_deviation: ", Standard_deviation]), 
   57     io:fwrite("~s~f~n", ["variance:           ", Variance]), 
   58     io:fwrite("~s~f~n", ["skew:               ", skew(N, As, Variance,
   59               Standard_deviation)]), 
   60     io:fwrite("~s~f~n", ["kurtosis:           ", kurtosis(N, Ak, Variance)]
   61               ). 
   62  
   63 main(A) -> 
   64     spawn(?MODULE,init,[]). 
   65  
   66  
   67 init() -> 
   68     Port = open_port({fd, 0, 1},[]), 
   69     receive 
   70     {Port, {data, Chars}} -> Chars 
   71     end, 
   72     L = string:tokens(Chars, "\n"), 
   73     Ints = map(fun(X) -> list_to_integer(X) end, L), 
   74     stats(Ints),  
   75     halt().
