    5 structure Test : sig
    6     val main : (string * string list) -> OS.Process.status
    7 end = struct
    8 
    9 open TextIO;
   10 
   11 fun chop str = String.substring (str, 0, (String.size str) - 1);
   12 
   13 fun spellcheck () =
   14     let
   15     val dict = HashTable.mkTable (HashString.hashString, op =) (40000, (Fail
   16             "not found"))
   17     in
   18     let val din = openIn "Usr.Dict.Words"
   19         fun init_dict din dict =
   20             case inputLine din of
   21             NONE => ()
   22           | SOME(instr) => ( HashTable.insert dict (chop instr, 1);
   23                                      init_dict din dict )
   24     in
   25         init_dict din dict
   26     end;
   27 
   28     let fun check_words dict =
   29         case inputLine stdIn of
   30             NONE => ()
   31           | SOME(instr) =>
   32               let val word = chop instr
   33           in
   34             case HashTable.find dict word of
   35             SOME _ => ()
   36               | NONE => (print word ; print "\n");
   37             check_words dict
   38           end
   39     in
   40         check_words dict
   41     end handle EndOfFile => ()
   42     end
   43 
   44 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   45 
   46 fun main(name, args) =
   47     let
   48     val arg = hd(args @ ["1"])
   49     val num = atoi arg
   50     in
   51     spellcheck ();
   52     OS.Process.success
   53     end;
   54 end
   55 
   56 val _ = SMLofNJ.exportFn("spellcheck", Test.main);
