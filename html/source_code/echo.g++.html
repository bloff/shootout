   18 #define DATA "Hello there sailor\n"

   20 void myabort (char *m) { fprintf(stderr, "%s\n", m); exit(1); }
   21 void sysabort (char *m) { perror(m); exit(1); }
   22 
   23 int sigchld = 0;
   24 void reaper (int sig) { sigchld = 1; }
   25 
   26 int
   27 server_sock () {
   28     int ss, optval = 1;
   29     struct sockaddr_in sin;
   30     if ((ss = socket(PF_INET, SOCK_STREAM, 0)) == -1)
   31     sysabort("server/socket");
   32     if (setsockopt(ss, SOL_SOCKET, SO_REUSEADDR, &optval, sizeof(optval))
   33         == -1)
   34     sysabort("server/setsockopt");
   35     memset(&sin,0,sizeof(sin));
   36     sin.sin_family = AF_INET;
   37     sin.sin_addr.s_addr = htonl(INADDR_LOOPBACK);
   38     sin.sin_port = 0;
   39     if (bind(ss, (sockaddr *)&sin, sizeof(sin)) == -1)
   40     sysabort("server/bind");
   41     listen(ss, 2);
   42     return(ss);
   43 }

   46 int
   47 get_port (int sock) {
   48     struct sockaddr_in sin;
   49     socklen_t slen = sizeof(sin);
   50     if (getsockname(sock, (sockaddr *)&sin, &slen) == -1)
   51     sysabort("server/getsockname");
   52     return(sin.sin_port);
   53 }

   55 
   56 int
   57 client_sock (int port) {
   58     struct sockaddr_in sin;
   59     int sock;
   60     if ((sock = socket(PF_INET, SOCK_STREAM, 0)) == -1)
   61     sysabort("client/socket");
   62     sin.sin_family = AF_INET;
   63     sin.sin_addr.s_addr = htonl(INADDR_LOOPBACK);
   64     sin.sin_port = port;
   65     if (connect(sock, (sockaddr *)&sin, sizeof(sin)) == -1)
   66     sysabort("client/connect");
   67     return(sock);
   68 }

   71 void
   72 echo_client (int n, int port) {
   73     int i, sock, olen, len, nwritten, nread;
   74     char *offset, obuf[64], ibuf[64];
   75     char *end = ibuf + sizeof(ibuf);
   76 
   77     sock = client_sock(port);
   78     strcpy(obuf, DATA);
   79     olen = strlen(obuf);
   80     for (i=0; i<n; i++) {
   81     len = olen;
   82     offset = obuf;
   83     while (len > 0) {
   84         if ((nwritten = write(sock, offset, len)) == -1)
   85         sysabort("client/write");
   86         offset += nwritten;
   87         len -= nwritten;
   88     }
   89     offset = ibuf;
   90     while ((nread = read(sock, offset, (end - offset))) > 0) {
   91         offset += nread;
   92         if (*(offset-1) == '\n') break;
   93     }
   94     if (nread == -1)
   95         sysabort("client/read");
   96     *offset = 0;
   97     if ((strcmp(obuf, ibuf)) != 0) {
   98         char mbuf[128];
   99         sprintf(mbuf, "client: \"%s\"  ne \"%s\"", obuf, ibuf);
  100         myabort(mbuf);
  101     }
  102     }
  103     close(sock);
  104 }

  107 void
  108 echo_server (int n) {
  109     int ssock, csock, len, nwritten, total_bytes;
  110     pid_t pid;
  111     char buf[64], *offset;
  112     struct sockaddr_in sin;
  113     socklen_t slen = sizeof(sin);
  114     int status;
  115 
  116     ssock = server_sock();
  117     signal(SIGCHLD, reaper);
  118     if ((pid = fork()) == -1)
  119     sysabort("server/fork");
  120     if (pid) {
  121     /* parent is server */
  122     if ((csock = accept(ssock, (sockaddr *)&sin, &slen)) == -1)
  123         sysabort("server/accept");
  124     total_bytes = 0;
  125     while ((len = read(csock, buf, sizeof(buf))) > 0) {
  126         if (sigchld) myabort("server/sigchld");
  127         offset = buf;
  128         total_bytes += len;
  129         while (len > 0) {
  130         if ((nwritten = write(csock, offset, len)) == -1)
  131             sysabort("server/write");
  132         offset += nwritten;
  133         len -= nwritten;
  134         }
  135     }
  136     if (len == -1)
  137         sysabort("server/read");
  138     close(csock);
  139     fprintf(stdout, "server processed %d bytes\n", total_bytes);
  140     } else {
  141     /* child is client */
  142     echo_client(n, get_port(ssock));
  143     }
  144     wait(&status);
  145 }

  149 main(int argc, char *argv[]) {
  150     int n = ((argc == 2) ? atoi(argv[1]) : 1);
  151     echo_server(n);
