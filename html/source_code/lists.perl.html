1 #!/usr/bin/perl
    2 # $Id: lists.perl.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    3 use strict;
    4 
    5 my $SIZE = 10000;
    6 
    7 my $ITER = $ARGV[0];
    8 $ITER = 1 if ($ITER < 1);
    9 
   10 my $result = 0;
   11 while ($ITER--) {
   12     $result = &test_lists();
   13 }
   14 print "$result\n";
   15 
   16 sub test_lists {
   17     # create a list of integers (Li1) from 1 to SIZE
   18     my @Li1 = (1..$SIZE);
   19     # copy the list to Li2 (not by individual items)
   20     my @Li2 = @Li1;
   21     my @Li3 = ();
   22     # remove each individual item from left side of Li2 and
   23     # append to right side of Li3 (preserving order)
   24     push(@Li3, shift @Li2) while (@Li2);
   25     # Li2 must now be empty
   26     # remove each individual item from right side of Li3 and
   27     # append to right side of Li2 (reversing list)
   28     push(@Li2, pop @Li3) while (@Li3);
   29     # Li3 must now be empty
   30     # reverse Li1 in place
   31     @Li1 = reverse @Li1;
   32     # check that first item is now SIZE
   33     return(0) if $Li1[0] != $SIZE;
   34     # compare Li1 and Li2 for equality
   35     my $len1 = scalar(@Li1);
   36     my $len2 = scalar(@Li2);
   37     my $lists_equal = ($len1 == $len2);
   38     return(0) if not $lists_equal;
   39     for my $i (0..($len1-1)) {
   40     if ($Li1[$i] != $Li2[$i]) {
   41         $lists_equal = 0;
   42         last;
   43     }
   44     }
   45     return(0) if not $lists_equal;
   46     # return the length of the list
   47     return($len1);
   48 }
