    5 module Main where
    6 
    7 import System(getArgs)
    8 
    9 main = getArgs >>= putStrLn . ("Count: "++) . show . mytest . read .
   10        headOr1
   11   where headOr1 x = if length x /= 1 then "1" else head x
   12 
   13 -- here we try to force it to recompute at each step.  Note
   14 -- that we are not naming `sieve [2..8192]' and we are forcing
   15 -- a comparison with -1.  Of course there is still no guarantee
   16 -- that any particular Haskell implementation will actually
   17 -- recompute the value.
   18 mytest :: Int -> Int
   19 mytest 1 = length (sieve [2..8192])
   20 mytest n | length (sieve [2..8192]) == -1 = error "doh"
   21          | otherwise                      = mytest (n-1)
   22 
   23 -- we use Int rather than let Haskell default to Integer,
   24 -- because we are trying to remain competetive with other
   25 -- languages that do not do arbitrary precision math by
   26 -- default...
   27 sieve :: [Int] -> [Int]
   28 sieve [] = []
   29 sieve (h:t) = h : sieve [x| x<-t, x`mod`h /= 0]
