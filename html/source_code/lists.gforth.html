    6 0. argc @ 1- arg >number 2drop drop constant NUM
    7 
    8 10000 constant SIZE
    9 
   10 struct
   11     cell% field list-next
   12     cell% field list-val
   13 end-struct list%
   14 
   15 : make-list ( -- list )
   16     0 0 SIZE -do ( tail )
   17     list% %alloc
   18     i over list-val !
   19     tuck list-next !
   20     1 -loop
   21 ;
   22 
   23 : copy-list ( list1 -- list2 )
   24     0 { w^ list2 }
   25     list2 begin ( list1' list2p' )
   26     over
   27     while
   28     list% %alloc dup >r swap ! ( list1' R: list2' )
   29     dup list-val @ r@ list-val !
   30     list-next @ r> list-next ( list1'' list2p'' )
   31     repeat
   32     off drop list2 @ ;
   33 
   34 : move-head-to-tail ( list1 -- list2 )
   35     \ somehow this is an expensive noop
   36     0 { w^ list2 }
   37     list2 begin ( list1' list2p' )
   38     over
   39     while
   40     \ move one element
   41     over list-next dup @ 2>r ( list1' list2p' R: list2p'' list1'' )
   42     over list-next off
   43     ! r> r>
   44     repeat
   45     off drop list2 @ ;
   46 
   47 : nreverse ( list1 -- list2 )
   48     \ destructive reverse
   49     0 swap begin ( list2' list1' )
   50     dup
   51     while
   52     dup list-next @ >r ( list2' list1' R: list1'' )
   53     tuck list-next ! r> ( list1' list1'' )
   54     repeat
   55     drop ;
   56 
   57 : move-tail-to-tail ( list1 -- list2 )
   58     \ use head-to-tail instead of head-to-head nreverse
   59     nreverse move-head-to-tail ;
   60 
   61 : list-equals ( list1 list2 -- f )
   62     begin ( l1' l2' )
   63     dup
   64     while
   65     over
   66     while
   67     over list-val @ over list-val @ <> if
   68         2drop false exit
   69     endif
   70     list-next @ swap list-next @ ( symmetric )
   71     repeat then
   72     = ;
   73 
   74 : list-length ( list -- n )
   75     0 begin ( l' n' )
   76     over
   77     while
   78     1+ swap list-next @ swap
   79     repeat
   80     nip ;
   81 
   82 s" wrong result" exception constant wrong-result
   83 
   84 : main ( -- n )
   85     0 NUM 0 ?do
   86     drop
   87     make-list dup copy-list ( list1 list2 )
   88     move-head-to-tail move-tail-to-tail swap nreverse ( list2 list1 )
   89     dup list-val @ SIZE <> wrong-result and throw
   90     tuck list-equals 0= wrong-result and throw
   91     list-length
   92     loop ;
   93 
   94 main 0 .r cr bye
