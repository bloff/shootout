    7 let nl = ref 0 and nw = ref 0 and nc = ref 0 and max = 4096
    8 let buf = String.create max
    9 
   10 let readblock scanfun =
   11   let nread = input stdin buf 0 max in
   12   if nread = 0 then ()
   13   else begin nc := !nc + nread; scanfun 0 nread end
   14 
   15 let rec scan_out_of_word i n =
   16   if i < n then
   17     match buf.[i] with
   18     | '\n' -> incr nl; scan_out_of_word (i + 1) n
   19     | ' ' | '\t' -> scan_out_of_word (i + 1) n
   20     | _ -> incr nw; scan_in_word (i+1) n
   21   else readblock scan_out_of_word
   22 
   23 and scan_in_word i n =
   24   if i < n then
   25     match buf.[i] with
   26     | '\n' -> incr nl; scan_out_of_word (i + 1) n
   27     | ' '|'\t' -> scan_out_of_word (i + 1) n
   28     | _ -> scan_in_word (i + 1) n
   29   else readblock scan_in_word;;
   30 
   31 scan_out_of_word 0 0;
   32 Printf.printf "%d %d %d\n" !nl !nw !nc
