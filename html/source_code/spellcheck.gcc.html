    7 #include <ctype.h>
    8 #include <stdio.h>
    9 #include <stdlib.h>
   10 #include <unistd.h>
   11 #include "simple_hash.h"
   12 
   13 #define MAXLINELEN 128
   14 
   15 struct ht_ht *dict = NULL;
   16 
   17 int handleInput(FILE *input,void (*hashManipFn)(char *))
   18 {
   19     int wordbufsize = 80,i = 0;
   20     char *cp, *wordbuf = (char *)malloc(wordbufsize + 1);
   21     char line[MAXLINELEN];
   22 
   23     if((wordbuf = malloc(wordbufsize+1)) == NULL)
   24         return(fprintf(stderr,"malloc\n"),0);
   25 
   26     while (fgets(line, MAXLINELEN, input))
   27     for (cp=line; *cp > 0; cp++) {
   28         if (isspace(*cp)) {
   29         if (i) {
   30             wordbuf[i] = '\0';
   31                     hashManipFn(wordbuf);
   32             i = 0;
   33         }
   34         } else {
   35         wordbuf[i++] = *cp;
   36         if (i == wordbufsize) {
   37             wordbufsize *= 2;
   38             if((wordbuf = realloc(wordbuf, wordbufsize + 1)) == NULL)
   39                         return(fprintf(stderr, "realloc\n"), 0);
   40         }
   41         }
   42         }
   43 
   44     free(wordbuf);
   45     return(1);
   46 }
   47 
   48 void spellCheck(char *key) {
   49     if (ht_find_new(dict,key)->val != 1) printf("%s\n",key);
   50 }
   51 
   52 void hashLoad(char *key) { ht_find_new(dict,key)->val = 1; }
   53 
   54 int main(int argc, char *argv[]) {
   55     FILE *fh;
   56     int rc;
   57 
   58     /*
   59         ht_create doesn't handle malloc and calloc failures
   60         so this is superfluous
   61     */
   62     if((dict = ht_create(40000)) == NULL)
   63         return(fprintf(stderr,"hash creation failed\n"),EXIT_FAILURE);
   64 
   65     if ((fh = fopen("Usr.Dict.Words", "r")) == NULL)
   66         return(fprintf(stderr,"couldn't open dictionary\n"),EXIT_FAILURE);
   67 
   68     rc = ((handleInput(fh,hashLoad) && handleInput(stdin,spellCheck)) ?
   69          EXIT_SUCCESS : EXIT_FAILURE);
   70 
   71     ht_destroy(dict);
   72     return(rc);
   73 }
