    6 proc main {} {
    7     set lines [split [read stdin] "\n"]
    8 
    9     fconfigure stdout -buffering full
   10 
   11     for {set i [expr {[llength $lines]-2}]} {$i >= 0} {incr i -1} {
   12         puts [lindex $lines $i]
   13     }
   14 }
   15 
   16 main
