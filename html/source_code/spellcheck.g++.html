   13 using namespace std;

   15 #define SGI_HASH_NAMESPACE __gnu_cxx
   16 
   17 namespace sgi {
   18     using SGI_HASH_NAMESPACE::hash;
   19     using SGI_HASH_NAMESPACE::hash_map;
   20 }
   21 
   22 typedef std::pair<const char*, const char*> span;
   23 
   24 namespace SGI_HASH_NAMESPACE {
   25     template<> struct hash<span> {
   26         inline size_t operator()(const span& s) const {
   27             size_t h = 0;
   28             const char* end = s.second;
   29             for (const char* begin = s.first; begin != end; ++begin) {
   30                 h = 5 * h + *begin;
   31             }
   32             return h;
   33         }
   34     };
   35 }
   36 
   37 namespace std {
   38     template<> struct std::equal_to<span> {
   39         inline bool operator()(const span& s1, const span& s2) const {
   40             return (s1.second - s1.first) == (s2.second - s2.first) &&
   41                 std::equal(s1.first, s1.second, s2.first);
   42         }
   43     };
   44 }
   45 class spell_checker {
   46 public:
   47     spell_checker() {
   48         std::ifstream in("Usr.Dict.Words");
   49         char line[32];
   50         while (in.getline(line, 32)) {
   51             const char* begin = line;
   52             const char* end = line + in.gcount() - 1;
   53             if (dict.end() == dict.find(span(begin, end))) {
   54                 const size_t len = end - begin;
   55                 char* word = new char[len];
   56                 copy(begin, end, word);
   57                 ++dict[span(word, word + len)];
   58             }
   59          }
   60     }
   61 
   62     void process(std::istream& in) {
   63         char line[32];
   64         while (in.getline(line, 32)) {
   65             if (dict.end() == dict.find(span(line, line + in.gcount() - 1))
   66                 ) {
   67                 cout << line << '\n';
   68             }
   69         }
   70     }
   71 
   72 private:
   73     __gnu_cxx::hash_map<span, int> dict;
   74 };
   75 
   76 int main() {
   77     spell_checker spell;
   78     char buff[4096];
   79     cin.rdbuf()->pubsetbuf(buff, 4096); // enable buffering
   80     spell.process(cin);
   81 }
