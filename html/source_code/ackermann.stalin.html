    5 (define (string->exact-integer string)
    6  (let ((n (string->number string)))
    7   (if (integer? n)
    8       (if (exact? n)
    9       n
   10       (panic "STRING->EXACT-INTEGER yields non-EXACT-INTEGER"))
   11       (panic "STRING->EXACT-INTEGER yields non-EXACT-INTEGER"))))

   13 (define (main args)
   14   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   15     (print-list "Ack(3," n "): " (ack 3 n))))
   16 
   17 (define (ack m n)
   18   (cond ((zero? m) (+ n 1))
   19     ((zero? n) (ack (- m 1) 1))
   20     (else      (ack (- m 1) (ack m (- n 1))))))
   21 
   22 (define (print-list . items) (for-each display items) (newline))
