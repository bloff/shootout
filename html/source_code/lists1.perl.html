    4 use strict;
    5 
    6 # member(item, list)
    7 sub member {
    8     foreach (@{$_[1]}) {
    9     return(1) if $_ == $_[0];
   10     }
   11     return(0);
   12 }
   13 
   14 # shift/unshift operate on "head" of list in Perl
   15 
   16 sub main {
   17     my $N = ($ARGV[0] < 1) ? 1 : int($ARGV[0]);
   18 
   19     # create list (A) of integers from 1 through N
   20     my @A = (1..$N);
   21 
   22     # create empty list B
   23     my @B = ();
   24 
   25     # move each individual item from A to B, in a loop, reversing order
   26     unshift(@B, shift(@A)) while (@A);
   27 
   28     # print first 2 elements of B
   29     print "$B[0] $B[1]\n";
   30 
   31     # reverse B (can be done in place)
   32     @B = reverse(@B);
   33 
   34     # is 0 a member of B?
   35     print member(0, \@B) ? "true\n" : "false\n";
   36 
   37     # is N a member of B?
   38     print member($N, \@B) ? "true\n" : "false\n";
   39 
   40     # filter values from B to A that are less than N/2, preserving order
   41     my $mid = int($N/2);
   42     my @A = grep($_ < $mid, @B);
   43 
   44     # print first ten items of A
   45     print "@A[0..9]\n";
   46 
   47     # print sum of items in A that are less than 1000
   48     my $sum = 0;
   49     foreach (@A) { $sum += $_ if $_ < 1000; }
   50     print "$sum\n";
   51 
   52     # append B to end of A
   53     push(@A, @B);
   54 
   55     # print length and last element of A
   56     print scalar(@A), " $A[-1]\n";
   57 }
   58 
   59 main();
