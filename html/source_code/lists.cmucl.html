    6 (defparameter *SIZE* 10000)
    7 
    8 (declaim (fixnum *SIZE*) (inline xcons push-queue))
    9 
   10 (defvar *free-conses*)
   11 
   12 (defun xcons (A B)
   13   (let ((x *free-conses*))
   14     (if x (progn (setf *free-conses* (cdr x) (car x) A (cdr x) B) x) (cons
   15      A B))))
   16 
   17 (defmacro xpop (X)
   18   `(prog1 (car ,x) (psetf ,x (cdr ,x) (cdr ,x) *free-conses* *free-conses*
   19     ,x)))
   20 
   21 (defun push-queue (item queue &aux (new (xcons item nil)))
   22   (if (cdr queue) (setf (cddr queue) new) (setf (car queue) new))
   23   (setf (cdr queue) new)
   24   (car queue))
   25 
   26 (defmacro with-collector ((name) &body body)
   27   (let ((collector (gensym)))
   28     `(let ((,collector (xcons nil nil)))
   29        (flet ((,name (value) (push-queue value ,collector)))
   30          ,@body
   31          (car ,collector)))))
   32 
   33 (defun test-list ()
   34   (let* ((L1 (with-collector (conc) (loop for x fixnum from 1 to *SIZE* do
   35    (conc x))))
   36          (L2 (with-collector (conc) (loop for x in L1 do (conc x))))
   37          (L3 nil))
   38     ;; Move items from left of L2 to right of L3 (preserve order)
   39     (setf L3 (with-collector (conc) (loop while L2 do (conc (xpop L2)))))
   40     ;; Move from tail of L3 to tail of L2 (reversing list)
   41     ;; start by reversing L3 so we can pop from the front
   42     (setf L3 (nreverse L3))
   43     (setf L2 (with-collector (conc) (loop while L3 do (conc (xpop L3)))))
   44     ;; Reverse L1 in place
   45     (setf L1 (nreverse L1))
   46     ;; Check that (first L1) == *SIZE*
   47     (assert (= (the fixnum (first L1)) *SIZE*))
   48     ;; Compare L1 and L2 for equality
   49     (assert (equal L1 L2))
   50     ;; Return the length -- and return the conses to the free list
   51     (prog1 (length (the list L1))
   52       (setf *free-conses* (nconc *free-conses* L3 L2 L1)))))
   53 
   54 (defun main ()
   55   (let ((n (parse-integer (or (car (last ext:*command-line-strings*)) "1"))
   56    )
   57         (num 0) (*free-conses* nil))
   58     (loop repeat n fixnum do (setf num (test-list)))
   59   (format t "~D~%" num)))
