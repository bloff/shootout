    2 # $Id: lists1.tcl.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 # based on code from Kristoffer Lawson and Tom Wilkason
    5 
    6 proc K {a b} {set a}
    7 
    8 proc ldelete {listName index} {
    9     upvar $listName list
   10     ;# Replace a deletion with null, much faster
   11     set list [lreplace [K $list [set list {}]] $index $index]
   12 }
   13 
   14 proc lreverse {_list} {
   15     upvar $_list List
   16     for {set i [expr {[llength $List] - 1}]} {$i >= 0} {incr i -1} {
   17     lappend Li1r [lindex $List $i]
   18     }
   19     set List $Li1r
   20     unset Li1r
   21 }
   22 
   23 proc member {target _list} {
   24     upvar $_list List
   25     foreach {item} $List {
   26     if {$item == $target} {
   27         puts "true"
   28         return
   29     }
   30     }
   31     puts "false"
   32 }
   33 
   34 proc main {args} {
   35     global argv
   36     set N [lindex $argv 0]
   37     if {$N < 1} {
   38     set N 1
   39     }
   40 
   41     # create list (A) of integers from 1 through N
   42     for {set i 1} {$i <= $N} {incr i} {lappend A $i}
   43 
   44     # move each individual item from A to B, in a loop, reversing order
   45     foreach {item} $A {
   46     lappend B [lindex $A end]
   47     ldelete A end
   48     }
   49 
   50     # print first 2 elements of B
   51     puts "[lindex $B 0] [lindex $B 1]"
   52 
   53     # reverse B (can be done in place)
   54     lreverse B
   55 
   56     # is 0 a member of B?
   57     member 0 B
   58 
   59     # is N a member of B?
   60     member $N B
   61 
   62     # filter values from B to A that are less than N/2, preserving order
   63     set mid [expr {$N / 2}]
   64     foreach {item} $B {
   65     if {$item < $mid} {
   66         lappend A $item
   67     }
   68     }
   69 
   70     # print first ten items of A
   71     puts [lrange $A 0 9]
   72 
   73     # print sum of items in A that are less than 1000
   74     set sum 0
   75     foreach {item} $A {
   76     if {$item < 1000} {
   77         incr sum $item
   78     }
   79     }
   80     puts $sum
   81 
   82     # append B to end of A
   83     foreach {item} $B {
   84     lappend A $item
   85     }
   86 
   87     # print length and last element of A
   88     puts "[llength $A] [lindex $A end]"
   89 }
   90 
   91 main
