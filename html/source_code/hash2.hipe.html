    5 %% Use ETS tables (Erlang's associative store).
    6 
    7 -module(hash2).
    8 -export([main/0, main/1]).
    9 
   10 main() -> main(['1']).
   11 main([Arg]) ->
   12     N = list_to_integer(atom_to_list(Arg)),
   13     H1 = ets:new(one, [set]),
   14     H2 = ets:new(two, [set]),
   15     doinserts1(0, H1),
   16     doinserts2(N, H1, H2),
   17     io:format("~w ~w ~w ~w~n", [value(H1, list_to_atom("foo_1")),
   18                 value(H1, list_to_atom("foo_9999")),
   19                 value(H2, list_to_atom("foo_1")),
   20                 value(H2, list_to_atom("foo_9999"))]),
   21     halt(0).
   22 
   23 doinserts1(10000, _) -> ok;
   24 doinserts1(I, H) ->
   25     Key = list_to_atom(lists:append("foo_", integer_to_list(I))),
   26     ets:insert(H, { Key, I }),
   27     doinserts1(I+1, H).
   28 
   29 doinserts2(0, _, _) -> ok;
   30 doinserts2(I, H1, H2) ->
   31     addTables(H1, H2),
   32     doinserts2(I-1, H1, H2).
   33 
   34 addTables(H1, H2) ->
   35     Key = ets:first(H1),
   36     addTables(Key, H1, H2).
   37 
   38 value(Tab, Key) -> { _, V } = hd(ets:lookup(Tab, Key)), V.
   39 
   40 addTables('$end_of_table', _, _) -> ok;
   41 addTables(Key, H1, H2) ->
   42     Val1 = value(H1, Key),
   43     case (catch ets:update_counter(H2, Key, Val1)) of
   44     {'EXIT', {badarg, _}} -> ets:insert(H2, {Key, Val1});
   45     _                     -> true
   46     end,
   47     addTables(ets:next(H1, Key), H1, H2).
