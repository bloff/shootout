   17 import StdEnv, StdOverloadedList, LanguageShootout 
   18  
   19 Start world   
   20    # (console, world) = stdio world 
   21    # (a,suma) = sumNumbers console 0.0 [] 
   22    = resultstring suma a  
   23  
   24    where 
   25    sumNumbers f suma a 
   26       # (ok,r,f) = freadr f 
   27       | not ok = (a, suma) 
   28       = sumNumbers f (suma+r) [r:a] 
   29   
   30    resultstring suma a =  
   31       "n:                  " +++ (toString n) +++ "\n" +++  
   32       "median:             " +++ (toStringWith 6 median) +++ "\n" +++  
   33       "mean:               " +++ (toStringWith 6 mean) +++ "\n" +++  
   34       "average_deviation:  " +++ (toStringWith 6 adev) +++ "\n" +++  
   35       "standard_deviation: " +++ (toStringWith 6 sdev) +++ "\n" +++  
   36       "variance:           " +++ (toStringWith 6 var) +++ "\n" +++  
   37       "skew:               " +++ (toStringWith 6 skew) +++ "\n" +++  
   38       "kurtosis:           " +++ (toStringWith 6 kurt) +++ "\n"      
   39        
   40       where 
   41       n = toReal (Length a) 
   42       mean = suma / n           
   43       dev = [x - mean \\ x <- a] 
   44       adev = (Sum [abs x \\ x <- dev]) / n       
   45       var = (Sum [x*x \\ x <- dev]) / (n-1.0) 
   46       sdev = sqrt var 
   47       skew = (Sum [x*x*x \\ x <- dev]) / (n*var*sdev) 
   48       kurt = (Sum [x*x*x*x \\ x <- dev]) / (n*var*var)-3.0 
   49  
   50       median = middle (sort a) 
   51          where 
   52          middle b 
   53             | isOdd (entier(n)) = b!!mid 
   54             = (b!!(mid-1) + b!!mid) / 2.0 
   55          mid = entier(n)/2
