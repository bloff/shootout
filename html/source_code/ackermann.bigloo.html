    5 (module ackermann
    6     (main main)
    7     (option (set! *genericity* #f)))
    8 
    9 (define (main args)
   10   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   11     (print-list "Ack(3," n "): " (ack 3 n))))
   12 
   13 (define (ack m n)
   14   (cond ((zero? m) (+ n 1))
   15     ((zero? n) (ack (- m 1) 1))
   16     (else      (ack (- m 1) (ack m (- n 1))))))
   17 
   18 (define (print-list . items) (for-each display items) (newline))
