    6 import System(getArgs, getProgName)

    8 lengthNHellos n = length (concat (replicate n "hello\n"))
    9 
   10 main = do
   11        arg <- getArgs
   12        case arg of
   13          [number] -> putStrLn $ show $ lengthNHellos (read number)
   14          _        -> do name <- getProgName; fail ("Usage: " ++ name ++
   15                                                    "number")
