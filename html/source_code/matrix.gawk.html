    4 function mkmatrix(mx, rows, cols) {
    5     count = 1;
    6     for (i=0; i<rows; i++) {
    7     for (j=0; j<cols; j++) {
    8         mx[i,j] = count++;
    9     }
   10     }
   11 }
   12 
   13 function mmult(rows, cols, m1, m2, m3) {
   14     for (i=0; i<rows; i++) {
   15     for (j=0; j<cols; j++) {
   16         val = 0;
   17         for (k=0; k<cols; k++) {
   18         val += m1[i,k] * m2[k,j];
   19         }
   20         m3[i,j] = val;
   21     }
   22     }
   23 }
