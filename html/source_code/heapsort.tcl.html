    2 # $Id: heapsort.tcl.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $

    4 # sped up by Miguel Sofer's function generator
    5 
    6 set IM 139968
    7 set IA   3877
    8 set IC  29573
    9 
   10 set last 42
   11 
   12 proc make_gen_random {} {
   13     global IM IA IC
   14     set params [list IM $IM IA $IA IC $IC]
   15     set body [string map $params {
   16         global last
   17         expr {($max * [set last [expr {($last * IA + IC) % IM}]]) / IM}
   18     }]
   19     proc gen_random {max} $body
   20 }
   21 
   22 proc heapsort {n ra_name} {
   23     upvar $ra_name ra
   24 
   25     set j 0
   26     set i 0
   27     set rra 0.0
   28     set l [expr {($n >> 1) + 1}]
   29     set ir $n
   30     while 1 {
   31         if {$l > 1} {
   32         incr l -1
   33             set rra $ra($l)
   34         } else {
   35         set rra $ra($ir)
   36         set ra($ir) $ra(1)
   37         incr ir -1
   38         if {$ir == 1} {
   39                 set ra(1) $rra
   40                 return
   41             }
   42         }
   43     set i $l
   44     set j [expr {$l << 1}]
   45         while {$j <= $ir} {
   46         if {($j < $ir) && ($ra($j) < $ra([expr {$j + 1}]))} {
   47         incr j
   48         }
   49             if {$rra < $ra($j)} {
   50         set ra($i) $ra($j)
   51         set i $j
   52                 set j [expr {$j + $i}]
   53             } else {
   54         set j [expr {$ir + 1}]
   55             }
   56         }
   57         set ra($i) $rra
   58     }
   59 }
   60 
   61 proc main {} {
   62     global argv
   63     set n [lindex $argv 0]
   64     make_gen_random
   65 
   66     for {set i 1} {$i <= $n} {incr i} {
   67     set ary($i) [gen_random 1.0]
   68     }
   69     heapsort $n ary
   70     puts [format "%.10f" $ary($n)]
   71 }
   72 
   73 main
