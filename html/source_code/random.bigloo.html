    5 (module random
    6     (main main)
    7     (extern (macro printf::int (string double) "printf")))
    8 
    9 (define IM   139968)
   10 (define IA     3877)
   11 (define IC    29573)
   12 
   13 (define LAST 42)
   14 (define (gen_random max)
   15   (set! LAST (modulo (+ (* LAST IA) IC) IM))
   16   (/ (* max LAST) IM))
   17 
   18 (define (main args)
   19   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   20     (let loop ((iter n))
   21       (cond ((> iter 1)
   22          (gen_random 100.0)
   23          (loop (- iter 1)))))
   24     (printf #"%.9f\n" (gen_random 100.0)))))
