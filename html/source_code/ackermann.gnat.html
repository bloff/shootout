    5 with Ada.Command_Line, Text_IO, Ada.Strings.Fixed, Interfaces;
    6 
    7 procedure Ackermann is
    8    function L_Trim (Source : String; Side : Ada.Strings.Trim_End :=
    9                Ada.Strings.Left) return String renames Ada.Strings.Fixed.
   10                Trim;
   11 
   12    subtype Nat is Interfaces.Integer_32 range 0 .. Interfaces.
   13    Integer_32'Last;
   14    subtype Pos is Nat range 1 .. Nat'Last;         --  Largest is
   15    2_147_483_647
   16    use type Nat;
   17 
   18    function Ack (M, N : Nat) return Pos is
   19       pragma Suppress (Overflow_Check);
   20       pragma Suppress (Range_Check);
   21    begin
   22       if M = 0 then
   23          return N + 1;
   24       elsif N = 0 then
   25          return Ack (M - 1, 1);
   26       else
   27          return Ack (M - 1, Ack (M, N - 1));
   28       end if;
   29    end Ack;
   30 
   31    N        : Pos := 1;
   32 begin
   33    if Ada.Command_Line.Argument_Count > 0 then
   34       N := Pos'Value (Ada.Command_Line.Argument (1));
   35    end if;
   36    Text_IO.Put_Line ("Ack(3," & L_Trim (Pos'Image (N)) & "):" &
   37             Pos'Image (Ack (3, N)));
   38 end Ackermann;
