    6 (defparameter *hi* 0)
    7 (defparameter *lo* 0)
    8 
    9 (defun some-fun (n)
   10   (catch t
   11     (hi-fun n)))
   12 
   13 (defun hi-fun (n)
   14   (catch 'Hi_Exception
   15     (lo-fun n)))
   16 
   17 (defun lo-fun (n)
   18   (catch 'Lo_Exception
   19     (blow-up n)))
   20 
   21 (defun blow-up (n)
   22   (if (evenp n)
   23       (throw 'Hi_Exception (setf *hi* (1+ *hi*)))
   24     (throw 'Lo_Exception (setf *lo* (1+ *lo*)))))
   25 
   26 (defun main ()
   27   (let ((n (parse-integer (or (car (last extensions:*command-line-strings*)
   28    ) "1"))))
   29   (setf *hi* 0
   30     *lo* 0)
   31   (do ((i 0 (1+ i)))
   32       ((= i n))
   33     (some-fun i)))
   34   (format t "Exceptions: HI=~A / LO=~A~%" *hi* *lo*))
