    8 -module(echo).
    9 -export([main/0, main/1, client/2, server/1]).
   10 
   11 -define(DATA, <<"Hello there sailor\n">>).
   12 
   13 main() -> main(['1']).
   14 main([Arg]) ->
   15     N = list_to_integer(atom_to_list(Arg)),
   16     ServerSock = create_server_sock(),
   17     spawn(?MODULE, client, [N, socket_port(ServerSock)]),
   18     server(ServerSock),
   19     init:stop().
   20 
   21 create_server_sock() ->
   22     {ok, LSock} = gen_tcp:listen(0, [binary]),
   23     LSock.
   24 
   25 socket_port(Sock) ->
   26     {ok, Port} = inet:port(Sock),
   27     Port.
   28 
   29 client(N, ServerPort) ->
   30     {ok, Sock} = gen_tcp:connect("localhost", ServerPort, [binary]),
   31     client_loop(N, Sock),
   32     gen_tcp:close(Sock).
   33 
   34 client_loop(0, Sock) -> ok;
   35 client_loop(N, Sock) ->
   36     ok = gen_tcp:send(Sock, ?DATA),
   37     receive
   38     {tcp, Sock, _} -> client_loop(N-1, Sock);
   39     {tcp_closed, Sock} -> ok
   40     end.
   41 
   42 server(LSock) ->
   43     {ok, Sock} = gen_tcp:accept(LSock),
   44     server_loop(Sock, 0),
   45     gen_tcp:close(LSock).
   46 
   47 server_loop(Sock, Bytes) ->
   48     receive
   49     {tcp, Sock, Packet} ->
   50         ok = gen_tcp:send(Sock, Packet),
   51         server_loop(Sock, Bytes + size(Packet));
   52     {tcp_closed, Sock} ->
   53         io:format("server processed ~w bytes~n", [Bytes]),
   54         gen_tcp:close(Sock)
   55     end.
