    5 # this program is modified from:
    6 #   http://cm.bell-labs.com/cm/cs/who/bwk/interps/pap.html
    7 # Timing Trials, or, the Trials of Timing: Experiments with Scripting
    8 # and User-Interface Languages</a> by Brian W. Kernighan and
    9 # Christopher J. Van Wyk.
   10 
   11 proc main {} {
   12     global argv
   13     set n [lindex $argv 0]
   14     set last [expr {$n - 1}]
   15     for {set i 0} {$i < $n} {incr i} {
   16     set x($i) [expr {$i + 1}]
   17     set y($i) 0
   18     }
   19     for {set k 0} {$k < 1000} {incr k} {
   20     for {set j $last} {$j >= 0} {incr j -1} {
   21         set y($j) [expr {$x($j) + $y($j)}]
   22     }
   23     }
   24     puts "$y(0) $y($last)"
   25 }
   26 
   27 main
