    7 #include <iostream>
    8 #include <algorithm>
    9 #include <vector>
   10 #include <cstdlib>
   11 
   12 using namespace std;
   13 
   14 int main(int argc, char *argv[]) {
   15     size_t NUM = (argc == 2 ? (atoi(argv[1]) < 1 ? 1 : atoi(argv[1])): 1);
   16 
   17     vector<char> primes(8192 + 1);
   18     vector<char>::iterator pbegin = primes.begin();
   19     vector<char>::iterator begin = pbegin + 2;
   20     vector<char>::iterator end = primes.end();
   21 
   22     while (NUM--) {
   23         fill(begin, end, 1);
   24         for (vector<char>::iterator i = begin; i < end; ++i) {
   25             if (*i) {
   26                 const size_t p = i - pbegin;
   27                 for (vector<char>::iterator k = i + p; k < end; k += p) {
   28                     *k = 0;
   29                 }
   30             }
   31         }
   32     }
   33 
   34     cout << "Count: " << count(begin, end, 1) << endl;
   35 }
