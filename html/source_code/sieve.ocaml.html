    8 let init_flags = String.make 8193 't'
    9 let flags = String.make 8193 't'
   10 
   11 let rec inner_loop f i k =
   12   if k < 8193 then begin f.[k] <- 'f'; inner_loop f i (k + i) end
   13 
   14 let rec middle_loop f i cnt =
   15   if i < 8193 then
   16     if f.[i] = 't' then begin
   17       inner_loop f i (i + i);
   18       middle_loop f (i + 1) (cnt + 1) end
   19     else middle_loop f (i + 1) cnt
   20   else cnt
   21 
   22 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
   23         1;;
   24 for iter = 2 to n do
   25   String.blit init_flags 0 flags 0 8193;
   26   ignore (middle_loop flags 2 0);
   27 done;
   28 Printf.printf "Count: %d\n" (middle_loop flags 2 0)
