    7 function main(num)
    8     for num=num,1,-1 do
    9         local flags = {}
   10         count = 0
   11         for i=2,8192 do
   12             if not flags[i] then
   13                 for k=i+i, 8192, i do
   14                     flags[k] = 1
   15                 end
   16                 count = count + 1
   17             end
   18         end
   19     end
   20 end
