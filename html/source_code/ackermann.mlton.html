    6 fun ack(0,n) = n+1
    7   | ack(m,0) = ack(m-1,1)
    8   | ack(m,n) = ack(m-1,ack(m,(n-1)));
    9 
   10 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   11 fun printl [] = print "\n" | printl(h::t) = ( print h ; printl t );
   12 
   13 fun main(name, args) =
   14   let
   15     val arg = hd(args @ ["1"]);
   16     val num = atoi arg;
   17     val ack = ack(3, num);
   18     val result = Int.toString ack;
   19   in (
   20       printl ["Ack(3,", arg, "): ", result];
   21       OS.Process.success
   22   ) end;
