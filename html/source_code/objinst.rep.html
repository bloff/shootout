1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: objinst.rep.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system rep.data.objects)
   14 
   15   (define (print-bool b)
   16     (format standard-output (if b "true\n" "false\n")))
   17 
   18   (define toggle
   19     (object nil
   20         ((new obj state)
   21          (put obj 'state state)
   22          obj)
   23         ((value tog) (get tog 'state))
   24         ((activate tog)
   25          (put tog 'state (not (get tog 'state)))
   26          tog)))
   27 
   28   (define nth-toggle
   29     (object toggle
   30         ((new obj state max)
   31          (toggle 'new obj state)
   32          (put obj 'count-max max)
   33          (put obj 'counter 0)
   34          obj)
   35         ((activate obj)
   36          (let ((c (get obj 'counter)))
   37            (if (= (+ c 1) (get obj 'count-max))
   38            (progn
   39              (put obj 'state (not (get obj 'state)))
   40              (put obj 'counter 0))
   41          (put obj 'counter (+ c 1)))
   42            obj))))
   43 
   44   (let* ((arg (get-command-line-option "--num" t))
   45      (n (if arg (read-from-string arg) 1)))
   46     (let ((tog (toggle 'new 'my-tog 't)))
   47       (do ((i 0 (+ i 1)))
   48       ((= i 5))
   49     (print-bool (toggle 'value (toggle 'activate tog))))
   50       (do ((i 0 (+ i 1)))
   51       ((= i n))
   52     (setq tog (toggle 'new 'my-tog 't))))
   53 
   54     (format standard-output "\n")
   55 
   56     (let ((ntog (nth-toggle 'new 'my-ntog 't 3)))
   57       (do ((i 0 (+ i 1)))
   58       ((= i 8))
   59     (print-bool (nth-toggle 'value (nth-toggle 'activate ntog))))
   60       (do ((i 0 (+ i 1)))
   61       ((= i n))
   62     (setq ntog (nth-toggle 'new 'my-ntog 't 3))))))
