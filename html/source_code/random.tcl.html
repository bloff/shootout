1 #!/usr/bin/tclsh
    2 # $Id: random.tcl.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 # from Miguel Sofer
    5 
    6 set IM 139968
    7 set IA   3877
    8 set IC  29573
    9 
   10 set last 42
   11 
   12 proc make_gen_random {} {
   13     global IM IA IC
   14     set params [list IM $IM IA $IA IC $IC]
   15     set body [string map $params {
   16         global last
   17         expr {($max * [set last [expr {($last * IA + IC) % IM}]]) / IM}
   18     }]
   19     proc gen_random {max} $body
   20 }
   21 
   22 proc main {} {
   23     global argv
   24 
   25     set N [expr {[lindex $argv 0] - 1}]
   26     make_gen_random
   27 
   28     while {$N} {
   29         gen_random 100.0
   30         incr N -1
   31     }
   32 
   33     puts [format "%.9f" [gen_random 100.0]]
   34 }
   35 
   36 main
