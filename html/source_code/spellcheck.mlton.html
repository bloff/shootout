    6 open TextIO;
    7 
    8 fun chop str = String.substring (str, 0, (String.size str) - 1);
    9 
   10 fun spellcheck () =
   11     let
   12     val dict = HashTable.mkTable (HashString.hashString, op =) (40000, (Fail
   13             "not found"))
   14     in
   15     let val din = openIn "Usr.Dict.Words"
   16         fun init_dict din dict =
   17             case inputLine din of
   18             NONE => ()
   19           | SOME(instr) => ( HashTable.insert dict (chop instr, 1);
   20                                      init_dict din dict )
   21     in
   22         init_dict din dict
   23     end;
   24 
   25     let fun check_words dict =
   26         case inputLine stdIn of
   27             NONE => ()
   28           | SOME(instr) =>
   29               let val word = chop instr
   30           in
   31             case HashTable.find dict word of
   32             SOME _ => ()
   33               | NONE => (print word ; print "\n");
   34             check_words dict
   35           end
   36     in
   37         check_words dict
   38     end handle EndOfFile => ()
   39     end
   40 
   41 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   42 
   43 fun main(name, args) =
   44     let
   45     val arg = hd(args @ ["1"])
   46     val num = atoi arg
   47     in
   48     spellcheck ();
   49     OS.Process.success
   50     end;
   51 
   52 
   53 val _ = main( CommandLine.name(), CommandLine.arguments() );
