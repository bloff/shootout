    2 # $Id: regexmatch.python.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
    3 # http://www.bagley.org/~doug/shootout/
    4 
    5 import sys, re
    6 
    7 def main():
    8     NUM = int(sys.argv[1])
    9     if NUM < 1:
   10         NUM = 1
   11 
   12     phones = sys.stdin.readlines()
   13 
   14     rx = re.compile(
   15         r'(?:^|[^\d\(])'
   16         r'(?:\((\d\d\d)\)|(\d\d\d))'
   17         r'[ ]'
   18         r'(\d\d\d)'
   19         r'[ -]'
   20         r'(\d\d\d\d)'
   21         r'\D'
   22         )
   23 
   24     findIt = rx.search
   25     count = 0
   26     for i in xrange(0,NUM):
   27         for line in phones:
   28             m = findIt(line)
   29             if m:
   30                 g = m.group
   31                 num = "(" + (g(1) or g(2)) + ") " + g(3) + "-" + g(4)
   32                 if 0 == i:
   33                     count = count + 1
   34                     print "%d: %s" % (count, num)
   35 
   36 main()
