1 #!/bin/bash
    2 # -*- mode: lisp -*-
    3 PROG=${0%.jl}
    4 [ \!  -f $PROG.jlc -o $0 -nt $PROG.jlc ] &&
    5     rep --batch --no-rc -l compiler -f compile-batch $0
    6 exec rep "$PROG.jlc" "$@"
    7 !#
    8 
    9 ;;; $Id: regexmatch.rep.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
   10 ;;; http://www.bagley.org/~doug/shootout/
   11 
   12 (define-structure main ()
   13   (open rep rep.system rep.regexp)
   14 
   15   (defvar regexp (concat
   16           "(^|[^0-9\\(])"    ; preceeding non-digit or bol
   17           "("            ; area code
   18           "\\(([0-9][0-9][0-9])\\)" ; is either 3 digits in parens
   19           "|"            ; or
   20           "([0-9][0-9][0-9])"    ; just 3 digits
   21           ")"            ; end of area code
   22           " "            ; area code is followed by one space
   23           "([0-9][0-9][0-9])"    ; exchange is 3 digits
   24           "[ -]"        ; separator is either space or dash
   25           "([0-9][0-9][0-9][0-9])" ; last 4 digits
   26           "[^0-9]"        ; must be followed by a non-digit
   27           ))
   28 
   29   (let* ((arg (get-command-line-option "--num" t))
   30      (n (if arg (read-from-string arg) 1))
   31      (phones '()))
   32     (let ((line))
   33       (while (setq line (read-line))
   34     (setq phones (append phones (list line)))))
   35     (while (> n 0)
   36       (setq n (1- n))
   37       (let ((count 0))
   38     (mapc (lambda (line)
   39         (if (string-match regexp line)
   40             (let ((num (expand-last-match "(\\3\\4) \\5-\\6")))
   41               (setq count (1+ count))
   42               (if (= 0 n)
   43               (format standard-output "%d: %s\n" count num)))))
   44           phones)))))
