    4 
    5 use strict;
    6 
    7 my $n = ($ARGV[0] > 0) ? $ARGV[0] : 1;
    8 my $x = 0;
    9 my $a = $n;
   10 while ($a--) {
   11     my $b = $n;
   12     while ($b--) {
   13     my $c = $n;
   14     while ($c--) {
   15         my $d = $n;
   16         while ($d--) {
   17         my $e = $n;
   18         while ($e--) {
   19             my $f = $n;
   20             while ($f--) {
   21             $x++;
   22             }
   23         }
   24         }
   25     }
   26     }
   27 }
   28 print "$x\n";
