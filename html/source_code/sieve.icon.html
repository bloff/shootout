    5 procedure main(argv)

    8     write("Count: ", count)
    9 end

   11 # algorithm from a test program that is distributed with

   14 procedure sieve()
   15    local limit, s, i
   16    limit := 8192
   17    s := set()
   18    every insert(s,1 to limit)
   19    every member(s,i := 2 to limit) do
   20       every delete(s,i + i to limit by i)
   21    delete(s,1)
   22    return(*s);
   23 end
