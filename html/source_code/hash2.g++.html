    5 #include <stdio.h>
    6 #include <iostream>
    7 #include <hash_map.h>
    8 
    9 using namespace std;
   10 
   11 struct eqstr {
   12     bool operator()(const char* s1, const char* s2) const {
   13     return strcmp(s1, s2) == 0;
   14     }
   15 };
   16 
   17 int
   18 main(int argc, char *argv[]) {
   19     int n = ((argc == 2) ? atoi(argv[1]) : 1);
   20     char buf[16];
   21     typedef hash_map<const char*, int, hash<const char*>, eqstr> HM;
   22     HM hash1, hash2;
   23 
   24     for (int i=0; i<10000; i++) {
   25     sprintf(buf, "foo_%d", i);
   26     hash1[strdup(buf)] = i;
   27     }
   28     for (int i=0; i<n; i++) {
   29     for (HM::iterator k = hash1.begin(); k != hash1.end(); ++k) {
   30         hash2[(*k).first] += hash1[(*k).first];
   31     }
   32     }
   33     cout << hash1["foo_1"] << " " << hash1["foo_9999"] << " "
   34      << hash2["foo_1"] << " " << hash2["foo_9999"] << endl;
   35 }
