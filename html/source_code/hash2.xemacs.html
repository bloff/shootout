    5 (defun main ()
    6   (let* ((arg (car command-line-args-left))
    7      (n (if arg (string-to-number arg) 1))
    8      (hash1 (make-hash-table :test 'equal :size 10000))
    9      (hash2 (make-hash-table :test 'equal :size 10000)))
   10     (do ((i 0 (+ i 1)))
   11     ((= i 10000))
   12       (puthash (format "foo_%d" i) i hash1))
   13     (do ((i 0 (+ i 1)))
   14     ((= i n))
   15       (maphash (lambda (k v)
   16          (puthash k (+ (gethash k hash1)
   17                    (or (gethash k hash2) 0))
   18               hash2)) hash1))
   19     (princ (format "%d %d %d %d\n"
   20            (gethash "foo_1" hash1)
   21            (gethash "foo_9999" hash1)
   22            (gethash "foo_1" hash2)
   23            (gethash "foo_9999" hash2)))))
