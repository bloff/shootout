    5 with Text_IO, Ada.Strings.Fixed, Ada.Command_Line;
    6 
    7 procedure Except is
    8    High_Exception : exception;
    9    Low_Exception  : exception;
   10    Low            : Integer := 0;
   11    High           : Integer := 0;
   12 
   13    procedure Blowup (K : Integer) is
   14       pragma Inline (Blowup);
   15    begin
   16       case 1 = (K mod 2) is
   17          when False => raise High_Exception;
   18          when True  => raise Low_Exception;
   19       end case;
   20    end Blowup;
   21 
   22    procedure Low_Function (K : Integer) is
   23       pragma Inline (Low_Function);
   24    begin
   25       Blowup (K);
   26    exception
   27       when Low_Exception => Low := Low + 1;
   28    end Low_Function;
   29 
   30    procedure High_Function (K : Integer) is
   31       pragma Inline (High_Function);
   32    begin
   33       Low_Function (K);
   34    exception
   35       when High_Exception => High := High + 1;
   36    end High_Function;
   37 
   38    procedure Some_Function (K : Integer) is
   39       pragma Inline (Some_Function);
   40    begin
   41       High_Function (K);
   42    exception
   43       when others => Text_IO.Put_Line ("We shouldn't get here");
   44    end Some_Function;
   45 
   46    function L_Trim (Source : String; Side : Ada.Strings.Trim_End :=
   47                Ada.Strings.Left) return String renames Ada.Strings.Fixed.
   48                Trim;
   49    N        : Natural := 0;
   50 begin
   51    begin
   52       N := Natural'Value (Ada.Command_Line.Argument (1));
   53    exception
   54       when Constraint_Error => null;
   55    end;
   56    for K in reverse 0 .. N - 1 loop
   57       Some_Function (K);
   58    end loop;
   59    Text_IO.Put_Line ("Exceptions: HI=" & L_Trim (Natural'Image (High)) &
   60             " / LO=" & L_Trim (Natural'Image (Low)));
   61 end Except;
