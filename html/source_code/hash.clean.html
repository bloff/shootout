   21 hashOps n 
   22    # hashtable = htNew n 
   23    # hashtable = addHexKeys n newHexConverter hashtable 
   24    =  countDecimalKeys n hashtable 0  
   25     
   26    where 
   27    addHexKeys i hc ht 
   28       | i==0 = ht 
   29       # hc = append i hc 
   30       #! s = toHexString hc 
   31       = addHexKeys (i-1) hc (htAdd s i ht)  
   32              
   33    countDecimalKeys i ht count  
   34       | i == 0    = count    
   35       = if (htHasKey (toString i) ht)  
   36          (countDecimalKeys (i-1) ht (count+1))  
   37          (countDecimalKeys (i-1) ht count)

   43 newHexConverter :: *HexConverter

   46 append n hc=:(Hc _ last _) 
   47    = append_ n last hc 
   48    where 
   49    append_ n i (Hc first last s)  
   50       | n==0 = (Hc i last s) 
   51       # i = i-1 
   52       # s = update s i hexchars.[n bitand 15]       
   53       = append_ (n>>4) i (Hc i last s)

   71 :: Item a = { key::!String 
   72             , val::a  
   73             } 
   74  
   75 :: HashTable a = { nBuckets::Int  
   76                  , table::!.{[Item a]} 
   77                  , nItems::Int  
   78                  } 
   79  
   80  
   81 hash :: !{#.Char} !(HashTable .a) -> Int 
   82 hash key ht=:{nBuckets}  
   83    = (abs (loop key (size key - 1) 0)) rem nBuckets    
   84    where 
   85    loop k n h 
   86       | n>(-1) = loop k (n-1) (5*h + toInt k.[n])          
   87       = h

   90 htNew n = { nBuckets = nprime 
   91         , table = {[] \\ i <- [1..nprime]} 
   92         , nItems = 0 
   93         } 
   94    where     
   95    nprime = hd (dropWhile (\x = x < n) primes)          
   96        
   97        
   98 htHasKey :: !{#.Char} !.(HashTable a) -> .Bool                
   99 htHasKey k ht=:{table}= findIn k table.[hash k ht]     
  100  
  101  
  102 htAdd k v ht=:{table,nItems}  
  103    #! i = hash k ht 
  104    #! (b,table) = uselect table i 
  105    = if (findIn k b)  
  106       {ht & table = update ht.table i (addItem k v b [])} 
  107       {ht & table = update ht.table i [{key=k,val=v}:b], nItems = nItems+1}
  108           
  109        
  110 findIn k [] = False 
  111 findIn k [item:ls]  
  112    | item.key == k   = True 
  113    = findIn k ls    
  114     
  115     
  116 addItem k v [] ls` = ls` 
  117 addItem k v [item:ls] ls` 
  118    | item.key == k    = [{item & val=v}:ls] ++ ls` 
  119    = addItem k v ls [item:ls`]
