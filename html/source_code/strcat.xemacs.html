    5 (defun main ()
    6   (let* ((arg (car command-line-args-left))
    7      (num (if arg (string-to-number arg) 1)))
    8     (let ((buf (get-buffer-create "mybuf"))
    9       (str ""))
   10       (dotimes (iter num)
   11     (insert-string "hello\n" buf))
   12       (princ (format "%d\n" (length (buffer-substring nil nil buf)))))))
