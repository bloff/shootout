   13 functor
   14 import System Application
   15 
   16 define
   17    IM = 139968
   18    IA = 3877
   19    IC = 29573
   20    Seed = 42
   21 
   22    fun {Random_heap H I S0}
   23       local R S in
   24      if I =< {Array.high H} then
   25         {Gen_random R S0 S}
   26         {Random_heap {Up_heap H I R} I+1 S}
   27      else
   28         H
   29      end
   30       end
   31    end
   32 
   33    fun {Up_heap H N Y}
   34       local HalfN X in
   35      HalfN = N div 2
   36      X = {Get H HalfN}
   37      if 0 < N andthen X < Y then
   38         {Put H N X}
   39         {Up_heap H HalfN Y}
   40      else
   41         {Put H N Y}
   42         H
   43      end
   44       end
   45    end
   46 
   47    fun {Heapsort H N}
   48       if N == 0 then H
   49       else {Heapsort {Remove_greatest H N} N-1} end
   50    end
   51 
   52    fun {Remove_greatest H N}
   53       local X Y in
   54      X = {Get H 0}
   55      Y = {Get H N}
   56      {Put H N X}
   57      {Down_heap H 0 N-1 Y}
   58       end
   59    end
   60 
   61    fun {Down_heap H I N X}
   62       local L R J Y in
   63      L = I + I + 1
   64      R = L + 1
   65      if N < L then
   66         {Put H I X}
   67         H
   68      else
   69         J = if R < N andthen {Get H R} > {Get H L}
   70         then  R else L end
   71         Y = {Get H J}
   72         if X > Y then
   73            {Put H I X}
   74            H
   75         else
   76            {Put H I Y}
   77            {Down_heap H J N X}
   78         end
   79      end
   80       end
   81    end
   82 
   83    proc {Gen_random R S0 S}
   84       S = (S0 * IA + IC) mod IM
   85       R = {IntToFloat S} / {IntToFloat IM}
   86    end
   87 
   88 in
   89    local Args N RandomHeap SortedHeap in
   90       [Args] = {Application.getArgs plain}
   91       N = {String.toInt Args}
   92       RandomHeap = {Random_heap {NewArray 0 N-1 0.0} 0 Seed}
   93       SortedHeap = {Heapsort RandomHeap N-1}
   94       {System.showInfo {Get SortedHeap N-1}}
   95    end
   96    {Application.exit 0}
   97 end
