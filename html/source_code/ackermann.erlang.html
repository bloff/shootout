    5 -module(ackermann).
    6 -export([main/1]).
    7 
    8 main() -> main(['1']).
    9 main([Arg]) ->
   10     Num = list_to_integer(atom_to_list(Arg)),
   11     io:fwrite("Ack(3,~w): ~w\n", [Num, ack(3, Num)]),
   12     halt(0).
   13 
   14 ack(0, N) -> N+1;
   15 ack(M, 0) -> ack(M-1, 1);
   16 ack(M, N) -> ack(M-1, ack(M, N-1)).
