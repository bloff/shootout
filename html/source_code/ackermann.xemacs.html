    5 (setq max-lisp-eval-depth 8000)
    6 (setq max-specpdl-size 5000)
    7 
    8 (defun Ack (M N)
    9   (cond ((= M 0) (+ N 1))
   10     ((= N 0) (Ack (- M 1) 1))
   11     (t (Ack (- M 1) (Ack M (- N 1))))))
   12 
   13 (defun main ()
   14   (let* ((arg (car command-line-args-left))
   15      (num (if arg (string-to-number arg) 1)))
   16     (princ (format "Ack(3,%d): %d\n" num (Ack 3 num)))))
