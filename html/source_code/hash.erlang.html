    5 %% Use ETS tables (Erlang's associative store).
    6 %%
    7 %% Optimizeations provided by Einar Karttunen.
    8 
    9 -module(hash).
   10 -export([main/0, main/1]).
   11 
   12 main() -> main(['1']).
   13 main([Arg]) ->
   14     N = list_to_integer(atom_to_list(Arg)),
   15     H = ets:new(i_am_a_carrot, [set]),
   16     doinserts(0, N, H),
   17     Count = dolookups(N, 0, H),
   18     io:format("~w~n", [Count]).
   19 
   20 doinserts(N, N, H) -> ok;
   21 doinserts(I, N, H) ->
   22     ets:insert(H, { erlang:integer_to_list(I, 16), I }),
   23     doinserts(I+1, N, H).
   24 
   25 dolookups(0, C, H) -> C;
   26 dolookups(I, C, H) ->
   27     Nx = integer_to_list(I),
   28     case ets:lookup(H, Nx) of
   29     [] ->    dolookups(I-1, C, H);
   30     Found -> dolookups(I-1, C+1, H)
   31     end.
