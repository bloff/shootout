    7 n = Integer(ARGV.shift || 1)
    8 x = 0
    9 n.times do
   10     n.times do
   11     n.times do
   12         n.times do
   13         n.times do
   14             n.times do
   15             x += 1
   16             end
   17         end
   18         end
   19     end
   20     end
   21 end
   22 puts x
