    5 ;;; $Id: nestedloop.guile.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $

    7 
    8 (define (main args)
    9   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   10      (x 0))
   11     (do ((a 0 (+ a 1)))
   12     ((= a n))
   13       (do ((b 0 (+ b 1)))
   14       ((= b n))
   15     (do ((c 0 (+ c 1)))
   16         ((= c n))
   17       (do ((d 0 (+ d 1)))
   18           ((= d n))
   19         (do ((e 0 (+ e 1)))
   20         ((= e n))
   21           (do ((f 0 (+ f 1)))
   22           ((= f n))
   23         (set! x (+ x 1))))))))
   24     (display x) (newline)))
