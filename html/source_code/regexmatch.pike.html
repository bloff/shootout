    7 constant area = "([0-9][0-9][0-9]|\\([0-9][0-9][0-9]\\))";
    8 constant exch = "([0-9][0-9][0-9])";
    9 constant last = "([0-9][0-9][0-9][0-9])";
   10 
   11 void main(int argc, array(string) argv)
   12 {
   13     Regexp r = Regexp("^[^0-9\\(]*"+area+" "+exch+"[ -]"+last+"[^0-9]*$");
   14     array(string) phones = Stdio.stdin->read()/"\n";
   15     int n = (int)argv[-1];
   16     int count = 0;
   17 
   18     while(n--)
   19     foreach(phones, string phone)
   20         if(array(string) parts = r->split(phone))
   21         if(n == 0)
   22             if(parts[0][0] == '(')
   23             write("%d: %s %s-%s\n", ++count, @parts);
   24             else
   25             write("%d: (%s) %s-%s\n", ++count, @parts);
   26 }
