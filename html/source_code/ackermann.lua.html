    4 function Ack(M, N)
    5     if (M == 0) then
    6     return( N + 1 )
    7     end
    8     if (N == 0) then
    9     return( Ack(M - 1, 1) )
   10     end
   11     return( Ack(M - 1, Ack(M, (N - 1))) )
   12 end
