    5 
    6 freq = Hash.new(0)
    7 loop {
    8     data = (STDIN.read(4095) or break) << (STDIN.gets || "")
    9     for word in data.downcase.tr_s('^A-Za-z',' ').split(' ')
   10     freq[word] += 1
   11     end
   12 }
   13 freq.delete("")
   14 
   15 lines = Array.new
   16 freq.each{|w,c| lines << sprintf("%7d\t%s\n", c, w) }
   17 print lines.sort.reverse
