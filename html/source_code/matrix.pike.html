    7 int size = 30;
    8 
    9 array(array(int))
   10 mkmatrix(int rows, int cols) {
   11     array(array(int)) m = allocate(rows);
   12     int count = 1;
   13     for (int i=0; i<rows; i++) {
   14     array(int) row = allocate(cols);
   15     for (int j=0; j<cols; j++) {
   16         row[j] = count++;
   17     }
   18     m[i] = row;
   19     }
   20     return(m);
   21 }
   22 
   23 void
   24 main(int argc, array(string) argv) {
   25     int n = (int)argv[-1];
   26     if (n < 1)
   27       n = 1;
   28 
   29     Math.Matrix m1 = Math.Matrix(mkmatrix(size, size));
   30     Math.Matrix m2 = Math.Matrix(mkmatrix(size, size));
   31     Math.Matrix mm;
   32     for( int i = n; i>0; i-- )
   33       mm = m1 * m2;
   34     array q = (array(array(int)))(array)mm;
   35     write( "%d %d %d %d\n", q[0][0], q[2][3], q[3][2], q[4][4] );
   36 }
