   11 
   12 (define-structure main ()
   13   (open rep rep.system)
   14 
   15   (define hello "hello
   16 ")
   17 
   18   (let* ((arg (get-command-line-option "--num" t))
   19      (n (if arg (read-from-string arg) 1))
   20      (str ""))
   21     (do ((i 0 (+ i 1)))
   22     ((= i n))
   23       (setq str (concat str hello)))
   24     (format standard-output "%d\n" (length str))))
