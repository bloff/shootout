    6 IM = 139968
    7 IA =   3877
    8 IC =  29573
    9 
   10 $last = 42.0
   11 def gen_random (max) (max * ($last = ($last * IA + IC) % IM)) / IM end
   12 
   13 def heapsort(n, ra)
   14     j = i = rra = 0
   15     l = (n >> 1) + 1
   16     ir = n
   17 
   18     while (1) do
   19     if (l > 1) then
   20         rra = ra.at(l -= 1)
   21     else
   22         rra = ra.at(ir)
   23         ra[ir] = ra.at(1)
   24         if ((ir -= 1) == 1) then
   25         ra[1] = rra
   26         return
   27         end
   28     end
   29     i = l
   30     j = l << 1
   31     while (j <= ir) do
   32         if ((j < ir) and (ra.at(j) < ra.at(j+1))) then
   33         j += 1
   34         end
   35         if (rra < ra.at(j)) then
   36         ra[i] = ra.at(j)
   37         j += (i = j)
   38         else
   39         j = ir + 1
   40         end
   41     end
   42     ra[i] = rra
   43     end
   44 end
   45 
   46 N = Integer(ARGV.shift || 1)
   47 ary = []
   48 for i in 1 .. N
   49     ary[i] = gen_random(1.0)
   50 end
   51 
   52 heapsort(N, ary)
   53 
   54 printf "%.10f\n", ary.at(N)
