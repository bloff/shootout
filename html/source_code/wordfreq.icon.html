    5 procedure main(argv)
    6     local wcount, buckets, w, c, pair, wl
    7 
    8     wcount := table(0)
    9     every wcount[words()] +:= 1
   10 
   11     buckets := table()
   12     every w := key(wcount) do {
   13     c := wcount[w]
   14     / buckets[c] := list()
   15     push(buckets[c], w)
   16     }
   17     buckets := sort(buckets,1)
   18     while pair := pull(buckets) do {
   19      c := pair[1]
   20      wl := sort(pair[2],1)
   21     while (write(right(c,7), "\t", pull(wl)))
   22     }
   23 end
   24 
   25 procedure words()
   26     local line, word
   27     while line := read() do {
   28     line := map(line)
   29     line ? while tab(upto(&letters)) do {
   30         word := tab(many(&letters))
   31         suspend(word)
   32     }
   33     }
   34 end
