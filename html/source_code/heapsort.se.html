    5 class HEAPSORT
    6 
    7 creation make
    8 
    9 feature
   10 
   11    make is
   12       local
   13      array: ARRAY[DOUBLE]
   14      n: INTEGER
   15       do
   16      n := argument(1).to_integer
   17      !!array.make(1, n)
   18      fill_array(array)
   19      sort_array(n, array)
   20      io.put_string(array.item(n).to_string_format(10))
   21      io.put_new_line
   22       end
   23 
   24    sort_array(n: INTEGER; ra: ARRAY[DOUBLE]) is
   25       local
   26      i, j, ir, l: INTEGER
   27      rra: DOUBLE
   28      done: BOOLEAN
   29       do
   30      j := 0
   31      i := 0
   32      rra := 0.0
   33      ir := n
   34      l := n // 2
   35      done := false
   36 
   37      from until done loop
   38         if l > 1 then
   39            l := l - 1
   40            rra := ra.item(l)
   41         else
   42            rra := ra.item(ir)
   43            ra.put(ra.item(1), ir)
   44            ir := ir - 1
   45            if ir = 1 then
   46           ra.put(rra, 1)
   47           -- should throw exception out of here instead of
   48           -- using boolean
   49           done := true
   50            end
   51         end
   52 
   53         if not done then
   54            i := l
   55            j := l * 2
   56 
   57            from until j > ir loop
   58           if (j < ir) and (ra.item(j) < ra.item(j+1)) then
   59              j := j + 1
   60           end
   61           if rra < ra.item(j) then
   62              ra.put(ra.item(j), i)
   63              i := j
   64              j := j + i
   65           else
   66              j := ir + 1
   67           end
   68            end
   69            ra.put(rra, i)
   70         end
   71      end
   72       end
   73 
   74 
   75    fill_array(an_array: ARRAY[DOUBLE]) is
   76       local
   77      rand: RANDOMNUMBER
   78      index: INTEGER
   79       do
   80      from
   81         !!rand.make
   82         index := an_array.lower
   83      until
   84         index > an_array.upper
   85      loop
   86         an_array.put(rand.next(1), index)
   87         index := index + 1
   88      end
   89       end
   90 
   91 end
