    5 %% Use ETS tables (Erlang's associative store).
    6 
    7 -module(wordfreq).
    8 -export([main/0, main/1]).
    9 
   10 %% ignore program argument
   11 main() -> main(['1']).
   12 main(Args) ->
   13     OutL = fun ({Word, Count}) ->
   14         io:format("~7w\t~s~n", [Count, Word])
   15        end,
   16     lists:foreach(OutL, sortedfreqs()),
   17     halt(0).
   18 
   19 
   20 % sort the results, descending
   21 sortedfreqs() ->
   22     Port = open_port({fd, 0, 1}, [eof, {line, 512}]),
   23     WordCountList = count_words_from_stream(ets:new(freqtab, [ordered_set])
   24                     , Port),
   25     lists:reverse(lists:keysort(2, WordCountList)).
   26 
   27 
   28 count_words_from_stream(Table, Port) ->
   29     receive
   30     {Port, eof} ->
   31         ets:delete(Table, ''),
   32         ets:tab2list(Table);
   33     {Port, {_, {_, Line}}} ->
   34         count_words([], Line, Table),
   35         count_words_from_stream(Table, Port)
   36     end.
   37 
   38 count_word(Word, Table) ->
   39     WordAtom = list_to_atom(Word),
   40     case (catch ets:update_counter(Table, WordAtom, 1)) of
   41     {'EXIT', {badarg, _}} ->
   42         ets:insert(Table, {WordAtom, 1});
   43     _ ->
   44         true
   45     end.
   46 
   47 % count_words(Word_Accumulator, Line_of_Chars, Table)
   48 count_words([], [], Table) -> true;
   49 count_words(Word, [], Table) ->
   50     count_word(Word, Table);
   51 count_words(Word, [H|T], Table) when H >= $a, H=< $z ->
   52     NewWord = lists:append(Word, [H]),
   53     count_words(NewWord, T, Table);
   54 count_words(Word, [H|T], Table) when H >= $A, H=< $Z ->
   55     NewWord = lists:append(Word, [(H - $A) + $a]),
   56     count_words(NewWord, T, Table);
   57 % we hit a non-word character so count previous word and continue
   58 count_words(Word, [H|T], Table) ->
   59     count_word(Word, Table),
   60     count_words([], T, Table).
