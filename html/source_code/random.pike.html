    2 // -*- mode: pike -*-
    3 // $Id: random.pike.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
    4 // http://www.bagley.org/~doug/shootout/
    5 
    6 int IM = 139968;
    7 int IA = 3877;
    8 int IC = 29573;
    9 
   10 int last = 42;
   11 
   12 float
   13 gen_random(float max) { return(max * (last = (last * IA + IC) % IM) / IM);
   14            }
   15 
   16 int
   17 main(int argc, array(string) argv) {
   18     int N = ((int)argv[-1] || 1) - 1;
   19     while (N--) {
   20     gen_random(100.0);
   21     }
   22     write("%.9f\n", gen_random(100.0));
   23     return(0);
   24 }
