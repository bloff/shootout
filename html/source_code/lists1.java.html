    4 import java.io.*;
    5 import java.util.*;
    6 import java.text.*;
    7 import java.lang.reflect.*;
    8 
    9 public class lists1 {
   10     public static void main(String args[]) {
   11     int n = Integer.parseInt(args[0]);
   12 
   13     // create empty list B
   14     LinkedList B = new LinkedList();
   15 
   16     // create list (A) of integers from 1 through N
   17     LinkedList A = new LinkedList();
   18     for (int i=n; i>0; i--) {
   19         A.addFirst(new Integer(i));
   20     }
   21 
   22     // move each individual item from A to B, in a loop, reversing order
   23     while (! A.isEmpty()) {
   24         B.addFirst(A.removeFirst());
   25     }
   26 
   27     // print first 2 elements of B
   28     System.out.println(B.get(0) + " " + B.get(1));
   29 
   30     // reverse B (can be done in place)
   31     LinkedList tmp = new LinkedList();
   32     while (! B.isEmpty()) {
   33         tmp.addFirst(B.removeFirst());
   34     }
   35     B = tmp;
   36     // java.util.Collections.reverse(B);
   37 
   38 
   39     // is 0 a member of B?
   40     System.out.println((B.contains(new Integer(0))) ? "true" : "false");
   41 
   42     // is N a member of B?
   43     System.out.println((B.contains(new Integer(n))) ? "true" : "false");
   44 
   45     // filter values from B to A that are less than N/2, preserving order
   46     int mid = n/2;
   47     for (ListIterator li = B.listIterator(); li.hasNext();) {
   48         int val = ((Integer)li.next()).intValue();
   49         if (val < mid) A.addLast(new Integer(val));
   50     }
   51 
   52     // print first ten items of A
   53     int i = 0;
   54     for (ListIterator li = A.listIterator(); i<10 && li.hasNext(); i++) {
   55         System.out.print(((Integer)li.next()).intValue() + ((i<9) ? " " :
   56                       "\n"));
   57     }
   58 
   59     // print sum of items in A that are less than 1000
   60     int sum = 0;
   61     for (ListIterator li = A.listIterator(); li.hasNext();) {
   62         int val = ((Integer)li.next()).intValue();
   63         if (val < 1000) sum += val;
   64     }
   65     System.out.println(sum);
   66 
   67     // append B to end of A
   68     A.addAll(B);
   69 
   70     // print length and last element of A
   71     System.out.println(A.size() + " " + ((Integer)A.getLast()).intValue());
   72     }
   73 }
