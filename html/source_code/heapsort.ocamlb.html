    7 let im = 139968 and ia = 3877 and ic = 29573 and last_ref = ref 42
    8 
    9 let gen_random max =
   10   let new_last = (!last_ref * ia + ic) mod im in
   11   last_ref := new_last;
   12   max *. float_of_int new_last /. float im
   13 
   14 
   15 let heapsort n ra =
   16   let l = ref ((n lsr 1) + 1)
   17   and rra = ref 0.0 and i = ref 0 and j = ref 0 and ir = ref n in
   18   try
   19     while true do
   20       if !l > 1 then begin
   21         decr l;
   22         rra := ra.(!l)
   23       end
   24       else begin
   25         rra := ra.(!ir);
   26         ra.(!ir) <- ra.(1);
   27         decr ir;
   28         if !ir = 1 then begin
   29           ra.(1) <- !rra;
   30           raise Exit
   31         end
   32       end;
   33       i := !l;
   34       j := !l lsl 1;
   35       while !j <= !ir do
   36         if !j < !ir && ra.(!j) < ra.(!j+1) then incr j;
   37         if !rra < ra.(!j) then begin
   38           ra.(!i) <- ra.(!j);
   39           i := !j;
   40           j := !j + !i
   41         end
   42         else j := !ir + 1;
   43       done;
   44       ra.(!i) <- !rra;
   45     done
   46   with Exit -> ()
   47 
   48 let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1) else
   49         1;;
   50 let ary = Array.init (n + 1) (fun _ -> gen_random 1.0) in
   51 heapsort n ary;
   52 Printf.printf "%.10f\n" ary.(n)
