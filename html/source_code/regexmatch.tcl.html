    5 
    6 proc main {} {
    7     global argv
    8     set NUM [lindex $argv 0]
    9     if {$NUM < 1} {
   10         set NUM 1
   11     }
   12 
   13     set data [read stdin]
   14     set count 0
   15     set rExp {(?:^|[^\d(])(\(\d{3}\)|\d{3}) (\d{3}[ -]\d{4})(?:$|[^\d])}
   16 
   17     while {$NUM} {
   18     incr NUM -1
   19     foreach {-- area num} [regexp -all -line -inline $rExp $data] {
   20         set pnum "([string trim $area () ]) [string map {" " -} $num]"
   21         if {!$NUM} {
   22         puts "[incr count]: $pnum"
   23         }
   24     }
   25     }
   26 }
   27 
   28 main
