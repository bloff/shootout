    6 %% Used destructive assignment in the process dictionary  
    7 %% to keep count of handled exceptions. 
    8 %% 
    9 %% Usage: start from command line with 
   10 %%     erlc except.erl 
   11 %%     erl -noinput -s except main 200000 
   12  
   13 -module(except).  
   14 -export([main/1]).  
   15  
   16  
   17 blowup(N) when N rem 2 == 0 -> throw({lo_exception, N}); 
   18 blowup(N) -> throw({hi_exception, N}). 
   19  
   20  
   21 lo_fun(N) -> 
   22     case catch blowup(N) of 
   23         {lo_exception, N1} -> put(lo_count, get(lo_count) + 1); 
   24         {hi_exception, N2} -> throw({hi_exception, N2}) 
   25     end. 
   26  
   27  
   28 hi_fun(N) ->  
   29     case catch lo_fun(N) of 
   30         {hi_exception, N1} -> put(hi_count, get(hi_count) + 1); 
   31         _ -> true 
   32     end. 
   33  
   34  
   35 some_fun(0) -> true; 
   36 some_fun(N) -> 
   37     case catch hi_fun(N) of 
   38         {lo_exception, N1} -> io:fwrite("~s~n", ["lo_exception should not
   39          get here."]); 
   40         {hi_exception, N2} -> io:fwrite("~s~n", ["hi_exception should not
   41          get here."]); 
   42         _ -> true 
   43     end, 
   44     some_fun(N - 1). 
   45  
   46  
   47 main([Arg]) -> 
   48     Num = list_to_integer(atom_to_list(Arg)), 
   49     put(hi_count, 0), 
   50     put(lo_count, 0), 
   51     some_fun(Num), 
   52     io:fwrite("~s~w ~s~w~n", ["Exceptions: HI=", get(hi_count),"/ LO=",
   53               get(lo_count)]), 
   54     halt(0).
