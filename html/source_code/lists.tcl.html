    6 
    7 set SIZE 10000
    8 
    9 proc K {a b} {set a}
   10 
   11 proc ldelete {listName index} {
   12     upvar $listName list
   13     ;# Replace a deletion with null, much faster
   14     set list [lreplace [K $list [set list {}]] $index $index]
   15 }
   16 
   17 proc lreverse {_list} {
   18     upvar $_list List
   19     for {set i [expr {[llength $List] - 1}]} {$i >= 0} {incr i -1} {
   20     lappend Li1r [lindex $List $i]
   21     }
   22     set List $Li1r
   23     unset Li1r
   24 }
   25 
   26 proc test_lists {args} {
   27     # create a list of integers (Li1) from 1 to SIZE
   28     for {set i 1} {$i <= $::SIZE} {incr i} {lappend Li1 $i}
   29     # copy the list to Li2 (not by individual items)
   30     set Li2 $Li1
   31     # remove each individual item from left side of Li2 and
   32     # append to right side of Li3 (preserving order)
   33     lreverse Li2
   34     foreach {item} $Li2 {
   35     lappend Li3 [lindex $Li2 end]
   36     ldelete Li2 end
   37     }
   38     # Li2 must now be empty
   39     # remove each individual item from right side of Li3 and
   40     # append to right side of Li2 (reversing list)
   41     foreach {item} $Li3 {
   42     lappend Li2 [lindex $Li3 end]
   43     ldelete Li3 end
   44     }
   45     # Li3 must now be empty
   46     # reverse Li1 in place
   47     lreverse Li1
   48     # check that first item is now SIZE
   49     if {[lindex $Li1 0] != $::SIZE} {
   50     return "fail size [lindex $Li1 0]"
   51     }
   52     # compare Li1 and Li2 for equality
   53     # and return length of the list
   54     if {$Li1 == $Li2} {
   55     return [llength $Li1]
   56     } else {
   57     return "fail compare"
   58     }
   59 }
   60 
   61 proc main {args} {
   62     global argv
   63     set NUM [lindex $argv 0]
   64     if {$NUM < 1} {
   65     set NUM 1
   66     }
   67     while {$NUM > 0} {
   68     set result [test_lists]
   69     incr NUM -1
   70     }
   71     puts $result
   72 }
   73 
   74 main
