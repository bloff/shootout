    5 class WC
    6 
    7 creation make
    8 
    9 feature
   10 
   11    make is
   12 
   13       local
   14      nl, nw, nc, state: INTEGER;
   15      c: CHARACTER;
   16       do
   17      nl := 0
   18      nw := 0
   19      nc := 0
   20      state := 0
   21      from
   22         io.read_character
   23      until
   24         io.end_of_input
   25      loop
   26         c := io.last_character
   27         nc := nc + 1
   28         if c = '%N' then
   29            nl := nl + 1
   30         end
   31         if c = ' ' or c = '%N' or c = '%T' then
   32            state := 0
   33         else
   34            if state = 0 then
   35           state := 1
   36           nw := nw + 1
   37            end
   38         end
   39         io.read_character
   40      end
   41          std_output.put_integer(nl)
   42          std_output.put_character(' ')
   43          std_output.put_integer(nw)
   44          std_output.put_character(' ')
   45          std_output.put_integer(nc)
   46          std_output.put_character('%N')
   47       end
   48 end
