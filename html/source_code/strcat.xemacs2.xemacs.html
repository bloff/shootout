    6 (defun main ()
    7   (let* ((arg (car command-line-args-left))
    8      (num (if arg (string-to-number arg) 1)))
    9     (let ((str ""))
   10       (dotimes (iter num)
   11     (setq str (concat str "hello\n")))
   12       (princ (format "%d\n" (length str))))))
