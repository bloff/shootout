    7 val size = 8193
    8 
    9 val flags = Array.array (size, false)
   10 
   11 fun init () =
   12    let
   13       fun loop i =
   14      if i < size
   15         then (Array.update (flags, i, true); loop (i + 1))
   16      else ()
   17    in loop 2
   18    end
   19 
   20 fun do_elts (i, count) =
   21   if i < size
   22      then
   23     if Array.sub (flags, i)
   24        then
   25           let
   26          fun loop k =
   27             if k < size
   28                then (Array.update (flags, k, false); loop (k + i))
   29             else ()
   30           in loop (i + i); do_elts(i + 1,count + 1)
   31           end
   32     else do_elts (i + 1, count)
   33   else count
   34 
   35 fun printl [] = print "\n" | printl(h::t) = ( print h ; printl t )
   36 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0
   37 
   38 fun doit () = (init (); do_elts (2, 0))
   39 
   40 fun repeat i =
   41    if i = 0
   42       then doit ()
   43    else (doit (); repeat (i - 1))
   44 
   45 fun main(name, param_list) =  let
   46     val arg = hd(param_list @ ["1"]);
   47     val num = atoi arg
   48     val count = repeat (num - 1)
   49     in  printl ["Count: ", Int.toString count];
   50     OS.Process.success
   51     end
   52 
   53 val _ = main( CommandLine.name(), CommandLine.arguments() );
