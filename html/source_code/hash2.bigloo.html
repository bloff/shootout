    5 (module hash
    6     (option (set! *genericity* #f))
    7     (main main))
    8 
    9 (define (main args)
   10   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   11      (max (do ((max 1 (* max 2))) ((> max 10000) max)))
   12      (x (make-hash-table max
   13                  (lambda (o) (string->0..2^x-1 o 14))
   14                  (lambda (c) (car c))
   15                  string=?
   16                  max))
   17      (y (make-hash-table max
   18                  (lambda (o) (string->0..2^x-1 o 14))
   19                  (lambda (c) (car c))
   20                  string=?
   21                  max)))
   22     (do ((i 0 (+ i 1)))
   23     ((= i 10000))
   24       (put-hash! (cons (string-append "foo_" (number->string i 10)) i) x))
   25     (do ((i 0 (+ i 1)))
   26     ((= i n))
   27       (for-each-hash (lambda (cell)
   28                (let* ((x-key (car cell))
   29                   (x-val (cdr cell))
   30                   (old-y-cell (get-hash x-key y))
   31                   (old-y-val (cond (old-y-cell
   32                         (rem-key-hash! x-key y)
   33                         (cdr old-y-cell))
   34                            (else
   35                         0))))
   36              (put-hash! (cons x-key (+ x-val old-y-val)) y)))
   37              x))
   38     (print (cdr (get-hash "foo_1" x)) " "
   39        (cdr (get-hash "foo_9999" x)) " "
   40        (cdr (get-hash "foo_1" y)) " "
   41        (cdr (get-hash "foo_9999" y)))))
