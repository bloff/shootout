    5 procedure main(argv)
    6     local n, a, b, c, d, e, f, x
    7     n := argv[1] | 1;
    8     x := 0
    9     every a := 1 to n do
   10     every b := 1 to n do
   11         every c := 1 to n do
   12         every d := 1 to n do
   13             every e := 1 to n do
   14             every f := 1 to n do
   15                 x +:= 1
   16     write(x)
   17 end
