    5 (defun main ()
    6   (let* ((arg (car command-line-args-left))
    7      (n (if arg (string-to-number arg) 1))
    8      (last (- n 1))
    9      (c 0)
   10      (x (make-hash-table :test 'equal :size n)))
   11     (do ((i 1 (+ i 1)))
   12     ((> i n))
   13       (puthash (format "%x" i) i x))
   14     (do ((i last (- i 1)))
   15     ((< i 0))
   16       (if (gethash (format "%d" i) x)
   17       (setq c (+ c 1))))
   18     (princ (format "%d\n" c))))
