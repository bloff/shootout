    6 #include <iostream>
    7 #include <vector>
    8 
    9 using namespace std;
   10 
   11 enum {
   12     OUT,            /* outside a word */
   13     IN                /* inside a word */
   14 };
   15 
   16 int
   17 main(int argc, char *argv[]) {
   18     char c;
   19     int nl, nw, nc, state;
   20     char buff[4096];
   21     cin.rdbuf()->pubsetbuf(buff, 4096); // enable buffering
   22 
   23     state = OUT;
   24     nl = nw = nc = 0;
   25     int intc;
   26     streambuf* sbuf = cin.rdbuf();
   27     while ((intc = sbuf->sbumpc()) != EOF) {
   28         c = (char)intc;
   29     ++nc;
   30     if (c == '\n')
   31         ++nl;
   32     if (c == ' ' || c == '\n' || c == '\t')
   33         state = OUT;
   34     else if (state == OUT) {
   35         state = IN;
   36         ++nw;
   37     }
   38     }
   39     cout << nl << " " << nw << " " << nc << endl;
   40 }
