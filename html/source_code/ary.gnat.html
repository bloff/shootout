    5 with Text_IO, Ada.Command_Line, Ada.Strings.Fixed;
    6 
    7 procedure Ary is
    8    function L_Trim (Source : String; Side : Ada.Strings.Trim_End :=
    9                Ada.Strings.Left) return String renames Ada.Strings.Fixed.
   10                Trim;
   11    N        : Positive := 1;
   12 begin
   13    begin
   14       N := Positive'Value (Ada.Command_Line.Argument (1));
   15    exception
   16       when Constraint_Error => null;
   17    end;
   18    declare
   19       type Vect is array (1 .. N) of Integer;
   20       X, Y     : Vect;
   21    begin
   22       for K in Vect'Range loop
   23          X (K) := K;
   24          Y (K) := 0;
   25       end loop;
   26 
   27       for Iter in 1 .. 1000 loop
   28          for K in reverse Vect'Range loop
   29             declare
   30                Y_K      : Integer renames Y (K);
   31             begin
   32                Y_K := Y_K + X (K);
   33             end;
   34          end loop;
   35       end loop;
   36       Text_IO.Put_Line (L_Trim (Integer'Image (Y (1))) &
   37                Integer'Image (Y (N)));
   38    end;
   39 end Ary;
