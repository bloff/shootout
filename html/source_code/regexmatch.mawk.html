    4 BEGIN {
    5     n = (ARGV[1] < 1) ? 1 : ARGV[1];
    6     delete ARGV;
    7 }
    8 
    9 { phones[p++] = $0; }
   10 
   11 END {
   12     for (i=0; i<n; i++) {
   13     for (j=0; j<p; j++) {
   14         line = phones[j];
   15         if (match(line, /(^|[^0-9])(\([0-9][0-9][0-9]\)|[0-9][0-9][0-9]) [0-9]
   16          [0-9][0-9][ -][0-9][0-9][0-9][0-9]($|[^0-9])/)) {
   17         m1 = substr(line, RSTART, RLENGTH);
   18         num = ""
   19         if (match(m1, /[0-9][0-9][0-9] [0-9][0-9][0-9][ -][0-9][0-9][0-9][0-9]/))
   20       {
   21             if (substr(m1, RSTART-1, 1) == "(") {
   22             break;
   23             }
   24             if (x = split(substr(m1, RSTART, RLENGTH), parts, /[ -]/)) {
   25             num = "(" parts[1] ") " parts[2] "-" parts[3];
   26             }
   27         } else if (match(m1, /\([0-9][0-9][0-9]\)  [0-9][0-9][0-9][ -][0-9][0-9][
   28              0-9][0-9]/)) {
   29             if (x = split(substr(m1, RSTART, RLENGTH), parts, /[ -]/)) {
   30             num = parts[1] " " parts[2] "-" parts[3];
   31             }
   32         }
   33         if (i == (n-1)) {
   34             count++;
   35             printf("%d: %s\n", count, num);
   36         }
   37         }
   38     }
   39     }
