    5 with Ada.Strings.Fixed, Ada.Command_Line, Text_IO;
    6 
    7 procedure ProdCons is
    8 
    9    type Data_Type is new Integer;
   10    End_Of_Data    : constant Data_Type := Data_Type'First;
   11 
   12    protected Queue is
   13       entry Put (Data : Data_Type);
   14       entry Get (Data_Out : out Data_Type);
   15    private
   16       Count          : Natural := 0;
   17       Buffer         : Data_Type;
   18    end Queue;
   19 
   20    protected body Queue is
   21       entry Put (Data : Data_Type)
   22          when Count = 0 is
   23       begin
   24          Buffer := Data;
   25          Count := Count + 1;
   26       end Put;
   27 
   28       entry Get (Data_Out : out Data_Type)
   29          when Count /= 0 is
   30       begin
   31          Data_Out := Buffer;
   32          Count := Count - 1;
   33       end Get;
   34    end Queue;
   35 
   36    Produced, Consumed : Natural := 0;
   37 
   38    task type Producer_Task (N : Natural);
   39    task type Consumer_Task (N : Natural);
   40 
   41    task body Producer_Task is
   42    begin
   43       for Data_K in 1 .. Data_Type (N) loop
   44          Queue.Put (Data => Data_K);
   45          Produced := Produced + 1;
   46       end loop;
   47       Queue.Put (Data => End_Of_Data);
   48    end Producer_Task;
   49 
   50    task body Consumer_Task is
   51       Data        : Data_Type;
   52    begin
   53       loop
   54          Queue.Get (Data_Out => Data);
   55          exit when Data = End_Of_Data;
   56          Consumed := Consumed + 1;
   57       end loop;
   58    end Consumer_Task;
   59 
   60    function L_Trim (Source : String; Side : Ada.Strings.Trim_End :=
   61                Ada.Strings.Left) return String renames Ada.Strings.Fixed.
   62                Trim;
   63    N        : Natural := 0;
   64 begin
   65    begin
   66       N := Natural'Value (Ada.Command_Line.Argument (1));
   67    exception
   68       when Constraint_Error => null;
   69    end;
   70    declare
   71       Producer    : Producer_Task (N => N);
   72       Consumer    : Consumer_Task (N => N);
   73    begin
   74       null;
   75    end;
   76    Text_IO.Put_Line (L_Trim (Natural'Image (Produced)) &
   77             Natural'Image (Consumed));
   78 end ProdCons;
