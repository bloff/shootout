1 #!/usr/bin/guile \
    2 -e main -s
    3 !#
    4 
    5 ;;; $Id: regexmatch.guile.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 
    8 (use-modules (ice-9 format))
    9 (use-modules (ice-9 regex))
   10 
   11 (define regexp
   12   (string-append
   13    "(^|[^0-9\\(])"            ; (1) preceeding non-digit or bol
   14    "("                    ; (2) area code
   15    "\\(([0-9][0-9][0-9])\\)"        ; (3) is either 3 digits in parens
   16    "|"                    ; or
   17    "([0-9][0-9][0-9])"            ; (4) just 3 digits
   18    ")"                    ; end of area code
   19    " "                    ; area code is followed by one space
   20    "([0-9][0-9][0-9])"            ; (5) exchange is 3 digits
   21    "[ -]"                ; separator is either space or dash
   22    "([0-9][0-9][0-9][0-9])"        ; (6) last 4 digits
   23    "([^0-9]|$)"                ; must be followed by a non-digit
   24    ))
   25 
   26 
   27 (define (main args)
   28   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   29     (phonelines '())
   30     (rx (make-regexp regexp))
   31     (count 0))
   32     (let loop ((line (read-line)))
   33       (cond ((eof-object? line) #f)
   34         (else
   35          (set! phonelines (append phonelines (list line)))
   36          (loop (read-line)))))
   37     (while (> n 0)
   38        (set! n (- n 1))
   39        (let loop ((phones phonelines)
   40               (count 0))
   41          (if (null? phones)
   42          count
   43          (let ((match (regexp-exec rx (car phones))))
   44            (if match
   45                (let* ((area (if (match:start match 3)
   46                     (match:substring match 3)
   47                     (match:substring match 4)))
   48                   (exch (match:substring match 5))
   49                   (numb (match:substring match 6))
   50                   (num (string-append "(" area ") " exch "-" numb)))
   51              (set! count (+ count 1))
   52              (if (= 0 n)
   53                   (display (format "~D: ~a\n" count num)))))
   54            (loop (cdr phones) count)))))))
   55
