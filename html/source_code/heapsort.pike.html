    7 #define IM 139968
    8 #define IA   3877
    9 #define IC  29573
   10 
   11 int last = 42;
   12 
   13 float
   14 gen_random(float max) { return(max * (last = (last * IA + IC) % IM) / IM);
   15            }
   16 
   17 void heapsort(int n, array(float) ra) {
   18     int l, j, ir, i;
   19     float rra;
   20 
   21     l = (n >> 1) + 1;
   22     ir = n;
   23     for (;;) {
   24         if (l > 1) {
   25             rra = ra[--l];
   26         } else {
   27             rra = ra[ir];
   28             ra[ir] = ra[1];
   29             if (--ir == 1) {
   30                 ra[1] = rra;
   31                 return;
   32             }
   33         }
   34         i = l;
   35         j = l << 1;
   36         while (j <= ir) {
   37             if (j < ir && ra[j] < ra[j+1]) { ++j; }
   38             if (rra < ra[j]) {
   39                 ra[i] = ra[j];
   40                 j += (i = j);
   41             } else {
   42                 j = ir + 1;
   43             }
   44         }
   45         ra[i] = rra;
   46     }
   47 }
   48 
   49 void main(int argc, array(string) argv) {
   50     int N = (int)argv[-1] || 1;
   51     array(float) ary;
   52     int i;
   53 
   54     // create an array of N random floats
   55     ary = allocate(N+1);
   56     for (i=1; i<=N; i++) {
   57         ary[i] = gen_random(1.0);
   58     }
   59 
   60     heapsort(N, ary);
   61 
   62     write("%.10f\n", ary[N]);
   63 }
