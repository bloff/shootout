    8 set_time_limit(0);

   10 $SIZE = 30;
   11 
   12 function mkmatrix ($rows, $cols) {
   13     $count = 1;
   14     $mx = array();
   15     for ($i=0; $i<$rows; $i++) {
   16     for ($j=0; $j<$cols; $j++) {
   17         $mx[$i][$j] = $count++;
   18     }
   19     }
   20     return($mx);
   21 }
   22 
   23 function mmult ($rows, $cols, $m1, $m2) {
   24     $m3 = array();
   25     for ($i=0; $i<$rows; $i++) {
   26     for ($j=0; $j<$cols; $j++) {
   27         $x = 0;
   28         for ($k=0; $k<$cols; $k++) {
   29         $x += $m1[$i][$k] * $m2[$k][$j];
   30         }
   31         $m3[$i][$j] = $x;
   32     }
   33     }
   34     return($m3);
   35 }
   36 
   37 $n = ($argc == 2) ? $argv[1] : 1;
   38 $m1 = mkmatrix($SIZE, $SIZE);
   39 $m2 = mkmatrix($SIZE, $SIZE);
   40 while ($n--) {
   41     $mm = mmult($SIZE, $SIZE, $m1, $m2);
   42 }
   43 print "{$mm[0][0]} {$mm[2][3]} {$mm[3][2]} {$mm[4][4]}\n";
