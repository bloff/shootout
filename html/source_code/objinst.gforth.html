    5 \ from Anton Ertl:
    6 \ I'm using objects.fs here, code using one of the other OO Forth
    7 \ extensions will look different.
    8 
    9 warnings off \ don't complain about redefining catch, state, value
   10 
   11 0. argc @ 1- arg >number 2drop drop constant NUM
   12 
   13 require objects.fs
   14 
   15 object class
   16     selector activate ( toggle -- toggle )
   17     selector value ( toggle -- f )
   18     cell% inst-var state
   19 
   20     m: ( f toggle -- )
   21     state ! ;m
   22     overrides construct
   23 
   24     m: ( toggle -- f )
   25     state @ ;m
   26     overrides value
   27 
   28     m: ( toggle -- toggle )
   29     state @ invert state !
   30     this ;m
   31     overrides activate
   32 
   33 end-class Toggle
   34 
   35 Toggle class
   36     cell% inst-var count-max
   37     cell% inst-var counter
   38 
   39     m: ( max-counter start-state nth-toggle -- )
   40     this [parent] construct
   41     count-max !
   42     0 counter ! ;m
   43     overrides construct
   44 
   45     m: ( toggle -- toggle )
   46     1 counter +!
   47     counter @ count-max @ >= if
   48         state @ invert state !
   49         0 counter !
   50     endif
   51     this ;m
   52     overrides activate
   53 
   54 end-class NthToggle
   55 
   56 : flag. ( f -- )
   57     if ." true" else ." false" endif cr ;
   58 
   59 : mainloop ( ... class n -- ) { class n }
   60     true class heap-new true n 0 ?do
   61     drop dup activate value dup flag.
   62     loop
   63     drop ;
   64 
   65 : main ( -- )
   66     Toggle 5 mainloop
   67     NUM 0 ?do
   68     true Toggle heap-new free drop \ like the C version
   69     loop
   70     cr
   71     3 NthToggle 8 mainloop
   72     NUM 0 ?do
   73     3 true NthToggle heap-new free drop \ like the C version
   74     loop ;
   75 
   76 main bye
