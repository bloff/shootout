    2 -e main -s
    3 !#
    4 
    5 ;;; $Id: methcall.guile.html,v 1.1 2004-06-14 23:09:27 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 ;;; from: Benedikt Rosenau
    8 
    9 (use-modules (oop goops))
   10 
   11 (define-method (print-bool (b <boolean>))
   12    (display (if b "true\n" "false\n")))
   13 
   14 
   15 (define-class <toggle> ()
   16   (state #:getter value? #:init-keyword #:state))
   17 
   18 (define-class <nth-toggle> (<toggle>)
   19   (count-max #:init-keyword #:count-max)
   20   (counter #:init-value 0))
   21 
   22 
   23 (define-method (activate! (t <toggle>))
   24   (slot-set! t 'state (not (slot-ref t 'state)))
   25    t)
   26 
   27 (define-method (activate! (n-t <nth-toggle>))
   28   (let ((counter (+ 1 (slot-ref n-t 'counter))))
   29     (slot-set! n-t 'counter counter)
   30     (if (>= counter (slot-ref n-t 'count-max))
   31       (begin (slot-set! n-t 'state (not (slot-ref n-t 'state)))
   32              (slot-set! n-t 'counter 0)))
   33      n-t))
   34 
   35 
   36 (define-method (main (l <list>))
   37   (let ((n (catch #t (lambda () (string->number (cadr l)))
   38                      (lambda ex 1))))
   39     (let ((tog (make <toggle> #:state #t)))
   40       (do ((i 1 (+ i 1))) ((= i n))
   41           (value? (activate! tog)))
   42       (print-bool (value? (activate! tog))))
   43     (let ((ntog (make <nth-toggle> #:state #t #:count-max 3)))
   44       (do ((i 1 (+ i 1))) ((= i n))
   45           (value? (activate! ntog)))
   46       (print-bool (value? (activate! ntog))))))
