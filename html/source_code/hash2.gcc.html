    6 #include <stdio.h>
    7 #include <stdlib.h>
    8 #include <unistd.h>
    9 #include "../../Include/simple_hash.h"
   10 
   11 int main(int argc, char *argv[]) {
   12     int i, n = ((argc == 2) ? atoi(argv[1]) : 1);
   13     char buf[32];
   14     struct ht_ht *ht1 = ht_create(10000);
   15     struct ht_ht *ht2 = ht_create(10000);
   16     struct ht_node *node;
   17 
   18     for (i=0; i<=9999; ++i) {
   19     sprintf(buf, "foo_%d", i);
   20     ht_find_new(ht1, buf)->val = i;
   21     }
   22 
   23     for (i=0; i<n; ++i) {
   24     for (node=ht_first(ht1); node; node=ht_next(ht1)) {
   25         ht_find_new(ht2, node->key)->val += node->val;
   26     }
   27     }
   28 
   29     printf("%d %d %d %d\n",
   30        (ht_find(ht1, "foo_1"))->val,
   31        (ht_find(ht1, "foo_9999"))->val,
   32        (ht_find(ht2, "foo_1"))->val,
   33        (ht_find(ht2, "foo_9999"))->val);
   34 
   35     ht_destroy(ht1);
   36     ht_destroy(ht2);
   37     return(0);
   38 }
