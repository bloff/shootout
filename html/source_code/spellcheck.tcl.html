    7 proc main {} {
    8     set 1 [open "Usr.Dict.Words" r]
    9     foreach  2 [read $1 [file size "Usr.Dict.Words"]] {
   10     set $2 1
   11     }
   12     close $1
   13 
   14     fconfigure stdout -buffering full
   15     while {[gets stdin 1] >= 0} {
   16         if {[catch {set $1}]} {
   17             puts $1
   18         }
   19     }
   20 }
   21 
   22 main
