    5 procedure hexstring(i)
    6     local s
    7     if i = 0 then s := "0"
    8     else {
    9     s := ""
   10     while i ~= 0 do {
   11         s := "0123456789abcdef"[iand(i,15) + 1] || s
   12         i := ishift(i,-4)
   13     }
   14     }
   15     return s
   16 end
   17 
   18 procedure main(argv)
   19     local n
   20     local X
   21     local c
   22     local i
   23     n := argv[1] | 1
   24     X := table(0)
   25     c := 0
   26     every i := 1 to n do {
   27     X[hexstring(i)] := i
   28     }
   29     every i := n to 1 by -1 do {
   30     if (member(X,string(i))) then { c +:= 1 }
   31     }
   32     write(c)
   33 end
