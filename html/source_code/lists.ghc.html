    5 module Main(main) where

    7 import Prelude
    8 
    9 copy [] = []
   10 copy (x:xs) = x:copy xs
   11 -- to be honest, in pure functional language the RIGHT
   12 -- thing is copy list = list, because it's not mutable anyway
   13 -- (and the price is paid when doing reverse or (++) anyway)
   14 
   15 -- `seq`s below force evaluation of isok1 and isok2
   16 test :: Int -> Int
   17 test size = isok1 `seq` length l3
   18   where single x = [x]
   19         l1 = [1..size]
   20         l2 = copy l1 -- Should be just: "l1"
   21         l3 = foldl (++) [] (map single l2)
   22         l2' = foldr (++) [] (map single l3)
   23         l1' = reverse l1
   24         isok1 = head l1' == size
   25         isok2 = l1' == l2'
   26 
   27 main = do s <- getLine
   28           putStrLn . show . test . read $ s
