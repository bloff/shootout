    5 (module random
    6     (main main)
    7     (extern (macro printf::int (string double) "printf")))
    8 
    9 (define IM   139968)
   10 (define IA     3877)
   11 (define IC    29573)
   12 
   13 (define LAST 42)
   14 (define (gen_random max)
   15   (set! LAST (modulo (+ (* LAST IA) IC) IM))
   16   (/ (* max LAST) IM))
   17 
   18 (define (heapsort n ra)
   19   (let ((ir n)
   20     (l (+ (bit-rsh n 1) 1))
   21     (i 0)
   22     (j 0)
   23     (rra 0.0))
   24     (bind-exit (return)
   25            (do ((bar #t))
   26            ((= 1 0))
   27          (cond ((> l 1)
   28             (set! l (- l 1))
   29             (set! rra (vector-ref ra l)))
   30                (else
   31             (set! rra (vector-ref ra ir))
   32             (vector-set! ra ir (vector-ref ra 1))
   33             (set! ir (- ir 1))
   34             (cond ((= ir 1)
   35                    (vector-set! ra 1 rra)
   36                    (return #t)))))
   37          (set! i l)
   38          (set! j (bit-lsh l 1))
   39          (do ((foo #t))
   40              ((> j ir))
   41            (cond ((and (< j ir) (< (vector-ref ra j) (vector-ref ra (+ j 1))))
   42               (set! j (+ j 1))))
   43            (cond ((< rra (vector-ref ra j))
   44               (vector-set! ra i (vector-ref ra j))
   45               (set! i j)
   46               (set! j (+ j i)))
   47              (else
   48               (set! j (+ ir 1)))))
   49          (vector-set! ra i rra)))))
   50 
   51 (define (main args)
   52   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   53      (last (+ n 1))
   54      (ary (make-vector last 0)))
   55     (do ((i 1 (+ i 1)))
   56     ((= i last))
   57       (vector-set! ary i (gen_random 1.0)))
   58     (heapsort n ary)
   59     (printf #"%.10f\n" (vector-ref ary n))))
