    2 # $Id: random.perl.html,v 1.1 2004-06-14 23:09:28 bfulgham Exp $

    4 
    5 use strict;
    6 
    7 use constant IM => 139968;
    8 use constant IA => 3877;
    9 use constant IC => 29573;
   10 
   11 my $LAST = 42;
   12 sub gen_random { ($_[0] * ($LAST = ($LAST * IA + IC) % IM)) / IM }
   13 
   14 my $N = ($ARGV[0] || 1) - 1;
   15 gen_random(100.0) while ($N--);
   16 printf "%.9f\n", gen_random(100.0);
