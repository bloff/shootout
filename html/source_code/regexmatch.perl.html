    5 use strict;
    6 
    7 my $re = qr{
    8     (?: ^ | [^\d\(])        # must be preceeded by non-digit
    9     ( \(  )?            # match 1: possible initial left paren
   10     (\d\d\d)            # match 2: area code is 3 digits
   11     (?(1) \)  )            # if match1 then match right paren
   12     [ ]                # area code followed by one space
   13     (\d\d\d)            # match 3: prefix of 3 digits
   14     [ -]            # separator is either space or dash
   15     (\d\d\d\d)            # match 4: last 4 digits
   16     \D    	    	    	    	# must be followed by a non-digit
   17 }x;
   18 
   19 my $NUM = $ARGV[0];
   20 $NUM = 1 if ($NUM < 1);
   21 
   22 my @phones = <STDIN>;
   23 my $count = 0;
   24 my $num;
   25 while ($NUM--) {
   26     foreach (@phones) {
   27     if (/$re/o) {
   28         $num = "($2) $3-$4";
   29         if (0 == $NUM) {
   30         $count++;
   31         print "$count: $num\n";
   32         }
   33     }
   34     }
   35 }
