    6 wordlist constant word-counts
    7 create word-pointers 10000 cells allot
    8 variable endwp word-pointers endwp !
    9 1024 constant max-line
   10 create line max-line 2 + allot
   11 
   12 struct
   13     cell% field    wf-count
   14     cell% 2* field wf-name
   15 end-struct wf%
   16 
   17 : count-word ( addr u -- )
   18     2dup word-counts search-wordlist if
   19     1 swap >body +! 2drop
   20     else
   21     nextname get-current word-counts set-current create set-current
   22     here endwp @ tuck ! cell+ endwp !
   23     1 , last @ name>string 2,
   24     endif ;
   25 
   26 : no-letter? ( c -- )
   27     dup 'a < swap 'z > or ;
   28 
   29 : process-word ( start end -- )
   30     2dup u< if
   31     over - count-word
   32     else
   33     2drop
   34     endif ;
   35 
   36 : process-line ( addr u -- )
   37     bounds 2dup ?do ( end wordstart )
   38     i c@ $20 or dup i c! ( end wordstart c )
   39     no-letter? if ( end wordstart )
   40         i process-word  i 1+
   41     endif
   42     loop ( end wordstart )
   43     swap process-word ;
   44 
   45 : process-file ( fid -- )
   46     >r begin
   47     line max-line r@ read-line throw
   48     while
   49     line swap process-line
   50     repeat
   51     rdrop ;
   52 
   53 : output ( -- )
   54     endwp @ word-pointers ?do
   55     i @ dup wf-count @ 7 .r #tab emit wf-name 2@ type cr
   56     cell +loop ;
   57 
   58 : wf< ( wf1 wf2 -- f )
   59     over wf-count @ over wf-count @ 2dup = if
   60     2drop >r wf-name 2@ r> wf-name 2@ compare 0>
   61     else
   62     u> nip nip
   63     endif ;
   64 
   65 (field) cell- -1 cells ,
   66 
   67 : partition ( first last lessthan-xt -- middle )
   68     \ partition array addr1 u1 into all elements less than pivot and all
   69     \ others, addr1 u2 and addr3 u3 are the two partitions.
   70     \ lessthan-xt ( elemptr1 elemptr2 -- f ) compares the two elements
   71     { lessthan-xt }
   72     over @ { pivot }
   73     begin ( i j )
   74     2dup u<
   75     while
   76     begin
   77         pivot over @ lessthan-xt execute
   78     while
   79         cell-
   80     repeat
   81     swap over @ over !
   82     begin ( j i )
   83         2dup u>
   84     while
   85         pivot over @ lessthan-xt execute 0=
   86     while
   87         cell+
   88     repeat then
   89     swap over @ over !
   90     repeat
   91     drop pivot over ! ;
   92 
   93 : sort1 ( first last lesstthan-xt -- ) recursive
   94     >r begin
   95     2dup u<
   96     while
   97     2dup r@ partition ( first last middle )
   98     rot over cell- r@ sort1
   99     cell+ swap
  100     repeat
  101     rdrop 2drop ;
  102 
  103 stdin process-file word-pointers endwp @ cell- ' wf< sort1 output bye
