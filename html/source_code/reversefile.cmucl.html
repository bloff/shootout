    6 (declaim (optimize (speed 3) (debug 0) (safety 0) (space 0) (compilation-
    7  speed 0)))
    8 
    9 (defconstant BLOCKSIZE 4096)
   10 
   11 (declaim (inline position-rev))
   12 ;; I wrote the the function below because the default cmucl image
   13 ;; doesn't seem to to have the inline expansion of position available
   14 (defun position-rev (string end char)
   15   (declare (simple-string string) (fixnum end))
   16   (declare (inline char=))
   17   (loop for i from (1- end) downto 0
   18     do (when (char= (aref string i) char)
   19      (return-from position-rev i))))
   20 
   21 (defun main ()
   22   (declare (inline read-sequence write-string position-rev replace))
   23   (let ((fp 0)
   24     (bufsize BLOCKSIZE)
   25     (buf (make-string BLOCKSIZE)))
   26     (declare (fixnum bufsize) (simple-string buf))
   27     (loop for i fixnum = (read-sequence buf *standard-input*
   28     :start fp :end (+ fp BLOCKSIZE))
   29       until (= i fp) do
   30       (setq fp i)
   31       (when (> (+ i BLOCKSIZE) bufsize)
   32     (setq bufsize (* 2 bufsize))
   33     (let ((tmpbuf (make-string bufsize)))
   34       (replace tmpbuf buf :start1 0 :end1 fp :start2 0 :end2 fp)
   35       (setq buf tmpbuf))))
   36     (loop for i = (1- fp) then j as j = (position-rev buf i #\Newline)
   37       do (write-string buf *standard-output* :start (if j (1+ j) 0) :end (
   38           1+ i))
   39       while j)))
