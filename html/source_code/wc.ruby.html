    7 nl = nw = nc = 0
    8 loop do
    9   data = (STDIN.read(4096) or break) << (STDIN.gets || "")
   10   nc += data.length
   11   nl += data.count("\n")
   12   ((data.strip! || data).tr!("\n", " ") || data).squeeze!
   13   nw += data.count(" ") + 1
   14 end
   15 puts "#{nl} #{nw} #{nc}"
