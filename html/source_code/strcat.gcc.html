    6 #include <stdio.h>
    7 #include <stdlib.h>
    8 #include <string.h>
    9 #include <unistd.h>
   10 
   11 #define STUFF "hello\n"
   12 
   13 int
   14 main(int argc, char *argv[]) {
   15     int n = ((argc == 2) ? atoi(argv[1]) : 1);
   16     int i, buflen = 32;
   17     char *strbuf = calloc(sizeof(char), buflen);
   18     char *strend = strbuf;
   19     int stufflen = strlen(STUFF);
   20 
   21     if (!strbuf) { perror("calloc strbuf"); exit(1); }
   22     for (i=0; i<n; i++) {
   23     if (((strbuf+buflen)-strend) < (stufflen+1)) {
   24         buflen = 2*buflen;
   25         strbuf = realloc(strbuf, buflen);
   26         if (!strbuf) { perror("realloc strbuf"); exit(1); }
   27         strend = strbuf + strlen(strbuf);
   28     }
   29     /* much faster to strcat to strend than to strbuf */
   30     strcat(strend, STUFF);
   31     strend += stufflen;
   32     }
   33     fprintf(stdout, "%d\n", strlen(strbuf));
   34     free(strbuf);
   35 
   36     sleep(1);
   37     return(0);
   38 }
