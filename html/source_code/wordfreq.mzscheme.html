1 #!/usr/bin/mzscheme -r
    2 
    3 ;
    4 ;  wordfreq.mzscheme by Grzegorz Chrupaa
    5 ;
    6 
    7 (require (lib "list.ss")
    8          (lib "string.ss"))
    9 
   10 (define port-for-each
   11   (lambda (processor reader)
   12     (let loop ((next (reader)))
   13       (unless (eof-object? next)
   14         (processor next)
   15         (loop (reader))))))
   16 
   17 (define (hash-table->alist table)
   18   (let ((alist '()))
   19     (hash-table-for-each
   20      table
   21      (lambda (key value) (set! alist (cons (cons key value) alist))))
   22     alist))
   23 
   24 
   25 (define (main . _)
   26   (define table (make-hash-table 'equal))
   27   (port-for-each (lambda (line) (parse-line line table)) read-line)
   28   (hash-table-remove! table "")
   29   (for-each (lambda (pair)
   30               (let ((word (car pair))
   31                     (count (cdr pair)))
   32                 (printf "~a~a~a~n" count #\tab word)))
   33             (quicksort (hash-table->alist table)
   34                   (lambda (pair1 pair2)
   35                     (or (> (cdr pair1) (cdr pair2))
   36                         (and (= (cdr pair1) (cdr pair2))
   37                              (string<? (car pair2) (car pair1))))))))
   38 
   39 (define rx (regexp "[^a-zA-Z]+"))
   40 
   41 (define (parse-line line table)
   42   (string-lowercase! line)
   43   (for-each
   44    (lambda (word)
   45      (let ((count (hash-table-get table word (lambda () 0))))
   46        (hash-table-put! table word (+ count 1))))
   47    (regexp-split rx line)))
   48 
   49 (main)
