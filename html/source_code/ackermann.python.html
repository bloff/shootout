    6 import sys
    7 
    8 def Ack(M, N):
    9     if (not M):
   10         return( N + 1 )
   11     if (not N):
   12         return( Ack(M-1, 1) )
   13     return( Ack(M-1, Ack(M, N-1)) )
   14 
   15 def main():
   16     NUM = int(sys.argv[1])
   17     sys.setrecursionlimit(3000)
   18     print "Ack(3,%d): %d" % (NUM, Ack(3, NUM))
   19
