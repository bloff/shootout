    6 (module lists
    7     (option (set! *genericity* #f))
    8     (main main))
    9 
   10 (define SIZE 10000)
   11 
   12 (define (test-lists)
   13   (let* ((l1 (integer-list 1 SIZE))
   14      (l2 (list-copy l1)))
   15     ;; move-left-to-right from l2 to l3
   16     (let ((l3 '()))
   17       (let iter ((l3-run l3))
   18     (cond ((pair? l2)
   19            (cond ((pair? l3-run)    ; not the first element moved
   20               (let ((l3-run-new (list (car l2))))
   21             (set! l2 (cdr l2))
   22             (set-cdr! l3-run l3-run-new)
   23             (iter l3-run-new)))
   24              (else        ; the first element moved
   25               (set! l3 (list (car l2)))
   26               (set! l2 (cdr l2))
   27               (iter l3))))))
   28       ;; move-right-to-left from l3 to l2
   29       (let ((l2-run #f))        ; the last pair of l2 during construction
   30     (let move-last ((l3-run l3)
   31             (l3-last-pair #f))
   32       (cond ((pair? l3-run)        ; last element of l3 not reached
   33          (let ((l3-last-element (car l3-run))) ; store current element
   34            (set! l2-run (move-last (cdr l3-run) l3-run))
   35            (let ((l2-run-new (list l3-last-element)))
   36              (cond (l2-run    ; not the first element moved
   37                 (set-cdr! l2-run l2-run-new)
   38                 (cond ((pair? l3-last-pair)
   39                    (set-cdr! l3-last-pair '()))
   40                   (else
   41                    (set! l3 '()))))
   42                (else    ; the first element moved
   43                 (set! l2 l2-run-new)))
   44              l2-run-new)))
   45         (else
   46          #f))))
   47       (set! l1 (reverse! l1))
   48       (cond ((every-2? = l1 l2)
   49          (length l1))
   50         (else
   51          #f)))))
   52 
   53 (define (main args)
   54   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   55     (let (result)
   56       (do ((i 0 (+ i 1)))
   57       ((= i n))
   58     (set! result (test-lists)))
   59       (display result)
   60       (newline))))
   61 
   62 (define every-2? (lambda (test l1 l2)
   63   (or (null? l1)
   64       (and (test (car l1) (car l2))
   65            (every-2? test (cdr l1) (cdr l2))))))
   66 
   67 (define integer-list (lambda (i j)
   68   (if (> i j)
   69       '()
   70       (cons i (integer-list (+ i 1) j)))))
   71 
   72 (define list-copy (lambda (l)
   73   (append l '())))
