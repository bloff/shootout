    5 local n = tonumber((arg and arg[1]) or 1)
    6 
    7 local size = 30
    8 
    9 function mkmatrix(rows, cols)
   10     local count = 1
   11     local mx = {}
   12     for i=0,(rows - 1) do
   13     local row = {}
   14     for j=0,(cols - 1) do
   15         row[j] = count
   16         count = count + 1
   17     end
   18     mx[i] = row
   19     end
   20     return(mx)
   21 end
   22 
   23 function mmult(rows, cols, m1, m2)
   24     local m3 = {}
   25     for i=0,(rows-1) do
   26         m3[i] = {}
   27         local m1_i = m1[i]              -- "cache" m1[i]
   28         for j=0,(cols-1) do
   29             local rowj = 0
   30             for k=0,(cols-1) do
   31                 rowj = rowj + m1_i[k] * m2[k][j]
   32             end
   33             m3[i][j] = rowj
   34         end
   35     end
   36     return(m3)
   37 end
   38 
   39 local m1 = mkmatrix(size, size)
   40 local m2 = mkmatrix(size, size)
   41 for i=1,n do
   42     mm = mmult(size, size, m1, m2)
   43 end
   44 io.write(format("%d %d %d %d\n", mm[0][0], mm[2][3], mm[3][2], mm[4][4]))
