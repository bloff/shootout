    5 ;;; $Id: hash.guile.html,v 1.1 2004-06-14 23:09:26 bfulgham Exp $
    6 ;;; http://www.bagley.org/~doug/shootout/
    7 
    8 (define (main args)
    9   (use-modules (ice-9 format))
   10   (let* ((n (or (and (= (length args) 2) (string->number (cadr args))) 1))
   11      (last (- n 1))
   12      (c 0)
   13      (x (make-hash-table n)))
   14     (do ((i 1 (+ i 1)))
   15     ((> i n))
   16       (hash-set! x (number->string i 16) i))
   17     (do ((i last (- i 1)))
   18     ((< i 0))
   19       (if (hash-ref x (number->string i 10))
   20       (set! c (+ c 1))))
   21     (display (format "~D\n" c))))
