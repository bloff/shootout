    5 import java.io.*;
    6 import java.util.*;
    7 import java.text.*;
    8 
    9 class Toggle {
   10     boolean state = true;
   11     public Toggle(boolean start_state) {
   12     this.state = start_state;
   13     }
   14     public boolean value() {
   15     return(this.state);
   16     }
   17     public Toggle activate() {
   18     this.state = !this.state;
   19     return(this);
   20     }
   21 }
   22 
   23 class NthToggle extends Toggle {
   24     int count_max = 0;
   25     int counter = 0;
   26 
   27     public NthToggle(boolean start_state, int max_counter) {
   28     super(start_state);
   29     this.count_max = max_counter;
   30     this.counter = 0;
   31     }
   32     public Toggle activate() {
   33     this.counter += 1;
   34     if (this.counter >= this.count_max) {
   35         this.state = !this.state;
   36         this.counter = 0;
   37     }
   38     return(this);
   39     }
   40 }
   41 
   42 public class methcall {
   43     public static void main(String args[]) throws IOException {
   44     int n = Integer.parseInt(args[0]);
   45 
   46     boolean val = true;
   47     Toggle toggle = new Toggle(val);
   48     for (int i=0; i<n; i++) {
   49         val = toggle.activate().value();
   50     }
   51     System.out.println((val) ? "true" : "false");
   52 
   53     val = true;
   54     NthToggle ntoggle = new NthToggle(true, 3);
   55     for (int i=0; i<n; i++) {
   56         val = ntoggle.activate().value();
   57     }
   58     System.out.println((val) ? "true" : "false");
   59     }
   60 }
