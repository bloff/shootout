    4 BEGIN {
    5     delete ARGV;
    6     sum = 0;
    7     n = 0;
    8 }
    9 
   10 {
   11     nums[n++] = $1;
   12     sum += $1;
   13 }
   14 
   15 END {
   16     mean = sum/n;
   17     for (num in nums) {
   18     dev = num - mean;
   19     if (dev > 0) { avg_dev += dev; } else { avg_dev -= dev; }
   20     vari += dev^2;
   21     skew += dev^3;
   22     kurt += dev^4;
   23     }
   24     avg_dev /= n;
   25     vari /= (n - 1);
   26     std_dev = sqrt(vari);
   27 
   28     if (vari > 0) {
   29     skew /= (n * vari * std_dev);
   30     kurt = kurt/(n * vari * vari) - 3.0;
   31     }
   32 
   33     nums[n] = nums[0];
   34     heapsort(n, nums);
   35 
   36     mid = int(n/2)+1;
   37     median = (n % 2) ? nums[mid] : (nums[mid] + nums[mid-1])/2;
   38 
   39     printf("n:                  %d\n", n);
   40     printf("median:             %f\n", median);
   41     printf("mean:               %f\n", mean);
   42     printf("average_deviation:  %f\n", avg_dev);
   43     printf("standard_deviation: %f\n", std_dev);
   44     printf("variance:           %f\n", vari);
   45     printf("skew:               %f\n", skew);
   46     printf("kurtosis:           %f\n", kurt);
   47 }
   48 
   49 function heapsort (n, ra) {
   50     l = int(0.5+n/2) + 1
   51     ir = n;
   52     for (;;) {
   53         if (l > 1) {
   54             rra = ra[--l];
   55         } else {
   56             rra = ra[ir];
   57             ra[ir] = ra[1];
   58             if (--ir == 1) {
   59                 ra[1] = rra;
   60                 return;
   61             }
   62         }
   63         i = l;
   64         j = l * 2;
   65         while (j <= ir) {
   66             if (j < ir && ra[j] < ra[j+1]) { ++j; }
   67             if (rra < ra[j]) {
   68                 ra[i] = ra[j];
   69                 j += (i = j);
   70             } else {
   71                 j = ir + 1;
   72             }
   73         }
   74         ra[i] = rra;
   75     }
   76 }
