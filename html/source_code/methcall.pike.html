    6 class Toggle {
    7     int bool;
    8 
    9     object create (int start_state) {
   10     bool = start_state;
   11     }
   12 
   13     int value () {
   14     return(bool);
   15     }
   16 
   17     object activate () {
   18     bool = !bool;
   19     return(this_object());
   20     }
   21 }
   22 
   23 class NthToggle {
   24     inherit Toggle;
   25 
   26     int count_max, count;
   27 
   28     object create (int start_state, int max_counter) {
   29     ::create(start_state);
   30     count_max = max_counter;
   31     count = 0;
   32     }
   33 
   34     object activate () {
   35     if (++count >= count_max) {
   36         bool = !bool;
   37         count = 0;
   38     }
   39     return(this_object());
   40     }
   41 }
   42 
   43 void main(int argc, array(string) argv) {
   44     int n = (int)argv[-1];
   45     if (n < 1) n = 1;
   46 
   47     int val = 1;
   48     object toggle = Toggle(val);
   49     for (int i=0; i<n; i++) {
   50     val = toggle->activate()->value();
   51     }
   52     write((toggle->value()) ? "true\n" : "false\n");
   53 
   54     val = 1;
   55     object ntoggle = NthToggle(val, 3);
   56     for (int i=0; i<n; i++) {
   57     val = ntoggle->activate()->value();
   58     }
   59     write((ntoggle->value()) ? "true\n" : "false\n");
   60 }
