    6 (proclaim '(optimize (speed 3) (space 0) (compilation-speed 0) (debug 0) (
    7  safety 0)))

    9 (defun matmul (a b c n m k)
   10   (declare (optimize (speed 3) (safety 0) (debug 0))
   11            (type (simple-array (unsigned-byte 32) (*)) a b c)
   12            (fixnum n m k))
   13   (let ((sum 0)
   14         (i1 (- m))
   15         (k2 0))
   16     (declare (type (unsigned-byte 32) sum) (type fixnum i1 k2))
   17     (dotimes (i n c)
   18       (declare (fixnum i))
   19       (setf i1 (+ i1 m)) ;; i1=i*m
   20       (dotimes (j k)
   21         (declare (fixnum j))
   22         (setf sum 0)
   23         (setf k2 (- k))
   24         (dotimes (l m)
   25           (declare (fixnum l))
   26           (setf k2 (+ k2 k)) ;; k2= l*k
   27           (setf sum (the (unsigned-byte 32) (+ (the (unsigned-byte 32) sum)
   28                                                (the (unsigned-byte 32) (* (
   29                                                 aref a (+ i1 l))
   30 
   31 aref b (+ k2 j))))))))
   32         (setf (aref c (+ i1 j)) sum)))))

   34 (defun main ()
   35   (let ((n (parse-integer (or (car (last extensions:*command-line-strings*)
   36    ) "1"))))
   37     (declare (fixnum n)
   38          (optimize (speed 3) (debug 0) (safety 0)))
   39     (let* ((m1 (make-matrix 30 30))
   40        (m2 (make-matrix 30 30))
   41        (m3 (make-matrix 30 30))
   42        (mm (make-array '(30 30) :element-type '(unsigned-byte 32) :displaced-
   43      to m3)))
   44       (loop repeat n do (matmul m1 m2 m3 30 30 30))
   45       (format t "~A ~A ~A ~A~%"
   46           (aref mm 0 0) (aref mm 2 3) (aref mm 3 2) (aref mm 4 4)))))

   48 (defun make-matrix (rows cols)
   49   (declare (type (unsigned-byte 32) rows cols)
   50            (optimize (speed 3) (safety 0))); (hcl:fixnum-safety 0)))
   51   (let* ((space (* rows cols))
   52          (matrix (make-array space
   53                              :element-type '(unsigned-byte 32))))
   54     (declare (type (simple-array (unsigned-byte 32) (*)) matrix)
   55              (fixnum space))
   56     (loop :for i :of-type fixnum :from 0 :below space
   57           :do (setf (aref matrix i) (1+ i)))
   58     matrix))
