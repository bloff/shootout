   10 (defun main ()
   11   (let ((n (parse-integer (or (car (last extensions:*command-line-strings*)
   12    ) "1"))))
   13     (format t "~A~%" (length (string-concat1 n)))))

   15 (defun string-concat1 (n)
   16   (declare (fixnum n))
   17   (let ((str "")
   18         (used-len 0)
   19         (string-leng 0)
   20         (i (1+ n)))
   21     (declare (fixnum i used-len string-leng))
   22     (declare (simple-base-string str))
   23     (declare (optimize (speed 3) (debug 0) (safety 0)))
   24     (dotimes (i (1- i) (replace (make-string used-len) str))
   25       (let ((required-length (+ used-len (length *string*))))
   26         (if (eq string-leng 0)
   27             (setq str (make-string required-length)
   28                   string-leng required-length)
   29           (if (> required-length string-leng)
   30               (let ((new-len (+ string-leng string-leng)))
   31                 (let ((new-str (make-string new-len)))
   32                   (replace new-str str :end2 used-len)
   33                   (setq str new-str string-leng new-len)))))
   34         (replace str *string* :start1 used-len)
   35         (setq used-len required-length)))))
