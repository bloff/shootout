   10 functor
   11 import System Application
   12 
   13 define
   14 
   15    fun {IntToHexString I Hex}
   16       if I =< 0 then Hex else
   17      local M D in
   18         D = I div 16
   19         M = I mod 16
   20         if M < 10 then
   21            {IntToHexString D M+&0|Hex}
   22         else
   23            {IntToHexString D (M-10)+&a|Hex}
   24         end
   25      end
   26       end
   27    end
   28 
   29    proc {InsertHexKeys H N}
   30       for I in 0..N-1 do
   31     {Dictionary.put H {String.toAtom {IntToHexString I nil}} I}
   32       end
   33    end
   34 
   35    proc {CountLookups H I S C}
   36       if I >= 0 then
   37      if {Dictionary.member H {String.toAtom {IntToString I}}} then
   38         {CountLookups H I-1 S+1 C}
   39      else
   40         {CountLookups H I-1 S C}
   41      end
   42       else C = S end
   43    end
   44 
   45 in
   46    local Args N H Count in
   47       [Args] = {Application.getArgs plain}
   48       N = {String.toInt Args}
   49 
   50       {NewDictionary H}
   51       {InsertHexKeys H N}
   52       {CountLookups H N 0 Count}
   53 
   54       {System.showInfo Count}
   55    end
   56    {Application.exit 0}
   57 end
