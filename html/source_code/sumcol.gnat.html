    5 with Text_IO, Ada.Strings.Fixed, Ada.IO_Exceptions;
    6 
    7 procedure SumCol is
    8    package IE renames Ada.IO_Exceptions;
    9    package AS renames Ada.Strings;
   10 
   11    function Read_Line return String is
   12       Buf      : String (1 .. 4096);
   13       Last     : Natural;
   14    begin    --  (End_Error here (in ObjectAda) on unusual missing final
   15                  "\n")
   16       Text_IO.Get_Line (Text_IO.Standard_Input, Item => Buf, Last => Last);
   17       if Last < Buf'Last or else
   18                      Text_IO.End_Of_File (Text_IO.Standard_Input) then
   19          return Buf (1 .. Last);
   20       else
   21          return Buf & Read_Line;
   22       end if;
   23    end Read_Line;
   24 
   25    Sum         : Integer := 0;
   26 begin
   27    while not Text_IO.End_Of_File loop
   28       declare
   29          Line     : String := AS.Fixed.Trim (Read_Line, Side => AS.Right);
   30       begin
   31          Sum := Sum + Integer'Value (Line);  --  Fail 0 or 2 numbers per
   32                 line
   33       end;
   34    end loop;
   35    Text_IO.Put_Line (AS.Fixed.Trim (Natural'Image (Sum), Side => AS.Left));
   36 exception                        --  Catch error from Get_Line
   37 Integer'Value
   38    when Constraint_Error | IE.Device_Error | IE.End_Error =>
   39       Text_IO.Put_Line ("> Error near line" &
   40                Text_IO.Count'Image (Text_IO.Line));
   41 end SumCol;
