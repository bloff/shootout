    5 local IM = 139968
    6 local IA = 3877
    7 local IC = 29573
    8 
    9 local LAST = 42
   10 function gen_random(max)
   11     LAST = math.mod((LAST * IA + IC), IM)
   12     return( (max * LAST) / IM )
   13 end
   14 
   15 local N = tonumber((arg and arg[1])) or 1
   16 local result = 0
   17 for i=1, N do
   18     result = gen_random(100)
   19 end
   20 io.write(string.format("%.9f\n", result))
