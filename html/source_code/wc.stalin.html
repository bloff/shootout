    5 (include "QobiScheme")

    7 ; x is a character
    8 ; nl is number of newlines
    9 ; nw is number of words
   10 ; nc in number of chsracters
   11 
   12 (define wcport
   13   (lambda (port)
   14     (letrec ((wc (lambda (x nl nw nc inword)
   15            (if (eof-object? x)
   16                (begin (display nl) (display " ") (display nw)
   17                   (display " ") (display nc) (newline))
   18                (let ((new-nl (if (char=? x #\newline) (+ nl 1) nl)))
   19              (if (char-whitespace? x)
   20                  (wc (read-char port) new-nl (if inword (+ nw 1) nw) (+ nc 1) #f)
   21                  (wc (read-char port) new-nl nw (+ nc 1) #t)))))))
   22       (wc (read-char port) 0 0 0 #f))))
   23 
   24 (wcport (current-input-port))
