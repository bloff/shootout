    5 class SUMCOL
    6 
    7 creation make
    8 
    9 feature
   10 
   11    make is
   12 
   13       local
   14      sum: INTEGER;
   15       do
   16      sum := 0
   17      from
   18         io.read_line
   19      until
   20         io.end_of_input
   21      loop
   22         sum := sum + io.last_string.to_integer
   23         io.read_line
   24      end
   25          std_output.put_integer(sum)
   26          std_output.put_character('%N')
   27       end
   28 end
