   16 #define QUICKIE_STRCMP(a, b)  (*(a) != *(b) ? *(a) - *(b) : strcmp((a), (b)
   17                        ))
   18 
   19 typedef int (*comparator)(const void *, const void *);
   20 
   21 static int cmp_hash(struct ht_node **a, struct ht_node **b) {
   22     int val = (*b)->val - (*a)->val;
   23     return((val == 0) ? QUICKIE_STRCMP((*b)->key, (*a)->key) : val);
   24 }
   25 
   26 int main() {
   27     unsigned long  hash_code=0;
   28     int readbufsize = 4096;
   29     int wordbufsize=16;
   30     char *readbuf = (char *)malloc(readbufsize + 1);
   31     char *wordbuf = (char *)malloc(wordbufsize + 1);
   32     int i = 0;
   33     struct ht_ht *ht = ht_create(2048);
   34     struct ht_node **sort_array, **sort_tmp, *node;
   35     /*new code*/
   36     int nread =0;
   37     int wordlen=0;
   38     readbuf[0]=0;
   39     while (readbuf[i] > 0 ||
   40        (nread = fread(readbuf, sizeof(char), readbufsize, stdin),
   41         readbuf[nread] = '\0',i=0,nread > 0) ) {
   42     if (isalpha(readbuf[i])){
   43         wordbuf[wordlen] = tolower(readbuf[i]);
   44         /*calculate the hash code byte by byte, to prevent looping through
   45         the word twice */
   46         HASHCODE_BYTE(hash_code,wordbuf[wordlen++]);
   47         if (wordlen == wordbufsize) {
   48         wordbufsize *= 2;
   49         wordbuf = realloc(wordbuf, wordbufsize + 1);
   50         }
   51     }
   52     else{
   53         if (wordlen > 0) {
   54         wordbuf[wordlen] = '\0';
   55         hash_code=HASHCODE_FINALIZE(hash_code,ht);
   56         ++(ht_find_new_prehashed(ht, wordbuf,hash_code)->val);
   57         wordlen = 0;
   58         hash_code=0;
   59         }
   60     }
   61     i++;
   62     }
   63     free(readbuf);
   64     free(wordbuf);
   65     sort_array = sort_tmp =
   66     malloc(sizeof(struct ht_node *) * ht_count(ht));
   67 
   68     for (node=ht_first(ht); (*sort_tmp++ = node) != 0; node=ht_next(ht)) ;
   69 
   70     qsort(sort_array, ht_count(ht), sizeof(struct ht_node *),
   71       (comparator)cmp_hash);
   72 
   73     for (i=0; i<ht_count(ht); i++)
   74     printf("%7d\t%s\n", ht_val(sort_array[i]), ht_key(sort_array[i]));
   75 
   76     ht_destroy(ht);
   77     return(0);
   78 }
