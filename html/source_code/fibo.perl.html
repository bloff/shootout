    5 use strict;
    6 use integer;
    7 
    8 # from Leif Stensson
    9 sub fib {
   10     return $_[0] < 2 ? 1 : fib($_[0]-2) + fib($_[0]-1);
   11 }
   12 
   13 my $N = ($ARGV[0] < 1) ? 1 : $ARGV[0];
   14 my $fib = fib($N);
   15 print "$fib\n";
