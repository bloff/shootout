    8 (defun fib (n)
    9   (cond ((< n 2) 1)
   10     (t (+ (fib (- n 2)) (fib (- n 1))))))
   11 
   12 (defun main ()
   13   (let* ((arg (car command-line-args-left))
   14      (num (if arg (string-to-number arg) 1)))
   15     (princ (format "%d\n" (fib num)))))
