    5 class NESTEDLOOP
    6 
    7 creation make
    8 
    9 feature
   10 
   11    make is
   12 
   13       local
   14      n,a,b,c,d,e,f,x: INTEGER;
   15       do
   16      if argument_count = 1 then
   17         n := argument(1).to_integer
   18      else
   19         n := 1
   20      end
   21 
   22      x := 0
   23      from
   24         a := 0
   25      until
   26         a = n
   27      loop
   28         from
   29            b := 0
   30         until
   31            b = n
   32         loop
   33            from
   34           c := 0
   35            until
   36           c = n
   37            loop
   38           from
   39              d := 0
   40           until
   41              d = n
   42           loop
   43              from
   44             e := 0
   45              until
   46             e = n
   47              loop
   48             from
   49                f := 0
   50             until
   51                f = n
   52             loop
   53                x := x + 1
   54                f := f + 1
   55             end
   56             e := e + 1
   57              end
   58              d := d + 1
   59           end
   60           c := c + 1
   61            end
   62            b := b + 1
   63         end
   64         a := a + 1
   65      end
   66 
   67          std_output.put_integer(x)
   68          std_output.put_character('%N')
   69       end;
   70 
   71 end
