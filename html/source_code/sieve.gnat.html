    5 with Text_IO, Ada.Command_Line;
    6 
    7 procedure Sieve is
    8    High        : constant := 8192;
    9    Is_Prime    : array (2 .. High) of Boolean;
   10    Count, K, N : Natural := 0;
   11 begin
   12    begin
   13       N := Natural'Value (Ada.Command_Line.Argument (1));
   14    exception
   15       when Constraint_Error => null;
   16    end;
   17    for Iter in 1 .. N loop
   18       declare
   19          pragma Suppress (Overflow_Check);
   20          pragma Suppress (Index_Check);
   21          pragma Suppress (Range_Check);
   22       begin
   23          Count := 0;
   24          Is_Prime := (others => True);
   25          for J in Is_Prime'Range loop
   26             if Is_Prime (J) then
   27                K := J + J;
   28                while K <= Is_Prime'Last loop
   29                   Is_Prime (K) := False;        --  K is not a prime since
   30                             a
   31                   K := K + J;                   --  multiple of prime J
   32                end loop;
   33                Count := Count + 1;
   34             end if;
   35          end loop;
   36       end;
   37    end loop;
   38    Text_IO.Put_Line ("Count:" & Natural'Image (Count));
   39 end Sieve;
