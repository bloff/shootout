    6 (defun fast-ack (m n)
    7   (declare (fixnum n m) (optimize (speed 3) (debug 0) (safety 0)))
    8   (the fixnum
    9     (cond
   10      ((zerop m) (the fixnum (1+ n)))
   11      ((zerop n) (the fixnum (fast-ack (1- m) 1)))
   12      (t (the fixnum (fast-ack (1- m) (the fixnum (fast-ack m (1- n)))))))))
