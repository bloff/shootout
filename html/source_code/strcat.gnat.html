    5 with Ada.Command_Line; use Ada.Command_Line;
    6 with Ada.Strings.Unbounded; use Ada.Strings.Unbounded;
    7 with Ada.Text_Io; use Ada.Text_Io;
    8 with Ada.Integer_text_Io; use Ada.Integer_Text_Io;
    9 
   10 procedure Strcat is
   11    N: Integer;
   12    Hello : String := "hello" & Ascii.Lf;
   13    Buffer : Unbounded_String := Null_Unbounded_String;
   14 begin
   15    N := Integer'Value(Argument(1));
   16    for Num in 1..N loop
   17       Append(Source => Buffer, New_Item => Hello);
   18    end loop;
   19    Put(Length(Buffer));
   20    New_Line;
   21 end Strcat;
