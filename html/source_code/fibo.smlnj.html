    6 structure Test : sig
    7     val main : (string * string list) -> OS.Process.status
    8 end = struct
    9 
   10 fun fib 0 = 1
   11   | fib 1 = 1
   12   | fib n = fib (n-2) + fib (n-1);
   13 
   14 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   15 
   16 fun main(name, args) =
   17   let
   18     val arg = hd(args @ ["1"]);
   19     val num = atoi arg;
   20     val fib = fib num;
   21     val result = Int.toString fib;
   22   in
   23       print result; print "\n";
   24       OS.Process.success
   25   end
   26 end
   27 
   28 val _ = SMLofNJ.exportFn("fibo", Test.main);
