    5 -- Friedrich Dominicus points out that it is about twice as fast
    6 -- when we substitute NATIVE_ARRAY for ARRAY.  I've commented out
    7 -- my original code below and now use NATIVE_ARRAY.
    8 
    9 class ARY
   10 
   11 creation make
   12 
   13 feature
   14 
   15    make is
   16 
   17       local
   18          x: NATIVE_ARRAY[INTEGER];
   19          y: NATIVE_ARRAY[INTEGER];
   20 --     x: ARRAY[INTEGER];
   21 --     y: ARRAY[INTEGER];
   22      i,k,n: INTEGER;
   23       do
   24      if argument_count = 1 then
   25         n := argument(1).to_integer
   26      else
   27         n := 1
   28      end
   29 
   30          x := x.calloc(n);
   31          y := y.calloc(n);
   32 --     !!x.make(0,n)
   33 --     !!y.make(0,n)
   34 
   35      from
   36         i := 0
   37      until
   38         i = n
   39      loop
   40         x.put(i + 1, i)
   41         i := i + 1
   42      end
   43      from
   44         k := 0
   45      until
   46         k = 1000
   47      loop
   48         from
   49            i := n-1
   50         until
   51            i < 0
   52         loop
   53            y.put(x.item(i) + y.item(i),i)
   54            i := i - 1
   55         end
   56         k := k + 1
   57      end
   58 
   59          std_output.put_integer(y.item(0))
   60          std_output.put_character(' ')
   61          std_output.put_integer(y.item(n-1))
   62          std_output.put_character('%N')
   63       end
   64 end
