    5 (defconst IM     139968)
    6 (defconst IA       3877)
    7 (defconst IC      29573)
    8 
    9 (defvar LAST 42)
   10 
   11 (defun gen_random (max)
   12   (/ (* max (setq LAST (% (+ (* LAST IA) IC) IM))) IM))
   13 
   14 (defun heapsort (n ra)
   15   (let ((ir n)
   16     (l (+ (lsh n -1) 1))
   17     (i 0)
   18     (j 0)
   19     (rra 0.0))
   20     (catch 'return
   21       (while t
   22     (cond ((> l 1)
   23            (setq rra (aref ra (setq l (- l 1)))))
   24           (t
   25            (setq rra (aref ra ir))
   26            (aset ra ir (aref ra 1))
   27            (setq ir (- ir 1))
   28            (if (= ir 1)
   29            (throw 'return (aset ra 1 rra)))))
   30     (setq i l)
   31     (setq j (lsh l 1))
   32     (while (<= j ir)
   33       (cond ((and (< j ir) (< (aref ra j) (aref ra (+ j 1))))
   34          (setq j (+ j 1))))
   35       (cond ((< rra (aref ra j))
   36          (aset ra i (aref ra j))
   37          (setq j (+ j (setq i j))))
   38         (t
   39          (setq j (+ ir 1)))))
   40     (aset ra i rra)))))
   41 
   42 (defun main ()
   43   (let* ((arg (car command-line-args-left))
   44      (n (if arg (string-to-number arg) 1))
   45      (last (+ n 1))
   46      (ary (make-vector last 0)))
   47     (dotimes (i last)
   48       (aset ary i (gen_random 1.0)))
   49     (heapsort n ary)
   50     (princ (format "%.10f\n" (aref ary n)))
   51   (kill-emacs 0)))
