    4 %%% Isaac Gouy suggested the use of a cell
    5 
    6 functor
    7 import System Application
    8 define
    9 
   10     local P X A B C D E F N in
   11         [P] = {Application.getArgs plain}
   12         N = {String.toInt P}
   13         X = {NewCell 0}
   14         for A in 1..N do
   15             for B in 1..N do
   16                 for C in 1..N do
   17                     for D in 1..N do
   18                         for E in 1..N do
   19                             for F in 1..N do
   20                                 {Assign X {Access X}+1}
   21                             end
   22                         end
   23                     end
   24                 end
   25             end
   26         end
   27         {System.printInfo {Access X}}
   28     end
   29     {Application.exit 0}
   30 end
