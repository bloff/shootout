    5 --------------------------------------------------------------
    6 -- List module
    7 -- defines a prototipe for lists
    8 --------------------------------------------------------------
    9 
   10 List = {first = 0, last = -1}
   11 
   12 function List:new ()
   13   local n = {}
   14   for k,v in self do n[k] = v end
   15   return n
   16 end
   17 
   18 function List:length ()
   19   return self.last - self.first + 1
   20 end
   21 
   22 function List:pushleft (value)
   23   local first = self.first - 1
   24   self.first = first
   25   self[first] = value
   26 end
   27 
   28 function List:pushright (value)
   29   local last = self.last + 1
   30   self.last = last
   31   self[last] = value
   32 end
   33 
   34 function List:popleft ()
   35   local first = self.first
   36   if first > self.last then error"list is empty" end
   37   local value = self[first]
   38   self[first] = nil  -- to allow collection
   39   self.first = first+1
   40   return value
   41 end
   42 
   43 function List:popright ()
   44   local last = self.last
   45   if self.first > last then error"list is empty" end
   46   local value = self[last]
   47   self[last] = nil  -- to allow collection
   48   self.last = last-1
   49   return value
   50 end
   51 
   52 function List:reverse ()
   53   local i, j = self.first, self.last
   54   while i<j do
   55     self[i], self[j] = self[j], self[i]
   56     i = i+1
   57     j = j-1
   58   end
   59 end
   60 
   61 function List:equal (otherlist)
   62   if self:length() ~= otherlist:length() then return nil end
   63   local diff = otherlist.first - self.first
   64   for i1=self.first,self.last do
   65     if self[i1] ~= otherlist[i1+diff] then return nil end
   66   end
   67   return 1
   68 end

   75 function test ()
   76   local SIZE = 10000
   77   -- create a list with elements 1..SIZE
   78   local l1 = List:new()
   79   for i=1,SIZE do
   80     l1:pushright(i)
   81   end
   82   -- creates a copy of l1
   83   local l2 = l1:new()
   84   -- remove each individual item from left side of l2 and
   85   -- append to right side of l3 (preserving order)
   86   local l3 = List:new()
   87   while l2:length() > 0 do
   88     l3:pushright(l2:popleft())
   89   end
   90   -- remove each individual item from right side of l3 and
   91   -- append to right side of l2 (reversing list)
   92   while l3:length() > 0 do
   93     l2:pushright(l3:popright())
   94   end
   95   -- reverse l1 in place
   96   l1:reverse()
   97   -- compare Li1 and Li2 for equality
   98   -- and return length of the list
   99   if not l1:equal(l2) then return nil
  100   else return l1:length()
  101   end
  102 end
