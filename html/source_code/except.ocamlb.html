    8 exception HiException of int
    9 exception LoException of int
   10 
   11 let hi = ref 0
   12 let lo = ref 0
   13 
   14 let blowup n =
   15   if n mod 2 = 0 then raise (LoException n)
   16   else raise (HiException n)
   17 
   18 let lo_fun n = try blowup n with LoException _ -> incr lo
   19 let hi_fun n = try lo_fun n with HiException _ -> incr hi
   20 
   21 let some_fun n =
   22   try hi_fun n with exc -> print_endline "Should not get here."; raise exc
   23 
   24 let _ =
   25   let n = if Array.length Sys.argv > 1 then int_of_string Sys.argv.(1)
   26           else 1 in
   27   for i = 1 to n do some_fun i done;
   28   Printf.printf "Exceptions: HI=%d / LO=%d\n" !hi !lo
