    6 structure Test : sig
    7     val main : (string * string list) -> OS.Process.status
    8 end = struct
    9 
   10 fun strcat 1 s0 s = s
   11   | strcat n s0 s = strcat (n-1) s0 (s0 ^ s);
   12 
   13 
   14 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   15 
   16 fun main(name, args) =
   17   let
   18     val arg = hd(args @ ["1"]);
   19     val num = atoi arg;
   20     val str = "hello\n";
   21     val result = String.size (strcat num str str);
   22   in
   23       print (Int.toString result); print "\n";
   24       OS.Process.success
   25   end
   26 end
   27 
   28 val _ = SMLofNJ.exportFn("strcat.smlnj2", Test.main);
