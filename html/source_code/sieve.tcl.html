    6 proc sieve {num} {
    7     while {$num > 0} {
    8     incr num -1
    9     set count 0
   10     for {set i 2} {$i <= 8192} {incr i 1} {
   11         set flags($i) 1
   12     }
   13     for {set i 2} {$i <= 8192} {incr i 1} {
   14         if {$flags($i) == 1} {
   15         # remove all multiples of prime: i
   16         for {set k [expr {$i+$i}]} {$k <= 8192} {incr k $i} {
   17             set flags($k) 0
   18         }
   19         incr count 1
   20         }
   21     }
   22     }
   23     return $count
   24 }
   25 
   26 set NUM [lindex $argv 0]
   27 if {$NUM < 1} {
   28     set NUM 1
   29 }
   30 
   31 set count [sieve $NUM]
   32 puts "Count: $count"
