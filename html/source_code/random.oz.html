    3 functor
    4 import System Application
    5 define
    6 fun {RandLoop N SEED R MAX}
    7     case N
    8     of 0 then R
    9     else
   10         local IA IC IM NEWSEED NEWRAND in
   11             IM = 139968
   12             IA =   3877
   13             IC =  29573
   14             NEWSEED = (SEED * IA + IC) mod IM
   15             NEWRAND = MAX * {Int.toFloat NEWSEED}/{Int.toFloat IM}
   16             {RandLoop N-1 NEWSEED NEWRAND MAX}
   17         end
   18     end
   19 end
   20 in
   21     local A NUM I in
   22         [A] = {Application.getArgs plain}
   23         NUM = {String.toInt A}
   24         {System.printInfo {RandLoop NUM 42 0 100.0}}
   25     end
   26     {Application.exit 0}
   27 end
