    5 with Ada.Command_Line, Ada.Text_IO, Ada.Integer_Text_IO;
    6 
    7 procedure Fibo is
    8    function Fib (N : Natural) return Integer is
    9       L        : Integer := N - 1;
   10       pragma Suppress (Range_Check, On => L);
   11    begin
   12       if L <= 0 then
   13          return 1;
   14       else
   15          return Fib (N - 2) + Fib (L);
   16       end if;
   17    end Fib;
   18    N        : Natural := 0;
   19 begin
   20    begin
   21       N := Natural'Value (Ada.Command_Line.Argument (1));
   22    exception
   23       when Constraint_Error => null;
   24    end;
   25    Ada.Integer_Text_IO.Put (Item => Fib (N), Width => 0);
   26    Ada.Text_IO.New_Line;
   27 end Fibo;
