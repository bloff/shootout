1 #!/usr/bin/guile-oops \
    2 -e main -s
    3 !#
    4 
    5 (use-modules (oop goops))
    6 
    7 (define-method (print-bool (b <boolean>))
    8   (display (if b "true\n" "false\n")))
    9 
   10 
   11 (define-class <toggle> ()
   12   (state #:getter value? #:init-keyword #:state))
   13 
   14 (define-class <nth-toggle> (<toggle>)
   15   (count-max #:init-keyword #:count-max)
   16   (counter #:init-value 0))
   17 
   18 
   19 (define-method (value? (t <toggle>)) (slot-ref t 'state))
   20 
   21 (define-method (activate! (t <toggle>))
   22   (slot-set! t 'state (not (slot-ref t 'state)))
   23   t)
   24 
   25 (define-method (activate! (n-t <nth-toggle>))
   26   (let ((counter (+ 1 (slot-ref n-t 'counter))))
   27     (slot-set! n-t 'counter counter)
   28     (if (>= counter (slot-ref n-t 'count-max))
   29     (begin (slot-set! n-t 'state (not (slot-ref n-t 'state)))
   30            (slot-set! n-t 'counter 0)))
   31     n-t))
   32 
   33 
   34 (define-method (main (args <list>))
   35   (let ((n (or (and (= (length args) 2) (string->number (cadr args))) 1)))
   36     (let ((tog (make <toggle> #:state #t)))
   37       (do ((i 0 (+ i 1))) ((= i 5))
   38     (print-bool (value? (activate! tog)))))
   39     (do ((i 0 (+ i 1))) ((= i n))
   40       (make <toggle> #:state #t))
   41     (newline)
   42     (let ((ntog (make <nth-toggle> #:state #t #:count-max 3)))
   43       (do ((i 0 (+ i 1))) ((= i 8))
   44     (print-bool (value? (activate! ntog)))))
   45     (do ((i 0 (+ i 1))) ((= i n))
   46       (make <nth-toggle> #:state #t #:count-max 3))))
