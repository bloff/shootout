    6 structure Test : sig
    7  val main : (string * string list) -> OS.Process.status
    8 end =
    9  struct
   10    exception Error of string
   11 
   12    val data = "Hello there sailor\n"
   13 
   14    fun mkSocks () = let
   15      val server = INetSock.TCP.socket()
   16      val client = INetSock.TCP.socket()
   17      val _ = Socket.bind(server,INetSock.any 0)
   18      val saddr = INetSock.fromAddr(Socket.Ctl.getSockName server)
   19      val _ = Socket.listen(server,2)
   20      val _ = Socket.connect(client,INetSock.toAddr saddr)
   21      val _ = INetSock.TCP.setNODELAY(server,true)
   22      val _ = INetSock.TCP.setNODELAY(client,true)
   23    in {client=client,server=server}
   24    end
   25 
   26    fun readString (s,n) = let
   27      fun loop(0) = []
   28        | loop(n) = let
   29        val data = Byte.unpackStringVec (Word8VectorSlice.full (Socket.recvVec(
   30                s,n)))
   31        val len = String.size data
   32      in if len = 0 then []
   33         else (data::(loop(n - len)))
   34      end
   35    in String.concat (loop n)
   36    end
   37 
   38    fun writeString (out,str) =
   39      Socket.sendVec(out, Word8VectorSlice.full( Byte.stringToBytes str))
   40 (*     Socket.sendVec(out,{buf=Byte.stringToBytes str,i=0,sz=NONE}) *)
   41 
   42    fun closeSock s =
   43      (Socket.shutdown(s,Socket.NO_RECVS_OR_SENDS);
   44       Socket.close s)
   45 
   46   fun main (_,args) = let
   47     val num =
   48       case args of
   49     nil => 1
   50       | n::_ => valOf (Int.fromString n)
   51     val {client=client_sock,server=server_sock} = mkSocks()
   52     fun server () = let
   53       val (sock,_) = Socket.accept(server_sock)
   54       fun s b =
   55     case readString(sock,19) of
   56        "" => (Posix.Process.wait ();
   57           TextIO.output(TextIO.stdOut,
   58                 concat ["server processed ",
   59                     Int.toString b,
   60                     " bytes\n"]);
   61           TextIO.flushOut(TextIO.stdOut))
   62      | i =>(writeString(sock,i);
   63         s (b + 19))
   64     in s 0
   65     end
   66     fun client () = let
   67       fun c 0 = closeSock(client_sock)
   68     | c n = let
   69         val _ = writeString(client_sock,data);
   70         val reply = readString(client_sock,19)
   71       in if reply = data then c(n - 1)
   72          else raise Error "Didn't receive the same data"
   73       end
   74     in c num
   75     end
   76   in
   77     case Posix.Process.fork () of
   78       SOME pid => server ()
   79     | NONE => client ();
   80      OS.Process.success
   81   end
   82 end
   83 (* Warning use this at the interactive toplevel has unexpected effects. *)
   84 val _ = SMLofNJ.exportFn("echo",Test.main);
