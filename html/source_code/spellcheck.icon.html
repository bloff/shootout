    5 procedure main(argv)
    6     local dict
    7     n := argv[1] | 1
    8     dict := table(0)
    9     f := open("Usr.Dict.Words") | {
   10     write(&errout, "Can't open \"Usr.Dict.Words\"")
   11     fail
   12     }
   13     while line := read(f) do line ? {
   14     dict[line] := 1
   15     }
   16     close(f)
   17     while line := read() do line ? {
   18     if (not member(dict,line)) then {
   19         write(line)
   20     }
   21     }
   22 end
