1 #!/usr/bin/mzscheme -r
    2 
    3 (define PORT 8888)
    4 (define DATA "Hello there sailor\n")
    5 (define n 10)
    6 
    7 (define (server)
    8   (thread client)
    9   (let-values ([(in out) (tcp-accept (tcp-listen PORT))])
   10     (let loop ([i (read in)]
   11                [bytes 0])
   12       (if (not (eof-object? i))
   13           (begin
   14             (write i out)
   15             (loop (read in)
   16                   (+ bytes (string-length i))))
   17           (begin
   18             (display "server processed ")
   19             (display bytes)
   20             (display " bytes\n"))))))
   21 
   22 (define (client)
   23   (let-values ([(in out) (tcp-connect "127.0.0.1" PORT)])
   24     (let loop ([n n])
   25       (if (> n 0)
   26           (begin
   27             (write DATA out)
   28             (let ([i (read in)])
   29               (begin
   30                 (if (equal? DATA i)
   31                     (loop (- n 1))
   32                     'error))))
   33           (close-output-port out)))))
   34 
   35 (define (main args)
   36   (set! n
   37     (if (= (vector-length args) 0)
   38         1
   39         (string->number (vector-ref  args 0))))
   40   (server))
   41 
   42 (main (current-command-line-arguments))
