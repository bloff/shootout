    5 (module moments
    6     (extern (macro printf::int (string double) "printf"))
    7     (main main))
    8 
    9 (define (main args)
   10   (let* ((sum 0.0)
   11      (numlist (let loop ((line (read-line)) (numlist '()))
   12             (cond ((eof-object? line) numlist)
   13               (else
   14                (let ((num (string->real line)))
   15                  (set! sum (+ num sum))
   16                  (loop (read-line) (cons num numlist))))))))
   17     (let ((n (length numlist)))
   18       (let ((mean (/ sum n))
   19         (average_deviation 0.0)
   20         (standard_deviation 0.0)
   21         (variance 0.0)
   22         (skew 0.0)
   23         (kurtosis 0.0)
   24         (median 0.0)
   25         (deviation 0.0))
   26     (let loop ((nums numlist))
   27          (if (not (null? nums))
   28          (begin
   29            (set! deviation (- (car nums) mean))
   30            (set! average_deviation (+ average_deviation (abs deviation)))
   31            (set! variance (+ variance (expt deviation 2.0)))
   32            (set! skew (+ skew (expt deviation 3.0)))
   33            (set! kurtosis (+ kurtosis (expt deviation 4)))
   34            (loop (cdr nums)))))
   35 
   36     (set! average_deviation (/ average_deviation (fixnum->flonum n)))
   37     (set! variance (/ variance (- n 1)))
   38     (set! standard_deviation (sqrt variance))
   39 
   40     (cond ((> variance 0.0)
   41            (set! skew (/ skew (* n variance standard_deviation)))
   42            (set! kurtosis (- (/ kurtosis (* n variance variance))
   43                  3.0))))
   44 
   45     (set! numlist (sort numlist (lambda (x y) (<fl x y))))
   46 
   47     (let ((mid (/ n 2)))
   48       (if (zero? (modulo n 2))
   49           (set! median (/ (+ (car (list-tail numlist mid))
   50                  (car (list-tail numlist (- mid 1))))
   51                   2.0))
   52         (set! median (car (list-tail numlist mid)))))
   53 
   54 
   55     (set! standard_deviation (/ (round (* standard_deviation 1000000))
   56   1000000))
   57 
   58     (print "n:                  " n)
   59     (printf #"median:             %f\n" median)
   60     (printf #"mean:               %f\n" mean)
   61     (printf #"average_deviation:  %f\n" average_deviation)
   62     (printf #"standard_deviation: %f\n" standard_deviation)
   63     (printf #"variance:           %f\n" variance)
   64     (printf #"skew:               %f\n" skew)
   65     (printf #"kurtosis:           %f\n" kurtosis)))))
