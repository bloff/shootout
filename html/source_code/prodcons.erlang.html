    6 %% Erlang by-design uses soft-real-time lightweight threads. 
    7 %% Message send/receive gives synchronization & data access. 
    8 %% 
    9 %% Usage: start from command line with 
   10 %%     erlc prodcons.erl 
   11 %%     erl -noinput -s prodcons main 100000 
   12  
   13 -module(prodcons).  
   14 -export([main/1, producer/3, consumer/4]).  
   15  
   16  
   17 producer(0, NP, Main) -> Main ! {self(), NP}; 
   18 producer(N, NP, Main) ->  
   19     receive                                           %% receive buffer
   20     empty 
   21         {From, next} ->  
   22             From ! {data, NP},                        %% signal buffer
   23                     full 
   24             producer(N-1, NP+1, Main) 
   25     end. 
   26  
   27  
   28 consumer(0, NC, _, Main) -> Main ! {self(), NC}; 
   29 consumer(N, NC, P, Main) -> 
   30     P ! {self(), next},                               %% signal buffer
   31          empty 
   32     receive                                           %% receive buffer
   33     full 
   34         {data, Value} -> consumer(N-1, NC+1, P, Main) %% empty buffer 
   35     end. 
   36  
   37  
   38 main([Arg]) -> 
   39     N = list_to_integer(atom_to_list(Arg)), 
   40     P = spawn(prodcon, producer, [N, 0, self()]),     %% spawn producer
   41         thread 
   42     C = spawn(prodcon, consumer, [N, 0, P, self()]),  %% spawn consumer
   43         thread 
   44     receive {P, NP} -> ok end,                        %% wait on producer
   45              thread 
   46     receive {C, NC} -> ok end,                        %% wait on consumer
   47              thread 
   48     io:format("~s ~w ~s ~w~n", ["Produced:", NP, "Consumed:", NC]), 
   49     halt(0).
