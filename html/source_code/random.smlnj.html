    6 structure Test : sig
    7     val main : (string * string list) -> OS.Process.status
    8 end = struct
    9 
   10 val im : int = 139968;
   11 val ia : int = 3877;
   12 val ic : int = 29573;
   13 
   14 fun randloop 0 seed rand max = rand
   15   | randloop n seed rand max =
   16     let
   17     val newseed : int = (seed * ia + ic) mod im;
   18     val newrand : real = max * (Real.fromInt newseed) / (Real.fromInt im);
   19     in
   20         randloop (n-1) newseed newrand max
   21     end;
   22 
   23 
   24 fun atoi s = case Int.fromString s of SOME num => num | NONE => 0;
   25 
   26 fun main(name, args) =
   27   let
   28     val arg = hd(args @ ["1"]);
   29     val num = atoi arg;
   30     val result = randloop num 42 0.0 100.0;
   31   in
   32       print (Real.fmt (StringCvt.FIX (SOME 9)) result); print "\n";
   33       OS.Process.success
   34   end
   35 end
   36 
   37 val _ = SMLofNJ.exportFn("random", Test.main);
