    5 (module sumcol
    6     (main main)
    7     (option (set! *genericity* #f)))
    8 
    9 (define (main args)
   10   (let ((sum 0))
   11     (let loop ((line (read-line)))
   12       (cond ((eof-object? line) #f)
   13         (else
   14          (set! sum (+ sum (string->integer line)))
   15          (loop (read-line)))))
   16     (display sum) (newline)))
