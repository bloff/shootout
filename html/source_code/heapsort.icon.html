    5 $define IM 139968
    6 $define IA 3877
    7 $define IC 29573
    8 
    9 procedure gen_random(max)
   10     static LAST; initial LAST := 42;
   11     repeat { suspend((max * (LAST := (LAST * IA + IC) % IM)) / IM) }
   12 end
   13 
   14 procedure heapsort(n, ra)
   15     local l, j, ir, i, rra
   16 
   17     l := ishift(n, -1) + 1
   18     ir := n
   19     repeat {
   20         if (l > 1) then {
   21         l := l - 1
   22             rra := ra[l]
   23         } else {
   24             rra := ra[ir]
   25             ra[ir] := ra[1]
   26         ir := ir - 1
   27             if (ir == 1) then {
   28                 ra[1] := rra
   29                 return
   30             }
   31         }
   32         i := l
   33         j := ishift(l, 1)
   34         while (j <= ir) do {
   35             if ((j < ir) & (ra[j] < ra[j+1])) then {
   36         j := j + 1
   37         }
   38             if (rra < ra[j]) then {
   39                 ra[i] := ra[j]
   40         i := j
   41                 j := j + i
   42             } else {
   43                 j := ir + 1
   44             }
   45         }
   46         ra[i] := rra
   47     }
   48 end
   49 
   50 procedure main(argv)
   51     n := argv[1] | 1
   52     ary := list(n)
   53     every i := 1 to n do ary[i] := gen_random(1.0)
   54     heapsort(n, ary)
   55     write(ary[n])
   56 end
