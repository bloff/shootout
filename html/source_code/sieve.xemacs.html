    5 (defun main ()
    6   (let* ((arg (car command-line-args-left))
    7      (num (if arg (string-to-number arg) 1))
    8      (count 0))
    9     (dotimes (iter num)
   10       (let ((flags (make-vector 8192 t)))
   11     (setq count 0)
   12     (do ((i 2 (1+ i)))
   13         ((>= i 8192))
   14       (if (aref flags i)
   15           (progn
   16         (do ((k (+ i i) (+ k i)))
   17             ((>= k 8192))
   18           (aset flags k nil))
   19         (setq count (1+ count)))))))
   20     (princ (format "Count: %d\n" count))))
