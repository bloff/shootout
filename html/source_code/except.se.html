   12 
   13 class
   14     EXCEPT
   15 
   16 inherit
   17     ARGUMENTS
   18 creation
   19 
   20     make
   21 
   22 feature -- Initialization
   23 
   24     high: INTEGER;
   25     low: INTEGER;
   26 
   27     high_exception: STRING is "high_exception";
   28     low_exception: STRING is "low_exception";
   29 
   30     run_n, outer_i: INTEGER;
   31 
   32 
   33     exc: EXCEPTIONS;
   34 
   35     make is
   36         do
   37             create exc;
   38             if argument_count /= 1 then
   39                 io.put_string("exception_test <integer>%N");
   40                 exc.die(1);
   41             end;
   42             if argument(1).is_integer then
   43                 run_n := argument(1).to_integer;
   44             else
   45                 io.put_string("Argument wasn't an integer, giving up%N");
   46                 exc.die(1);
   47             end;
   48             outer_i := 1;
   49             some_fun(run_n);
   50         end;
   51 
   52     some_fun(n: INTEGER) is
   53         do
   54             from
   55             invariant
   56                 i_in_bounds: outer_i >= 1 and then outer_i <= n + 1;
   57             variant
   58                 really_decreasing: n - outer_i + 1
   59             until outer_i > run_n
   60             loop
   61                 high_fun(outer_i);
   62                 -- an exception should be raised somwehere below
   63                 -- `high_fun' in the call chain
   64                 exc.raise("should not come here%N")
   65                 -- outer_i := outer_i + 1
   66             end;
   67             io.put_string("Exceptions: HI=");
   68             io.put_integer(high);
   69             io.put_string(" / LO=");
   70             io.put_integer(low);
   71             io.put_character('%N');
   72         rescue
   73             outer_i := outer_i + 1;
   74             retry;
   75         end;
   76 
   77 
   78     high_fun (i: INTEGER) is
   79         do
   80             low_fun(i);
   81         rescue
   82             if exc.developer_exception_name.is_equal(high_exception)
   83 then
   84                 high := high + 1;
   85             end
   86         end;
   87 
   88 
   89     low_fun (i: INTEGER) is
   90         do
   91             -- exc.catch(exc.Developer_exception);
   92             blow_up(i);
   93         rescue
   94             if exc.developer_exception_name.is_equal(low_exception) then
   95                 low := low + 1;
   96             end
   97         end;
   98 
   99 
  100     blow_up (i : INTEGER) is
  101         do
  102             if ((i \\ 2) = 0) then
  103                 exc.raise(low_exception);
  104             else
  105                 exc.raise(high_exception);
  106             end;
  107         end;
  108 
  109 end -- class EXCEPT
