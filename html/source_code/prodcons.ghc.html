    5 module Main where
    6 
    7 import CVar
    8 import MVar
    9 import Concurrent
   10 import Control.Exception
   11 import GHC.IOBase
   12 import System
   13 
   14 producer :: Int -> IORef Int -> CVar Int -> IO ()
   15 producer n p ch = sequence_ (map send [1..n])
   16   where send i = do writeCVar ch i
   17                     prod <- readIORef p
   18                     writeIORef p (prod+1)
   19 
   20 consumer :: Int -> IORef Int -> CVar Int -> IO ()
   21 consumer n c ch = cons 1
   22   where cons i | n <= i = return ()
   23         cons i
   24             = do i <- readCVar ch
   25                  con <- readIORef c
   26                  writeIORef c (con+1)
   27                  cons i
   28 
   29 myForkIO :: IO () -> IO (MVar ())
   30 myForkIO io = do
   31    mvar <- newEmptyMVar
   32    forkIO (io `finally` putMVar mvar ())
   33    return mvar
   34 
   35 join :: MVar () -> IO ()
   36 join mvar = readMVar mvar
   37 
   38 main = do (a:_) <- getArgs
   39           let n = read a
   40           produced <- newIORef 0
   41           consumed <- newIORef 0
   42           channel <- newCVar
   43           p <- myForkIO (producer n produced channel)
   44           c <- myForkIO (consumer n consumed channel)
   45           join p; join c
   46           prod <- readIORef produced
   47           cons <- readIORef consumed
   48           putStrLn (show prod ++ " " ++ show cons)
